18  import java.util.Set;

20  import org.kuali.kfs.sys.document.authorization.FinancialSystemMaintenanceDocumentPresentationControllerBase;
21  import org.kuali.kfs.vnd.VendorPropertyConstants;
22  import org.kuali.kfs.vnd.businessobject.VendorDetail;
23  import org.kuali.rice.kns.bo.BusinessObject;
24  import org.kuali.rice.kns.document.MaintenanceDocument;
25  import org.kuali.rice.kns.util.ObjectUtils;

27  public class VendorDocumentPresentationController extends FinancialSystemMaintenanceDocumentPresentationControllerBase {
28  
29      @Override
30      public Set<String> getConditionallyReadOnlySectionIds(MaintenanceDocument document) {
31          Set<String> conditionallyReadOnlySectionIds = super.getConditionallyReadOnlySectionIds(document);
32          VendorDetail vendor = (VendorDetail)document.getNewMaintainableObject().getBusinessObject();
33  
34          if (!vendor.isVendorParentIndicator()) {
35              // make some sections read only, e.g. supplier diversity cause they're on the header
36              conditionallyReadOnlySectionIds.add(VendorPropertyConstants.VENDOR_SUPPLIER_DIVERSITIES);
37          }
38  
39          return conditionallyReadOnlySectionIds;
40      }
41  
42      @Override
43      public Set<String> getConditionallyReadOnlyPropertyNames(MaintenanceDocument document) {
44          Set<String> conditionallyReadonlyPropertyNames = super.getConditionallyReadOnlyPropertyNames(document);
45          VendorDetail vendor = (VendorDetail)document.getNewMaintainableObject().getBusinessObject();
46  
47          if (vendor.isVendorParentIndicator()) {
48              // Vendor Parent Indicator should be readOnly if the vendor is a parent.
49              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_PARENT_INDICATOR);
50  
51              // For existing vendors, don't allow vendor type code to be changed if maint table indicates it shouldn't be changed 
52              if (ObjectUtils.isNotNull(vendor.getVendorHeaderGeneratedIdentifier()) && 
53                      !vendor.getVendorHeader().getVendorType().isVendorTypeChangeAllowedIndicator()) {
54                  conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_TYPE_CODE);
55              }
56          }
57  
58          // If the vendor is not a parent, there are certain fields that should be readOnly
59          else {
60              // All the fields in VendorHeader should be readOnly if the vendor is not a parent.
61              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_TYPE_CODE);
62              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_TAX_NUMBER);
63              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_TAX_TYPE_CODE);
64              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_OWNERSHIP_CODE);
65              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_OWNERSHIP_CATEGORY_CODE);
66              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_FEDERAL_WITHOLDING_TAX_BEGINNING_DATE);
67              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_FEDERAL_WITHOLDING_TAX_END_DATE);
68              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_W9_RECEIVED_INDICATOR);
69              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_W8_BEN_RECEIVED_INDICATOR);
70              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_DEBARRED_INDICATOR);
71              conditionallyReadonlyPropertyNames.add(VendorPropertyConstants.VENDOR_FOREIGN_INDICATOR);
72          }
73  
74          return conditionallyReadonlyPropertyNames;
75      }
76      
77      @Override
78      public Set<String> getConditionallyHiddenPropertyNames(BusinessObject businessObject) {
79          Set<String> conditionallyHiddenPropertyNames = super.getConditionallyHiddenPropertyNames(businessObject);
80          MaintenanceDocument document = (MaintenanceDocument) businessObject;
81          VendorDetail vendor = (VendorDetail)document.getNewMaintainableObject().getBusinessObject();
82          // If the vendor is a parent then the vendor parent name should be hidden.
83          if (vendor.isVendorParentIndicator()) {
84              conditionallyHiddenPropertyNames.add(VendorPropertyConstants.VENDOR_PARENT_NAME);
85          }
86          
87          return conditionallyHiddenPropertyNames;
88      }
