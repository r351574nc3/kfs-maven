75      /**
76       * Attributes: Chart Code Account Number Requirements: - KFS-COA Account Supervisor: CA_ACCOUNT_T.ACCT_SPVSR_UNVL_ID - KFS-SYS
77       * Fiscal Officer: CA_ACCOUNT_T.ACCT_FSC_OFC_UID - KFS-SYS Fiscal Officer Primary Delegate: CA_ACCT_DELEGATE_T.ACCT_DLGT_UNVL_ID
78       * where ACCT_DLGT_PRMRT_CD = Y - KFS-SYS Fiscal Officer Secondary Delegate: CA_ACCT_DELEGATE_T.ACCT_DLGT_UNVL_ID where
79       * ACCT_DLGT_PRMRT_CD = N - KFS-SYS Award Project Director: use the getProjectDirectorForAccount(String chartOfAccountsCode,
80       * String accountNumber) method on the contracts and grants module service
81       * 
82       * @see org.kuali.rice.kim.service.support.impl.KimRoleTypeServiceBase#getPrincipalIdsFromApplicationRole(java.lang.String,
83       *      java.lang.String, org.kuali.rice.kim.bo.types.dto.AttributeSet)
84       */
85      @Override
86      public List<RoleMembershipInfo> getRoleMembersFromApplicationRole(String namespaceCode, String roleName, AttributeSet qualification) {
87          // validate received attributes
88          validateRequiredAttributesAgainstReceived(qualification);
89          List<RoleMembershipInfo> members = new ArrayList<RoleMembershipInfo>();
90          if(qualification!=null && !qualification.isEmpty()){
91              String chartOfAccountsCode = qualification.get(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE);
92              String accountNumber = qualification.get(KfsKimAttributes.ACCOUNT_NUMBER);
93      
94              String financialSystemDocumentTypeCodeCode = qualification.get(KfsKimAttributes.DOCUMENT_TYPE_NAME);
95              String totalDollarAmount = qualification.get(KfsKimAttributes.FINANCIAL_DOCUMENT_TOTAL_AMOUNT);
96      
97              String fiscalOfficerPrincipalID = qualification.get(KFSPropertyConstants.ACCOUNT_FISCAL_OFFICER_SYSTEM_IDENTIFIER);
98              String accountSupervisorPrincipalID = qualification.get(KFSPropertyConstants.ACCOUNTS_SUPERVISORY_SYSTEMS_IDENTIFIER);
99              String principalId = qualification.get(KimAttributes.PRINCIPAL_ID);
100     
101             // Default to 0 total amount
102             if (StringUtils.isEmpty(totalDollarAmount)) {
103                 totalDollarAmount = getDefaultTotalAmount();
104             }
105     
106             AttributeSet roleQualifier = new AttributeSet();
107             roleQualifier.put(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE, chartOfAccountsCode);
108             roleQualifier.put(KfsKimAttributes.ACCOUNT_NUMBER, accountNumber);
109     
110             if (chartOfAccountsCode == null && accountNumber == null && StringUtils.isNotBlank(principalId)) {
111                 RoleMembershipInfo roleMembershipInfo = getRoleMembershipInfoWhenAccountInfoUnavailable(roleName, principalId, roleQualifier);
112                 if(ObjectUtils.isNotNull(roleMembershipInfo)) {
113                     members.add(roleMembershipInfo);
114                 }
115             }
116     
117             if (KFSConstants.SysKimConstants.ACCOUNT_SUPERVISOR_KIM_ROLE_NAME.equals(roleName)) {
118                 Account account = getAccount(chartOfAccountsCode, accountNumber);
119                 if (account != null) {
120                     members.add(new RoleMembershipInfo(null, null, account.getAccountsSupervisorySystemsIdentifier(), Role.PRINCIPAL_MEMBER_TYPE, roleQualifier));
121                 }
122                 // only add the additional approver if they are different
123                 if (StringUtils.isNotBlank(accountSupervisorPrincipalID) && (account == null || !StringUtils.equals(accountSupervisorPrincipalID, account.getAccountsSupervisorySystemsIdentifier()))) {
124                     members.add(new RoleMembershipInfo(null, null, accountSupervisorPrincipalID, Role.PRINCIPAL_MEMBER_TYPE, roleQualifier));
125                 }
126             }
127             else if (KFSConstants.SysKimConstants.FISCAL_OFFICER_KIM_ROLE_NAME.equals(roleName)) {
128                 Account account = getAccount(chartOfAccountsCode, accountNumber);
129                 if (account != null) {
130                     members.add(new RoleMembershipInfo(null, null, account.getAccountFiscalOfficerSystemIdentifier(), Role.PRINCIPAL_MEMBER_TYPE, roleQualifier));
131                 }
132                 // only add the additional approver if they are different
133                 if (StringUtils.isNotBlank(fiscalOfficerPrincipalID) && (account == null || !StringUtils.equals(fiscalOfficerPrincipalID, account.getAccountFiscalOfficerSystemIdentifier()))) {
134                     members.add(new RoleMembershipInfo(null, null, fiscalOfficerPrincipalID, Role.PRINCIPAL_MEMBER_TYPE, roleQualifier));
135                 }
136             }
137             else if (KFSConstants.SysKimConstants.FISCAL_OFFICER_PRIMARY_DELEGATE_KIM_ROLE_NAME.equals(roleName)) {
138                 AccountDelegate delegate = getPrimaryDelegate(chartOfAccountsCode, accountNumber, financialSystemDocumentTypeCodeCode, totalDollarAmount);
139                 if (delegate != null) {
140                     roleQualifier.put(KfsKimAttributes.FINANCIAL_SYSTEM_DOCUMENT_TYPE_CODE, delegate.getFinancialDocumentTypeCode());
141                     roleQualifier.put(KfsKimAttributes.FROM_AMOUNT, (delegate.getFinDocApprovalFromThisAmt() == null) ? "0" : delegate.getFinDocApprovalFromThisAmt().toString());
142                     roleQualifier.put(KfsKimAttributes.TO_AMOUNT, (delegate.getFinDocApprovalToThisAmount() == null) ? "NOLIMIT" : delegate.getFinDocApprovalToThisAmount().toString());
143                     members.add(new RoleMembershipInfo(null, null, delegate.getAccountDelegateSystemId(), Role.PRINCIPAL_MEMBER_TYPE, roleQualifier));
144                 }
145             }
146             else if (KFSConstants.SysKimConstants.FISCAL_OFFICER_SECONDARY_DELEGATE_KIM_ROLE_NAME.equals(roleName)) {
147                 List<AccountDelegate> delegates = getSecondaryDelegates(chartOfAccountsCode, accountNumber, financialSystemDocumentTypeCodeCode, totalDollarAmount);
148                 for (AccountDelegate delegate : delegates) {
149                     roleQualifier.put(KfsKimAttributes.FINANCIAL_SYSTEM_DOCUMENT_TYPE_CODE, delegate.getFinancialDocumentTypeCode());
150                     roleQualifier.put(KfsKimAttributes.FROM_AMOUNT, (delegate.getFinDocApprovalFromThisAmt() == null) ? "0" : delegate.getFinDocApprovalFromThisAmt().toString());
151                     roleQualifier.put(KfsKimAttributes.TO_AMOUNT, (delegate.getFinDocApprovalToThisAmount() == null) ? "NOLIMIT" : delegate.getFinDocApprovalToThisAmount().toString());
152                     members.add(new RoleMembershipInfo(null, null, delegate.getAccountDelegateSystemId(), Role.PRINCIPAL_MEMBER_TYPE, roleQualifier));
153                 }
154             }
155             else if (KFSConstants.SysKimConstants.AWARD_SECONDARY_DIRECTOR_KIM_ROLE_NAME.equals(roleName)) {
156                 Person person = getProjectDirectorForAccount(chartOfAccountsCode, accountNumber);
157                 if (person != null) {
158                     members.add(new RoleMembershipInfo(null, null, person.getPrincipalId(), Role.PRINCIPAL_MEMBER_TYPE, roleQualifier));
159                 }
160             }
161         }
162         return members;
163     }

165     // build memebership information based on the given role name for the given principal when account information is unavailable
166     protected RoleMembershipInfo getRoleMembershipInfoWhenAccountInfoUnavailable(String roleName, String principalId, AttributeSet roleQualifier) {
167         BusinessObjectService businessObjectService = SpringContext.getBean(BusinessObjectService.class);
168         Map<String, Object> fieldValues = new HashMap<String, Object>();
169         RoleMembershipInfo roleMembershipInfo = null;
170 
171         if ((KFSConstants.SysKimConstants.FISCAL_OFFICER_KIM_ROLE_NAME.equals(roleName))) {
172             fieldValues.put(KFSPropertyConstants.ACCOUNT_FISCAL_OFFICER_SYSTEM_IDENTIFIER, principalId);
173 
174             if (businessObjectService.countMatching(Account.class, fieldValues) > 0) {
175                 roleMembershipInfo = new RoleMembershipInfo(null, null, principalId, Role.PRINCIPAL_MEMBER_TYPE, roleQualifier);
176             }
177         }
178         else if (KFSConstants.SysKimConstants.FISCAL_OFFICER_PRIMARY_DELEGATE_KIM_ROLE_NAME.equals(roleName)) {
179             fieldValues.put(KFSPropertyConstants.ACCOUNT_DELEGATE_SYSTEM_ID, principalId);
180             fieldValues.put(KFSPropertyConstants.ACCOUNTS_DELEGATE_PRMRT_INDICATOR, Boolean.TRUE);
181 
182             if (businessObjectService.countMatching(AccountDelegate.class, fieldValues) > 0) {
183                 roleMembershipInfo = new RoleMembershipInfo(null, null, principalId, Role.PRINCIPAL_MEMBER_TYPE, roleQualifier);
184             }
185         }
186         else if (KFSConstants.SysKimConstants.FISCAL_OFFICER_SECONDARY_DELEGATE_KIM_ROLE_NAME.equals(roleName)) {
187             fieldValues.put(KFSPropertyConstants.ACCOUNT_DELEGATE_SYSTEM_ID, principalId);
188             fieldValues.put(KFSPropertyConstants.ACCOUNTS_DELEGATE_PRMRT_INDICATOR, Boolean.FALSE);
189 
190             if (businessObjectService.countMatching(AccountDelegate.class, fieldValues) > 0) {
191                 roleMembershipInfo = new RoleMembershipInfo(null, null, principalId, Role.PRINCIPAL_MEMBER_TYPE, roleQualifier);
192             }
193         }
194         
195         return roleMembershipInfo;
196     }

316     @Override
317     public void principalInactivated(String principalId, String namespaceCode, String roleName) {
318         super.principalInactivated(principalId, namespaceCode, roleName);
319         if (KFSConstants.SysKimConstants.ACCOUNT_SUPERVISOR_KIM_ROLE_NAME.equals(roleName)) {
320             if (getAccountService().isPrincipalInAnyWayShapeOrFormAccountSupervisor(principalId)) {
321                 handleAccountSupervisorInactivation(principalId, namespaceCode, roleName);
322             }
323         }
324         else if (KFSConstants.SysKimConstants.FISCAL_OFFICER_KIM_ROLE_NAME.equals(roleName)) {
325             if (getAccountService().isPrincipalInAnyWayShapeOrFormFiscalOfficer(principalId)) {
326                 handleFiscalOfficerInactivation(principalId, namespaceCode, roleName);
327             }
328         }
329         else if (KFSConstants.SysKimConstants.FISCAL_OFFICER_PRIMARY_DELEGATE_KIM_ROLE_NAME.equals(roleName)) {
330             if (getAccountDelegateService().isPrincipalInAnyWayShapeOrFormPrimaryAccountDelegate(principalId)) {
331                 handlePrimaryAccountDelegateInactivation(principalId, namespaceCode, roleName);
332             }
333         }
334         else if (KFSConstants.SysKimConstants.FISCAL_OFFICER_SECONDARY_DELEGATE_KIM_ROLE_NAME.equals(roleName)) {
335             if (getAccountDelegateService().isPrincipalInAnyWayShapeOrFormSecondaryAccountDelegate(principalId)) {
336                 handleSecondaryAccountDelegateInactivation(principalId, namespaceCode, roleName);
337             }
338         }
339     }

341     /**
342      * Notifies the specified list that the given principal is being inactivated, listing any active or expired accounts the principal is fiscal officer of
343      * @param principalId the principal id of the inactivating person
344      * @param roleNamespace the namespace of the role the inactivated principal is leaving
345      * @param roleName the name of the role the inactivated principal is leaving
346      */
347     protected void handleFiscalOfficerInactivation(String principalId, String roleNamespace, String roleName) {
348         // build the message
349         final KimPrincipalInfo principal = getIdentityManagementService().getPrincipal(principalId);
350         
351         Iterator<Account> activeAccounts = getAccountService().getActiveAccountsForFiscalOfficer(principalId);
352         final String joinedActiveAccounts = joinAccounts(activeAccounts);
353         Iterator<Account> inactiveAccounts = getAccountService().getExpiredAccountsForFiscalOfficer(principalId);
354         final String joinedExpiredAccounts = joinAccounts(inactiveAccounts);
355         
356         final String message = getMessage(KFSKeyConstants.MESSAGE_ACCOUNT_DERIVED_ROLE_PRINCIPAL_INACTIVATED_FISCAL_OFFICER_NOTIFICATION, roleNamespace, roleName, principal.getPrincipalName(), joinedActiveAccounts, joinedExpiredAccounts);
357         
358         // get listserv to mail to
359         String toAddress = getDeactivationInterestAddress();
360         try {
361             
362             // get mail message
363             MailMessage mailMessage = buildMailMessage(toAddress, message, roleNamespace, roleName, principalId);
364             // send it
365             getMailService().sendMessage(mailMessage);
366         }
367         catch (InvalidAddressException iae) {
368             throw new RuntimeException("Could not mail principal inactivation notification to "+toAddress);
369         }
370         
371     }

424     /**
425      * Handles the inactivation of account delegates when their principal is inactivated
426      * @param principalId the principal id of the inactivating person
427      * @param roleNamespace the namespace of the role the inactivated principal is leaving
428      * @param roleName the name of the role the inactivated principal is leaving
429      * @param primary whether primary delegates should be inactivated, rather than secondary
430      */
431     protected void handleAccountDelegateInactivation(String principalId, String roleNamespace, String roleName, boolean primary) {
432         List<String> accountDisdelegations = new ArrayList<String>();
433         List<String> blockedAccountDisdelegations = new ArrayList<String>();
434         
435         Iterator<AccountDelegate> accountDelegationsToInactivate = getAccountDelegateService().retrieveAllActiveDelegationsForPerson(principalId, primary);
436         while (accountDelegationsToInactivate.hasNext()) {
437             final AccountDelegate accountDelegate = accountDelegationsToInactivate.next();
438             final String blockingDocumentId = lookupDocumentBlockingInactivation(accountDelegate);
439             if (StringUtils.isBlank(blockingDocumentId)) {
440                 accountDisdelegations.add(getAccountDisdelegationInformationForAccountDelegate(accountDelegate));
441                 inactivateDelegation(accountDelegate);
442             } else {
443                 blockedAccountDisdelegations.add(getBlockedAccountDisdelegationInformationForAccountDelegate(accountDelegate, blockingDocumentId));
444             }            
445         }
446         notifyListAboutAccountDelegateInactivations(principalId, roleNamespace, roleName, accountDisdelegations, blockedAccountDisdelegations, primary);
447     }

531      * @return a String of account information
532      */
533     protected String joinAccounts(Iterator<Account> accounts) {
534         StringBuilder s = new StringBuilder();
535         int count = 0;
536         
537         while (accounts.hasNext()) {
538             final Account account = accounts.next();
539             if (count > 0) {
540                 s.append('\n');
541             }
542             s.append(account.getChartOfAccountsCode()+"-"+account.getAccountNumber());
543             count += 1;
544         }
545         return s.toString();
