38      private static final Logger LOG = Logger.getLogger(AccountingOrganizationHierarchyReviewRoleTypeServiceImpl.class);
39      /**
40       * Create role type service - org.kuali.kfs.coa.identity.AccountingOrganizationHierarchyReviewRoleTypeService for
41       * KFS-COA/"Organization: Always Hierarchical, Document Type & Accounting" Attributes: Chart Code (required) Organization Code
42       * Document Type Name From Amount To Amount Override Code - total amount will be passed in as qualification and role type
43       * service will need to compare it to from and to amount qualifier values for assignees Requirements: - Traverse the org
44       * hierarchy but not the document type hierarchy - from amount must be null or <= total amount supplied / to amount must be null
45       * or >= total amount supplied, and null override code on assignment matches all override codes
46       * 
47       * @see org.kuali.kfs.coa.identity.OrganizationOptionalHierarchyRoleTypeServiceImpl#performMatch(org.kuali.rice.kim.bo.types.dto.AttributeSet,
48       *      org.kuali.rice.kim.bo.types.dto.AttributeSet)
49       */
50      public boolean performMatch(AttributeSet qualification, AttributeSet roleQualifier) {
51          return doesOverrideCodeMatch(qualification, roleQualifier) 
52                  && isValidTotalAmount(qualification, roleQualifier) 
53                  && super.performMatch(qualification, roleQualifier);
54      }

56      protected boolean doesOverrideCodeMatch(AttributeSet qualification, AttributeSet roleQualifier) {
57          return qualification==null || roleQualifier==null || StringUtils.isBlank(qualification.get(KfsKimAttributes.ACCOUNTING_LINE_OVERRIDE_CODE)) 
58                  || StringUtils.isBlank(roleQualifier.get(KfsKimAttributes.ACCOUNTING_LINE_OVERRIDE_CODE)) 
59                  || qualification.get(KfsKimAttributes.ACCOUNTING_LINE_OVERRIDE_CODE).equals(roleQualifier.get(KfsKimAttributes.ACCOUNTING_LINE_OVERRIDE_CODE));
60      }
61  
62      protected boolean isValidTotalAmount(AttributeSet qualification, AttributeSet roleQualifier) {
63          boolean isValidTotalAmount = false;
64          if(qualification==null || roleQualifier==null) {
65              return false;
66          }
67          String totalAmountStr = qualification.get(KfsKimAttributes.FINANCIAL_DOCUMENT_TOTAL_AMOUNT);
68          if ( StringUtils.isBlank(totalAmountStr) ) {
69              return false;
70          }
71          try {
72              KualiDecimal totalAmount = new KualiDecimal(totalAmountStr);
73              String toAmountStr = roleQualifier.get(KfsKimAttributes.TO_AMOUNT);
74              String fromAmountStr = roleQualifier.get(KfsKimAttributes.FROM_AMOUNT);
75              if ((StringUtils.isBlank(toAmountStr) || new KualiDecimal(toAmountStr).isGreaterEqual(totalAmount) ) 
76                      && (StringUtils.isBlank(fromAmountStr) || new KualiDecimal(fromAmountStr).isLessEqual(totalAmount) )) {
77                  isValidTotalAmount = true;
78              }
79          } catch (Exception ex) {
80              isValidTotalAmount = false;
81              LOG.error( "Exception comparing document amount to role qualifiers.", ex );
82          }
83          return isValidTotalAmount;
84      }

136             }
137             attributeErrors = null;
138         }

143     protected boolean isLesserNumber(String numberStr1, String numberStr2){
144         if(StringUtils.isBlank(numberStr1) ) {
145             numberStr1 = "0";
146         }
147         if(StringUtils.isBlank(numberStr2) ) {
148             numberStr2 = "0";
149         }
150         int number1 = KNSUtils.getIntegerValue(numberStr1);
151         int number2 = KNSUtils.getIntegerValue(numberStr2);
152         return number1 < number2;
153     }
