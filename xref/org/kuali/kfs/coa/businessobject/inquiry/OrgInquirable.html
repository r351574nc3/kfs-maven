36  /**
37   * This class adds in some new sections for {@link Org} inquiries, specifically Org Hierarchy Org Review Hierarchy
38   */
39  public class OrgInquirable extends KfsInquirableImpl {
40  
41      public void addAdditionalSections(List sections, BusinessObject bo) {
42          if (bo instanceof Organization) {
43              Organization org = (Organization) bo;
44  
45              List rows = new ArrayList();
46  
47              Field f = new Field();
48              f.setPropertyName("Organization Hierarchy");
49              f.setFieldLabel("Organization Hierarchy");
50              f.setPropertyValue(org.getOrganizationHierarchy());
51              f.setFieldType(Field.TEXT);
52              rows.add(new Row(f));
53  
54              f = new Field();
55              f.setPropertyName("Organization Review Hierarchy");
56              f.setFieldLabel("Organization Review Hierarchy");
57              f.setPropertyValue("run search");
58              f.setFieldType(Field.TEXT);
59              Properties parameters = new Properties();
60              parameters.put(OrgReviewRoleLookupableHelperServiceImpl.MEMBER_ATTRIBUTE_CHART_OF_ACCOUNTS_CODE, org.getChartOfAccountsCode());
61              parameters.put(OrgReviewRoleLookupableHelperServiceImpl.MEMBER_ATTRIBUTE_ORGANIZATION_CODE, org.getOrganizationCode());
62              parameters.put(KNSConstants.BUSINESS_OBJECT_CLASS_ATTRIBUTE, OrgReviewRole.class.getName());
63              parameters.put(KNSConstants.RETURN_LOCATION_PARAMETER, KNSConstants.PORTAL_ACTION);
64              parameters.put(KNSConstants.DOC_FORM_KEY, KimConstants.KimUIConstants.KIM_ROLE_DOCUMENT_SHORT_KEY);
65              String hrefStr = UrlFactory.parameterizeUrl(KNSConstants.LOOKUP_ACTION, parameters);
66              HtmlData hRef = new AnchorHtmlData(hrefStr, KNSConstants.EMPTY_STRING);
67              f.setInquiryURL(hRef);
68              rows.add(new Row(f));
69  
70              Section section = new Section();
71              section.setRows(rows);
72              section.setSectionTitle("Organization Hierarchy");
73              sections.add(section);
74          }
75      }
