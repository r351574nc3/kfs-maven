16  package org.kuali.kfs.coa.service;
17  
18  import java.util.List;
19  
20  import org.kuali.kfs.coa.businessobject.Organization;
21  
22  /**
23   * This interface defines methods that an Org Service must provide.
24   */
25  public interface OrganizationService {
26      /**
27       * This method retrieves an organization instance by its composite primary keys (parameters passed in).
28       * 
29       * @param chartOfAccountsCode
30       * @param organizationCode
31       * @return An Org instance.
32       */
33      public Organization getByPrimaryId(String chartOfAccountsCode, String organizationCode);
34  
35      /**
36       * Method is used by KualiOrgReviewAttribute to enable caching of orgs for routing.
37       * 
38       * @see org.kuali.kfs.coa.service.OrganizationService#getByPrimaryId(java.lang.String, java.lang.String)
39       */
40      public Organization getByPrimaryIdWithCaching(String chartOfAccountsCode, String organizationCode);
41  
42      /**
43       * Retrieves a List of Accounts that are active, and are tied to this Org. If there are no Accounts that meet this criteria, an
44       * empty list will be returned.
45       * 
46       * @param chartOfAccountsCode - chartCode for the Org you want Accounts for
47       * @param organizationCode - orgCode for the Org you want Accounts for
48       * @return A List of Accounts that are active, and tied to this Org
49       */
50      public List getActiveAccountsByOrg(String chartOfAccountsCode, String organizationCode);
51  
52      /**
53       * Retrieves a List of Orgs that are active, and that ReportTo this Org If there are no Orgs that meet this criteria, an empty
54       * list will be returned.
55       * 
56       * @param chartOfAccountsCode - chartCode for the Org you want Child Orgs for
57       * @param organizationCode - orgCode for the Org you want Child Orgs for
58       * @return A List of Orgs that are active, and report to this Org
59       */
60      public List getActiveChildOrgs(String chartOfAccountsCode, String organizationCode);
61      
62      /**
63       * Returns a list of active organizations with the given organization type code.
64       * 
65       * @param organizationTypeCode
66       * @return
67       */
68      public List<Organization> getActiveOrgsByType(String organizationTypeCode);
69  
70      
71      /**
72       * Returns a list of active financial processing organizations.
73       * 
74       * @return A List of Orgs that are active and financial processing.
75       */
76      public List<Organization> getActiveFinancialOrgs();
77      
78      /**
79       * returns the chart and organization of the ACTIVE root-level organization
80       */
81      public String[] getRootOrganizationCode();
82      
83      /**
84       * This method traverses the hierarchy to see if the organization represented by the potentialChildChartCode and potentialChildOrganizationCode 
85       * reports to the organization represented by the potentialParentChartCode and potentialParentOrganizationCode
86       * 
87       * @param potentialChildChartCode
88       * @param potentialChildOrganizationCode
89       * @param potentialParentChartCode
90       * @param potentialParentOrganizationCode
91       * @return boolean indicating whether the organization represented by the first two parameters reports to one represented by the last two parameters
92       */
93      public boolean isParentOrganization(String potentialChildChartCode, String potentialChildOrganizationCode, String potentialParentChartCode, String potentialParentOrganizationCode);
94  }
