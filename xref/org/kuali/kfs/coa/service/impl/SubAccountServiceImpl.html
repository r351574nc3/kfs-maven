27  /**
28   * This class is the service implementation for the SubAccount structure. This is the default implementation that gets delivered
29   * with Kuali.
30   */
31  
32  @NonTransactional
33  public class SubAccountServiceImpl implements SubAccountService {
34      private static final Logger LOG = Logger.getLogger(SubAccountServiceImpl.class);
35  
36      private SubAccountDao subAccountDao;
37  
38      /**
39       * @see org.kuali.kfs.coa.service.SubAccountService#getByPrimaryId(java.lang.String, java.lang.String, java.lang.String)
40       */
41      public SubAccount getByPrimaryId(String chartOfAccountsCode, String accountNumber, String subAccountNumber) {
42          return subAccountDao.getByPrimaryId(chartOfAccountsCode, accountNumber, subAccountNumber);
43      }
44  
45      /**
46       * Method is used by KualiAccountAttribute to enable caching of accounts for routing.
47       * 
48       * @see org.kuali.kfs.coa.service.impl.SubAccountServiceImpl#getByPrimaryId(String, String, String)
49       */
50      @Cached
51      public SubAccount getByPrimaryIdWithCaching(String chartOfAccountsCode, String accountNumber, String subAccountNumber) {
52          return subAccountDao.getByPrimaryId(chartOfAccountsCode, accountNumber, subAccountNumber);
53      }
54  
55      /**
56       * Retrieves SubAccount objects associated with the given chart-org-subAccount code combination
57       * 
58       * @param chartOfAccountsCode - 'Reports To' Chart of Accounts Code
59       * @param organizationCode - 'Reports To' Organization Code
60       * @param subAccountNumber - Sub Account Number
61       * @return a list of SubAccount objects
62       */
63      public List getSubAccountsByReportsToOrganization(String chartOfAccountsCode, String organizationCode, String subAccountNumber) {
64          return subAccountDao.getSubAccountsByReportsToOrganization(chartOfAccountsCode, organizationCode, subAccountNumber);
65      }
66  
67      /**
68       * @return SubAccountDao
69       */
70      public SubAccountDao getSubAccountDao() {
71          return subAccountDao;
72      }
73  
74      /**
75       * @param subAccountDao
76       */
77      public void setSubAccountDao(SubAccountDao subAccountDao) {
78          this.subAccountDao = subAccountDao;
79      }
