28  /**
29   * This class is the default implementation of the A21SubAccountService
30   */
31  
32  @NonTransactional
33  public class A21SubAccountServiceImpl implements A21SubAccountService {
34  
35      private A21SubAccountDao a21SubAccountDao;
36      private AccountService accountService;
37  
38      /**
39       * @see org.kuali.kfs.coa.service.A21SubAccountService#getByPrimaryKey(java.lang.String, java.lang.String, java.lang.String)
40       */
41      public A21SubAccount getByPrimaryKey(String chartOfAccountsCode, String accountNumber, String subAccountNumber) {
42          return a21SubAccountDao.getByPrimaryKey(chartOfAccountsCode, accountNumber, subAccountNumber);
43      }
44  
45      /**
46       * @see org.kuali.kfs.coa.service.A21SubAccountService#buildCgIcrAccount(java.lang.String, java.lang.String, java.lang.String,
47       *      java.lang.String)
48       */
49      public A21SubAccount buildCgIcrAccount(String chartOfAccountsCode, String accountNumber, String subAccountNumber, String subAccountTypeCode) {
50          if (StringUtils.isEmpty(chartOfAccountsCode) || StringUtils.isEmpty(accountNumber) || StringUtils.equals(subAccountTypeCode, KFSConstants.SubAccountType.COST_SHARE)) {
51              return null;
52          }
53  
54          A21SubAccount a21SubAccount = new A21SubAccount();
55          a21SubAccount.setSubAccountNumber(subAccountNumber);
56          a21SubAccount.setSubAccountTypeCode(subAccountTypeCode);
57  
58          this.populateCgIcrAccount(a21SubAccount, chartOfAccountsCode, accountNumber);
59  
60          return a21SubAccount;
61      }
62  
63      /**
64       * @see org.kuali.kfs.coa.service.A21SubAccountService#populateCgIcrAccount(org.kuali.kfs.coa.businessobject.A21SubAccount,
65       *      java.lang.String, java.lang.String)
66       */
67      public void populateCgIcrAccount(A21SubAccount a21SubAccount, String chartOfAccountsCode, String accountNumber) {
68          Account account = accountService.getByPrimaryIdWithCaching(chartOfAccountsCode, accountNumber);
69  
70          if (ObjectUtils.isNotNull(account) && ObjectUtils.isNotNull(a21SubAccount) && !StringUtils.equals(a21SubAccount.getSubAccountTypeCode(), KFSConstants.SubAccountType.COST_SHARE)) {
71              a21SubAccount.setChartOfAccountsCode(account.getChartOfAccountsCode());
72              a21SubAccount.setAccountNumber(account.getAccountNumber());
73              a21SubAccount.setFinancialIcrSeriesIdentifier(account.getFinancialIcrSeriesIdentifier());
74  
75              a21SubAccount.setIndirectCostRcvyFinCoaCode(account.getIndirectCostRcvyFinCoaCode());
76              a21SubAccount.setIndirectCostRecoveryAcctNbr(account.getIndirectCostRecoveryAcctNbr());
77              a21SubAccount.setIndirectCostRecoveryTypeCode(account.getAcctIndirectCostRcvyTypeCd());
78  
79              a21SubAccount.setOffCampusCode(account.isAccountOffCampusIndicator());
80          }
81      }
82  
83      /**
84       * @param subAccountDao The a21SubAccountDao to set.
85       */
86      public void setA21SubAccountDao(A21SubAccountDao subAccountDao) {
87          this.a21SubAccountDao = subAccountDao;
88      }

92       *

94       */
95      public void setAccountService(AccountService accountService) {
96          this.accountService = accountService;
97      }
