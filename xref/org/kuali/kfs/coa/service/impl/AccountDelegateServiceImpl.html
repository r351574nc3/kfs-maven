51      /**
52       * 
53       * @see org.kuali.kfs.coa.service.AccountDelegateService#getLockingDocumentId(org.kuali.kfs.coa.document.AccountDelegateGlobalMaintainableImpl, java.lang.String)
54       */
55      @NonTransactional
56      public String getLockingDocumentId(AccountDelegateGlobalMaintainableImpl global, String docNumber) {
57          String lockingDocId = null;
58          List<MaintenanceLock> maintenanceLocks = global.generateMaintenanceLocks();
59          for (MaintenanceLock maintenanceLock : maintenanceLocks) {
60              lockingDocId = accountDelegateGlobalDao.getLockingDocumentNumber(maintenanceLock.getLockingRepresentation(),docNumber);
61              if (StringUtils.isNotBlank(lockingDocId)) {
62                  break;
63              }
64          }
65          return lockingDocId;
66      }

68      /**
69       * 
70       * @see org.kuali.kfs.coa.service.AccountDelegateService#getLockingDocumentId(org.kuali.kfs.coa.document.AccountDelegateMaintainableImpl, java.lang.String)
71       */
72      @NonTransactional
73      public String getLockingDocumentId(AccountDelegateMaintainableImpl delegate, String docNumber) {
74          String lockingDocId = null;
75          List<MaintenanceLock> maintenanceLocks = delegate.generateMaintenanceLocks();
76          maintenanceLocks.add(delegate.createGlobalAccountLock());
77          
78          for (MaintenanceLock maintenanceLock : maintenanceLocks) {
79              lockingDocId = accountDelegateDao.getLockingDocumentNumber(maintenanceLock.getLockingRepresentation(),docNumber);
80              if (StringUtils.isNotBlank(lockingDocId)) {
81                  break;
82              }
83          }
84          return lockingDocId;
85      }

179     /**
180      * Updates the role that this delegate is part of, to account for the changes in this delegate
181      */
182     @Transactional
183     public void updateDelegationRole() {
184         final RoleManagementService roleManagementService = SpringContext.getBean(RoleManagementService.class);
185         final String roleId = roleManagementService.getRoleIdByName(KFSConstants.ParameterNamespaces.KFS, KFSConstants.SysKimConstants.FISCAL_OFFICER_KIM_ROLE_NAME);
186         if (!StringUtils.isBlank(roleId)) {
187             roleManagementService.applicationRoleMembershipChanged(roleId);
188         }
