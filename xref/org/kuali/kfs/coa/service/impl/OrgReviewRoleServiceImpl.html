39      public void populateOrgReviewRoleFromRoleMember(OrgReviewRole orr, String roleMemberId) {
40          List<RoleMemberCompleteInfo> roleMembers = (List<RoleMemberCompleteInfo>)KIMServiceLocator.getRoleManagementService().findRoleMembersCompleteInfo(Collections.singletonMap(KimConstants.PrimaryKeyConstants.ROLE_MEMBER_ID, roleMemberId));
41          RoleMemberCompleteInfo roleMember = new RoleMemberCompleteInfo();
42          if(roleMembers!=null && roleMembers.size()>0){
43              roleMember = roleMembers.get(0);
44          }
45          orr.setRoleMemberId(roleMember.getRoleMemberId());
46          orr.setKimDocumentRoleMember(roleMember);
47  
48          KimRoleInfo roleInfo = KIMServiceLocator.getRoleManagementService().getRole(roleMember.getRoleId());
49          KimTypeInfo typeInfo = KIMServiceLocator.getTypeInfoService().getKimType(roleInfo.getKimTypeId());
50          List<KfsKimDocumentAttributeData> attributes = orr.getAttributeSetAsQualifierList(typeInfo, roleMember.getQualifier());
51          orr.setAttributes(attributes);
52          orr.setRoleRspActions(getRoleRspActions(roleMember.getRoleMemberId()));
53          orr.setRoleId(roleMember.getRoleId());
54          orr.setActiveFromDate(roleMember.getActiveFromDate());
55          orr.setActiveToDate(roleMember.getActiveToDate());
56          populateObjectExtras(orr);
57      }

59      public void populateOrgReviewRoleFromDelegationMember(OrgReviewRole orr, String delegationMemberId) {
60          DelegateMemberCompleteInfo delegationMember = KIMServiceLocator.getRoleManagementService().getDelegationMemberById(delegationMemberId);
61          DelegateTypeInfo delegation = KIMServiceLocator.getRoleManagementService().getDelegateTypeInfoById(delegationMember.getDelegationId());
62          KimRoleInfo roleInfo = KIMServiceLocator.getRoleManagementService().getRole(delegation.getRoleId());
63          KimTypeInfo typeInfo = KIMServiceLocator.getTypeInfoService().getKimType(roleInfo.getKimTypeId());
64          orr.setDelegationMemberId(delegationMember.getDelegationMemberId());
65          orr.setRoleMemberId(delegationMember.getRoleMemberId());
66          orr.setRoleRspActions(getRoleRspActions(delegationMember.getRoleMemberId()));
67          orr.setAttributes(orr.getAttributeSetAsQualifierList(typeInfo, delegationMember.getQualifier()));
68          orr.setRoleId(delegation.getRoleId());
69          orr.setDelegationTypeCode(delegationMember.getDelegationTypeCode());
70          orr.setRoleDocumentDelegationMember(delegationMember);
71          populateObjectExtras(orr);
72      }

74      protected void populateObjectExtras( OrgReviewRole orr ) {
75          Role role = orr.getRole(orr.getRoleId());
76          //Set the role details
77          orr.setRoleName(role.getRoleName());
78          orr.setNamespaceCode(role.getNamespaceCode());
79          
80          orr.setChartOfAccountsCode(orr.getAttributeValue(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE));
81          orr.setOrganizationCode(orr.getAttributeValue(KfsKimAttributes.ORGANIZATION_CODE));
82          orr.setOverrideCode(orr.getAttributeValue(KfsKimAttributes.ACCOUNTING_LINE_OVERRIDE_CODE));
83          orr.setFromAmount(orr.getAttributeValue(KfsKimAttributes.FROM_AMOUNT));
84          orr.setToAmount(orr.getAttributeValue(KfsKimAttributes.TO_AMOUNT));
85          orr.setFinancialSystemDocumentTypeCode(orr.getAttributeValue(KfsKimAttributes.DOCUMENT_TYPE_NAME));
86          
87          orr.getChart().setChartOfAccountsCode(orr.getChartOfAccountsCode());
88          orr.getOrganization().setOrganizationCode(orr.getOrganizationCode());
89  
90          if(orr.getRoleRspActions()!=null && orr.getRoleRspActions().size()>0){
91              orr.setActionTypeCode(orr.getRoleRspActions().get(0).getActionTypeCode());
92              orr.setPriorityNumber(orr.getRoleRspActions().get(0).getPriorityNumber()==null?"":orr.getRoleRspActions().get(0).getPriorityNumber()+"");
93              orr.setActionPolicyCode(orr.getRoleRspActions().get(0).getActionPolicyCode());
94              orr.setForceAction(orr.getRoleRspActions().get(0).isForceAction());
95          }
96          if(orr.getPerson()!=null){
97              orr.setPrincipalMemberPrincipalId(orr.getPerson().getPrincipalId());
98              orr.setPrincipalMemberPrincipalName(orr.getPerson().getPrincipalName());
99          }
100         if(orr.getRole()!=null){
101             orr.setRoleMemberRoleId(orr.getRole().getRoleId());
102             orr.setRoleMemberRoleNamespaceCode(orr.getRole().getNamespaceCode());
103             orr.setRoleMemberRoleName(orr.getRole().getRoleName());
104         }
105         if(orr.getGroup()!=null){
106             orr.setGroupMemberGroupId(orr.getGroup().getGroupId());
107             orr.setGroupMemberGroupNamespaceCode(orr.getGroup().getNamespaceCode());
108             orr.setGroupMemberGroupName(orr.getGroup().getGroupName());
109         }
110     }
