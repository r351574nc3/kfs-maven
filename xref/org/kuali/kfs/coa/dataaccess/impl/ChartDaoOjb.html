34  /**
35   * This class is the OJB implementation of the ChartDao interface.
36   */
37  
38  public class ChartDaoOjb extends PlatformAwareDaoBaseOjb implements ChartDao {
39      private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(ChartDaoOjb.class);
40      private ParameterService parameterService;
41  
42      /**
43       * @see org.kuali.kfs.coa.dataaccess.ChartDao#getAll()
44       */
45      public Collection getAll() {
46          QueryByCriteria qbc = QueryFactory.newQuery(Chart.class, (Criteria) null);
47          qbc.addOrderByAscending("chartOfAccountsCode");
48          Collection charts = getPersistenceBrokerTemplate().getCollectionByQuery(qbc);
49          List chartList = new ArrayList(charts);
50          return chartList;
51      }
52  
53      /**
54       * @see org.kuali.kfs.coa.dataaccess.ChartDao#getUniversityChart()
55       */
56      public Chart getUniversityChart() {
57          // 1. find the organization with the type which reports to itself
58          final String organizationReportsToSelfParameterValue = getParameterService().getParameterValue(Organization.class, KFSConstants.ChartApcParms.ORG_MUST_REPORT_TO_SELF_ORG_TYPES);
59          Criteria orgCriteria = new Criteria();
60          orgCriteria.addEqualTo("organizationTypeCode", organizationReportsToSelfParameterValue);
61          orgCriteria.addEqualTo("active", KFSConstants.ACTIVE_INDICATOR);
62          
63          Iterator organizations = getPersistenceBrokerTemplate().getCollectionByQuery(QueryFactory.newQuery(Organization.class, orgCriteria)).iterator();
64          Organization topDogOrg = null;
65          while (organizations.hasNext()) {
66              topDogOrg = (Organization)organizations.next();
67          }
68  
69          return getByPrimaryId(topDogOrg.getChartOfAccountsCode());
70      }
71  
72      /**
73       * @see org.kuali.kfs.coa.dataaccess.ChartDao#getByPrimaryId(java.lang.String)
74       */
75      public Chart getByPrimaryId(String chartOfAccountsCode) {
76          Criteria criteria = new Criteria();
77          criteria.addEqualTo("chartOfAccountsCode", chartOfAccountsCode);
78  
79          return (Chart) getPersistenceBrokerTemplate().getObjectByQuery(QueryFactory.newQuery(Chart.class, criteria));
80      }
81  
82      /**
83       * fetch the charts that the user is manager for
84       * 
85       * @param kualiUser
86       * @return a list of Charts that the user has responsibility for
87       */
88      public List getChartsThatUserIsResponsibleFor(Person person) {
89          List chartResponsibilities = new ArrayList();
90          Criteria criteria = new Criteria();
91          criteria.addEqualTo("finCoaManagerUniversalId", person.getPrincipalId());
92          Collection charts = getPersistenceBrokerTemplate().getCollectionByQuery(QueryFactory.newQuery(Chart.class, criteria));
93          for (Iterator iter = charts.iterator(); iter.hasNext();) {
94              Chart chart = (Chart) iter.next();
95              chartResponsibilities.add(chart);
96          }
97          return chartResponsibilities;
98      }
