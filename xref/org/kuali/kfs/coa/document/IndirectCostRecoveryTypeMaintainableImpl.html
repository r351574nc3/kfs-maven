41      public static final String MAINTAINABLE_ERROR_PATH = DOCUMENT_ERROR_PREFIX + "newMaintainableObject";

44      @Override
45      public void addMultipleValueLookupResults(MaintenanceDocument document, String collectionName, Collection<PersistableBusinessObject> rawValues, boolean needsBlank, PersistableBusinessObject bo) {
46          // PersistableBusinessObject bo = document.getNewMaintainableObject().getBusinessObject();
47          Collection maintCollection = (Collection) ObjectUtils.getPropertyValue(bo, collectionName);
48          String docTypeName = document.getDocumentHeader().getWorkflowDocument().getDocumentType();
49          
50          List<String> duplicateIdentifierFieldsFromDataDictionary = getDuplicateIdentifierFieldsFromDataDictionary(docTypeName, collectionName);
51          
52          List<String> existingIdentifierList = getMultiValueIdentifierList(maintCollection, duplicateIdentifierFieldsFromDataDictionary);
53          
54          Class collectionClass = getMaintenanceDocumentDictionaryService().getCollectionBusinessObjectClass(docTypeName, collectionName);
55  
56          List<MaintainableSectionDefinition> sections = getMaintenanceDocumentDictionaryService().getMaintainableSections(docTypeName);
57          Map<String, String> template = MaintenanceUtils.generateMultipleValueLookupBOTemplate(sections, collectionName);
58          try {
59              int collectionItemNumber = 0; // is there a better way to do this? ie- old school i=0;i<rawValues.size();i++ and rawValues.get(i)?
60              boolean isValid = true;
61              GlobalVariables.getMessageMap().addToErrorPath(DETAIL_ERROR_PATH);
62              for (PersistableBusinessObject nextBo : rawValues) {
63                  IndirectCostRecoveryExclusionType templatedBo = (IndirectCostRecoveryExclusionType) ObjectUtils.createHybridBusinessObject(collectionClass, nextBo, template);
64                  templatedBo.setNewCollectionRecord(true);
65                  prepareBusinessObjectForAdditionFromMultipleValueLookup(collectionName, templatedBo);
66                  if(!hasBusinessObjectExisted(templatedBo, existingIdentifierList, duplicateIdentifierFieldsFromDataDictionary)) {
67                      maintCollection.add(templatedBo);
68                  }
69                  collectionItemNumber++;
70                  templatedBo.setActive(true); // TODO remove after active indicator work is complete
71              }
72              GlobalVariables.getMessageMap().removeFromErrorPath(DETAIL_ERROR_PATH);
73              // putGlobalError(KFSKeyConstants.ERROR_DOCUMENT_ACCTDELEGATEMAINT_PRIMARY_ROUTE_ALREADY_EXISTS_FOR_DOCTYPE);
74          } 
75          catch (Exception e) {
76              LOG.error("Unable to add multiple value lookup results " + e.getMessage());
77              throw new RuntimeException("Unable to add multiple value lookup results " + e.getMessage());
78          }
79      }
