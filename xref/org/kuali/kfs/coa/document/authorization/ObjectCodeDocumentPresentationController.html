18  import java.util.Set;

20  import org.kuali.kfs.coa.businessobject.ObjectCode;

22  import org.kuali.kfs.sys.context.SpringContext;
23  import org.kuali.kfs.sys.document.authorization.FinancialSystemMaintenanceDocumentPresentationControllerBase;
24  import org.kuali.rice.kns.bo.BusinessObject;

27  public class ObjectCodeDocumentPresentationController extends FinancialSystemMaintenanceDocumentPresentationControllerBase {
28      
29      /**
30       * Determins whether to display the research admin attributes or not
31       * @see org.kuali.rice.kns.document.authorization.MaintenanceDocumentPresentationControllerBase#getConditionallyHiddenSectionIds(org.kuali.rice.kns.document.MaintenanceDocument)
32       */
33      @Override
34      public Set<String> getConditionallyHiddenSectionIds(BusinessObject businessObject) {
35          Set<String> hiddenSectionIds = super.getConditionallyHiddenSectionIds(businessObject);
36          
37          //  consult the system param on whether to hide the research admin attributes tab
38          if (!"Y".equalsIgnoreCase(SpringContext.getBean(ParameterService.class).getParameterValue(ObjectCode.class, KFSConstants.ObjectCodeConstants.PARAMETER_KC_ENABLE_RESEARCH_ADMIN_OBJECT_CODE_ATTRIBUTE_IND))) {
39              hiddenSectionIds.add(KFSConstants.ObjectCodeConstants.SECTION_ID_RESEARCH_ADMIN_ATTRIBUTES);
40          }
41  
42          return hiddenSectionIds;
43      }
