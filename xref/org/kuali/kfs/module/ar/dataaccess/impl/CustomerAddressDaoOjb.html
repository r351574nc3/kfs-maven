34      public CustomerAddress getPrimaryAddress(String customerNumber) {
35          Criteria criteria = new Criteria();
36          criteria.addEqualTo("customerNumber", customerNumber==null?customerNumber:customerNumber.toUpperCase());
37          criteria.addEqualTo("customerAddressTypeCode", ArKeyConstants.CustomerConstants.CUSTOMER_ADDRESS_TYPE_CODE_PRIMARY);
38  
39          return (CustomerAddress) getPersistenceBrokerTemplate().getObjectByQuery(QueryFactory.newQuery(CustomerAddress.class, criteria));
40      }
41  
42      public Integer getMaxSquenceNumber(String customerNumber) {
43          Criteria criteria = new Criteria();
44  
45          criteria.addEqualTo(ArPropertyConstants.CustomerFields.CUSTOMER_NUMBER, customerNumber==null?customerNumber:customerNumber.toUpperCase());
46          ReportQueryByCriteria query = QueryFactory.newReportQuery(CustomerAddress.class, criteria);
47          query.setAttributes(new String[] { "max(" + ArPropertyConstants.CustomerFields.CUSTOMER_ADDRESS_IDENTIFIER + ")" });
48          Iterator<?> iterator = getPersistenceBrokerTemplate().getReportQueryIteratorByQuery(query);
49          Integer maxSequenceNumber = Integer.valueOf(0);
50  
51          if (iterator.hasNext()) {
52              Object[] data = (Object[]) TransactionalServiceUtils.retrieveFirstAndExhaustIterator(iterator);
53              if (data[0] != null) {
54                  maxSequenceNumber = ((BigDecimal) data[0]).intValue();
55              }
56          }
57          return maxSequenceNumber;
58      }
