16  package org.kuali.kfs.module.ar.batch.report;
17  
18  import org.apache.commons.lang.StringUtils;
19  
20  public class CustomerLoadBatchError {
21  
22      private String customerName;
23      private String propertyName;
24      private Class<?> propertyClass;
25      private String value;
26      private String description;
27      
28      /**
29       * 
30       * Default constructor with all internal fields defaulting to empty-string or null.
31       */
32      public CustomerLoadBatchError() {
33          this.customerName = "";
34          this.propertyName = "";
35          this.propertyClass = null;
36          this.value = "";
37          this.description = "";
38      }
39      
40      /**
41       * 
42       * Constructor to initialize with just the customerName.
43       * @param customerName The customer name of the document being imported.
44       */
45      public CustomerLoadBatchError(String customerName) {
46          this.customerName = customerName;
47      }
48      
49      /**
50       * 
51       * Constructs a CustomerLoadBatchError.java.
52       * @param customerName The customer name of the document being imported.
53       * @param propertyName The name of the property on the Customer BO.
54       * @param propertyClass The class of the property named by propertyName.
55       * @param value The original value of the field from the batch import document.
56       * @param description The description of the error that occurred.
57       */
58      public CustomerLoadBatchError(String customerName, String propertyName, Class<?> propertyClass, String value, String description) {
59          this.customerName = customerName;
60          this.propertyName = propertyName;
61          this.propertyClass = propertyClass;
62          this.value = value;
63          this.description = description;
64      }
65  
66      public String toString() {
67          return "[" + customerName + "] " + 
68                  ("class java.lang.Object".equals(propertyClass.toString()) || propertyClass == null ? "" : "(" + propertyClass.toString() + ") ") + 
69                  (StringUtils.isBlank(propertyName) ? "" : getPropertyNameLastElement() + ": ") + 
70                  (StringUtils.isBlank(value) || "N/A".equalsIgnoreCase(value) ? "" : "'" + value + "' - ") +  
71                  description;
72      }
73      
74      public String getPropertyNameLastElement() {
75          if (StringUtils.isBlank(propertyName)) return propertyName;
76          
77          String[] propertyNameElements = propertyName.split("\\.");
78          if (propertyNameElements.length <= 0) {
79              return propertyName;
80          }
81          else {
82              return propertyNameElements[propertyNameElements.length - 1];
83          }
84          
85      }
86      
87      public String getCustomerName() {
88          return customerName;
89      }
90  
91      public void setCustomerName(String customerName) {
92          this.customerName = customerName;
93      }
94  
95      public String getPropertyName() {
96          return propertyName;
97      }
98  
99      public void setPropertyName(String propertyName) {
100         this.propertyName = propertyName;
101     }
102 
103     public Class<?> getPropertyClass() {
104         return propertyClass;
105     }
106 
107     public void setPropertyClass(Class<?> propertyClass) {
108         this.propertyClass = propertyClass;
109     }
110 
111     public String getValue() {
112         return value;
113     }
114 
115     public void setValue(String value) {
116         this.value = value;
117     }
118 
119     public String getDescription() {
120         return description;
121     }
122 
123     public void setDescription(String description) {
124         this.description = description;
125     }

127 }
