19  
20  import org.kuali.kfs.module.ar.ArKeyConstants;

26  import org.kuali.rice.kns.util.GlobalVariables;
27  import org.kuali.rice.kns.util.KualiDecimal;
28  import org.kuali.rice.kns.util.ObjectUtils;
29  
30  public class CustomerInvoiceDetailUnitPriceValidation extends GenericValidation {
31  
32      private CustomerInvoiceDetail customerInvoiceDetail;
33      private CustomerInvoiceDocument customerInvoiceDocument;
34      
35      public boolean validate(AttributedDocumentEvent event) {
36          BigDecimal unitPrice = customerInvoiceDetail.getInvoiceItemUnitPrice();
37  
38          // if amount is = 0
39          if (ObjectUtils.isNull(unitPrice) || BigDecimal.ZERO.equals(unitPrice)) {
40              GlobalVariables.getMessageMap().putError(ArPropertyConstants.CustomerInvoiceDocumentFields.INVOICE_ITEM_UNIT_PRICE, ArKeyConstants.ERROR_CUSTOMER_INVOICE_DETAIL_UNIT_PRICE_LESS_THAN_OR_EQUAL_TO_ZERO);
41              return false;
42          }
43          else {
44              // else if unit price is greater than or less than zero
45  
46              if (customerInvoiceDocument.isInvoiceReversal()) {
47                  // if invoice is reversal, discount should be positive, non-discounts should be negative
48                  if (customerInvoiceDetail.isDiscountLine() && unitPrice.compareTo(BigDecimal.ZERO) == -1) {
49                      GlobalVariables.getMessageMap().putError(ArPropertyConstants.CustomerInvoiceDocumentFields.INVOICE_ITEM_UNIT_PRICE, ArKeyConstants.ERROR_CUSTOMER_INVOICE_DETAIL_UNIT_PRICE_LESS_THAN_OR_EQUAL_TO_ZERO);
50                      return false;
51                  }
52                  else if (!customerInvoiceDetail.isDiscountLine() && unitPrice.compareTo(BigDecimal.ZERO) == 1) {
53                      GlobalVariables.getMessageMap().putError(ArPropertyConstants.CustomerInvoiceDocumentFields.INVOICE_ITEM_UNIT_PRICE, ArKeyConstants.ERROR_CUSTOMER_INVOICE_DETAIL_UNIT_PRICE_LESS_THAN_OR_EQUAL_TO_ZERO);
54                      return false;
55                  }
56              }
57              else {
58                  //No need to check for positive because if the line is a discount and the amount is negative, we will negate the line.
59                  if (!customerInvoiceDetail.isDiscountLine() && unitPrice.compareTo(BigDecimal.ZERO) == -1) {
60                      GlobalVariables.getMessageMap().putError(ArPropertyConstants.CustomerInvoiceDocumentFields.INVOICE_ITEM_UNIT_PRICE, ArKeyConstants.ERROR_CUSTOMER_INVOICE_DETAIL_UNIT_PRICE_LESS_THAN_OR_EQUAL_TO_ZERO);
61                      return false;
62                  }
63              }
64          }
65          
66          return true;
67      }
68      
69      public CustomerInvoiceDetail getCustomerInvoiceDetail() {
70          return customerInvoiceDetail;
71      }
72  
73      public void setCustomerInvoiceDetail(CustomerInvoiceDetail customerInvoiceDetail) {
74          this.customerInvoiceDetail = customerInvoiceDetail;
75      }
76  
77      public CustomerInvoiceDocument getCustomerInvoiceDocument() {
78          return customerInvoiceDocument;
79      }
80  
81      public void setCustomerInvoiceDocument(CustomerInvoiceDocument customerInvoiceDocument) {
82          this.customerInvoiceDocument = customerInvoiceDocument;
83      }        
84  }
