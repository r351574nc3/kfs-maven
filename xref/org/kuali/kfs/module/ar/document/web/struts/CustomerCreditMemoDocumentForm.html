41      @Override

46      /**
47       * Setup workflow doc in the document.
48       */
49      @Override
50      public void populate(HttpServletRequest request) {
51          
52          //populate document using request
53          super.populate(request);
54          
55          CustomerCreditMemoDocument customerCreditMemoDocument = (CustomerCreditMemoDocument)getDocument();
56          String customerInvoiceNumber = customerCreditMemoDocument.getFinancialDocumentReferenceInvoiceNumber();
57          
58          //this will make sure that every action has fully populated invoice
59          if(StringUtils.isNotEmpty(customerInvoiceNumber) /*&& customerCreditMemoDocument.getInvoice() == null*/){
60              customerCreditMemoDocument.refreshReferenceObject("invoice");
61          }        
62      }
63  
64      /**
65       * Build additional customer credit memo specific buttons and set extraButtons list.
66       * 
67       * @return - list of extra buttons to be displayed to the user
68       */
69      @Override
70      public List<ExtraButton> getExtraButtons() {
71          
72          // clear out the extra buttons array
73          extraButtons.clear();
74  
75          CustomerCreditMemoDocument creditMemoDoc = (CustomerCreditMemoDocument) getDocument();
76          DocumentHelperService docHelperService = SpringContext.getBean(DocumentHelperService.class);
77          TransactionalDocumentPresentationController presoController = 
78                  (TransactionalDocumentPresentationController) docHelperService.getDocumentPresentationController(creditMemoDoc);
79          Set<String> editModes = presoController.getEditModes(creditMemoDoc);
80  
81          //  special buttons for the first 'init' screen
82          if (editModes.contains(ArAuthorizationConstants.CustomerCreditMemoEditMode.DISPLAY_INIT_TAB)) {
83              String externalImageURL = SpringContext.getBean(KualiConfigurationService.class).getPropertyString(KFSConstants.RICE_EXTERNALIZABLE_IMAGES_URL_KEY);
84              addExtraButton("methodToCall.continueCreditMemo", externalImageURL + "buttonsmall_continue.gif", "Continue");
85              addExtraButton("methodToCall.clearInitTab", externalImageURL + "buttonsmall_clear.gif", "Clear");
86          }
87          
88          //  show the print button if appropriate 
89          if (editModes.contains(ArAuthorizationConstants.CustomerCreditMemoEditMode.DISPLAY_PRINT_BUTTON)) {
90              String printButtonURL = SpringContext.getBean(KualiConfigurationService.class).getPropertyString(KFSConstants.EXTERNALIZABLE_IMAGES_URL_KEY);
91              addExtraButton("methodToCall.print", printButtonURL + "buttonsmall_genprintfile.gif", "Print");
92          }
93  
94          return extraButtons;
95      }
96      
97      /**
98       * Adds a new button to the extra buttons collection.
99       * 
100      * @param property - property for button
101      * @param source - location of image
102      * @param altText - alternate text for button if images don't appear
103      */
104     protected void addExtraButton(String property, String source, String altText) {
105 
106         ExtraButton newButton = new ExtraButton();
107 
108         newButton.setExtraButtonProperty(property);
109         newButton.setExtraButtonSource(source);
110         newButton.setExtraButtonAltText(altText);
111 
112         extraButtons.add(newButton);
113     }
