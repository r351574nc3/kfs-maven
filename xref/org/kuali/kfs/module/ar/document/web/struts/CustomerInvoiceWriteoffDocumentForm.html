34  public class CustomerInvoiceWriteoffDocumentForm extends FinancialSystemTransactionalDocumentFormBase {
35      
36      public CustomerInvoiceWriteoffDocumentForm() {
37          super();
38      }
39  
40      @Override
41      protected String getDefaultDocumentTypeName() {
42          return "INVW";
43      }
44      
45      /**
46       * Setup workflow doc in the document.
47       */
48      @Override
49      public void populate(HttpServletRequest request) {
50          
51          //populate document using request
52          super.populate(request);
53          
54          CustomerInvoiceWriteoffDocument customerInvoiceWriteoffDocument = (CustomerInvoiceWriteoffDocument)getDocument();
55          String customerInvoiceNumber = customerInvoiceWriteoffDocument.getFinancialDocumentReferenceInvoiceNumber();
56          
57          //this will make sure that every action has fully populated invoice
58          if(StringUtils.isNotEmpty(customerInvoiceNumber)){
59              customerInvoiceWriteoffDocument.refreshReferenceObject("customerInvoiceDocument");
60          }        
61      }
62  
63      /**
64       * Build additional customer credit memo specific buttons and set extraButtons list.
65       * 
66       * @return - list of extra buttons to be displayed to the user
67       */
68      @Override
69      public List<ExtraButton> getExtraButtons() {
70          
71          // clear out the extra buttons array
72          extraButtons.clear();
73  
74          CustomerInvoiceWriteoffDocument writeoffDoc = (CustomerInvoiceWriteoffDocument) getDocument();
75          DocumentHelperService documentHelperService = SpringContext.getBean(DocumentHelperService.class);
76          TransactionalDocumentPresentationController presoController = 
77                  (TransactionalDocumentPresentationController) documentHelperService.getDocumentPresentationController(writeoffDoc);
78          Set<String> editModes = presoController.getEditModes(writeoffDoc);
79  
80          if (editModes.contains(ArAuthorizationConstants.CustomerCreditMemoEditMode.DISPLAY_INIT_TAB)) {
81              String externalImageURL = SpringContext.getBean(KualiConfigurationService.class).getPropertyString(KFSConstants.RICE_EXTERNALIZABLE_IMAGES_URL_KEY);
82              addExtraButton("methodToCall.continueCustomerInvoiceWriteoff", externalImageURL + "buttonsmall_continue.gif", "Continue");
83              addExtraButton("methodToCall.clearInitTab", externalImageURL + "buttonsmall_clear.gif", "Clear");
84          }
85          return extraButtons;
86      }
87      
88      /**
89       * Adds a new button to the extra buttons collection.
90       * 
91       * @param property - property for button
92       * @param source - location of image
93       * @param altText - alternate text for button if images don't appear
94       */
95      protected void addExtraButton(String property, String source, String altText) {
96  
97          ExtraButton newButton = new ExtraButton();
98  
99          newButton.setExtraButtonProperty(property);
100         newButton.setExtraButtonSource(source);
101         newButton.setExtraButtonAltText(altText);
102 
103         extraButtons.add(newButton);
104     }
