51      /**
52       * @see org.kuali.kfs.module.ar.report.service.CustomerAgingReportService#calculateAgingReportAmounts()
53       */
54      public CustomerAgingReportDataHolder calculateAgingReportAmounts(Collection<CustomerInvoiceDetail> details, Date reportRunDate) {
55          CustomerAgingReportDataHolder agingData = new CustomerAgingReportDataHolder();
56          
57          KualiDecimal total0to30 = KualiDecimal.ZERO;
58          KualiDecimal total31to60 = KualiDecimal.ZERO;
59          KualiDecimal total61to90 = KualiDecimal.ZERO;
60          KualiDecimal total91toSYSPR = KualiDecimal.ZERO;
61          KualiDecimal totalSYSPRplus1orMore = KualiDecimal.ZERO;
62  
63          String nbrDaysForLastBucket = SpringContext.getBean(ParameterService.class).getParameterValue(CustomerAgingReportDetail.class, "CUSTOMER_INVOICE_AGE");     // ArConstants.CUSTOMER_INVOICE_AGE); // default is 120
64  
65          Date cutoffdate30 = DateUtils.addDays(reportRunDate, -30);
66          Date cutoffdate60 = DateUtils.addDays(reportRunDate, -60);
67          Date cutoffdate90 = DateUtils.addDays(reportRunDate, -90);
68          Date cutoffdate120 = DateUtils.addDays(reportRunDate, -1*Integer.parseInt(nbrDaysForLastBucket));
69          
70          Map<String, Object> knownCustomers = new HashMap<String, Object>(details.size());
71          Map<String, Object> retrievedInvoices = new HashMap<String, Object>(); // Simple caching mechanism to try to lower overhead for retrieving invoices that have multiple details
72          
73          KualiDecimal totalBalance = new KualiDecimal(0.00);
74          CustomerAgingReportDetail custDetail = null;
75          
76          // iterate over all invoices consolidating balances for each customer
77          for (CustomerInvoiceDetail cid : details) {
78              String invoiceDocumentNumber = cid.getDocumentNumber();
79              CustomerInvoiceDocument custInvoice;
80              // check cache for already retrieved invoices before attempting to retrieve it from the DB
81              if(retrievedInvoices.containsKey(invoiceDocumentNumber)) {
82                  custInvoice = (CustomerInvoiceDocument) retrievedInvoices.get(invoiceDocumentNumber);
83              } else {
84                  custInvoice = SpringContext.getBean(CustomerInvoiceDocumentService.class).getInvoiceByInvoiceDocumentNumber(invoiceDocumentNumber);
85                  retrievedInvoices.put(invoiceDocumentNumber, custInvoice);
86              }
87              Date approvalDate=custInvoice.getBillingDate(); 
88              if (ObjectUtils.isNull(approvalDate)) {
89                  continue;
90              }
91  
92              if(custInvoice!=null && cid.getAmountOpen().isNonZero()) {
93  
94                  Customer customerobj = custInvoice.getCustomer();                        
95                  String customerNumber = customerobj.getCustomerNumber();    // tested and works
96                  String customerName = customerobj.getCustomerName();  // tested and works
97  
98                  if (knownCustomers.containsKey(customerNumber)) { 
99                      custDetail = (CustomerAgingReportDetail) knownCustomers.get(customerNumber);
100                 } else {
101                     custDetail = new CustomerAgingReportDetail();
102                     custDetail.setCustomerName(customerName);
103                     custDetail.setCustomerNumber(customerNumber);
104                     knownCustomers.put(customerNumber, custDetail);
105                 }
106                 KualiDecimal amountOpenOnReportRunDate = cid.getAmountOpenByDateFromDatabase(reportRunDate);
107                 if (!approvalDate.after(reportRunDate) && !approvalDate.before(cutoffdate30)) {                                
108                     custDetail.setUnpaidBalance0to30(amountOpenOnReportRunDate.add(custDetail.getUnpaidBalance0to30()));
109                     total0to30 = total0to30.add(amountOpenOnReportRunDate);
110                 }
111                 else if (approvalDate.before(cutoffdate30) && !approvalDate.before(cutoffdate60)) {               
112                     custDetail.setUnpaidBalance31to60(amountOpenOnReportRunDate.add(custDetail.getUnpaidBalance31to60()));
113                     total31to60 = total31to60.add(amountOpenOnReportRunDate);
114                 }
115                 else if (approvalDate.before(cutoffdate60) && !approvalDate.before(cutoffdate90)) {
116                     custDetail.setUnpaidBalance61to90(amountOpenOnReportRunDate.add(custDetail.getUnpaidBalance61to90()));
117                     total61to90 = total61to90.add(amountOpenOnReportRunDate);
118                 }
119                 else if (approvalDate.before(cutoffdate90) && !approvalDate.before(cutoffdate120)) {
120                     custDetail.setUnpaidBalance91toSYSPR(amountOpenOnReportRunDate.add(custDetail.getUnpaidBalance91toSYSPR()));
121                     total91toSYSPR = total91toSYSPR.add(amountOpenOnReportRunDate);
122                 }
123                 else if (approvalDate.before(cutoffdate120)) {
124                     custDetail.setUnpaidBalanceSYSPRplus1orMore(amountOpenOnReportRunDate.add(custDetail.getUnpaidBalanceSYSPRplus1orMore()));
125                     totalSYSPRplus1orMore = totalSYSPRplus1orMore.add(amountOpenOnReportRunDate);
126                 }            
127                 totalBalance = totalBalance.add(amountOpenOnReportRunDate);
128             }        
129         } 
130         
131         agingData.setTotal0to30(total0to30);
132         agingData.setTotal31to60(total31to60);
133         agingData.setTotal61to90(total61to90);
134         agingData.setTotal91toSYSPR(total91toSYSPR);
135         agingData.setTotalSYSPRplus1orMore(totalSYSPRplus1orMore);
136         agingData.setTotalAmountDue(totalBalance);
137         
138         agingData.setKnownCustomers(knownCustomers);
139         
140         return agingData;
141     }
