40      /**
41       * Context:
42       *  the EquipmentLoanOrReturnDocument document can be accessed by clicking "loan" on the Asset lookup results. 
43       *  With this document, a user may decide to lend an asset to someone. 
44       *  Therefore, the borrower will want to approve the document confirming that he indeed wants to borrow the item.
45       *  
46       * Requirement:
47       *  The service will take in a document number as a role qualifier. 
48       *  It will retrieve the document given the doc ID, extract out the user ID of the borrower 
49       *  (the borrowerUniversalIdentifier property of the document object), 
50       *  and return this principal ID as the only role member for a given document.
51       * 
52       * @see org.kuali.rice.kim.service.support.impl.KimRoleTypeServiceBase#getPrincipalIdsFromApplicationRole(java.lang.String, java.lang.String, org.kuali.rice.kim.bo.types.dto.AttributeSet)
53       */
54      @Override
55      public List<RoleMembershipInfo> getRoleMembersFromApplicationRole(String namespaceCode, String roleName, AttributeSet qualification) {
56          validateRequiredAttributesAgainstReceived(qualification);
57          List<RoleMembershipInfo> members = new ArrayList<RoleMembershipInfo>(1);
58          if(qualification!=null && !qualification.isEmpty()){
59              String documentNumber = qualification.get(KimAttributes.DOCUMENT_NUMBER);
60              if ( StringUtils.isNotBlank( documentNumber ) ) {
61                  EquipmentLoanOrReturnDocument document = getEquipmentLoanOrReturnDocument(documentNumber);
62                  if(document!=null){
63                      members.add( new RoleMembershipInfo(null,null,document.getBorrowerUniversalIdentifier(),Role.PRINCIPAL_MEMBER_TYPE,null) );
64                  }
65              }
66          }
67          return members;
68      }
69  
70      /**
71       * @param documentNumber
72       * @return
73       */
74      protected EquipmentLoanOrReturnDocument getEquipmentLoanOrReturnDocument(String documentNumber){
75          try{
76              return (EquipmentLoanOrReturnDocument)getDocumentService().getByDocumentHeaderId(documentNumber);
77          } catch (WorkflowException e) {
78              throw new RuntimeException("Workflow problem while trying to get document using doc id '" + documentNumber + "'", e);
79          }
80      }
81  
82      protected DocumentService getDocumentService() {
83          if ( documentService == null ) {
84              documentService = SpringContext.getBean(DocumentService.class);
85          }
86          return documentService;
87      }
