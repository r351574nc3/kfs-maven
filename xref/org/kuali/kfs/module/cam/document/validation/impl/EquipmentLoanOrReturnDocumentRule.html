49       * Retrieve asset number need to be locked.
50       * 
51       * @param document
52       * @return
53       */
54      protected List<Long> retrieveAssetNumberForLocking(Document document) {
55          EquipmentLoanOrReturnDocument equipmentLoanOrReturnDocument = (EquipmentLoanOrReturnDocument) document;
56          List<Long> assetNumbers = new ArrayList<Long>();
57          if (equipmentLoanOrReturnDocument.getCapitalAssetNumber() != null) {
58              assetNumbers.add(equipmentLoanOrReturnDocument.getCapitalAssetNumber());
59          }
60          return assetNumbers;
61      }

63      /**
64       * @see org.kuali.rice.kns.rules.DocumentRuleBase#processCustomRouteDocumentBusinessRules(org.kuali.rice.kns.document.Document)
65       */
66      @Override
67      protected boolean processCustomRouteDocumentBusinessRules(Document document) {
68          if (!super.processCustomRouteDocumentBusinessRules(document)) {
69              return false;
70          }
71  
72          EquipmentLoanOrReturnDocument equipmentLoanOrReturnDocument = (EquipmentLoanOrReturnDocument) document;
73          boolean valid = processValidation(equipmentLoanOrReturnDocument);
74  
75          // check if asset is locked by other document and display error message when save
76          valid &= !getAssetLockService().isAssetLocked(retrieveAssetNumberForLocking(document), CamsConstants.DocumentTypeName.ASSET_EQUIPMENT_LOAN_OR_RETURN, document.getDocumentNumber());
77          return valid;
78      }

80      /**
81       * This method applies business rules
82       * 
83       * @param document equipmentLoanOrReturnDocument Document
84       * @return true if all rules are pass
85       */
86      protected boolean processValidation(EquipmentLoanOrReturnDocument equipmentLoanOrReturnDocument) {
87          boolean valid = true;
88          if (equipmentLoanOrReturnDocument.getBorrowerPerson() == null) {
89              valid &= false;
90              GlobalVariables.getMessageMap().putError(KFSConstants.DOCUMENT_PROPERTY_NAME + "." + CamsPropertyConstants.EquipmentLoanOrReturnDocument.BORROWER_PRINCIPAL_NAME, CamsKeyConstants.EquipmentLoanOrReturn.ERROR_INVALID_BORROWER_ID);
91          }
92          // validate campus tag number
93          valid &= validateTagNumber(equipmentLoanOrReturnDocument);
94  
95          // validate both loan return date and expected loan return date
96          valid &= validateLoanDate(equipmentLoanOrReturnDocument);
97  
98          // validate borrower and storage state codes
99          valid &= validStateZipCode(equipmentLoanOrReturnDocument);
100 
101         return valid;
102     }

114         map.put(CamsPropertyConstants.EquipmentLoanOrReturnDocument.CAPITAL_ASSET_NUMBER, equipmentLoanOrReturnDocument.getCapitalAssetNumber());
115         Asset asset = (Asset) SpringContext.getBean(BusinessObjectService.class).findByPrimaryKey(Asset.class, map);

118             valid &= false;

120         }

125     /**
126      * Implementation of the rule that if a document has a valid expect loan date and loan return date, the both dates should come
127      * before the 2 years limit.
128      * 
129      * @param equipmentLoanOrReturnDocument the equipmentLoanOrReturn document to be validated
130      * @return boolean false if the expect loan date or loan return date is not before the 2 years limit.
131      */
132     protected boolean validateLoanDate(EquipmentLoanOrReturnDocument equipmentLoanOrReturnDocument) {
133         boolean valid = true;
134         Date loanDate = DateUtils.clearTimeFields(equipmentLoanOrReturnDocument.getLoanDate());
135         Calendar cal = GregorianCalendar.getInstance();
136         cal.setTime(loanDate);
137         cal.add(Calendar.YEAR, 2);
138         Date maxDate = new Date(cal.getTime().getTime());
139 
140         // Loan can not be before today
141         Date loanReturnDate = equipmentLoanOrReturnDocument.getLoanReturnDate();
142         if (equipmentLoanOrReturnDocument.isNewLoan() && loanDate.before(DateUtils.clearTimeFields(new java.util.Date()))) {
143             GlobalVariables.getMessageMap().putError(KFSConstants.DOCUMENT_PROPERTY_NAME + "." + CamsPropertyConstants.EquipmentLoanOrReturnDocument.LOAN_DATE, CamsKeyConstants.EquipmentLoanOrReturn.ERROR_INVALID_LOAN_DATE);
144         }
145 
146         // expect return date must be >= loan date and within 2 years limit
147         Date expectReturnDate = equipmentLoanOrReturnDocument.getExpectedReturnDate();
148         if (expectReturnDate != null) {
149             DateUtils.clearTimeFields(expectReturnDate);
150             if (expectReturnDate.before(loanDate)) {
151                 valid &= false;
152                 GlobalVariables.getMessageMap().putError(KFSConstants.DOCUMENT_PROPERTY_NAME + "." + CamsPropertyConstants.EquipmentLoanOrReturnDocument.EXPECTED_RETURN_DATE, CamsKeyConstants.EquipmentLoanOrReturn.ERROR_INVALID_EXPECTED_RETURN_DATE);
153             }
154             if (maxDate.before(expectReturnDate)) {
155                 valid &= false;
156                 GlobalVariables.getMessageMap().putError(KFSConstants.DOCUMENT_PROPERTY_NAME + "." + CamsPropertyConstants.EquipmentLoanOrReturnDocument.EXPECTED_RETURN_DATE, CamsKeyConstants.EquipmentLoanOrReturn.ERROR_INVALID_EXPECTED_MAX_DATE);
157             }
158         }
159 
160         // loan return date must be >= loan date and within 2 years limit
161         if (loanReturnDate != null) {
162             DateUtils.clearTimeFields(loanReturnDate);
163             if (loanDate.after(loanReturnDate) || maxDate.before(loanReturnDate)) {
164                 valid &= false;
165                 GlobalVariables.getMessageMap().putError(KFSConstants.DOCUMENT_PROPERTY_NAME + "." + CamsPropertyConstants.EquipmentLoanOrReturnDocument.LOAN_RETURN_DATE, CamsKeyConstants.EquipmentLoanOrReturn.ERROR_INVALID_LOAN_RETURN_DATE);
166             }
167         }
168 
169         return valid;
170     }

172     /**
173      * Implementation of the rule that if borrower and storage state codes are valid
174      * 
175      * @param equipmentLoanOrReturnDocument the equipmentLoanOrReturn document to be validated
176      * @return boolean false if the borrower or storage state code is not valid
177      */
178     protected boolean validStateZipCode(EquipmentLoanOrReturnDocument equipmentLoanOrReturnDocument) {
179         boolean valid = true;
180         // validate borrower state and postal zip code
181         if (StringUtils.isBlank(equipmentLoanOrReturnDocument.getBorrowerCountryCode())) {
182             equipmentLoanOrReturnDocument.setBorrowerCountryCode(KFSConstants.COUNTRY_CODE_UNITED_STATES);
183         }
184 
185         if (equipmentLoanOrReturnDocument.getBorrowerCountryCode().equals(KFSConstants.COUNTRY_CODE_UNITED_STATES)) {
186             State borrowStateCode = equipmentLoanOrReturnDocument.getBorrowerState();
187             if (ObjectUtils.isNull(borrowStateCode)) {
188                 GlobalVariables.getMessageMap().putError(KFSConstants.DOCUMENT_PROPERTY_NAME + "." + CamsPropertyConstants.EquipmentLoanOrReturnDocument.BORROWER_STATE_CODE, CamsKeyConstants.EquipmentLoanOrReturn.ERROR_INVALID_BORROWER_STATE, equipmentLoanOrReturnDocument.getBorrowerStateCode());
189                 valid &= false;
190             }
191         }
192 
193         // validate borrower storage state and postal zip code
194         if (StringUtils.isNotBlank(equipmentLoanOrReturnDocument.getBorrowerStorageStateCode())) {
195             if (StringUtils.isBlank(equipmentLoanOrReturnDocument.getBorrowerStorageCountryCode())) {
196                 equipmentLoanOrReturnDocument.setBorrowerStorageCountryCode(KFSConstants.COUNTRY_CODE_UNITED_STATES);
197             }
198         }
199 
200         String storageCountryCode = equipmentLoanOrReturnDocument.getBorrowerStorageCountryCode();
201         String storageAddress = equipmentLoanOrReturnDocument.getBorrowerStorageAddress();
202         if (StringUtils.isNotBlank(storageAddress)&& StringUtils.isNotBlank(storageCountryCode) && storageCountryCode.equals(KFSConstants.COUNTRY_CODE_UNITED_STATES)){
203             if (StringUtils.isBlank(equipmentLoanOrReturnDocument.getBorrowerStorageStateCode())) {
204                 GlobalVariables.getMessageMap().putError(KFSConstants.DOCUMENT_PROPERTY_NAME + "." + CamsPropertyConstants.EquipmentLoanOrReturnDocument.BORROWER_STAORAGE_STATE_CODE, CamsKeyConstants.EquipmentLoanOrReturn.ERROR_BORROWER_STORAGE_STATE_REQUIRED, equipmentLoanOrReturnDocument.getBorrowerCountryCode());
205                 valid = false;
206             }
207             else {
208                 State borrowStorageStateCode = equipmentLoanOrReturnDocument.getBorrowerStorageState();
209                 if (ObjectUtils.isNull(borrowStorageStateCode)) {
210                     GlobalVariables.getMessageMap().putError(KFSConstants.DOCUMENT_PROPERTY_NAME + "." + CamsPropertyConstants.EquipmentLoanOrReturnDocument.BORROWER_STAORAGE_STATE_CODE, CamsKeyConstants.EquipmentLoanOrReturn.ERROR_INVALID_BORROWER_STORAGE_STATE, equipmentLoanOrReturnDocument.getBorrowerStorageStateCode());
211                     valid = false;
212                 }
213             }
214 
215             if (StringUtils.isBlank(equipmentLoanOrReturnDocument.getBorrowerStorageZipCode())) {
216                 GlobalVariables.getMessageMap().putError(KFSConstants.DOCUMENT_PROPERTY_NAME + "." + CamsPropertyConstants.EquipmentLoanOrReturnDocument.BORROWER_STORAGE_ZIP_CODE, CamsKeyConstants.EquipmentLoanOrReturn.ERROR_BORROWER_STORAGE_ZIP_REQUIRED, equipmentLoanOrReturnDocument.getBorrowerCountryCode());
217                 valid = false;
218             }
219 
220         }
221 
222         return valid;
223     }
