40      @Override
41      public Set<String> getConditionallyHiddenPropertyNames(BusinessObject businessObject) {
42          Set<String> fields = super.getConditionallyHiddenPropertyNames(businessObject);
43          MaintenanceDocument document = (MaintenanceDocument) businessObject;
44          AssetGlobal assetGlobal = (AssetGlobal) document.getNewMaintainableObject().getBusinessObject();
45          MaintainableCollectionDefinition maintCollDef = SpringContext.getBean(MaintenanceDocumentDictionaryService.class).getMaintainableCollection("AA", "assetPaymentDetails");
46          boolean isAssetSeparate = SpringContext.getBean(AssetGlobalService.class).isAssetSeparate(assetGlobal);
47  
48          if (assetGlobal.isCapitalAssetBuilderOriginIndicator() || isAssetSeparate) {
49              // do not include payment add section within the payment details collection
50              maintCollDef.setIncludeAddLine(false);
51              // No update could be made to payment if it's created from CAB. Here, disable delete button if payment already added into
52              // collection.
53              for (AssetPaymentDetail payment : assetGlobal.getAssetPaymentDetails()) {
54                  payment.setNewCollectionRecord(false);
55              }
56          }
57          else {
58              // conversely allow add during any other case. This is important because the attribute is set on the DD and the DD is
59              // only loaded on project startup. Hence setting is important to avoid state related bugs
60              maintCollDef.setIncludeAddLine(true);
61          }
62  
63          AssetGlobalService assetGlobalService = SpringContext.getBean(AssetGlobalService.class);
64          if (!assetGlobalService.isAssetSeparateByPayment(assetGlobal)) {
65              // Show payment sequence number field only if a separate by payment was selected
66              fields.add(CamsPropertyConstants.AssetGlobal.SEPERATE_SOURCE_PAYMENT_SEQUENCE_NUMBER);
67          }
68  
69          if (!isAssetSeparate) {
70              fields.addAll(getAssetGlobalLocationHiddenFields(assetGlobal));
71  
72              if (assetGlobalService.existsInGroup(assetGlobalService.getNonNewAcquisitionCodeGroup(), assetGlobal.getAcquisitionTypeCode())) {
73                  // Fields in the add section
74                  fields.add(KFSConstants.ADD_PREFIX + "." + CamsPropertyConstants.AssetGlobal.ASSET_PAYMENT_DETAILS + "." + CamsPropertyConstants.AssetPaymentDetail.DOCUMENT_POSTING_DATE);
75                  fields.add(KFSConstants.ADD_PREFIX + "." + CamsPropertyConstants.AssetGlobal.ASSET_PAYMENT_DETAILS + "." + CamsPropertyConstants.AssetPaymentDetail.DOCUMENT_NUMBER);
76                  fields.add(KFSConstants.ADD_PREFIX + "." + CamsPropertyConstants.AssetGlobal.ASSET_PAYMENT_DETAILS + "." + CamsPropertyConstants.AssetPaymentDetail.DOCUMENT_TYPE_CODE);
77                  fields.add(KFSConstants.ADD_PREFIX + "." + CamsPropertyConstants.AssetGlobal.ASSET_PAYMENT_DETAILS + "." + CamsPropertyConstants.AssetPaymentDetail.ORIGINATION_CODE);
78  
79                  // Hiding some fields when the status of the document is not final.
80                  if (!document.getDocumentHeader().getWorkflowDocument().stateIsFinal()) {
81                      fields.addAll(getAssetGlobalPaymentsHiddenFields(assetGlobal));
82                  }
83              }
84          }
85          // if no separate asset is added, hide the button
86          if (isAssetSeparate && (assetGlobal.getAssetSharedDetails().isEmpty() || assetGlobal.getAssetSharedDetails().get(0).getAssetGlobalUniqueDetails().isEmpty())) {
87              fields.add("calculateEqualSourceAmountsButton");
88          }
89  
90  
91  
92          return fields;
93      }

95      @Override
96      public Set<String> getConditionallyReadOnlyPropertyNames(MaintenanceDocument document) {
97          Set<String> fields = super.getConditionallyReadOnlyPropertyNames(document);
98  
99          AssetGlobal assetGlobal = (AssetGlobal) document.getNewMaintainableObject().getBusinessObject();
100 
101         // "Asset Separate" document functionality
102         if (SpringContext.getBean(AssetGlobalService.class).isAssetSeparate(assetGlobal)) {
103             fields.addAll(getAssetGlobalDetailsReadOnlyFields());
104             fields.addAll(getAssetGlobalPaymentsReadOnlyFields(assetGlobal));
105         }
106         else if (assetGlobal.isCapitalAssetBuilderOriginIndicator()) {
107             // If asset global document is created from CAB, disallow add payment to collection.
108             fields.addAll(getAssetGlobalPaymentsReadOnlyFields(assetGlobal));
109         }
110 
111         return fields;
112     }
