115     /**
116      * Validates Pretag and its PretagDetail.
117      * 
118      * @return boolean false or true
119      */
120     public boolean processPretagValidation() {
121         boolean success = true;
122         boolean newDetailLine = false;
123 
124         setupConvenienceObjects();
125         success &= checkPurchaseOrderItemExists();
126         success &= checkAssetRepresentativePrincipalNameExists();
127         
128         if (newPretag.isActive()) {
129             success &= checkTotalDetailCount(newPretag, newDetailLine);
130             success &= isAllCampusBuildingRoomValid(newPretag.getPretagDetails());
131         }
132         else {
133             deactivePretagDetails(newPretag);
134         }
135 
136         return success;
137     }

139     /**
140      * validate the asset representative principal name.
141      * 
142      * @return boolean false or true
143      */
144     protected boolean checkAssetRepresentativePrincipalNameExists() {
145         boolean valid = true;
146         if (StringUtils.isNotBlank(newPretag.getPersonUniversal().getPrincipalName())) {
147             PersonService personService = SpringContext.getBean(org.kuali.rice.kim.service.PersonService.class);
148             Person person = personService.getPersonByPrincipalName(newPretag.getPersonUniversal().getPrincipalName());
149             if (person != null) {
150                 newPretag.setPersonUniversal(person);
151                 newPretag.setRepresentativeUniversalIdentifier(person.getPrincipalId());
152             }
153             else {
154                 putFieldError(CabPropertyConstants.Pretag.REPRESENTATIVE_ID, CamsKeyConstants.PreTag.ERROR_PRE_TAG_INVALID_REPRESENTATIVE_ID, newPretag.getPersonUniversal().getPrincipalName());
155                 newPretag.setPersonUniversal(null);
156                 newPretag.setRepresentativeUniversalIdentifier(null);
157                 valid &= false;
158             }
159         }
160         return valid;
161     }

163     /**
164      * validate the purchase order item existence in PurAp.
165      * 
166      * @return boolean false or true
167      */
168     protected boolean checkPurchaseOrderItemExists() {
169         boolean valid = true;
170         if (StringUtils.isNotBlank(newPretag.getPurchaseOrderNumber()) && newPretag.getItemLineNumber() != null) {
171             PurchaseOrderDocument purchaseOrderDoc = getPurApInfoService().getCurrentDocumentForPurchaseOrderIdentifier(Integer.valueOf(newPretag.getPurchaseOrderNumber()));
172             if (purchaseOrderDoc == null) {
173                 String label = SpringContext.getBean(DataDictionaryService.class).getDataDictionary().getBusinessObjectEntry(Pretag.class.getName()).getAttributeDefinition(CabPropertyConstants.Pretag.PURCHASE_ORDER_NUMBER).getLabel();
174                 putFieldError(CabPropertyConstants.Pretag.PURCHASE_ORDER_NUMBER, RiceKeyConstants.ERROR_EXISTENCE, label);
175                 valid = false;
176             }
177             else if (getItemByLineNumber(purchaseOrderDoc, newPretag.getItemLineNumber()) == null) {
178                 String label = SpringContext.getBean(DataDictionaryService.class).getDataDictionary().getBusinessObjectEntry(Pretag.class.getName()).getAttributeDefinition(CabPropertyConstants.Pretag.ITEM_LINE_NUMBER).getLabel();
179                 putFieldError(CabPropertyConstants.Pretag.ITEM_LINE_NUMBER, RiceKeyConstants.ERROR_EXISTENCE, label);
180                 valid = false;
181             }
182         }
183         return valid;
184     }

189      * @param items

193     protected PurApItem getItemByLineNumber(PurchaseOrderDocument purchaseOrderDocument, int lineNumber) {
194         List items = purchaseOrderDocument.getItems();
195         for (Iterator iter = items.iterator(); iter.hasNext();) {
196             PurApItem item = (PurApItem) iter.next();
197             if (item.getItemLineNumber() != null && item.getItemLineNumber().intValue() == lineNumber) {

199             }
200         }
201         return null;

206      * rule analysis

208      * @param document
209      * @param details
210      * @return true if the collection of {@link pretagDetail}s passes the sub-rules

212     public boolean isAllCampusBuildingRoomValid(List<PretagDetail> details) {
213         boolean success = true;
214 
215         // check if there are any pretag details
216         if (details.size() != 0) {
217 
218             // check each CampusBuildingRoom
219             int index = 0;
220             for (PretagDetail dtl : details) {
221                 String errorPath = MAINTAINABLE_ERROR_PREFIX + "pretagDetails[" + index + "]";
222                 GlobalVariables.getMessageMap().addToErrorPath(errorPath);
223                 success &= isCampusBuildingRoomValid(dtl);
224                 GlobalVariables.getMessageMap().removeFromErrorPath(errorPath);
225                 index++;
226             }
227         }
228 
229         return success;
230     }

240         setupConvenienceObjects();
241 
242         Pretag pretag = (Pretag) document.getNewMaintainableObject().getBusinessObject();
243         PretagDetail detail = (PretagDetail) bo;
244 
245         boolean success = true;
246         boolean newDetailLine = true;
247         if (detail.isActive()) {
248 
249             detail.setPurchaseOrderNumber(pretag.getPurchaseOrderNumber());
250             detail.setItemLineNumber(pretag.getItemLineNumber());
251 
252             success &= checkDuplicateTagNumber(pretag, detail.getCampusTagNumber());
253             success &= checkTotalDetailCount(pretag, newDetailLine);
254             success &= isCampusTagNumberValid(detail);
255             success &= isCampusBuildingRoomValid(detail);
256         }
257 
258         return success;
259     }

263      * 
264      * @return boolean indicating if validation succeeded

266     protected boolean checkDuplicateTagNumber(Pretag pretag, String tagNumber) {
267         boolean success = true;
268 
269         for (PretagDetail dtl : pretag.getPretagDetails()) {
270             if (dtl.getCampusTagNumber().equals(tagNumber) && dtl.isActive()) {
271                 GlobalVariables.getMessageMap().putError(CabPropertyConstants.Pretag.CAMPUS_TAG_NUMBER, CamsKeyConstants.ERROR_TAG_NUMBER_DUPLICATE, new String[] { tagNumber });

279     /**
280      * This method ensures that total {@link pretagDetail} tag details does not excees in quantity invoiced
281      * 
282      * @param dtl
283      * @return true if the detail tag doesn't exist in Asset
284      */
285     public boolean checkTotalDetailCount(Pretag pretag, boolean addLine) {
286         boolean success = true;
287         if (pretag.getQuantityInvoiced() != null) {
288             int totalActiveDetails = getActiveDetailsCount(pretag, addLine);
289             KualiDecimal totalNumerOfDetails = new KualiDecimal(totalActiveDetails);
290 
291             if (pretag.getQuantityInvoiced().compareTo(totalNumerOfDetails) < 0) {
292                 GlobalVariables.getMessageMap().putError(CabPropertyConstants.Pretag.CAMPUS_TAG_NUMBER, CamsKeyConstants.PreTag.ERROR_PRE_TAG_DETAIL_EXCESS, new String[] { pretag.getQuantityInvoiced().toString() + "" + " Total number of detail lines " + totalNumerOfDetails.toString() });
293                 success &= false;
294             }
295         }
296 
297         return success;
298     }

317     /**
318      * This method ensures that each {@link pretagDetail} tag number does not exist in Asset table
319      * 
320      * @param dtl
321      * @return true if the detail tag doesn't exist in Asset
322      */
323     public boolean isCampusTagNumberValid(PretagDetail dtl) {
324         boolean success = true;
325 
326         if ((dtl.getCampusTagNumber() != null) && (dtl.isActive() && !dtl.getCampusTagNumber().equalsIgnoreCase("N"))) {
327             Map<String, String> tagMap = new HashMap<String, String>();
328             tagMap.put(CabPropertyConstants.Pretag.CAMPUS_TAG_NUMBER, dtl.getCampusTagNumber());
329             int matchDetailCount = getMatchDetailCount(tagMap);
330             if ((getBoService().countMatching(Asset.class, tagMap) != 0) || (matchDetailCount > 0)) {
331                 GlobalVariables.getMessageMap().putError(CabPropertyConstants.Pretag.CAMPUS_TAG_NUMBER, CamsKeyConstants.PreTag.ERROR_PRE_TAG_NUMBER, new String[] { dtl.getCampusTagNumber() });
332                 success &= false;
333             }
334         }
335 
336         return success;
337     }

339     /**
340      * This method ensures that each {@link pretagDetail} buildingCode and buildingRoomNumber does exist in bulding and room tables
341      * 
342      * @param dtl
343      * @return true if the detail buildingCode and buildingRoomNumber does exist in building and room
344      */
345     public boolean isCampusBuildingRoomValid(PretagDetail dtl) {
346         boolean success = true;
347 
348         int originalErrorCount = GlobalVariables.getMessageMap().getErrorCount();
349         getDictionaryValidationService().validateBusinessObject(dtl);
350 
351         if (StringUtils.isNotBlank(dtl.getCampusCode()) && StringUtils.isNotBlank(dtl.getBuildingCode())) {
352             Map<String, String> preTagMap = new HashMap<String, String>();
353             preTagMap.put(KFSPropertyConstants.CAMPUS_CODE, dtl.getCampusCode());
354             preTagMap.put(KFSPropertyConstants.BUILDING_CODE, dtl.getBuildingCode());
355 
356             bo = (Building) SpringContext.getBean(BusinessObjectService.class).findByPrimaryKey(Building.class, preTagMap);
357             if (bo == null) {
358                 GlobalVariables.getMessageMap().putError(KFSPropertyConstants.BUILDING_CODE, CamsKeyConstants.ERROR_INVALID_BUILDING_CODE, new String[] { dtl.getCampusCode(), dtl.getBuildingCode() });
359             }
360 
361             if (StringUtils.isNotBlank(dtl.getBuildingRoomNumber())) {
362                 preTagMap.put(KFSPropertyConstants.BUILDING_ROOM_NUMBER, dtl.getBuildingRoomNumber());
363                 bo = (Room) SpringContext.getBean(BusinessObjectService.class).findByPrimaryKey(Room.class, preTagMap);
364                 if (bo == null) {
365                     GlobalVariables.getMessageMap().putError(KFSPropertyConstants.BUILDING_ROOM_NUMBER, CamsKeyConstants.ERROR_INVALID_ROOM_NUMBER, new String[] { dtl.getCampusCode(), dtl.getBuildingCode(), dtl.getBuildingRoomNumber() });
366                 }
367             }
368         }
369         success &= GlobalVariables.getMessageMap().getErrorCount() == originalErrorCount;
370 
371         return success;
372     }
