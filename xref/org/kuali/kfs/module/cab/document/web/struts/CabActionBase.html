36  public class CabActionBase extends KualiAction {
37      @Override
38      public ActionForward execute(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
39          ActionForward returnForward = mapping.findForward(RiceConstants.MAPPING_BASIC);
40  
41          // if found methodToCall, pass control to that method
42          try {
43              returnForward = super.execute(mapping, form, request, response);
44          }
45          catch (OjbOperationException e) {
46              // special handling for OptimisticLockExceptions
47              OjbOperationException ooe = (OjbOperationException) e;
48  
49              Throwable cause = ooe.getCause();
50              if (cause instanceof OptimisticLockException) {
51                  OptimisticLockException ole = (OptimisticLockException) cause;
52                  GlobalVariables.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, CabKeyConstants.DATA_EDIT_LOCK_ERROR);
53              }
54              else {
55                  // if exceptions are from 'save'
56                  throw e;
57              }
58          }
59          return returnForward;
60      }

63      /**
64       * This method will process the view document request by clicking on a specific document.
65       * 
66       * @param mapping ActionMapping
67       * @param form ActionForm
68       * @param request HttpServletRequest
69       * @param response HttpServletResponse
70       * @return ActionForward
71       * @throws Exception
72       */
73      public ActionForward viewDoc(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
74          String documentId = request.getParameter("documentNumber");
75          KualiWorkflowInfo kualiWorkflowInfo = SpringContext.getBean(KualiWorkflowInfo.class);
76          try {
77              RouteHeaderDTO routeHeader = kualiWorkflowInfo.getRouteHeader(Long.valueOf(documentId));
78              String docHandlerUrl = routeHeader.getDocumentUrl();
79              if (docHandlerUrl.indexOf("?") == -1) {
80                  docHandlerUrl += "?";
81              }
82              else {
83                  docHandlerUrl += "&";
84              }
85  
86              docHandlerUrl += "docId=" + documentId + "&" + "command=displayDocSearchView";
87              return new ActionForward(docHandlerUrl, true);
88          }
89          catch (WorkflowException e) {
90              throw new RuntimeException("Caught WorkflowException trying to get document handler URL from Workflow", e);
91          }
92      }
