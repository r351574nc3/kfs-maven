220     /**
221      * @see org.kuali.kfs.module.ld.document.service.LaborCorrectionDocumentService#persistLaborOriginEntries(java.lang.String,
222      *      java.util.Iterator)
223      */
224     protected void persistLaborOriginEntries(String fullPathUniqueFileName, Iterator<LaborOriginEntry> entries) {
225         File fileOut = new File(fullPathUniqueFileName);
226         FileOutputStream streamOut = null;
227         BufferedOutputStream bufferedStreamOut = null;
228         try {
229             streamOut = new FileOutputStream(fileOut);
230             bufferedStreamOut = new BufferedOutputStream(streamOut);
231 
232             byte[] newLine = "\n".getBytes();
233             while (entries.hasNext()) {
234                 LaborOriginEntry entry = entries.next();
235                 bufferedStreamOut.write(entry.getLine().getBytes());
236                 bufferedStreamOut.write(newLine);
237             }
238         }
239         catch (IOException e) {
240             LOG.error("unable to persist labor origin entries to file: " + fullPathUniqueFileName, e);
241             throw new RuntimeException("unable to persist origin entries to file.", e);
242         }
243         finally {
244             try {
245                 bufferedStreamOut.close();
246                 streamOut.close();
247             }
248             catch (IOException e) {
249                 LOG.error("unable to close output streams for file: " + fullPathUniqueFileName, e);
250                 throw new RuntimeException("unable to close output streams", e);
251             }
252         }
253     }

322     /**
323      * @see org.kuali.kfs.module.ld.document.service.LaborCorrectionDocumentService#retrievePersistedLaborOriginEntries(java.lang.String, int)
324      */
325     protected List<LaborOriginEntry> retrievePersistedLaborOriginEntries(String fullPathUniqueFileName, int abortThreshold) {
326         if ( LOG.isInfoEnabled() ) {
327             LOG.info( "Retrieving Entries from file " + fullPathUniqueFileName);
328         }
329         File fileIn = new File(fullPathUniqueFileName);
330         if (!fileIn.exists()) {
331             LOG.error("File " + fullPathUniqueFileName + " does not exist.");
332             throw new RuntimeException("File does not exist");
333         }
334         BufferedReader reader = null;
335         FileReader fReader = null;
336 
337         List<LaborOriginEntry> entries = new ArrayList<LaborOriginEntry>();
338         int lineNumber = 0;
339         try {
340             fReader = new FileReader(fileIn);
341             reader = new BufferedReader(fReader);
342             String line;
343             while ((line = reader.readLine()) != null) {
344                 LaborOriginEntry entry = new LaborOriginEntry();
345                 entry.setFromTextFileForBatch(line, lineNumber);
346                 if (abortThreshold != UNLIMITED_ABORT_THRESHOLD && lineNumber >= abortThreshold) {
347                     return null;
348                 }
349                 lineNumber++;
350                 entries.add(entry);
351             }
352         }
353         catch (IOException e) {
354             LOG.error("retrievePersistedOriginEntries() Error reading file " + fileIn.getAbsolutePath(), e);
355             throw new RuntimeException("Error reading file");
356         }
357         finally {
358             try {
359                 if (fReader != null) {
360                     fReader.close();
361                 }
362                 if (reader != null) {
363                     reader.close();
364                 }
365             }
366             catch (IOException e) {
367                 LOG.error("Unable to close file " + fileIn.getAbsolutePath(), e);
368                 throw new RuntimeException("Error closing file");
369             }
370         }
371         return entries;
372     }

477         } finally {
478             fileIn.close();
479         }

482     /**
483      * @see org.kuali.kfs.module.ld.document.service.LaborCorrectionDocumentService#writePersistedOriginEntriesToStream(org.kuali.kfs.module.ld.document.LaborCorrectionDocument,
484      *      org.kuali.module.gl.util)
485      */
486     public void persistOriginEntryGroupsForDocumentSave(LaborCorrectionDocument document, CorrectionDocumentEntryMetadata correctionDocumentEntryMetadata) {
487         if (correctionDocumentEntryMetadata.getAllEntries() == null && !correctionDocumentEntryMetadata.isRestrictedFunctionalityMode()) {
488             // if we don't have origin entries loaded and not in restricted functionality mode, then there's nothing worth
489             // persisting
490             removePersistedInputOriginEntries(document);
491             removePersistedOutputOriginEntries(document);
492             return;
493         }
494 
495         if (!correctionDocumentEntryMetadata.getDataLoadedFlag() && !correctionDocumentEntryMetadata.isRestrictedFunctionalityMode()) {
496             // data is not loaded (maybe user selected a new group with no rows)
497             // clear out existing data
498             removePersistedInputOriginEntries(document);
499             removePersistedOutputOriginEntries(document);
500             return;
501         }
502 
503         // reload the group from the origin entry service
504         Iterator<LaborOriginEntry> inputGroupEntries;
505         KualiWorkflowDocument workflowDocument = document.getDocumentHeader().getWorkflowDocument();
506         if ((workflowDocument.stateIsSaved() && !(correctionDocumentEntryMetadata.getInputGroupIdFromLastDocumentLoad() != null && correctionDocumentEntryMetadata.getInputGroupIdFromLastDocumentLoad().equals(document.getCorrectionInputFileName()))) || workflowDocument.stateIsInitiated()) {
507             // we haven't saved the origin entry group yet, so let's load the entries from the DB and persist them for the document
508             // this could be because we've previously saved the doc, but now we are now using a new input group, so we have to
509             // repersist the input group
510             
511             
512             //OriginEntryGroup group = originEntryGroupService.getExactMatchingEntryGroup(document.getCorrectionInputGroupId());
513             File file = new File(document.getCorrectionInputFileName());
514             
515             inputGroupEntries = new LaborOriginEntryFileIterator(file);
516             persistInputOriginEntriesForInitiatedOrSavedDocument(document, inputGroupEntries);
517 
518             // we've exhausted the iterator for the origin entries group
519             // reload the iterator from the file
520             inputGroupEntries = retrievePersistedInputOriginEntriesAsIterator(document);
521         }
522         else if (workflowDocument.stateIsSaved() && correctionDocumentEntryMetadata.getInputGroupIdFromLastDocumentLoad().equals(document.getCorrectionInputFileName())) {
523             // we've saved the origin entries before, so just retrieve them
524             inputGroupEntries = retrievePersistedInputOriginEntriesAsIterator(document);
525         }
526         else {
527             LOG.error("Unexpected state while trying to persist/retrieve GLCP origin entries during document save: document status is " + workflowDocument.getStatusDisplayValue() + " selected input group: " + document.getCorrectionInputFileName() + " last saved input group: " + correctionDocumentEntryMetadata.getInputGroupIdFromLastDocumentLoad());
528             throw new RuntimeException("Error persisting GLCP document origin entries.");
529         }
530 
531         OriginEntryStatistics statistics;
532         if (LaborCorrectionDocumentService.CORRECTION_TYPE_MANUAL.equals(correctionDocumentEntryMetadata.getEditMethod())) {
533             // persist the allEntries element as the output group, since it has all of the modifications made by during the manual
534             // edits
535             Collection allEntries = new ArrayList();
536             allEntries.addAll(correctionDocumentEntryMetadata.getAllEntries());
537             persistOutputLaborOriginEntriesForInitiatedOrSavedDocument(document, allEntries.iterator());
538 
539             // even though the struts action handler may have computed the doc totals, let's recompute them
540             statistics = CorrectionDocumentUtils.getStatistics(correctionDocumentEntryMetadata.getAllEntries());
541         }
542         else if (LaborCorrectionDocumentService.CORRECTION_TYPE_CRITERIA.equals(correctionDocumentEntryMetadata.getEditMethod())) {
543             // we want to persist the values of the output group. So reapply all of the criteria on each entry, one at a time
544 
545             BufferedOutputStream bufferedOutputStream = null;
546             try {
547                 bufferedOutputStream = openEntryOutputStreamForOutputGroup(document);
548                 statistics = new OriginEntryStatistics();
549                 byte[] newLine = "\n".getBytes();
550 
551                 while (inputGroupEntries.hasNext()) {
552                     LaborOriginEntry entry = inputGroupEntries.next();
553 
554                     entry = (LaborOriginEntry) CorrectionDocumentUtils.applyCriteriaToEntry(entry, correctionDocumentEntryMetadata.getMatchCriteriaOnly(), document.getCorrectionChangeGroup());
555                     if (entry != null) {
556                         CorrectionDocumentUtils.updateStatisticsWithEntry(entry, statistics);
557                         bufferedOutputStream.write(entry.getLine().getBytes());
558                         bufferedOutputStream.write(newLine);
559                     }
560                     // else it was null, which means that the match criteria only flag was set, and the entry didn't match the
561                     // criteria
562                 }
563             }
564             catch (IOException e) {
565                 LOG.error("Unable to persist persisted output entry", e);
566                 throw new RuntimeException("Unable to persist output entry",e);
567             }
568             finally {
569                 if (bufferedOutputStream != null) {
570                     try {
571                         bufferedOutputStream.close();
572                     }
573                     catch (IOException e) {
574                         LOG.error("Unable to close output stream for persisted output entries", e);
575                         throw new RuntimeException("Unable to close output entry file",e);
576                     }
577                 }
578             }
579         }
580         else if (LaborCorrectionDocumentService.CORRECTION_TYPE_REMOVE_GROUP_FROM_PROCESSING.equals(correctionDocumentEntryMetadata.getEditMethod())) {
581             // just wipe out the previous output entries
582             removePersistedOutputOriginEntries(document);
583             statistics = new OriginEntryStatistics();
584         }
585         else {
586             throw new RuntimeException("Unrecognized edit method: " + correctionDocumentEntryMetadata.getEditMethod());
587         }
588 
589         CorrectionDocumentUtils.copyStatisticsToDocument(statistics, document);
590     }
