53      /**
54       * Populates source a accounting line encumbrance code (D, R or null) for a given source accounting line 
55       * 
56       * @see org.kuali.kfs.fp.document.web.struts.JournalVoucherForm#populateSourceAccountingLineEncumbranceCode(org.kuali.kfs.sys.businessobject.SourceAccountingLine)
57       */
58      @Override
59      protected void populateSourceAccountingLineEncumbranceCode(SourceAccountingLine sourceLine) {
60          BalanceType selectedBalanceType = getSelectedBalanceType();
61          if (selectedBalanceType != null && StringUtils.isNotBlank(selectedBalanceType.getCode())) {
62              sourceLine.setBalanceTyp(selectedBalanceType);
63              sourceLine.setBalanceTypeCode(selectedBalanceType.getCode());
64  
65              /*
66               * set the encumbrance update code appropriately. If balance type is an external encumbrance, update code can be D or R.
67               * For all other encumbrance types it must be D. If the balance type is not an encumbrance type the update code is set
68               * to null
69               */
70              if (selectedBalanceType.isFinBalanceTypeEncumIndicator()) {
71                  if (StringUtils.isBlank(sourceLine.getEncumbranceUpdateCode()) || !KFSConstants.ENCUMB_UPDT_DOCUMENT_CD.equals(sourceLine.getEncumbranceUpdateCode())) {
72                      sourceLine.setEncumbranceUpdateCode(KFSConstants.ENCUMB_UPDT_REFERENCE_DOCUMENT_CD);
73                  }
74              }
75              else {
76                  sourceLine.setEncumbranceUpdateCode(null);
77              }
78          }
79          else {
80              // it's the first time in, the form will be empty the first time in set up default selection value
81              selectedBalanceType = SpringContext.getBean(BalanceTypeService.class).getBalanceTypeByCode(KFSConstants.BALANCE_TYPE_ACTUAL);
82              setSelectedBalanceType(selectedBalanceType);
83              setOriginalBalanceType(selectedBalanceType.getCode());
84  
85              sourceLine.setEncumbranceUpdateCode(null);
86          }
87      }
