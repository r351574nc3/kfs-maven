34  /**
35   * Data dictionary definition that includes metadata for an accounting document about one of its groups of accounting lines
36   * (typically source vs. target, but this should open things up).
37   */
38  public class LaborExpenseTransferAccountingLineAuthorizer extends AccountingLineAuthorizerBase {
39     
40      /**
41       * @see org.kuali.kfs.sys.document.authorization.AccountingLineAuthorizerBase#determineEditPermissionOnField(org.kuali.kfs.sys.document.AccountingDocument, org.kuali.kfs.sys.businessobject.AccountingLine, java.lang.String, java.lang.String)
42       */
43      @Override
44      public boolean determineEditPermissionOnField(AccountingDocument accountingDocument, AccountingLine accountingLine, String accountingLineCollectionProperty, String fieldName, boolean editablePage) {
45          
46          if (fieldName.equals(KFSPropertyConstants.AMOUNT)) {
47              return true;
48          }
49          
50          if (fieldName.equals(LaborPropertyConstants.PAYROLL_TOTAL_HOURS)) {
51              return true;
52          }
53          
54          return super.determineEditPermissionOnField(accountingDocument, accountingLine, accountingLineCollectionProperty, fieldName, editablePage);
55      }
56      
57      /**
58       * @see org.kuali.kfs.sys.document.authorization.AccountingLineAuthorizerBase#getActionMap(org.kuali.kfs.sys.businessobject.AccountingLine,
59       *      java.lang.String, java.lang.Integer, java.lang.String)
60       */
61      @Override
62      protected Map<String, AccountingLineViewAction> getActionMap(AccountingLineRenderingContext accountingLineRenderingContext, String accountingLinePropertyName, Integer accountingLineIndex, String groupTitle) {
63          Map<String, AccountingLineViewAction> actionMap = super.getActionMap(accountingLineRenderingContext, accountingLinePropertyName, accountingLineIndex, groupTitle);
64  
65          if (accountingLineRenderingContext.getAccountingLine().isSourceAccountingLine()) {
66              AccountingLineViewAction copyAction = this.getCopyAction(accountingLineRenderingContext.getAccountingLine(), accountingLinePropertyName, accountingLineIndex, groupTitle);
67              actionMap.put(KFSConstants.COPY_METHOD, copyAction);
68          }
69  
70          return actionMap;
71      }
72  
73      /**
74       * construct the copy action for the given accounting line, typically, a new accounting line
75       * 
76       * @param accountingLine the given accounting line
77       * @param accountingLinePropertyName the property name of the given account line, typically, the form name
78       * @param accountingLineIndex the index of the given accounting line in its accounting line group
79       * @param groupTitle the title of the accounting line group
80       * @return the copy action for the given accounting line
81       */
82      protected AccountingLineViewAction getCopyAction(AccountingLine accountingLine, String accountingLinePropertyName, Integer accountingLineIndex, String groupTitle) {
83          String actionMethod = this.getCopyLineMethod(accountingLine, accountingLinePropertyName, accountingLineIndex);
84          String actionLabel = this.getActionLabel(KFSKeyConstants.AccountingLineViewRendering.ACCOUNTING_LINE_COPY_ACTION_LABEL, groupTitle, accountingLineIndex + 1);
85  
86          KualiConfigurationService kualiConfigurationService = SpringContext.getBean(KualiConfigurationService.class);
87          String imagesPath = kualiConfigurationService.getPropertyString(KNSConstants.APPLICATION_EXTERNALIZABLE_IMAGES_URL_KEY);
88          String actionImageName = imagesPath + "tinybutton-copy2.gif";
89  
90          return new AccountingLineViewAction(actionMethod, actionLabel, actionImageName);
91      }
92  
93      /**
94       * Builds the action method name of the method that deletes accounting lines for this group
95       * 
96       * @param accountingLine the accounting line an action is being checked for
97       * @param accountingLinePropertyName the property name of the accounting line
98       * @param accountingLineIndex the index of the given accounting line within the the group being rendered
99       * @return the action method name of the method that deletes accounting lines for this group
100      */
101     protected String getCopyLineMethod(AccountingLine accountingLine, String accountingLineProperty, Integer accountingLineIndex) {
102         final String infix = getActionInfixForExtantAccountingLine(accountingLine, accountingLineProperty);
103         return "copyAccountingLine.line" + accountingLineIndex + ".anchoraccounting" + infix + "Anchor";
104     }
