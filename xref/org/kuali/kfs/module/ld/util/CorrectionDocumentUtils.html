31  /**
32   * This class provides utility methods for the Labor correction document
33   */
34  public class CorrectionDocumentUtils {
35      /**
36       * Returns whether an origin entry matches the passed in criteria. If both the criteria and actual value are both String types
37       * and are empty, null, or whitespace only, then they will match.
38       * 
39       * @param cc correction criteria to test against origin entry
40       * @param oe origin entry to test
41       * @return true if origin entry matches the passed in criteria
42       */
43      public static boolean laborEntryMatchesCriteria(CorrectionCriteria cc, OriginEntryFull oe) {
44          LaborOriginEntryFieldFinder loeff = new LaborOriginEntryFieldFinder();
45          LaborOriginEntry loe = (LaborOriginEntry) oe;
46          Object fieldActualValue = loe.getFieldValue(cc.getCorrectionFieldName());
47          String fieldTestValue = StringUtils.isBlank(cc.getCorrectionFieldValue()) ? "" : cc.getCorrectionFieldValue();
48          String fieldType = loeff.getFieldType(cc.getCorrectionFieldName());
49          String fieldActualValueString = org.kuali.kfs.gl.document.CorrectionDocumentUtils.convertToString(fieldActualValue, fieldType);
50  
51          if ("String".equals(fieldType) || "sw".equals(cc.getCorrectionOperatorCode()) || "ew".equals(cc.getCorrectionOperatorCode()) || "ct".equals(cc.getCorrectionOperatorCode())) {
52              return org.kuali.kfs.gl.document.CorrectionDocumentUtils.compareStringData(cc, fieldTestValue, fieldActualValueString);
53          }
54          int compareTo = 0;
55          try {
56              if (fieldActualValue == null) {
57                  return false;
58              }
59              if ("Integer".equals(fieldType)) {
60                  compareTo = ((Integer) fieldActualValue).compareTo(Integer.parseInt(fieldTestValue));
61              }
62              if ("KualiDecimal".equals(fieldType)) {
63                  compareTo = ((KualiDecimal) fieldActualValue).compareTo(new KualiDecimal(Double.parseDouble(fieldTestValue)));
64              }
65              if ("BigDecimal".equals(fieldType)) {
66                  compareTo = ((BigDecimal) fieldActualValue).compareTo(new BigDecimal(Double.parseDouble(fieldTestValue)));
67  
68              }
69              if ("Date".equals(fieldType)) {
70                  SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd");
71                  compareTo = ((Date) fieldActualValue).compareTo(df.parse(fieldTestValue));
72              }
73          }
74          catch (Exception e) {
75              // any exception while parsing data return false
76              return false;
77          }
78          return org.kuali.kfs.gl.document.CorrectionDocumentUtils.compareTo(compareTo, cc.getCorrectionOperatorCode());
79      }
80  
81      /**
82       * Returns whether the labor entry matches any of the criteria groups
83       * 
84       * @param entry labor origin entry
85       * @param groups collection of correction change group
86       * @return true if labor origin entry matches any of the criteria groups
87       */
88      public static boolean doesLaborEntryMatchAnyCriteriaGroups(OriginEntryFull entry, Collection<CorrectionChangeGroup> groups) {
89          boolean anyGroupMatch = false;
90          for (CorrectionChangeGroup ccg : groups) {
91              int matches = 0;
92              for (CorrectionCriteria cc : ccg.getCorrectionCriteria()) {
93                  if (CorrectionDocumentUtils.laborEntryMatchesCriteria(cc, entry)) {
94                      matches++;
95                  }
96              }
97  
98              // If they all match, change it
99              if (matches == ccg.getCorrectionCriteria().size()) {
100                 anyGroupMatch = true;
101                 break;
102             }
103         }
104         return anyGroupMatch;
105     }
