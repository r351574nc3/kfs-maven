36  public class PositionFundingLookupableHelperServiceImpl extends AbstractLookupableHelperServiceImpl {
37      /**
38       * @see org.kuali.rice.kns.lookup.Lookupable#getInquiryUrl(org.kuali.rice.kns.bo.BusinessObject, java.lang.String)
39       */
40      @Override
41      public HtmlData getInquiryUrl(BusinessObject businessObject, String propertyName) {
42          if (KFSPropertyConstants.POSITION_NUMBER.equals(propertyName)) {
43              LaborCalculatedSalaryFoundationTracker CSFTracker = (LaborCalculatedSalaryFoundationTracker) businessObject;
44              AbstractLaborInquirableImpl positionDataDetailsInquirable = new PositionDataDetailsInquirableImpl();
45  
46              Map<String, String> fieldValues = new HashMap<String, String>();
47              fieldValues.put(propertyName, CSFTracker.getPositionNumber());
48  
49              BusinessObject positionData = positionDataDetailsInquirable.getBusinessObject(fieldValues);
50  
51              return positionData == null ? new AnchorHtmlData(KFSConstants.EMPTY_STRING, KFSConstants.EMPTY_STRING) : positionDataDetailsInquirable.getInquiryUrl(positionData, propertyName);
52          }
53          return (new PositionFundingInquirableImpl()).getInquiryUrl(businessObject, propertyName);
54      }

56      /**
57       * @see org.kuali.rice.kns.lookup.AbstractLookupableHelperServiceImpl#getSearchResults(java.util.Map)
58       */
59      @Override
60      public List<? extends BusinessObject> getSearchResults(Map<String, String> fieldValues) {
61          // remove hidden fields
62          LookupUtils.removeHiddenCriteriaFields(getBusinessObjectClass(), fieldValues);
63  
64          setBackLocation(fieldValues.get(KFSConstants.BACK_LOCATION));
65          setDocFormKey(fieldValues.get(KFSConstants.DOC_FORM_KEY));
66          setReferencesToRefresh(fieldValues.get(KFSConstants.REFERENCES_TO_REFRESH));
67  
68          List searchResults = (List) getLookupService().findCollectionBySearchHelper(getBusinessObjectClass(), fieldValues, false);
69          // sort list if default sort column given
70          List defaultSortColumns = getDefaultSortColumns();
71          if (defaultSortColumns.size() > 0) {
72              Collections.sort(searchResults, new BeanPropertyComparator(getDefaultSortColumns(), true));
73          }
74          return searchResults;
75      }
