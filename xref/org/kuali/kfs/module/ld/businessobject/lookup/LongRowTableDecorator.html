16  package org.kuali.kfs.module.ld.businessobject.lookup;
17  
18  import java.util.List;
19  
20  import org.apache.commons.lang.StringUtils;
21  import org.displaytag.decorator.TableDecorator;
22  import org.displaytag.properties.MediaTypeEnum;
23  import org.kuali.rice.kns.web.ui.Column;
24  import org.kuali.rice.kns.web.ui.ResultRow;
25  
26  /**
27   * Decorator class for the Long Row Table.
28   */
29  
30  public class LongRowTableDecorator extends TableDecorator {
31  
32      private static final int numOfColumnInEachRow = 12;
33      private static final int numOfColumnInFirstRow = 14;
34  
35      /**
36       * @see org.displaytag.decorator.TableDecorator#finishRow()
37       */
38      @Override
39      public String finishRow() {
40          MediaTypeEnum mediaType = (MediaTypeEnum) getPageContext().getAttribute("mediaType");
41          ResultRow row = (ResultRow) getCurrentRowObject();
42  
43          if (MediaTypeEnum.HTML.equals(mediaType)) {
44              StringBuffer rowBuffer = new StringBuffer();
45  
46              List columns = row.getColumns();
47              int columnCount = columns.size();
48              int numOfRows = (int) Math.ceil(1.0 * (columnCount - numOfColumnInFirstRow) / numOfColumnInEachRow);
49  
50              rowBuffer.append("<tr>").append("<td colspan='" + numOfColumnInFirstRow + "'><br/><center>");
51              rowBuffer.append("<table class='datatable-80' cellspacing='0' cellpadding='0' >");
52  
53              for (int rowIndex = 0; rowIndex < numOfRows; rowIndex++) {
54  
55                  rowBuffer.append("<tr>");
56                  for (int columnIndex = 0; columnIndex < numOfColumnInEachRow; columnIndex++) {
57  
58                      int currentPosition = rowIndex * numOfColumnInEachRow + columnIndex + numOfColumnInFirstRow;
59                      String title = currentPosition < columnCount ? ((Column) columns.get(currentPosition)).getColumnTitle() : null;
60  
61                      rowBuffer.append("<th>");
62                      rowBuffer.append(!StringUtils.isBlank(title) ? title : "&nbsp;");
63                      rowBuffer.append("</th>");
64                  }
65                  rowBuffer.append("</tr>");
66  
67                  rowBuffer.append("<tr>");
68                  for (int columnIndex = 0; columnIndex < numOfColumnInEachRow; columnIndex++) {
69                      int currentPosition = rowIndex * numOfColumnInEachRow + columnIndex + numOfColumnInFirstRow;
70                      String value = currentPosition < columnCount ? ((Column) columns.get(currentPosition)).getPropertyValue() : null;
71  
72                      rowBuffer.append("<td class='infocell'>");
73                      rowBuffer.append(!StringUtils.isBlank(value) ? value : "&nbsp;");
74                      rowBuffer.append("</td>");
75                  }
76                  rowBuffer.append("</tr>");
77              }
78              rowBuffer.append("</table></center><br/><br/></td></tr>");
79              return rowBuffer.toString();
80          }
81          return super.finishRow();
82      }
83  }
