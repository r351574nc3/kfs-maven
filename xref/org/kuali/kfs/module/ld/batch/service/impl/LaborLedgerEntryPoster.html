31  /**
32   * The class is used to post a transaction to labor ledger entry table
33   */
34  @Transactional
35  public class LaborLedgerEntryPoster implements PostTransaction {
36      private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(LaborLedgerEntryPoster.class);
37      private LaborAccountingCycleCachingService laborAccountingCycleCachingService;
38  
39      /**
40       * @see org.kuali.kfs.gl.batch.service.PostTransaction#post(org.kuali.kfs.gl.businessobject.Transaction, int, java.util.Date)
41       */
42      public String post(Transaction transaction, int mode, java.util.Date postDate, ReportWriterService posterReportWriterService) {
43          String operationType = KFSConstants.OperationType.INSERT;
44          LedgerEntry ledgerEntry = new LedgerEntry((LaborTransaction) transaction);
45          // ObjectUtil.buildObject(ledgerEntry, transaction);
46  
47          try {
48              ledgerEntry.setTransactionLedgerEntrySequenceNumber(getLaborAccountingCycleCachingService().getMaxLaborSequenceNumber(ledgerEntry) + 1);
49              ledgerEntry.setTransactionPostingDate(new Date(postDate.getTime()));
50              getLaborAccountingCycleCachingService().insertLedgerEntry(ledgerEntry);
51          }
52          catch (Exception e) {
53              throw new RuntimeException(e);
54          }
55          
56          return operationType;
57      }
58  
59      /**
60       * @see org.kuali.kfs.gl.batch.service.PostTransaction#getDestinationName()
61       */
62      public String getDestinationName() {
63          return LaborConstants.DestinationNames.LEDGER_ENRTY;
64      }
65  
66      /**
67       * Gets the laborAccountingCycleCachingService attribute. 
68       * @return Returns the laborAccountingCycleCachingService.
69       */
70      public LaborAccountingCycleCachingService getLaborAccountingCycleCachingService() {
71          return laborAccountingCycleCachingService;
72      }
73  
74      /**
75       * Sets the laborAccountingCycleCachingService attribute value.
76       * @param laborAccountingCycleCachingService The laborAccountingCycleCachingService to set.
77       */
78      public void setLaborAccountingCycleCachingService(LaborAccountingCycleCachingService laborAccountingCycleCachingService) {
79          this.laborAccountingCycleCachingService = laborAccountingCycleCachingService;
80      }
