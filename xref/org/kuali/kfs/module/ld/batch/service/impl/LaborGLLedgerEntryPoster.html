33  /**
34   * This class is used to post a transaction into labor GL entry table
35   */
36  @Transactional
37  public class LaborGLLedgerEntryPoster implements PostTransaction {
38      private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(LaborGLLedgerEntryPoster.class);
39      
40      private LaborGeneralLedgerEntryService laborGeneralLedgerEntryService;
41  
42      /**
43       * @see org.kuali.kfs.gl.batch.service.PostTransaction#post(org.kuali.kfs.gl.businessobject.Transaction, int, java.util.Date)
44       */
45      public String post(Transaction transaction, int mode, java.util.Date postDate, ReportWriterService posterReportWriterService) {
46          String operationType = KFSConstants.OperationType.INSERT;
47          LaborGeneralLedgerEntry laborGeneralLedgerEntry = new LaborGeneralLedgerEntry();
48          ObjectUtil.buildObject(laborGeneralLedgerEntry, transaction);
49  
50          laborGeneralLedgerEntry.setTransactionDate(new Date(postDate.getTime()));
51  
52          if(! transaction.getFinancialBalanceTypeCode().equalsIgnoreCase("CB")){
53              laborGeneralLedgerEntry.setTransactionDebitCreditCode(this.getDebitCreditCode(transaction));
54          }
55          laborGeneralLedgerEntry.setTransactionLedgerEntryAmount(this.getTransactionAmount(transaction));
56          
57          String encumbranceUpdateCode = this.getEncumbranceUpdateCode(transaction);
58          if(StringUtils.isNotEmpty(encumbranceUpdateCode)) {
59              laborGeneralLedgerEntry.setTransactionEncumbranceUpdateCode(encumbranceUpdateCode);
60          }
61  
62          Integer sequenceNumber = laborGeneralLedgerEntryService.getMaxSequenceNumber(laborGeneralLedgerEntry) + 1;
63          laborGeneralLedgerEntry.setTransactionLedgerEntrySequenceNumber(sequenceNumber);
64          try {
65              laborGeneralLedgerEntryService.save(laborGeneralLedgerEntry);
66          }
67          catch (Exception e) {
68              throw new RuntimeException(e);
69          }
70          return operationType;
71      }
72  
73      /**
74       * @return the debit credit code
75       */
76      protected String getDebitCreditCode(Transaction transaction) {
77          KualiDecimal transactionAmount = transaction.getTransactionLedgerEntryAmount();
78          return DebitCreditUtil.getDebitCreditCode(transactionAmount, false);
79      }
80  
81      /**
82       * @return the transaction amount
83       */
84      protected KualiDecimal getTransactionAmount(Transaction transaction) {
85          KualiDecimal transactionAmount = transaction.getTransactionLedgerEntryAmount();
86          return transactionAmount.abs();
87      }
88  
89      /**
90       * @return the encumbrance update code
91       */
92      protected String getEncumbranceUpdateCode(Transaction transaction) {
93          String encumbranceUpdateCode = transaction.getTransactionEncumbranceUpdateCode();
94          if (KFSConstants.ENCUMB_UPDT_DOCUMENT_CD.equals(encumbranceUpdateCode) || KFSConstants.ENCUMB_UPDT_REFERENCE_DOCUMENT_CD.equals(encumbranceUpdateCode)) {
95              return encumbranceUpdateCode;
96          }
97          return null;
98      }
99  
100     /**
101      * @see org.kuali.kfs.gl.batch.service.PostTransaction#getDestinationName()
102      */
103     public String getDestinationName() {
104         return LaborConstants.DestinationNames.LABOR_GL_ENTRY;
105     }
106 
107     /**
108      * Sets the laborGeneralLedgerEntryService attribute value.
109      * 
110      * @param laborGeneralLedgerEntryService The laborGeneralLedgerEntryService to set.
111      */
112     public void setLaborGeneralLedgerEntryService(LaborGeneralLedgerEntryService laborGeneralLedgerEntryService) {
113         this.laborGeneralLedgerEntryService = laborGeneralLedgerEntryService;
114     }
