38   * To provide its clients with access to the benefit calculation.

40  @Transactional
41  public class LaborBenefitsCalculationServiceImpl implements LaborBenefitsCalculationService {
42      private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(LaborBenefitsCalculationServiceImpl.class);

44      private BusinessObjectService businessObjectService;
45      private LaborPositionObjectBenefitService laborPositionObjectBenefitService;

47      /**
48       * @see org.kuali.kfs.module.ld.service.LaborBenefitsCalculationService#getBenefitsCalculation(java.lang.Integer,
49       *      java.lang.String, java.lang.String)
50       */
51      public BenefitsCalculation getBenefitsCalculation(Integer universityFiscalYear, String chartOfAccountsCode, String benefitTypeCode) {
52          Map<String, Object> fieldValues = new HashMap<String, Object>();
53          fieldValues.put(KFSPropertyConstants.UNIVERSITY_FISCAL_YEAR, universityFiscalYear);
54          fieldValues.put(KFSPropertyConstants.CHART_OF_ACCOUNTS_CODE, chartOfAccountsCode);
55          fieldValues.put(LaborPropertyConstants.POSITION_BENEFIT_TYPE_CODE, benefitTypeCode);
56  
57          return (BenefitsCalculation) businessObjectService.findByPrimaryKey(BenefitsCalculation.class, fieldValues);
58      }

60      /**
61       * @see org.kuali.kfs.module.ld.service.LaborBenefitsCalculationService#calculateFringeBenefit(java.lang.Integer,
62       *      java.lang.String, java.lang.String, org.kuali.rice.kns.util.KualiDecimal)
63       */
64      public KualiDecimal calculateFringeBenefit(Integer fiscalYear, String chartCode, String objectCode, KualiDecimal salaryAmount) {
65          LaborObject laborObject = new LaborObject();
66          
67          laborObject.setUniversityFiscalYear(fiscalYear);
68          laborObject.setChartOfAccountsCode(chartCode);
69          laborObject.setFinancialObjectCode(objectCode);
70  
71          laborObject = (LaborObject) businessObjectService.retrieve(laborObject);
72  
73          return calculateFringeBenefit(laborObject, salaryAmount);
74      }
75  
76      /**
77       * @see org.kuali.kfs.module.ld.service.LaborBenefitsCalculationService#calculateFringeBenefit(org.kuali.kfs.module.ld.businessobject.LaborObject,
78       *      org.kuali.rice.kns.util.KualiDecimal)
79       */
80      public KualiDecimal calculateFringeBenefit(LaborLedgerObject laborLedgerObject, KualiDecimal salaryAmount) {
81          KualiDecimal fringeBenefit = KualiDecimal.ZERO;
82  
83          if (salaryAmount == null || salaryAmount.isZero() || ObjectUtils.isNull(laborLedgerObject)) {
84              return fringeBenefit;
85          }
86  
87          String FringeOrSalaryCode = laborLedgerObject.getFinancialObjectFringeOrSalaryCode();
88          if (!LaborConstants.SalaryExpenseTransfer.LABOR_LEDGER_SALARY_CODE.equals(FringeOrSalaryCode)) {
89              return fringeBenefit;
90          }
91  
92          Integer fiscalYear = laborLedgerObject.getUniversityFiscalYear();
93          String chartOfAccountsCode = laborLedgerObject.getChartOfAccountsCode();
94          String objectCode = laborLedgerObject.getFinancialObjectCode();
95  
96          Collection<PositionObjectBenefit> positionObjectBenefits = laborPositionObjectBenefitService.getPositionObjectBenefits(fiscalYear, chartOfAccountsCode, objectCode);
97          for (PositionObjectBenefit positionObjectBenefit : positionObjectBenefits) {
98              KualiDecimal benefitAmount = this.calculateFringeBenefit(positionObjectBenefit, salaryAmount);
99              fringeBenefit = fringeBenefit.add(benefitAmount);
100         }
101 
102         return fringeBenefit;
103     }
104 
105     /**
106      * @see org.kuali.kfs.module.ld.service.LaborBenefitsCalculationService#calculateFringeBenefit(org.kuali.kfs.module.ld.businessobject.PositionObjectBenefit,
107      *      org.kuali.rice.kns.util.KualiDecimal)
108      */
109     public KualiDecimal calculateFringeBenefit(PositionObjectBenefit positionObjectBenefit, KualiDecimal salaryAmount) {
110         if (salaryAmount == null || salaryAmount.isZero() || ObjectUtils.isNull(positionObjectBenefit)) {
111             return KualiDecimal.ZERO;
112         }
113 
114         // calculate the benefit amount (ledger amt * (benfit pct/100) )
115         KualiDecimal fringeBenefitPercent = positionObjectBenefit.getBenefitsCalculation().getPositionFringeBenefitPercent();
116         return fringeBenefitPercent.multiply(salaryAmount).divide(KFSConstants.ONE_HUNDRED.kualiDecimalValue());
117     }
