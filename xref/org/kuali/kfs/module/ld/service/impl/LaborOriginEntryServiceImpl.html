55      public OriginEntryStatistics getStatistics(String fileName) {
56          LOG.debug("getStatistics() started");
57          OriginEntryStatistics oes = new OriginEntryStatistics();
58          KualiDecimal totalCredit = KualiDecimal.ZERO;
59          KualiDecimal totalDebit = KualiDecimal.ZERO;
60          Integer rowCount = 0;
61          FileReader INPUT_FILE = null;
62          BufferedReader INPUT_FILE_br;
63          try {
64              INPUT_FILE = new FileReader(fileName);
65          }
66          catch (FileNotFoundException e) {
67              throw new RuntimeException(e);
68          }
69          Collection<LaborOriginEntry> entryCollection = new ArrayList();
70          INPUT_FILE_br = new BufferedReader(INPUT_FILE);
71  
72          try {
73              String currentLine = INPUT_FILE_br.readLine();
74              while (currentLine != null) {
75                  KualiDecimal amount = KualiDecimal.ZERO;
76                  if (!currentLine.substring(109, 126).trim().equals(GeneralLedgerConstants.EMPTY_CODE)) {
77                      try {
78                          amount = new KualiDecimal(currentLine.substring(109, 126).trim());
79  
80                          // TODO:- Check with FIS (Row count should be all rows?)
81                          rowCount++;
82                      }
83                      catch (NumberFormatException e) {
84                      }
85                  }
86                  else {
87                      amount = KualiDecimal.ZERO;
88                  }
89                  String debitOrCreditCode = currentLine.substring(126, 127);
90  
91                  if (debitOrCreditCode.equals(KFSConstants.GL_CREDIT_CODE)) {
92                      totalCredit.add(amount);
93                  }
94                  else if (debitOrCreditCode.equals(KFSConstants.GL_DEBIT_CODE)) {
95                      totalDebit.add(amount);
96                  }
97                  currentLine = INPUT_FILE_br.readLine();
98              }
99              INPUT_FILE_br.close();
100         }
101         catch (IOException e) {
102             // FIXME: do whatever should be done here
103             throw new RuntimeException(e);
104         }
105 
106         oes.setCreditTotalAmount(totalCredit);
107         oes.setDebitTotalAmount(totalDebit);
108         oes.setRowCount(rowCount);
109 
110         return oes;
111 
112     }
