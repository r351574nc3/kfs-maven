43      /**
44       * @see org.kuali.rice.kim.service.support.impl.KimTypeServiceBase#performMatch(org.kuali.rice.kim.bo.types.dto.AttributeSet,
45       *      org.kuali.rice.kim.bo.types.dto.AttributeSet)
46       */
47      @Override
48      protected boolean performMatch(AttributeSet qualification, AttributeSet roleQualifier) {
49          // if no qualification given but the user is assigned an organization then return they have the role
50          if ((qualification == null || qualification.isEmpty()) && roleQualifier != null && !roleQualifier.isEmpty()) {
51              return true;
52          }
53  
54          // if they don't have a qualification then return false for the role
55          if (qualification == null || qualification.isEmpty() || roleQualifier == null || roleQualifier.isEmpty()) {
56              return false;
57          }
58  
59          String orgChartOfAccountsCode = qualification.get(BCPropertyConstants.ORGANIZATION_CHART_OF_ACCOUNTS_CODE);
60          String organizationCode = qualification.get(KfsKimAttributes.ORGANIZATION_CODE);
61          String roleChartOfAccountsCode = roleQualifier.get(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE);
62          String roleOrganizationCode = roleQualifier.get(KfsKimAttributes.ORGANIZATION_CODE);
63  
64          String descendHierarchy = DESCEND_HIERARCHY_FALSE_VALUE;
65          if (qualification.containsKey(KfsKimAttributes.DESCEND_HIERARCHY)) {
66              descendHierarchy = qualification.get(KfsKimAttributes.DESCEND_HIERARCHY);
67          }
68  
69          if (DESCEND_HIERARCHY_TRUE_VALUE.equals(descendHierarchy)) {
70              Integer universityFiscalYear = Integer.valueOf(qualification.get(KFSPropertyConstants.UNIVERSITY_FISCAL_YEAR));
71              String chartOfAccountsCode = qualification.get(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE);
72              String accountNumber = qualification.get(KfsKimAttributes.ACCOUNT_NUMBER);
73              Integer organizationLevelCode = Integer.parseInt(qualification.get(BCPropertyConstants.ORGANIZATION_LEVEL_CODE));
74              String accountReportExists = qualification.get(BCPropertyConstants.ACCOUNT_REPORTS_EXIST);
75              
76              // if account report mapping does not exist, want to give special view access
77              if (Boolean.FALSE.toString().equals(accountReportExists)) {
78                  return true;
79              }
80              
81              Integer roleOrganizationLevelCode = -1;
82              List<BudgetConstructionAccountOrganizationHierarchy> accountOrganizationHierarchy = (List<BudgetConstructionAccountOrganizationHierarchy>) budgetDocumentService.retrieveOrBuildAccountOrganizationHierarchy(universityFiscalYear, chartOfAccountsCode, accountNumber);
83              for (BudgetConstructionAccountOrganizationHierarchy accountOrganization : accountOrganizationHierarchy) {
84                  if (accountOrganization.getOrganizationChartOfAccountsCode().equals(roleChartOfAccountsCode) && accountOrganization.getOrganizationCode().equals(roleOrganizationCode)) {
85                      roleOrganizationLevelCode = accountOrganization.getOrganizationLevelCode();
86                  }
87              }
88              
89              if (roleOrganizationLevelCode == -1) {
90                  return false;
91              }
92  
93              return roleOrganizationLevelCode.intValue() >= organizationLevelCode.intValue();
94          }
95  
96          return roleChartOfAccountsCode.equals(orgChartOfAccountsCode) && roleOrganizationCode.equals(organizationCode);
97      }
