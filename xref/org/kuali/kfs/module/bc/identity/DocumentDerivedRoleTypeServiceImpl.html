45      @Override
46      public AttributeSet convertQualificationForMemberRoles(String namespaceCode, String roleName, String memberRoleNamespaceCode, String memberRoleName, AttributeSet qualification) {
47          AttributeSet newQualification = new AttributeSet();
48  
49          if(qualification!=null && !qualification.isEmpty()){
50              String universityFiscalYear = qualification.get(KFSPropertyConstants.UNIVERSITY_FISCAL_YEAR);
51              String chartOfAccountsCode = qualification.get(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE);
52              String accountNumber = qualification.get(KfsKimAttributes.ACCOUNT_NUMBER);
53              String orgChartOfAccountsCode = qualification.get(BCPropertyConstants.ORGANIZATION_CHART_OF_ACCOUNTS_CODE);
54              String organizationCode = qualification.get(KfsKimAttributes.ORGANIZATION_CODE);
55              String accountReportExists = qualification.get(BCPropertyConstants.ACCOUNT_REPORTS_EXIST);
56      
57              Integer organizationLevelCode = qualification.get(BCPropertyConstants.ORGANIZATION_LEVEL_CODE)!=null?Integer.parseInt(qualification.get(BCPropertyConstants.ORGANIZATION_LEVEL_CODE)):null;
58              if (BCConstants.KimConstants.BC_PROCESSOR_ROLE_NAME.equals(memberRoleName)) {
59                  if (BCConstants.KimConstants.DOCUMENT_EDITOR_ROLE_NAME.equals(roleName) && (organizationLevelCode!=null && organizationLevelCode.intValue() == 0)) {
60                      organizationCode = UNMATCHABLE_QUALIFICATION;
61                  }
62              }
63              else {
64                  if (organizationLevelCode!=null && organizationLevelCode.intValue() != 0) {
65                      if (BCConstants.KimConstants.DOCUMENT_EDITOR_ROLE_NAME.equals(roleName) || Boolean.TRUE.toString().equals(accountReportExists)) {
66                          accountNumber = UNMATCHABLE_QUALIFICATION;
67                      }
68                  }
69              }
70      
71              String descendHierarchy = OrganizationOptionalHierarchyRoleTypeServiceImpl.DESCEND_HIERARCHY_FALSE_VALUE;
72              if (BCConstants.KimConstants.DOCUMENT_VIEWER_ROLE_NAME.equals(roleName)) {
73                  descendHierarchy = OrganizationOptionalHierarchyRoleTypeServiceImpl.DESCEND_HIERARCHY_TRUE_VALUE;
74      
75                  newQualification.put(KFSPropertyConstants.UNIVERSITY_FISCAL_YEAR, universityFiscalYear);
76                  newQualification.put(BCPropertyConstants.ORGANIZATION_LEVEL_CODE, organizationLevelCode!=null?organizationLevelCode.toString():null);
77              }
78      
79              newQualification.put(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE, chartOfAccountsCode);
80              newQualification.put(KfsKimAttributes.ACCOUNT_NUMBER, accountNumber);
81              newQualification.put(BCPropertyConstants.ORGANIZATION_CHART_OF_ACCOUNTS_CODE, orgChartOfAccountsCode);
82              newQualification.put(KfsKimAttributes.ORGANIZATION_CODE, organizationCode);
83              newQualification.put(KfsKimAttributes.DESCEND_HIERARCHY, descendHierarchy);
84              newQualification.put(BCPropertyConstants.ACCOUNT_REPORTS_EXIST, accountReportExists);
85              if (qualification.containsKey(KfsKimAttributes.DOCUMENT_TYPE_NAME)) {
86                  newQualification.put(KfsKimAttributes.DOCUMENT_TYPE_NAME, qualification.get(KfsKimAttributes.DOCUMENT_TYPE_NAME));
87              }
88          }
89          return newQualification;
90      }

96      /**
97       * @see org.kuali.kfs.module.bc.identity.BudgetConstructionNoAccessMessageSetting#setNoAccessMessage(org.kuali.kfs.module.bc.document.BudgetConstructionDocument,
98       *      org.kuali.rice.kim.bo.Person, org.kuali.rice.kns.util.MessageMap)
99       */
100     public void setNoAccessMessage(BudgetConstructionDocument document, Person user, MessageMap messageMap) {
101         AttributeSet qualification = new AttributeSet();
102         qualification.put(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE, document.getChartOfAccountsCode());
103         qualification.put(KfsKimAttributes.ACCOUNT_NUMBER, document.getAccountNumber());
104         qualification.put(KfsKimAttributes.DOCUMENT_TYPE_NAME, BCConstants.BUDGET_CONSTRUCTION_DOCUMENT_NAME);
105 
106         List<String> roleId = new ArrayList<String>();
107         roleId.add(roleManagementService.getRoleIdByName(KFSConstants.ParameterNamespaces.KFS, KFSConstants.SysKimConstants.FISCAL_OFFICER_KIM_ROLE_NAME));
108 
109         boolean isFiscalOfficerOrDelegate = roleManagementService.principalHasRole(user.getPrincipalId(), roleId, qualification);
110         boolean isBCProcessor = false;
111         boolean isProcessorInAccountHierarchy = false;
112 
113         List<Organization> userProcessingOrgs = budgetConstructionProcessorService.getProcessorOrgs(user); 
114         if (userProcessingOrgs != null && !userProcessingOrgs.isEmpty()) {
115             isBCProcessor = true;
116             
117             List<BudgetConstructionAccountOrganizationHierarchy> accountOrganizationHierarchy = (List<BudgetConstructionAccountOrganizationHierarchy>) budgetDocumentService.retrieveOrBuildAccountOrganizationHierarchy(document.getUniversityFiscalYear(), document.getChartOfAccountsCode(), document.getAccountNumber());
118             for (BudgetConstructionAccountOrganizationHierarchy accountOrganization : accountOrganizationHierarchy) {
119                 if (userProcessingOrgs.contains(accountOrganization.getOrganization())) {
120                     isProcessorInAccountHierarchy = true;
121                 }
122             }
123         }
124 
125         if (document.getOrganizationLevelCode().intValue() == 0) {
126             messageMap.putError(KFSConstants.DOCUMENT_ERRORS, BCKeyConstants.ERROR_BUDGET_USER_NOT_IN_HIERARCHY);
127         }
128         else {
129             if (isFiscalOfficerOrDelegate || isProcessorInAccountHierarchy) {
130                 messageMap.putError(KFSConstants.DOCUMENT_ERRORS, BCKeyConstants.ERROR_BUDGET_USER_BELOW_DOCLEVEL);
131             }
132             else if (isBCProcessor) {
133                 messageMap.putError(KFSConstants.DOCUMENT_ERRORS, BCKeyConstants.ERROR_BUDGET_USER_NOT_IN_HIERARCHY);
134             }
135             else {
136                 messageMap.putError(KFSConstants.DOCUMENT_ERRORS, BCKeyConstants.ERROR_BUDGET_USER_NOT_ORG_APPROVER);
137             }
138         }
139     }
