28      /**
29       * 
30       * @see org.kuali.kfs.module.bc.batch.service.BudgetConstructionHumanResourcesPayrollInterfaceService#refreshBudgetConstructionPosition(java.lang.Integer, boolean, boolean)
31       */
32      public void refreshBudgetConstructionPosition(Integer baseYear, boolean positionSynchOK, boolean CSFUpdateOK) {
33       /**
34        *  base year positions are built only if current payroll information is still flowing into budget construction
35        *  otherwise, the base year positions are frozen
36        */
37          Integer requestYear = baseYear+1;
38          if (positionSynchOK && CSFUpdateOK)
39          {
40              budgetConstructionHumanResourcesPayrollInterfaceDao.buildBudgetConstructionPositionBaseYear(baseYear);
41          }
42          /**
43           *  request year positions are updated as long as human resources information is still flowing into budget construction
44           */
45          if (positionSynchOK)
46          {
47              budgetConstructionHumanResourcesPayrollInterfaceDao.buildBudgetConstructionPositonRequestYear(requestYear);
48              budgetConstructionHumanResourcesPayrollInterfaceDao.buildBudgetConstructionAdministrativePosts();
49          }
50      }
51  
52      /**
53       * 
54       * @see org.kuali.kfs.module.bc.batch.service.BudgetConstructionHumanResourcesPayrollInterfaceService#refreshBudgetConstructionIntendedIncumbent(java.lang.Integer, boolean, boolean)
55       */
56      public void refreshBudgetConstructionIntendedIncumbent(Integer baseYear, boolean positionSynchOK, boolean CSFUpdateOK, boolean BCUpdatesAllowed) {
57          Integer requestYear = baseYear+1;
58          /**
59           *  the intended incumbent table is updated when human resources information is still flowing into budget construction.
60           *  when this is no longer the case, only the names are updated (unless all of budget construction is no longer in update mode).
61           */
62          if (positionSynchOK)
63          {
64              if (CSFUpdateOK)
65              {
66                  // we update the faculty level (full, associate, assistant, etc.) only if base payroll information is still flowing into budget construction.
67                  // otherwise, we assume that the base payroll is "frozen" as a base-line for salary setting, and we stop allowing people to move between faculty levels.
68                  // this version builds intended incumbent and updates faculty ranks.
69                  budgetConstructionHumanResourcesPayrollInterfaceDao.buildBudgetConstructionIntendedIncumbentWithFacultyAttributes(requestYear);
70              }
71              else
72              {
73                  // this version builds intended incumbent without adding anyone to the faculty levels.
74                  budgetConstructionHumanResourcesPayrollInterfaceDao.buildBudgetConstructionIntendedIncumbent(requestYear);
75              }
76          }
77          else
78          {
79             // the name is always updated if the budget is in update mode, even if intended incumbent was not rebuilt because position synchronization was off.
80              if (BCUpdatesAllowed)
81              {
82                 budgetConstructionHumanResourcesPayrollInterfaceDao.updateNamesInBudgetConstructionIntendedIncumbent();
83              }
84          }
85      }
