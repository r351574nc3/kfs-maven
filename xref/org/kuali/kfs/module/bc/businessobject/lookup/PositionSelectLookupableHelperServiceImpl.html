32  /**
33   * Lookupable helper service implementation for the position select screen.
34   */
35  public class PositionSelectLookupableHelperServiceImpl extends SelectLookupableHelperServiceImpl {
36  
37      /**
38       * @see org.kuali.rice.kns.lookup.LookupableHelperService#getCustomActionUrls(org.kuali.rice.kns.bo.BusinessObject, java.util.List, java.util.List pkNames)
39       */
40      @Override
41      public List<HtmlData> getCustomActionUrls(BusinessObject businessObject, List pkNames) {
42          BudgetConstructionPositionSelect positionSelect = (BudgetConstructionPositionSelect) businessObject;
43  
44          Properties parameters = new Properties();
45          parameters.put(KFSConstants.DISPATCH_REQUEST_PARAMETER, BCConstants.POSITION_SALARY_SETTING_METHOD);
46  
47          parameters.put(KFSPropertyConstants.UNIVERSITY_FISCAL_YEAR, positionSelect.getUniversityFiscalYear().toString());
48          parameters.put(BCPropertyConstants.POSITION_NUMBER, positionSelect.getPositionNumber());
49          parameters.put(BCPropertyConstants.SINGLE_ACCOUNT_MODE, "false");
50          parameters.put(BCPropertyConstants.ADD_LINE, "false");
51  
52          String href = UrlFactory.parameterizeUrl(BCConstants.POSITION_SALARY_SETTING_ACTION, parameters);
53          List<HtmlData> anchorHtmlDataList = new ArrayList<HtmlData>();
54          AnchorHtmlData anchorHtmlData = new AnchorHtmlData(href, BCConstants.POSITION_SALARY_SETTING_METHOD, "Posn Salset");
55          anchorHtmlData.setTarget(KFSConstants.NEW_WINDOW_URL_TARGET);
56          anchorHtmlDataList.add(anchorHtmlData);
57          
58          return anchorHtmlDataList;
59      }
