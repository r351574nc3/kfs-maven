49      /**
50       * @see org.kuali.kfs.module.bc.service.BudgetConstructionPositionService#pullNewPositionFromExternal(java.lang.Integer,
51       *      java.lang.String)
52       */
53      @Transactional
54      public synchronized void pullNewPositionFromExternal(Integer universityFiscalYear, String positionNumber) throws BudgetPositionAlreadyExistsException {
55          // call humanResourcesPayrollService service to pull record
56          Position position = humanResourcesPayrollService.getPosition(universityFiscalYear, positionNumber);
57  
58          // check if position already exists in budget position table, if not add position
59          BudgetConstructionPosition retrievedPosition = getByPrimaryId(universityFiscalYear.toString(), positionNumber);
60          if (retrievedPosition != null) {
61              throw new BudgetPositionAlreadyExistsException(universityFiscalYear, positionNumber);
62          }
63          else {
64              retrievedPosition = new BudgetConstructionPosition();
65          }
66          
67          // populate BudgetConstructionPosition
68          BudgetConstructionPosition budgetConstructionPosition = buildBudgetPosition(position, retrievedPosition);
69  
70          // insert position record
71          businessObjectService.save(budgetConstructionPosition);
72      }

75       * @see org.kuali.kfs.module.bc.service.BudgetConstructionPositionService#refreshPositionFromExternal(java.lang.Integer,
76       *      java.lang.String)
77       */
78      @Transactional(propagation = Propagation.REQUIRES_NEW)
79      public synchronized void refreshPositionFromExternal(Integer universityFiscalYear, String positionNumber) {
80          // call humanResourcesPayrollService service to pull record
81          Position position = humanResourcesPayrollService.getPosition(universityFiscalYear, positionNumber);
82  
83          // update budget record
84          BudgetConstructionPosition retrievedPosition = getByPrimaryId(universityFiscalYear.toString(), positionNumber);
85          
86          // populate BudgetConstructionPosition
87          BudgetConstructionPosition budgetConstructionPosition = buildBudgetPosition(position, retrievedPosition);
88  
89          // update position record
90          businessObjectService.save(budgetConstructionPosition);
91  
92          // update funding position change indicators
93          updateFundingPositionChangeIndicators(universityFiscalYear, positionNumber);
94      }

96      /**
97       * Retrieves all funding lines for the position that are not marked as delete and sets the position change indicator fields to
98       * true.
99       * 
100      * @param universityFiscalYear budget fiscal year for the position
101      * @param positionNumber position number for the record
102      */
103     protected void updateFundingPositionChangeIndicators(Integer universityFiscalYear, String positionNumber) {
104         // retrieve funding records for the position
105         List<PendingBudgetConstructionAppointmentFunding> allPositionFunding = budgetConstructionDao.getAllFundingForPosition(universityFiscalYear, positionNumber);
106 
107         // update indicators if the line is not marked for delete
108         for (PendingBudgetConstructionAppointmentFunding appointmentFunding : allPositionFunding) {
109             if (!appointmentFunding.isAppointmentFundingDeleteIndicator()) {
110                 appointmentFunding.setPositionObjectChangeIndicator(true);
111                 appointmentFunding.setPositionSalaryChangeIndicator(true);
112                 appointmentFunding.setVersionNumber(appointmentFunding.getVersionNumber());
113                 
114                 businessObjectService.save(appointmentFunding);
115             }
116         }
117     }

161     @NonTransactional
162     public BudgetConstructionPosition getByPrimaryId(String fiscalYear, String positionNumber) {
163         Map<String, Object> primaryKeys = new HashMap<String, Object>();
164         primaryKeys.put(KFSPropertyConstants.UNIVERSITY_FISCAL_YEAR, fiscalYear);
165         primaryKeys.put(KFSPropertyConstants.POSITION_NUMBER, positionNumber);
166 
167         return (BudgetConstructionPosition) businessObjectService.findByPrimaryKey(BudgetConstructionPosition.class, primaryKeys);
168     }
