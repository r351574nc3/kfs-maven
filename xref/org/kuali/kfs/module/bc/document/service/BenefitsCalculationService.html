16  package org.kuali.kfs.module.bc.document.service;

18  /**
19   * This class defines methods that a Benefits Calculation Service must provide The Benefits Calculation Service supports
20   * functionality related to calculating benefits request amounts for a Budget Construction Document (account/sub-account). This
21   * includes calculations for annual and monthly amounts. Monthly amounts are only calculated when associated monthly request amounts
22   * exist.
23   */
24  public interface BenefitsCalculationService {
25  
26      /**
27       * This method returns the disabled setting of the System Parameter controlling Budget module Benefits Calculation. Disabling
28       * Benefits Calculation will cause any UI controls related to the Benefits Calculation functionality to not be displayed.
29       * Disabling will also cause associated business rules checks to behave differently or not be run.
30       * 
31       * @return
32       */
33      public boolean isBenefitsCalculationDisabled();
34      
35      /**
36       * 
37       * calculates the annual benefits for the budget construction general ledger key passed in, and stores them in the database
38       * @param documentNumber  the string containing the document number
39       * @param fiscalYear      the integer value of the fiscal year
40       * @param chartOfAccounts the string containing the chart of accounts
41       * @param accountNumber   the string containing the account number
42       * @param subAccountNumber the string containing the sub account number
43       */
44      public void calculateAnnualBudgetConstructionGeneralLedgerBenefits(String documentNumber,
45                                                                         Integer fiscalYear,
46                                                                         String chartOfAccounts,
47                                                                         String accountNumber,
48                                                                         String subAccountNumber);
49      /**
50       * 
51       * calculates the monthly budget benefits for the budget construction general ledger key passed in, and stores them in the database
52       * @param documentNumber  the string containing the document number
53       * @param fiscalYear      the integer value of the fiscal year
54       * @param chartOfAccounts the string containing the chart of accounts
55       * @param accountNumber   the string containing the account number
56       * @param subAccountNumber the string containing the sub account number 
57       * 
58       */
59      public void calculateMonthlyBudgetConstructionGeneralLedgerBenefits(String documentNumber,
60                                                                          Integer fiscalYear,
61                                                                          String chartOfAccounts,
62                                                                          String accountNumber,
63                                                                          String subAccountNumber);
64  
65  
66      /**
67       * 
68       * calculates both the monthly budget and the annual budget budget construction general ledger benefits for the key passed in, and 
69       * stores them in the database.
70       * @param documentNumber  the string containing the document number
71       * @param fiscalYear      the integer value of the fiscal year
72       * @param chartOfAccounts the string containing the chart of accounts
73       * @param accountNumber   the string containing the account number
74       * @param subAccountNumber
75       */
76      public void calculateAllBudgetConstructionGeneralLedgerBenefits(String documentNumber,
77                                                                      Integer fiscalYear,
78                                                                      String chartOfAccounts,
79                                                                      String accountNumber,
80                                                                      String subAccountNumber);
81  
82  }
