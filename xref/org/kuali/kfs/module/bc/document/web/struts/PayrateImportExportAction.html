51      public ActionForward performImport(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
52          PayrateImportExportForm payrateImportExportForm = (PayrateImportExportForm) form;
53          PayrateImportService payrateImportService = SpringContext.getBean(PayrateImportService.class);
54          List<ExternalizedMessageWrapper> messageList = new ArrayList<ExternalizedMessageWrapper>();
55          Integer budgetYear = payrateImportExportForm.getUniversityFiscalYear();
56          ByteArrayOutputStream baos = new ByteArrayOutputStream();
57          String principalId = GlobalVariables.getUserSession().getPerson().getPrincipalId();
58          
59          SimpleDateFormat dateFormatter = new SimpleDateFormat("dd-MMM-yyyy ' ' HH:mm:ss", Locale.US);
60          
61          boolean isValid = validateImportFormData(payrateImportExportForm);
62          
63          if (!isValid) {
64              return mapping.findForward(KFSConstants.MAPPING_BASIC);
65          }
66          //get start date for log file
67          Date startTime = new Date();
68          messageList.add(new ExternalizedMessageWrapper(BCKeyConstants.MSG_PAYRATE_IMPORT_LOG_FILE_HEADER_LINE, dateFormatter.format(startTime)));
69          
70          //parse file
71          if (!payrateImportService.importFile(payrateImportExportForm.getFile().getInputStream(), messageList, principalId) ) {
72              payrateImportService.generatePdf(messageList, baos);
73              WebUtils.saveMimeOutputStreamAsFile(response, ReportGeneration.PDF_MIME_TYPE, baos, BCConstants.PAYRATE_IMPORT_LOG_FILE);
74              return null;
75          }
76          
77          messageList.add(new ExternalizedMessageWrapper(BCKeyConstants.MSG_PAYRATE_IMPORT_COMPLETE));
78          
79          Person user = GlobalVariables.getUserSession().getPerson();
80          
81          //perform updates
82          payrateImportService.update(budgetYear, user, messageList, principalId);
83          
84          messageList.add(new ExternalizedMessageWrapper(BCKeyConstants.MSG_PAYRATE_IMPORT_LOG_FILE_FOOTER, dateFormatter.format(new Date())));
85          
86          //write messages to log file
87          payrateImportService.generatePdf(messageList, baos);
88          
89          WebUtils.saveMimeOutputStreamAsFile(response, ReportGeneration.PDF_MIME_TYPE, baos, BCConstants.PAYRATE_IMPORT_LOG_FILE);
90          
91          return null;
92      }

94      public ActionForward performExport(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
95          PayrateImportExportForm payrateImportExportForm = (PayrateImportExportForm) form;
96          PayrateExportService payrateExportService = SpringContext.getBean(PayrateExportService.class);
97          HumanResourcesPayrollService payrollPerimeterService = SpringContext.getBean(HumanResourcesPayrollService.class);
98          Integer budgetYear = payrateImportExportForm.getUniversityFiscalYear();
99          String positionUnionCode = payrateImportExportForm.getPositionUnionCode();
100         MessageMap errorMap = GlobalVariables.getMessageMap();
101         String principalId = GlobalVariables.getUserSession().getPerson().getPrincipalId();
102         
103         //form validation
104         boolean isValidPositionUnionCode = validateExportFormData(payrateImportExportForm);
105         if (!isValidPositionUnionCode) {
106             return mapping.findForward(KFSConstants.MAPPING_BASIC);
107         }
108         
109         //position union code validation
110         isValidPositionUnionCode = payrollPerimeterService.validatePositionUnionCode(positionUnionCode);
111         if (!isValidPositionUnionCode) {
112             errorMap.putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.ERROR_PAYRATE_EXPORT_INVALID_POSITION_UNION_CODE, positionUnionCode);
113             return mapping.findForward(KFSConstants.MAPPING_BASIC);
114         }
115         
116         StringBuilder fileContents = payrateExportService.buildExportFile(budgetYear, positionUnionCode, payrateImportExportForm.getCsfFreezeDateFormattedForExportFile(), principalId);
117         
118         ByteArrayOutputStream baos = new ByteArrayOutputStream();
119         baos.write(fileContents.toString().getBytes());
120         WebUtils.saveMimeOutputStreamAsFile(response, ReportGeneration.TEXT_MIME_TYPE, baos, BCConstants.PAYRATE_EXPORT_FILE);
121         
122         return null;
123     }

129     /**
130      * Performs form validation
131      * 
132      * @param form
133      * @return
134      */
135     public boolean validateImportFormData(PayrateImportExportForm form) {
136         boolean isValid = true;
137         PayrateImportExportForm importForm = (PayrateImportExportForm) form;
138         MessageMap errorMap = GlobalVariables.getMessageMap();
139         
140         FiscalYearFunctionControlService fiscalYearFunctionControlService = SpringContext.getBean(FiscalYearFunctionControlService.class);
141         boolean budgetUpdatesAllowed = fiscalYearFunctionControlService.isBudgetUpdateAllowed(form.getUniversityFiscalYear());
142         
143         if ( importForm.getFile() == null || importForm.getFile().getFileSize() == 0 ) {
144             errorMap.putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.ERROR_FILE_IS_REQUIRED);
145             isValid = false;
146         }
147         if ( importForm.getFile() != null && importForm.getFile().getFileSize() == 0 ) {
148             errorMap.putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.ERROR_FILE_EMPTY);
149             isValid = false;
150         }
151         if (importForm.getFile() != null && (StringUtils.isBlank(importForm.getFile().getFileName())) ) {
152             errorMap.putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.ERROR_FILENAME_REQUIRED);
153             isValid = false;
154         }
155         if ( !budgetUpdatesAllowed ) {
156             errorMap.putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.ERROR_PAYRATE_IMPORT_UPDATE_NOT_ALLOWED);
157             isValid = false;
158         }
159         
160         
161         return isValid;
162     }

164     /**
165      * Performs export form validation
166      * 
167      * @param form
168      * @return
169      */
170     public boolean validateExportFormData(PayrateImportExportForm form) {
171         boolean isValid = true;
172         PayrateImportExportForm importForm = (PayrateImportExportForm) form;
173         MessageMap errorMap = GlobalVariables.getMessageMap();
174         PayrateExportService payrateExportService = SpringContext.getBean(PayrateExportService.class);
175         
176         if (StringUtils.isBlank(importForm.getPositionUnionCode()) ) {
177             errorMap.putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.ERROR_PAYRATE_EXPORT_POSITION_UNION_CODE_REQUIRED);
178             isValid = false;
179         }
180         if (StringUtils.isBlank(importForm.getCsfFreezeDate()) ) {
181             errorMap.putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.ERROR_PAYRATE_EXPORT_CSF_FREEZE_DATE_REQUIRED);
182             isValid = false;
183         }
184         if (!payrateExportService.isValidPositionUnionCode(form.getPositionUnionCode()) ) {
185             errorMap.putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.ERROR_PAYRATE_EXPORT_INVALID_POSITION_UNION_CODE);
186             isValid = false;
187         }
188         SimpleDateFormat validDateFormatter = new SimpleDateFormat("MM/dd/yyyy", Locale.US);
189         SimpleDateFormat exportFileFormat = new SimpleDateFormat("yyyyMMdd", Locale.US);
190         try {
191             Date validDate = validDateFormatter.parse(form.getCsfFreezeDate());
192             importForm.setCsfFreezeDateFormattedForExportFile(exportFileFormat.format(validDate));
193         }
194         catch (ParseException e) {
195             errorMap.putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.ERROR_PAYRATE_EXPORT_CSF_FREEZE_DATE_INCORRECT_FORMAT);
196             isValid = false;
197         }
198         
199         return isValid;
200     }
