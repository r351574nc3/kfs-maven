80      protected boolean checkFiscalYearIsCurrent(MaintenanceDocument document) {
81          boolean success = true;
82  
83          if ((ObjectUtils.isNotNull(newCalculatedSalaryFoundationTrackerOverride.getUniversityFiscalYear())) && (document.isNew())) {
84  
85              Integer currentFiscalYear = SpringContext.getBean(UniversityDateService.class).getCurrentFiscalYear();
86              Integer universityFiscalYear = newCalculatedSalaryFoundationTrackerOverride.getUniversityFiscalYear();
87              if (!(universityFiscalYear.equals(currentFiscalYear))) {
88                  putFieldError("universityFiscalYear", BCKeyConstants.ERROR_FISCAL_YEAR_NOT_CURRENT, "Fiscal Year");
89                  success &= false;
90              }
91          }
92          return success;
93      }
94  
95      protected boolean checkAppointmentIsValid() {
96          boolean success = true;
97  
98          if ((ObjectUtils.isNotNull(newCalculatedSalaryFoundationTrackerOverride.getEmplid())) && (ObjectUtils.isNotNull(newCalculatedSalaryFoundationTrackerOverride.getPositionNumber())) && (ObjectUtils.isNotNull(newCalculatedSalaryFoundationTrackerOverride.getUniversityFiscalYear()))) {
99  
100             String emplid = newCalculatedSalaryFoundationTrackerOverride.getEmplid();
101             String positionNumber = newCalculatedSalaryFoundationTrackerOverride.getPositionNumber();
102             Integer universityFiscalYear = newCalculatedSalaryFoundationTrackerOverride.getUniversityFiscalYear();
103             boolean result = calculatedSalaryFoundationTrackerOverrideService.isValidAppointment(universityFiscalYear, positionNumber, emplid);
104             if (!result) {
105                 putFieldError("emplid", BCKeyConstants.ERROR_INVALID_APPOINTMENT, "Employee Id");
106                 success &= false;
107             }
108         }
109         else {
110             putFieldError("emplid", BCKeyConstants.ERROR_INVALID_APPOINTMENT, "Employee Id");
111             success &= false;
112         }
113         return success;
114     }
