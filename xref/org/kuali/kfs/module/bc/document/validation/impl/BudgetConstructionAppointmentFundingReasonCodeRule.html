29  public class BudgetConstructionAppointmentFundingReasonCodeRule extends MaintenanceDocumentRuleBase {
30      protected static final org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(BudgetConstructionAppointmentFundingReasonCodeRule.class);
31  
32      protected BudgetConstructionAppointmentFundingReasonCode oldBudgetConstructionAppointmentFundingReasonCode;
33      protected BudgetConstructionAppointmentFundingReasonCode newBudgetConstructionAppointmentFundingReasonCode;
34  
35      protected SalarySettingService salarySettingService;
36  
37      public BudgetConstructionAppointmentFundingReasonCodeRule() {
38          
39          this.setSalarySettingService(SpringContext.getBean(SalarySettingService.class));
40      }
41  
42      /**
43       * @see org.kuali.rice.kns.maintenance.rules.MaintenanceDocumentRuleBase#processCustomRouteDocumentBusinessRules(org.kuali.rice.kns.document.MaintenanceDocument)
44       */
45      @Override
46      protected boolean processCustomRouteDocumentBusinessRules(MaintenanceDocument document) {
47          LOG.info("processCustomRouteDocumentBusinessRules called");
48  
49          boolean success = true;
50          
51          // checks when editing existing code
52          if (document.isEdit()) {
53              success &= checkInactivateReason(document);
54          }
55  
56          return success;
57      }
58  
59      protected boolean checkInactivateReason(MaintenanceDocument document){
60          LOG.info("checkInactivateReason called");
61  
62          boolean success = true;
63          
64          // just return when not being inactivated 
65          if (!(oldBudgetConstructionAppointmentFundingReasonCode.isActive() && !newBudgetConstructionAppointmentFundingReasonCode.isActive())) {
66              return true;
67          }
68  
69          // disallow inactivation if appointment funding reasons exist using this editable code
70          if (salarySettingService.hasExistingFundingReason(newBudgetConstructionAppointmentFundingReasonCode)){
71              putGlobalError(BCKeyConstants.ERROR_BUDGET_REASONMAINT_INACTIVATE_REASONEXIST);
72              success &= false;
73          }
74  
75          return success;
76      }
77

85          newBudgetConstructionAppointmentFundingReasonCode = (BudgetConstructionAppointmentFundingReasonCode) super.getNewBo();
86      }
