25  /**
26   * builds the report for accounts which are out of balance because benefits have been calculated on salaries and the amounts don't match what is in GL.  the discrepancy is in an object code called '2PLG'.
27   */
28  
29  public class BudgetConstructionList2PLGReportDaoJdbc extends BudgetConstructionDaoJdbcBase implements BudgetConstructionList2PLGReportDao {
30      private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(BudgetConstructionList2PLGReportDaoJdbc.class);
31  
32      private static ArrayList<SQLForStep> updateReportsList2PLGTable = new ArrayList<SQLForStep>(1);
33      
34      private PersistenceService persistenceService;
35  
36      public BudgetConstructionList2PLGReportDaoJdbc() {
37  
38          ArrayList<Integer> insertionPoints = new ArrayList<Integer>(1);
39          
40          // builds and updates List2PLGReports
41          /* get accounts, sub-accounts that have an imbalance between detailed salary and general ledger, displayed in the object class 2PLG */
42          StringBuilder sqlText = new StringBuilder(500);
43          sqlText.append("INSERT INTO LD_BCN_2PLG_LIST_MT \n");
44          sqlText.append("(PERSON_UNVL_ID, ORG_FIN_COA_CD, ORG_CD, FIN_COA_CD, ACCOUNT_NBR, SUB_ACCT_NBR, ACLN_ANNL_BAL_AMT) \n");
45          sqlText.append("SELECT ?, ctrl.sel_org_fin_coa, ctrl.sel_org_cd, ctrl.fin_coa_cd, ctrl.account_nbr, ctrl.sub_acct_nbr, pbgl.acln_annl_bal_amt \n");
46          sqlText.append("FROM LD_PND_BCNSTR_GL_T pbgl, LD_BCN_CTRL_LIST_T ctrl \n");
47          sqlText.append("WHERE ctrl.person_unvl_id = ? \n");
48          sqlText.append(" AND pbgl.fdoc_nbr = ctrl.fdoc_nbr \n");
49          sqlText.append(" AND pbgl.univ_fiscal_yr = ctrl.univ_fiscal_yr \n");
50          sqlText.append(" AND pbgl.fin_coa_cd = ctrl.fin_coa_cd \n");
51          sqlText.append(" AND pbgl.account_nbr = ctrl.account_nbr \n");
52          sqlText.append(" AND pbgl.sub_acct_nbr = ctrl.sub_acct_nbr \n");
53          sqlText.append(" AND pbgl.fin_object_cd = '");
54          // the 2PLG constant
55          insertionPoints.add(sqlText.length());
56          sqlText.append("'");
57  
58          updateReportsList2PLGTable.add(new SQLForStep(sqlText,insertionPoints));
59          insertionPoints.clear();
60          sqlText.delete(0, sqlText.length());
61  
62      }
63  
64      /**
65       *  remove from the table rows for any previous report requested by this user
66       */
67      protected void cleanReportsList2PLGTable(String principalName) {
68          clearTempTableByUnvlId("LD_BCN_2PLG_LIST_MT", "PERSON_UNVL_ID", principalName);
69      }
70  
71      /**
72       * 
73       * @see org.kuali.kfs.module.bc.document.dataaccess.BudgetConstructionList2PLGReportDao#updateReportsList2PLGTable(java.lang.String)
74       */
75      public void updateList2PLGReportsTable(String principalName) {
76          // get rid of any rows from a previous report requested by this user
77          cleanReportsList2PLGTable(principalName);
78          // fetch the constant naming the 2PLG object class
79          ArrayList<String> stringToInsert = new ArrayList<String>(1); 
80          stringToInsert.add(BudgetConstructionConstants.OBJECT_CODE_2PLG);
81          // fill the table
82          getSimpleJdbcTemplate().update(updateReportsList2PLGTable.get(0).getSQL(stringToInsert), principalName, principalName);
83          /**
84           * this is necessary to clear any rows for the tables we have just updated from the OJB cache.  otherwise, subsequent calls to OJB will fetch the old, unupdated cached rows.
85           */
86          persistenceService.clearCache();
87      }
88      
89      public void setPersistenceService(PersistenceService persistenceService)
90      {
91          this.persistenceService = persistenceService;
92      }
