16  package org.kuali.kfs.module.bc.document.dataaccess.impl;
17  
18  import org.kuali.kfs.module.bc.document.dataaccess.ReportDumpDao;
19  
20  /**
21   * JCBC implementation of ReportDumpDao
22   * 
23   * @see org.kuali.kfs.module.bc.document.dataaccess.ReportDumpDao
24   */
25  public class ReportDumpDaoJdbc extends BudgetConstructionDaoJdbcBase implements ReportDumpDao {
26      private static String updateAccountDump = new String();
27  
28      public ReportDumpDaoJdbc() {
29          super();
30  
31          // SQL to update account dump table from control list and sub-fund selections
32          StringBuilder sqlText = new StringBuilder(500);
33          sqlText.append("INSERT INTO LD_BCN_ACCT_DUMP_T\n");
34          sqlText.append("SELECT ?, ctrl.univ_fiscal_yr, ctrl.fin_coa_cd, ctrl.account_nbr, ctrl.sub_acct_nbr, ctrl.ver_nbr \n");
35          sqlText.append("FROM LD_BCN_CTRL_LIST_T ctrl, LD_BCN_SUBFUND_PICK_T pick \n");
36          sqlText.append("WHERE ctrl.person_unvl_id = ? \n");
37          sqlText.append("AND pick.person_unvl_id = ctrl.person_unvl_id \n");
38          sqlText.append("AND pick.sub_fund_grp_cd = ctrl.sel_sub_fund_grp \n");
39          sqlText.append("AND pick.report_flag > 0 \n");
40  
41          updateAccountDump = sqlText.toString();
42          sqlText.delete(0, sqlText.length());
43      }
44  
45      /**
46       * @see org.kuali.kfs.module.bc.document.dataaccess.ReportDumpDao#cleanAccountDump(java.lang.String)
47       */
48      public void cleanAccountDump(String principalName) {
49          // clear out previous account dump data for user
50          clearTempTableByUnvlId("LD_BCN_ACCT_DUMP_T", "PERSON_UNVL_ID", principalName);
51      }
52  
53      /**
54       * @see org.kuali.kfs.module.bc.document.dataaccess.ReportDumpDao#updateAccountDump(java.lang.String)
55       */
56      public void updateAccountDump(String principalId) {
57          cleanAccountDump(principalId);
58          
59          // rebuild account dump table
60          getSimpleJdbcTemplate().update(updateAccountDump, principalId, principalId);
61      }
62  
63  }
