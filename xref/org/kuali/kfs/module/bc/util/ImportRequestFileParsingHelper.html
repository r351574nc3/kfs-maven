34      /**
35       * Parses line and creates BudgetConstructionRequestMove object. 
36       * 
37       * @param lineToParse
38       * @param fieldSeperator
39       * @param textDelimiter
40       * @return the BudgetConstructionRequestMove or null if there was an error parsing the line
41       */
42      public static BudgetConstructionRequestMove parseLine(String lineToParse, String fieldSeperator, String textDelimiter, boolean isAnnual) {
43          List<String> attributes = new ArrayList<String>();
44          BudgetConstructionRequestMove budgetConstructionRequestMove = new BudgetConstructionRequestMove();
45          
46          int expectedNumberOfSeparators = isAnnual ? 5 : 16;
47          
48          lineToParse = lineToParse.trim();
49          
50          //check if line is in correct format
51          if (!isLineCorrectlyFormatted(lineToParse, fieldSeperator, textDelimiter, isAnnual)) return null;
52          
53          if (textDelimiter.equalsIgnoreCase(BCConstants.RequestImportTextFieldDelimiter.NOTHING.getDelimiter())) {
54              attributes.addAll(Arrays.asList(lineToParse.split(isFieldSeparatorSpecialCharacter(fieldSeperator) ? "\\" + fieldSeperator : fieldSeperator)));
55          } else if ( getEscapedFieldSeparatorCount(lineToParse, fieldSeperator, textDelimiter, isAnnual) == 0) {
56              lineToParse = StringUtils.remove(lineToParse, textDelimiter);
57              attributes.addAll(Arrays.asList(lineToParse.split(isFieldSeparatorSpecialCharacter(fieldSeperator) ? "\\" + fieldSeperator : fieldSeperator)));
58          } else {
59              int firstIndexOfTextDelimiter = 0;
60              int nextIndexOfTextDelimiter = lineToParse.indexOf(textDelimiter, firstIndexOfTextDelimiter + 1);
61              int expectedNumberOfTextDelimiters = 10;
62              
63              for (int i = 0; i < expectedNumberOfTextDelimiters/2; i++) {
64                  attributes.add(lineToParse.substring(firstIndexOfTextDelimiter, nextIndexOfTextDelimiter).replaceAll(textDelimiter, ""));
65                  firstIndexOfTextDelimiter = lineToParse.indexOf(textDelimiter, nextIndexOfTextDelimiter + 1);
66                  nextIndexOfTextDelimiter = lineToParse.indexOf(textDelimiter, firstIndexOfTextDelimiter + 1);
67              }
68              
69              String remainingNonStringValuesToParse = lineToParse.substring(lineToParse.lastIndexOf(textDelimiter + 1));
70              attributes.addAll(Arrays.asList(remainingNonStringValuesToParse.split(isFieldSeparatorSpecialCharacter(fieldSeperator) ? "\\" + fieldSeperator : fieldSeperator)));
71          }
72          
73          budgetConstructionRequestMove.setChartOfAccountsCode(attributes.get(0));
74          budgetConstructionRequestMove.setAccountNumber(attributes.get(1));
75          budgetConstructionRequestMove.setSubAccountNumber(attributes.get(2));
76          budgetConstructionRequestMove.setFinancialObjectCode(attributes.get(3));
77          budgetConstructionRequestMove.setFinancialSubObjectCode(attributes.get(4));
78          
79          try {
80              if (isAnnual) {
81                  budgetConstructionRequestMove.setAccountLineAnnualBalanceAmount(new KualiInteger(Integer.parseInt(attributes.get(5))));
82                  
83              } else {
84                  budgetConstructionRequestMove.setFinancialDocumentMonth1LineAmount(new KualiInteger(Integer.parseInt(attributes.get(5))));
85                  budgetConstructionRequestMove.setFinancialDocumentMonth2LineAmount(new KualiInteger(Integer.parseInt(attributes.get(6))));
86                  budgetConstructionRequestMove.setFinancialDocumentMonth3LineAmount(new KualiInteger(Integer.parseInt(attributes.get(7))));
87                  budgetConstructionRequestMove.setFinancialDocumentMonth4LineAmount(new KualiInteger(Integer.parseInt(attributes.get(8))));
88                  budgetConstructionRequestMove.setFinancialDocumentMonth5LineAmount(new KualiInteger(Integer.parseInt(attributes.get(9))));
89                  budgetConstructionRequestMove.setFinancialDocumentMonth6LineAmount(new KualiInteger(Integer.parseInt(attributes.get(10))));
90                  budgetConstructionRequestMove.setFinancialDocumentMonth7LineAmount(new KualiInteger(Integer.parseInt(attributes.get(11))));
91                  budgetConstructionRequestMove.setFinancialDocumentMonth8LineAmount(new KualiInteger(Integer.parseInt(attributes.get(12))));
92                  budgetConstructionRequestMove.setFinancialDocumentMonth9LineAmount(new KualiInteger(Integer.parseInt(attributes.get(13))));
93                  budgetConstructionRequestMove.setFinancialDocumentMonth10LineAmount(new KualiInteger(Integer.parseInt(attributes.get(14))));
94                  budgetConstructionRequestMove.setFinancialDocumentMonth11LineAmount(new KualiInteger(Integer.parseInt(attributes.get(15))));
95                  budgetConstructionRequestMove.setFinancialDocumentMonth12LineAmount(new KualiInteger(Integer.parseInt(attributes.get(16))));
96                  
97              }
98          }
99          catch (NumberFormatException e) {
100             return null;
101         }
102 
103         return budgetConstructionRequestMove;
104     }

106     /**
107      * Checks for the correct number of field separators and text delimiters on line (either annual or monthly file). Does not check if the correct field separator and text delimiter are being used (form level validation should do this)
108      * 
109      * @param lineToParse
110      * @param fieldSeperator
111      * @param textDelimiter
112      * @param isAnnual
113      * @return
114      */
115     public static boolean isLineCorrectlyFormatted(String lineToParse, String fieldSeperator, String textDelimiter, boolean isAnnual) {
116         int fieldSeparatorCount = StringUtils.countMatches(lineToParse, fieldSeperator);
117         int expectedNumberOfSeparators = isAnnual ? 5 : 16;
118         
119         if (textDelimiter.equalsIgnoreCase(BCConstants.RequestImportTextFieldDelimiter.NOTHING.getDelimiter())) {
120             
121             if (isAnnual) {
122                 if (fieldSeparatorCount != 5) {
123                     return false;
124                 } 
125             } else {
126                 if (fieldSeparatorCount != 16) {
127                     return false;
128                 } 
129             }
130         } else if (StringUtils.countMatches(lineToParse, textDelimiter) != 10) {
131             return false;
132         } else if ( getEscapedFieldSeparatorCount(lineToParse, fieldSeperator, textDelimiter, isAnnual) == -1 || ( fieldSeparatorCount - getEscapedFieldSeparatorCount(lineToParse, fieldSeperator, textDelimiter, isAnnual) != expectedNumberOfSeparators ) ) {
133             return false;
134         }
135         
136         return true;
137     }

139     /**
140      * Checks if a line of an annual file contains an escaped field separator (convience method to aid in file parsing)
141      * Will not work correctly if text delimiters are not correctly placed in lineToParse (method does not check file formatting)
142      * 
143      * @param lineToParse
144      * @param fieldSeperator
145      * @param textDelimiter
146      * @return number of escaped separators or -1 if file is incorrectly formatted
147      */
148     private static int getEscapedFieldSeparatorCount(String lineToParse, String fieldSeperator, String textDelimiter, boolean isAnnual) {
149         int firstIndexOfTextDelimiter = 0;
150         int nextIndexOfTextDelimiter = lineToParse.indexOf(textDelimiter, firstIndexOfTextDelimiter + 1);
151         int expectedNumberOfSeparators = isAnnual ? 5 : 16;
152         int expectedTextDelimitersCount = 10;
153         int actualNumberOfTextDelimiters = StringUtils.countMatches(lineToParse, textDelimiter);
154         int totalSeparatorsInLineToParse = StringUtils.countMatches(lineToParse, fieldSeperator);
155         int escapedSeparatorsCount = 0;
156         
157         //line does not use text delimiters
158         if (textDelimiter.equalsIgnoreCase(BCConstants.RequestImportTextFieldDelimiter.NOTHING.getDelimiter())) return 0;
159         
160         //line does not contain escaped field separators
161         if (totalSeparatorsInLineToParse == expectedNumberOfSeparators) return 0;
162         
163         //line is incorrectly formatted
164         if ( actualNumberOfTextDelimiters != expectedTextDelimitersCount) return -1;
165         
166         for (int i = 0; i < expectedTextDelimitersCount/2; i++) {
167             String escapedString = lineToParse.substring(firstIndexOfTextDelimiter, nextIndexOfTextDelimiter);
168             escapedSeparatorsCount += StringUtils.countMatches(escapedString, fieldSeperator);
169             firstIndexOfTextDelimiter = lineToParse.indexOf(textDelimiter, nextIndexOfTextDelimiter + 1);
170             nextIndexOfTextDelimiter = lineToParse.indexOf(textDelimiter, firstIndexOfTextDelimiter + 1);
171         }
172         
173         return escapedSeparatorsCount;
174     }

176     /**
177      * Determines if the field delimiter is a regular expression special character
178      * 
179      * @param delimiter
180      * @return true if special character
181      */
182     private static boolean isFieldSeparatorSpecialCharacter(String delimiter) {
183         if (delimiter.equals(".")) return true;
184         if (delimiter.equals("[")) return true;
185         if (delimiter.equals("\\")) return true;
186         if (delimiter.equals("*")) return true;
187         if (delimiter.equals("^")) return true;
188         if (delimiter.equals("$")) return true;
189         
190         return false;
191     }
