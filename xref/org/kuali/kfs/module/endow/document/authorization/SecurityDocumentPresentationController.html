35      /**
36       * @see org.kuali.rice.kns.document.authorization.MaintenanceDocumentPresentationControllerBase#getConditionallyHiddenPropertyNames(org.kuali.rice.kns.bo.BusinessObject)
37       */
38      @Override
39      public Set<String> getConditionallyHiddenPropertyNames(BusinessObject businessObject) {
40          Set<String> fields = super.getConditionallyHiddenPropertyNames(businessObject);
41          MaintenanceDocument document = (MaintenanceDocument) businessObject;
42          Security security = (Security) document.getNewMaintainableObject().getBusinessObject();
43          String incomePayFrequencyCode = security.getIncomePayFrequency();
44          
45          if (StringUtils.isNotEmpty(incomePayFrequencyCode)) {
46              FrequencyCodeServiceImpl frequencyCodeServiceImpl = (FrequencyCodeServiceImpl) SpringContext.getBean(FrequencyCodeServiceImpl.class);
47              security.setIncomeNextPayDate(frequencyCodeServiceImpl.calculateProcessDate(incomePayFrequencyCode));
48          }
49          //KFSMI-6674
50          //If SEC_INC_PAY_FREQ entered then the SEC_INC_NEXT_PAY_DT is 
51          //automatically calculated.
52          //if class code type is stocks and SEC_DIV_PAY_DT is entered then 
53          //copy the date value to SEC_INC_NEXT_PAY_DT.
54          //We do not want to overwrite the date if it already exists.
55          if (EndowConstants.ClassCodeTypes.STOCKS.equalsIgnoreCase(security.getClassCode().getClassCodeType())) {
56              if (security.getDividendPayDate() != null) {
57                  security.setIncomeNextPayDate(security.getDividendPayDate());
58              }
59          }
60  
61          // when we create or copy a new Security, only certain fields are displayed; the following code is used to hide the unwanted
62          // fields
63          if (KNSConstants.MAINTENANCE_NEW_ACTION.equals(document.getNewMaintainableObject().getMaintenanceAction()) || KNSConstants.MAINTENANCE_COPY_ACTION.equals(document.getNewMaintainableObject().getMaintenanceAction())) {
64  
65              // the security ID hidded on creation and a dummy field is used for user input (userEnteredSecurityIDprefix)
66              String routeStatus = document.getDocumentHeader().getWorkflowDocument().getRouteHeader().getDocRouteStatus();
67              if (KEWConstants.ROUTE_HEADER_INITIATED_CD.equalsIgnoreCase(routeStatus) || KEWConstants.ROUTE_HEADER_SAVED_CD.equalsIgnoreCase(routeStatus)) {
68                  fields.add(EndowPropertyConstants.SECURITY_ID);
69              }
70              else {
71                  fields.add(EndowPropertyConstants.SECURITY_USER_ENTERED_ID_PREFIX);
72              }
73  
74              fields.add(EndowPropertyConstants.SECURITY_UNIT_VALUE);
75              fields.add(EndowPropertyConstants.SECURITY_UNITS_HELD);
76              fields.add(EndowPropertyConstants.SECURITY_VALUATION_DATE);
77              fields.add(EndowPropertyConstants.SECURITY_UNIT_VALUE_SOURCE);
78              fields.add(EndowPropertyConstants.SECURITY_PREVIOUS_UNIT_VALUE);
79              fields.add(EndowPropertyConstants.SECURITY_PREVIOUS_UNIT_VALUE_DATE);
80              fields.add(EndowPropertyConstants.SECURITY_CARRY_VALUE);
81              fields.add(EndowPropertyConstants.SECURITY_MARKET_VALUE);
82              fields.add(EndowPropertyConstants.SECURITY_LAST_TRANSACTION_DATE);
83              fields.add(EndowPropertyConstants.SECURITY_INCOME_NEXT_PAY_DATE);
84              fields.add(EndowPropertyConstants.SECURITY_INCOME_CHANGE_DATE);
85              fields.add(EndowPropertyConstants.SECURITY_DIVIDEND_RECORD_DATE);
86              fields.add(EndowPropertyConstants.SECURITY_EX_DIVIDEND_DATE);
87              fields.add(EndowPropertyConstants.SECURITY_DIVIDEND_PAY_DATE);
88              fields.add(EndowPropertyConstants.SECURITY_DIVIDEND_AMOUNT);
89              fields.add(EndowPropertyConstants.REPORTING_GROUP_DESC);
90              fields.add(EndowPropertyConstants.ACCRUAL_METHOD_DESC);
91              fields.add(EndowPropertyConstants.SECURITY_NEXT_FISCAL_YEAR_DISTRIBUTION_AMOUNT);
92              fields.add(EndowPropertyConstants.SECURITY_VALUE_BY_MARKET);
93  
94          }
95          // if action is not new or copy the userEnteredSecurityIDprefix shall not be displayed
96          else {
97              fields.add(EndowPropertyConstants.SECURITY_USER_ENTERED_ID_PREFIX);
98          }
99  
100         return fields;
101     }

103     /**
104      * @see org.kuali.rice.kns.document.authorization.MaintenanceDocumentPresentationControllerBase#getConditionallyReadOnlyPropertyNames(org.kuali.rice.kns.document.MaintenanceDocument)
105      */
106     @Override
107     public Set<String> getConditionallyReadOnlyPropertyNames(MaintenanceDocument document) {
108 
109         Set<String> fields = super.getConditionallyReadOnlyPropertyNames(document);
110         Security security = (Security) document.getNewMaintainableObject().getBusinessObject();
111         ClassCode classCode = security.getClassCode();
112 
113         // If the class code type = "P" -- pooled investment:
114         // - the unit value and value date in the security can't be modified through editing that maintenance doc
115         // - END_SEC_T: SEC_RT should NOT be modified through edit that maintenance doc
116         if (classCode != null && EndowConstants.ClassCodeTypes.POOLED_INVESTMENT.equalsIgnoreCase(classCode.getClassCodeType()) && KNSConstants.MAINTENANCE_EDIT_ACTION.equals(document.getNewMaintainableObject().getMaintenanceAction())) {
117             fields.add(EndowPropertyConstants.SECURITY_UNIT_VALUE);
118             fields.add(EndowPropertyConstants.SECURITY_MARKET_VALUE);
119             fields.add(EndowPropertyConstants.SECURITY_VALUATION_DATE);
120             fields.add(EndowPropertyConstants.SECURITY_INCOME_RATE);
121         }
122 
123         // The default unit value for a new security is 1 EXCEPT for Liabilities (CLS_CD_T: CLS_CD_TYP = L) which will be negative 1
124         // (-1). The unit value for these securities must remain -1. The unit value for these securities cannot be edited.
125         if (classCode != null && EndowConstants.ClassCodeTypes.LIABILITY.equalsIgnoreCase(classCode.getClassCodeType())) {
126             fields.add(EndowPropertyConstants.SECURITY_UNIT_VALUE);
127         }
128 
129         return fields;
130     }
