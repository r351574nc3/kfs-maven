46      /**
47       * Overrides the parent method to check the status of the Pooled Fund Value Maintenance document. Business Rule 1: Saving a new
48       * or changed unit value will result in changes to the Security unit values and the associated holding market values. Once the
49       * document has been approved, If it is in the edit-maintenance doc mode, compare unit value between old and new pooled fund
50       * value BOs. If it is in the create-new or copy maintenance doc mode, compare unit value in the new pooled fund value BO with
51       * unit value in the referred Security record. When saving a new or changed unit value 1) Copy current Security Unit Value to
52       * Previous Unit Value 2) Copy current Security Price Date to Previous Price dAte 3) Copy PooledFundValue.POOL_FND_UNIT_VAL to
53       * current Security Unit Price 4) Copy PooledFundValue.VAL_EFF_DT to current Security Price Date 5) Update Security.Security
54       * Source to Pooled Fund Value Business Rule 2:When the DSTRB_AMT is updated, the amount of the distribution is multiplied by
55       * the number of times per year that the institution will distribute income to the account holders (this is an institution
56       * specified parameter: DISTRIBUTION_TIMES_PER_YEAR stored in the KFS parameter table) and the result is copied to END_SEC_T:
57       * SEC_RT for the Pooled Fund SEC_ID.
58       * 
59       * @see org.kuali.rice.kns.maintenance.KualiMaintainableImpl#doRouteStatusChange(org.kuali.rice.kns.bo.DocumentHeader)
60       */
61      @Override
62      public void doRouteStatusChange(DocumentHeader header) {
63          super.doRouteStatusChange(header);
64          KualiWorkflowDocument workflowDoc = header.getWorkflowDocument();
65  
66          DocumentService documentService = SpringContext.getBean(DocumentService.class);
67          try {
68              MaintenanceDocument maintDoc = (MaintenanceDocument) documentService.getByDocumentHeaderId(header.getDocumentNumber());
69  
70              initializeAttributes(maintDoc);
71  
72              // Use the stateIsFinal() method so this code is only executed when the final approval occurs
73              if (workflowDoc.stateIsFinal()) {
74                  BigDecimal oldUnitValue = null;
75                  BigDecimal oldDistributionPerUnit = null;
76                  BigDecimal newUnitValue = newPooledFundValue.getUnitValue();
77                  BigDecimal newDistributionPerUnit = newPooledFundValue.getIncomeDistributionPerUnit();
78                  BigDecimal oldIncomeDistributionPerUnit = oldPooledFundValue.getIncomeDistributionPerUnit();
79                  BigDecimal newIncomeDistributionPerUnit = newPooledFundValue.getIncomeDistributionPerUnit();
80                  Date newValueDate = newPooledFundValue.getValueEffectiveDate();
81                  String newUnitValueSource = EndowParameterKeyConstants.POOLED_FUND_VALUE;
82                  String pooledSecurityID = newPooledFundValue.getPooledSecurityID();
83  
84                  // Why the following method didn't work?
85                  // newPooledFundValue.refreshNonUpdateableReferences();
86                  // Security theSecurity = newPooledFundValue.getPooledFundControl().getSecurity();
87                  SecurityService securityService = SpringContext.getBean(SecurityService.class);
88                  Security theSecurity = securityService.getByPrimaryKey(pooledSecurityID);
89  
90                  ParameterService parameterService = SpringContext.getBean(ParameterService.class);
91                  BigDecimal numOfDistributions = new BigDecimal(new Double(parameterService.getParameterValue(PooledFundValue.class, EndowParameterKeyConstants.DISTRIBUTION_TIMES_PER_YEAR)).doubleValue());
92                  BigDecimal interestRate = newDistributionPerUnit.multiply(numOfDistributions);
93  
94                  /*
95                   * Note: There is an assumption here that the unitValue in the oldPooledFundValue is always equal to the unitValue
96                   * in the referred Security object. A business rule has to be added for Security impl -- when a Security has the
97                   * class code type equals to Pooled Investment, the unit value can't be modify directly through Security maintenance
98                   * doc.
99                   */
100                 if (KNSConstants.MAINTENANCE_EDIT_ACTION.equals(getMaintenanceAction())) {
101                     oldUnitValue = oldPooledFundValue.getUnitValue();
102                     oldDistributionPerUnit = oldPooledFundValue.getIncomeDistributionPerUnit();
103                     if (newUnitValue.compareTo(oldUnitValue) != 0) {
104                         compareAndUpdateUnitValue(oldUnitValue, newUnitValue, theSecurity, newValueDate, newUnitValueSource);
105                     }
106                     if (newDistributionPerUnit.compareTo(oldDistributionPerUnit) != 0) {
107                         updateInterestRate(theSecurity, interestRate);
108                     }                    
109                     if (newIncomeDistributionPerUnit.compareTo(oldIncomeDistributionPerUnit) != 0) {
110                         updateIncomeChangeDate(theSecurity);    
111                     }
112                 }
113                 else if (KNSConstants.MAINTENANCE_COPY_ACTION.equals(getMaintenanceAction()) || KNSConstants.MAINTENANCE_NEW_ACTION.equals(getMaintenanceAction())) {
114                     oldUnitValue = theSecurity.getUnitValue();
115                     compareAndUpdateUnitValue(oldUnitValue, newUnitValue, theSecurity, newValueDate, newUnitValueSource);
116                     updateInterestRate(theSecurity, interestRate);
117                     updateIncomeChangeDate(theSecurity);                    
118                 }
119             }
120         }
121         catch (WorkflowException we) {
122             LOG.error("encountered workflow exception while attempting to retrieve PooledFundValueMaintenanceDocument: " + header.getDocumentNumber() + " " + we);
123             throw new RuntimeException(we.getMessage());
124         }
125 
126     }

128     private void compareAndUpdateUnitValue(BigDecimal oldUnitValue, BigDecimal newUnitValue, Security theSecurity, Date newValueDate, String newUnitValueSource) {
129         SecurityService securityService = SpringContext.getBean(SecurityService.class);
130         if (newUnitValue.compareTo(oldUnitValue) != 0) {
131             Security security = securityService.updateUnitValue(theSecurity, newUnitValue, newValueDate, newUnitValueSource);
132             SpringContext.getBean(BusinessObjectService.class).save(security);
133         }
134     }
135 
136     private void updateInterestRate(Security theSecurity, BigDecimal interestRate) {
137         SecurityService securityService = SpringContext.getBean(SecurityService.class);
138         Security security = securityService.updateInterestRate(theSecurity, interestRate);
139         SpringContext.getBean(BusinessObjectService.class).save(security);
140     }
