48      /**
49       * 
50       * @see org.kuali.kfs.module.endow.report.service.TrialBalanceReportService#getTrialBalanceReportsByKemidByIds(java.util.List, java.lang.String)
51       */
52      public List<TrialBalanceReportDataHolder> getTrialBalanceReportsByKemidByIds(List<String> kemids, String endownmentOption, String closedIndicator) {
53          
54          List<TrialBalanceReportDataHolder> trialBalanceReportList = new ArrayList<TrialBalanceReportDataHolder>();
55          List<KEMID> kemidRecords = kemidDao.getKemidRecordsByIds(kemids, endownmentOption, closedIndicator);
56          if (kemidRecords == null || kemidRecords.isEmpty()) {
57              return null;
58          }
59          
60          for (KEMID kemidObj : kemidRecords) {
61              
62              // get the kemid and the title
63              TrialBalanceReportDataHolder trialBalanceReport = new TrialBalanceReportDataHolder(); 
64              trialBalanceReport.setKemid(kemidObj.getKemid());
65              trialBalanceReport.setKemidName(kemidObj.getShortTitle());
66  
67              Map<String, Object> primaryKeys = new HashMap<String, Object>();
68              primaryKeys.put(EndowPropertyConstants.KEMID, kemidObj.getKemid());
69              
70              // get income cash balance, principal cash balance            
71              KemidCurrentCash kemidCurrentCash = (KemidCurrentCash) businessObjectService.findByPrimaryKey(KemidCurrentCash.class, primaryKeys);
72              BigDecimal subTotalKemidTotalMarketValue = BigDecimal.ZERO;  
73              if (ObjectUtils.isNotNull(kemidCurrentCash)) {
74                  trialBalanceReport.setInocmeCashBalance(kemidCurrentCash.getCurrentIncomeCash());
75                  trialBalanceReport.setPrincipalcashBalance(kemidCurrentCash.getCurrentPrincipalCash());
76                  // add the cash amount to market value
77                  subTotalKemidTotalMarketValue = subTotalKemidTotalMarketValue.add(kemidCurrentCash.getCurrentIncomeCash().bigDecimalValue()).add(kemidCurrentCash.getCurrentPrincipalCash().bigDecimalValue());
78              } else {
79                  trialBalanceReport.setInocmeCashBalance(KualiDecimal.ZERO);
80                  trialBalanceReport.setPrincipalcashBalance(KualiDecimal.ZERO);  
81              }
82              
83              // get available expendable funds
84              KEMIDCurrentAvailableBalance kemidCurrentAvailableBalance = (KEMIDCurrentAvailableBalance) businessObjectService.findByPrimaryKey(KEMIDCurrentAvailableBalance.class, primaryKeys);
85              if (ObjectUtils.isNotNull(kemidCurrentAvailableBalance)) {
86                  trialBalanceReport.setAvailableExpendableFunds(kemidCurrentAvailableBalance.getAvailableTotalCash());
87              } else {
88                  trialBalanceReport.setAvailableExpendableFunds(BigDecimal.ZERO);
89              }
90              
91              // get sub total market value, FY remainder estimated income                       
92              BigDecimal subTotalRemainderEstimatedIncome = BigDecimal.ZERO;            
93              List<CurrentTaxLotBalance> CurrentTaxLotBalanceRecords = (List<CurrentTaxLotBalance>) businessObjectService.findMatching(CurrentTaxLotBalance.class, primaryKeys);
94              if (ObjectUtils.isNotNull(CurrentTaxLotBalanceRecords)) {
95                  for (CurrentTaxLotBalance currentTaxLotBalance : CurrentTaxLotBalanceRecords) {
96                      subTotalKemidTotalMarketValue = subTotalKemidTotalMarketValue.add(currentTaxLotBalance.getHoldingMarketValue());
97                      subTotalRemainderEstimatedIncome = subTotalRemainderEstimatedIncome.add(currentTaxLotBalance.getRemainderOfFYEstimatedIncome());
98                      trialBalanceReport.setKemidTotalMarketValue(subTotalKemidTotalMarketValue);
99                      trialBalanceReport.setFyRemainderEstimatedIncome(subTotalRemainderEstimatedIncome);
100                 }                
101             }
102             
103             // add this new one
104             trialBalanceReportList.add(trialBalanceReport);
105         }
106         
107         return trialBalanceReportList;
108     }
