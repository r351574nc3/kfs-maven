56      /**
57       * 
58       * @see org.kuali.kfs.module.endow.report.service.AssetStatementReportService#getAssetStatementReportsByKemidByIds(java.util.List, java.lang.String)
59       */
60      public List<AssetStatementReportDataHolder> getAssetStatementReportsByKemidByIds(List<String> kemids, String monthEndDate, String endowmentOption, String reportOption, String closedIndicator) {
61          
62          List<AssetStatementReportDataHolder> assetStatementReportList = null;
63  
64          // we need a valid month end date
65          Date date = convertStringToDate(monthEndDate);
66          if (ObjectUtils.isNull(date)) {
67              return null;
68          }
69  
70          // get valid kemids
71          List<String> kemidsSelected = kemidDao.getKemidsByAttributeWithEndowmentOption(EndowPropertyConstants.KEMID, kemids, endowmentOption, closedIndicator);
72          if (kemidsSelected == null || kemidsSelected.isEmpty()) {
73              return null;
74          }
75          // remove kemids that do not exist in historical cash
76          kemidsSelected = getKemidsInHistoryCash(kemidsSelected, monthEndDate);
77          
78          // get historical cash records
79          MonthEndDate endingMED = getMonthEndDate(convertStringToDate(monthEndDate));
80          KualiInteger medId = endingMED.getMonthEndDateId();
81          List<KemidHistoricalCash> historyCashRecords = getKemidHistoricalCashRecords(kemidsSelected, medId);
82          
83          if (KFSConstants.ParameterValues.YES.equalsIgnoreCase(endowmentOption)) {
84              // create a data holder for endowment
85              assetStatementReportList = createDataHoldersForEndowment(historyCashRecords, monthEndDate, medId, reportOption);            
86          } else if (KFSConstants.ParameterValues.NO.equalsIgnoreCase(endowmentOption)) {
87              // create a data holder for non-endowed
88              assetStatementReportList = createDataHoldersForNonEndowed(historyCashRecords, monthEndDate, medId, reportOption);
89          }    
90          return assetStatementReportList;      
91      }

93      /**
94       * Creates a collection of data holder for endowment
95       * 
96       * @param historyCashRecords
97       * @param monthEndDate
98       * @param medId
99       * @param reportOption
100      * @return
101      */
102     protected List<AssetStatementReportDataHolder> createDataHoldersForEndowment(List<KemidHistoricalCash> historyCashRecords, String monthEndDate, KualiInteger medId, String reportOption) {
103         
104         List<AssetStatementReportDataHolder> assetStatementReportList = new ArrayList<AssetStatementReportDataHolder>();
105         
106         for (KemidHistoricalCash historyCash : historyCashRecords) {
107             
108             KEMID kemidOjb = getKemid(historyCash.getKemid());
109             List<HoldingHistory> holdingHistoryRecordsForIncome = holdingHistoryDao.getHoldingHistoryByKemidIdAndMonthEndIdAndIpInd(historyCash.getKemid(), medId, IncomePrincipalIndicator.INCOME);
110             List<HoldingHistory> holdingHistoryRecordsForPrincipal = holdingHistoryDao.getHoldingHistoryByKemidIdAndMonthEndIdAndIpInd(historyCash.getKemid(), medId, IncomePrincipalIndicator.PRINCIPAL);
111             
112             // create a data holder
113             if ((holdingHistoryRecordsForIncome != null && !holdingHistoryRecordsForIncome.isEmpty()) || (holdingHistoryRecordsForPrincipal != null && !holdingHistoryRecordsForPrincipal.isEmpty())) {
114                 AssetStatementReportDataHolder dataHolder = new AssetStatementReportDataHolder();
115                 dataHolder.setInstitution(getInstitutionName());
116                 dataHolder.setKemid(historyCash.getKemid());
117                 dataHolder.setKemidLongTitle(kemidOjb.getLongTitle());
118                 dataHolder.setMonthEndDate(monthEndDate);
119                 
120                 dataHolder.setHistoryIncomeCash(historyCash.getHistoricalIncomeCash().bigDecimalValue());
121                 dataHolder.setHistoryPrincipalCash(historyCash.getHistoricalPrincipalCash().bigDecimalValue());
122                 
123                 // for income
124                 if (holdingHistoryRecordsForIncome != null) {
125                     for (HoldingHistory holdingHistory : holdingHistoryRecordsForIncome) {
126                         Security security = getSecurityById(holdingHistory.getSecurityId());
127                         SecurityReportingGroup securityReportingGroup = getSecurityReportingGroupBySecurityId(holdingHistory.getSecurityId());
128                         ReportGroupData rgIncome = dataHolder.createReportGroupData(securityReportingGroup, security, IncomePrincipalIndicator.INCOME);
129     
130                         rgIncome.addSumOfUnits(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_UNITS, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.INCOME));
131                         rgIncome.addSumOfMarketValue(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_MARKET_VALUE, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.INCOME));
132                         rgIncome.addSumOfEstimatedIncome(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.INCOME));
133                         rgIncome.addSumOfRemainderOfFYEstimated(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_REMAIDER_OF_FY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.INCOME));
134                         rgIncome.addSumOfNextFYEstimatedIncome(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_NEXT_FY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.INCOME));
135                     }
136                 }
137                 
138                 // for principal
139                 if (holdingHistoryRecordsForPrincipal != null) {
140                     for (HoldingHistory holdingHistory : holdingHistoryRecordsForPrincipal) {
141                         Security security = getSecurityById(holdingHistory.getSecurityId());
142                         SecurityReportingGroup securityReportingGroup = getSecurityReportingGroupBySecurityId(holdingHistory.getSecurityId());
143                         ReportGroupData rgPrincipal = dataHolder.createReportGroupData(securityReportingGroup, security, IncomePrincipalIndicator.PRINCIPAL);
144                     
145                         rgPrincipal.addSumOfUnits(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_UNITS, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.PRINCIPAL));
146                         rgPrincipal.addSumOfMarketValue(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_MARKET_VALUE, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.PRINCIPAL));
147                         rgPrincipal.addSumOfEstimatedIncome(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.PRINCIPAL));
148                         rgPrincipal.addSumOfRemainderOfFYEstimated(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_REMAIDER_OF_FY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.PRINCIPAL));
149                         rgPrincipal.addSumOfNextFYEstimatedIncome(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_NEXT_FY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.PRINCIPAL));                
150                     }
151                 }
152                 
153                 // add footer data
154                 if (!reportOption.equalsIgnoreCase(EndowConstants.EndowmentReport.TOTAL)) {     
155                     dataHolder.setFooter(createFooterData(kemidOjb));
156                 }
157                 
158                 // add this new one
159                 assetStatementReportList.add(dataHolder);
160             }
161         }
162         
163         return assetStatementReportList;
164     }
