60      /**
61       * 
62       * @see org.kuali.kfs.module.endow.dataaccess.UpdateCorpusDao#updateCorpusAmountsFromTransactionArchive()
63       */
64      public void updateCorpusAmountsFromTransactionArchive(Date currentDate){
65  
66          LOG.info("updateCorpusAmountsFromTransactionArchive() started");
67          
68          Map<String, Object> currentCorpus = null;
69          BigDecimal currentCorpusAmount = null;
70          Date tomrrowsDate = getNextDay(currentDate);
71  
72          try {
73              String selectTransactionArchiveSql = "SELECT TRAN_KEMID, SUM(TRAN_CORPUS_AMT) TRAN_CORPUS_AMT FROM END_TRAN_ARCHV_T WHERE TRAN_CORPUS_IND = ? AND (TRAN_PSTD_DT >= ? AND TRAN_PSTD_DT < ?) GROUP BY TRAN_KEMID";
74              
75              String selectCurrentCorpusSql = "SELECT CRNT_CORPUS_VAL FROM END_CRNT_ENDOW_CORPUS_T WHERE KEMID = ?";
76              String updateCurrentCorpusSql = "UPDATE END_CRNT_ENDOW_CORPUS_T SET CRNT_CORPUS_VAL = ? WHERE KEMID = ?";
77              String insertCurrentCorpusSql = "INSERT INTO END_CRNT_ENDOW_CORPUS_T (KEMID, CRNT_CORPUS_VAL, CRNT_PRIN_MVAL, PRIOR_FY_CORPUS_VAL, PRIOR_FY_PRIN_MVAL, VER_NBR, OBJ_ID) VALUES (?, ?, 0, 0, 0, 1, ?)";
78                          
79              String updateKemIdCorpusSql = "UPDATE END_KEMID_CORPUS_VAL_T SET CRNT_CORPUS_VAL = ? WHERE KEMID = ?";            
80              
81              //Summarize transaction archive records by KemID on current date for CorpusAmount.
82              SqlRowSet pendingEntryRowSet = getJdbcTemplate().queryForRowSet(selectTransactionArchiveSql, new Object[] { EndowConstants.YES, currentDate, tomrrowsDate});
83  
84              int updateCount = 0;
85              int insertCount = 0;
86              
87              //Loop over transaction archive
88              while (pendingEntryRowSet.next()) {
89                  
90                  //store kem id and transaction corpus amount
91                  String kemId = pendingEntryRowSet.getString(EndowPropertyConstants.ColumnNames.UpdateCorpus.TRANSACTION_ARCHIVE_KEMID);
92                  BigDecimal corpusAmount = pendingEntryRowSet.getBigDecimal(EndowPropertyConstants.ColumnNames.UpdateCorpus.TRANSACTION_CORPUS_AMT);
93                                  
94                  try {
95                      //Try and get an Endow Current Transaction
96                      currentCorpus = getSimpleJdbcTemplate().queryForMap(selectCurrentCorpusSql, kemId);
97                  }
98                  catch (IncorrectResultSizeDataAccessException ex) {
99                      if (ex.getActualSize() != 0) {
100                         LOG.error("current corpus sql returned more than one row, aborting", ex);
101                         throw ex;
102                     }
103                     // no rows returned - that's ok
104                 }
105 
106                 if (currentCorpus != null) {
107                     //if exists update Current Corpus (trans amt + current amt)
108                     updateCount++;
109 
110                     currentCorpusAmount = (BigDecimal) currentCorpus.get(EndowPropertyConstants.ColumnNames.UpdateCorpus.CRNT_ENDOW_CORPUS_VAL);
111                     currentCorpusAmount = currentCorpusAmount.add(corpusAmount);
112                     
113                     // A current corpus exists, so we need to update it
114                     getSimpleJdbcTemplate().update(updateCurrentCorpusSql, currentCorpusAmount, kemId);
115                     
116                 } else {
117                      //if doesn't exist insert Current Corpus(trans amt)
118                     insertCount++;
119 
120                     currentCorpusAmount = new BigDecimal("0");
121                     currentCorpusAmount = currentCorpusAmount.add(corpusAmount);
122 
123                     // No current corpus exists, so we need to insert one
124                     getSimpleJdbcTemplate().update(insertCurrentCorpusSql, kemId, currentCorpusAmount, new Guid().toString());
125                 }
126                 
127                 // Now update KEM ID Corpus table
128                 getSimpleJdbcTemplate().update(updateKemIdCorpusSql, currentCorpusAmount, kemId);
129 
130             }
131             
132             LOG.info ("updateCorpusAmountsFromTransactionArchive has inserted Corpus Amount values for " + insertCount + " records in the END_CRNT_ENDOW_CORPUS_T table.");
133             LOG.info ("updateCorpusAmountsFromTransactionArchive has updated Corpus Amount values for " + updateCount + " records in the END_CRNT_ENDOW_CORPUS_T table.");            
134             LOG.info ("updateCorpusAmountsFromTransactionArchive has updated Corpus Amount values for " + (updateCount + insertCount) + " records in the END_KEMID_CORPUS_VAL_T table.");
135 
136         }
137         catch (Exception e) {
138             LOG.error("updateCorpusAmountsFromTransactionArchive() Exception running sql", e);
139             throw new RuntimeException("Unable to execute: " + e.getMessage(), e);
140         }        
141         
142         LOG.info("updateCorpusAmountsFromTransactionArchive() completed");
143     }
