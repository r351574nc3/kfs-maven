16  package org.kuali.kfs.module.ec.batch.dataaccess.impl;
17  
18  import org.kuali.kfs.module.ec.businessobject.EffortCertificationReportDefinition;
19  import org.kuali.kfs.sys.KFSConstants;
20  import org.kuali.kfs.sys.batch.dataaccess.impl.FiscalYearMakerImpl;
21  import org.kuali.rice.kns.bo.PersistableBusinessObject;
22  
23  /**
24   * Performs custom population of report definition records for a new year being created in the fiscal year maker process
25   */
26  public class EffortCertificationReportDefinitionFiscalYearMakerImpl extends FiscalYearMakerImpl {
27  
28      /**
29       * @see org.kuali.kfs.coa.dataaccess.impl.FiscalYearMakerImpl#changeForNewYear(java.lang.Integer,
30       *      org.kuali.rice.kns.bo.PersistableBusinessObject)
31       */
32      @Override
33      public void changeForNewYear(Integer baseFiscalYear, PersistableBusinessObject currentRecord) {
34          super.changeForNewYear(baseFiscalYear, currentRecord);
35  
36          EffortCertificationReportDefinition reportDefinition = (EffortCertificationReportDefinition) currentRecord;
37          reportDefinition.setEffortCertificationReportReturnDate(null);
38  
39          // set the various fiscal year fields up by 1
40          reportDefinition.setExpenseTransferFiscalYear(reportDefinition.getExpenseTransferFiscalYear() + 1);
41          reportDefinition.setEffortCertificationReportBeginFiscalYear(reportDefinition.getEffortCertificationReportBeginFiscalYear() + 1);
42          reportDefinition.setEffortCertificationReportEndFiscalYear(reportDefinition.getEffortCertificationReportEndFiscalYear() + 1);
43  
44          // all reporting period status codes to "not yet opened--updates allowed" at start of year
45          reportDefinition.setEffortCertificationReportPeriodStatusCode(KFSConstants.PeriodStatusCodes.NOT_OPEN);
46      }
47  
48  }
