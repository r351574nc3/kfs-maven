63      /**

65       * 
66       * @return true if the importing field values are valid; otherwsie, add errors into error map and return false
67       */
68      public boolean validateImportingFieldValues(EffortCertificationDocument document) {
69          DictionaryValidationService dictionaryValidationService = SpringContext.getBean(DictionaryValidationService.class);
70  
71          for (String fieldName : this.getImportingFields()) {
72              dictionaryValidationService.validateDocumentAttribute(document, fieldName, EffortConstants.DOCUMENT_PREFIX);
73          }

75          return GlobalVariables.getMessageMap().isEmpty();
76      }
