36  public class RelatedDocumentDerivedRoleTypeServiceImpl extends KimDerivedRoleTypeServiceBase {
37      protected static final String SOURCE_DOCUMENT_ROUTER_ROLE_NAME = "Source Document Router";
38      protected static final String SENSITIVE_RELATED_DOCUMENT_INITATOR_OR_REVIEWER_ROLE_NAME = "Sensitive Related Document Initiator Or Reviewer";
39  
40      private DocumentService documentService;
41      private PurapService purapService;

44      /**
45       * This service will accept the following attributes: Document Number Context: An fyi to the initiator - in the case of
46       * Automatic Purchase Orders (apo), the fyi is supposed to go to the requisition router. Otherwise, it should go to the PO
47       * router. Requirements: - KFS-PURAP Source Document Router - for Automated Purchase Order, Requisition router according to
48       * KR-WKFLW Router role / for normal Purchase Order, Purchase Order router according to KR-WKFLW Router
49       * 
50       * @see org.kuali.rice.kim.service.support.impl.KimRoleTypeServiceBase#getPrincipalIdsFromApplicationRole(java.lang.String,
51       *      java.lang.String, org.kuali.rice.kim.bo.types.dto.AttributeSet)
52       */
53      @Override
54      public List<RoleMembershipInfo> getRoleMembersFromApplicationRole(String namespaceCode, String roleName, AttributeSet qualification) {
55          List<RoleMembershipInfo> members = new ArrayList<RoleMembershipInfo>();
56          if(qualification!=null && !qualification.isEmpty()){
57              if (SOURCE_DOCUMENT_ROUTER_ROLE_NAME.equals(roleName)) {
58                  try {
59                      PurchasingAccountsPayableDocument document = (PurchasingAccountsPayableDocument) getDocumentService().getByDocumentHeaderId(qualification.get(PurapKimAttributes.DOCUMENT_NUMBER));
60                      if (document != null) {
61                          PurchasingAccountsPayableDocument sourceDocument = document.getPurApSourceDocumentIfPossible();
62                          if (sourceDocument != null && StringUtils.isNotBlank(sourceDocument.getDocumentHeader().getWorkflowDocument().getRoutedByPrincipalId()) ) {
63                              AttributeSet roleQualifier = new AttributeSet(1);
64                              roleQualifier.put(KfsKimAttributes.DOCUMENT_NUMBER, sourceDocument.getDocumentNumber() );
65                              members.add( new RoleMembershipInfo(null,null,sourceDocument.getDocumentHeader().getWorkflowDocument().getRoutedByPrincipalId(),Role.PRINCIPAL_MEMBER_TYPE,roleQualifier) );
66                          }
67                      }
68                  }
69                  catch (WorkflowException e) {
70                      throw new RuntimeException("Unable to load document in getPrincipalIdsFromApplicationRole", e);
71                  }
72              }
73              else if (SENSITIVE_RELATED_DOCUMENT_INITATOR_OR_REVIEWER_ROLE_NAME.equals(roleName)) {
74                  for (String documentId : getPurapService().getRelatedDocumentIds(new Integer(qualification.get(PurapKimAttributes.ACCOUNTS_PAYABLE_PURCHASING_DOCUMENT_LINK_IDENTIFIER)))) {
75                      AttributeSet tempQualification = new AttributeSet(1);
76                      tempQualification.put(PurapKimAttributes.DOCUMENT_NUMBER, documentId);
77                      for ( String principalId : getRoleManagementService().getRoleMemberPrincipalIds(KNSConstants.KUALI_RICE_WORKFLOW_NAMESPACE, RouteLogDerivedRoleTypeServiceImpl.INITIATOR_OR_REVIEWER_ROLE_NAME, tempQualification) ) {
78                          members.add( new RoleMembershipInfo(null,null,principalId,Role.PRINCIPAL_MEMBER_TYPE,tempQualification) );
79                      }
80                  }
81              }
82          }
83          return members;
84      }
