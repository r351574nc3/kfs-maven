20  package org.kuali.kfs.module.purap.util.cxml;
21  
22  import java.text.SimpleDateFormat;
23  import java.util.Date;
24  
25  import org.kuali.kfs.module.purap.PurapConstants;
26  import org.kuali.kfs.module.purap.businessobject.B2BInformation;
27  import org.kuali.kfs.module.purap.util.PurApDateFormatUtils;
28  import org.kuali.kfs.sys.context.SpringContext;
29  import org.kuali.rice.kim.bo.Person;
30  import org.kuali.rice.kns.service.DateTimeService;
31  
32  public class PunchOutSetupCxml {
33    private Person user;
34    private B2BInformation b2bInformation;
35  
36    public PunchOutSetupCxml(Person u,B2BInformation b) {
37      user = u;
38      b2bInformation = b;
39    }
40  
41    /**
42     * Get cxml punch out request xml
43     * @return xml for punch out request
44     */
45    public String getPunchOutSetupRequestMessage() {
46      StringBuffer cxml = new StringBuffer();
47      Date d = SpringContext.getBean(DateTimeService.class).getCurrentDate();
48      SimpleDateFormat date = PurApDateFormatUtils.getSimpleDateFormat(PurapConstants.NamedDateFormats.CXML_SIMPLE_DATE_FORMAT);
49      SimpleDateFormat time = PurApDateFormatUtils.getSimpleDateFormat(PurapConstants.NamedDateFormats.CXML_SIMPLE_TIME_FORMAT);
50  
51      // doing as two parts b/c they want a T instead of space
52      // between them, and SimpleDateFormat doesn't allow putting the
53      // constant "T" in the string
54  
55      cxml.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
56      cxml.append("<!DOCTYPE cXML SYSTEM \"cXML.dtd\">\n");
57      cxml.append("<cXML payloadID=\"irrelevant\" xml:lang=\"en-US\" timestamp=\"").append(date.format(d)).append("T")
58          .append(time.format(d)).append("-05:00").append("\">\n");
59  
60      // note that timezone is hard coded b/c this is the format
61      // they wanted, but SimpleDateFormat returns -0500, so rather than
62      // parse it just hard-coded
63  
64      cxml.append("  <Header>\n");
65      cxml.append("    <From>\n");
66      cxml.append("      <Credential domain=\"NetworkId\">\n");
67      cxml.append("        <Identity>").append(b2bInformation.getIdentity()).append("</Identity>\n");
68      cxml.append("      </Credential>\n");
69      cxml.append("    </From>\n");
70      cxml.append("    <To>\n");
71      cxml.append("      <Credential domain=\"DUNS\">\n");
72      cxml.append("        <Identity>").append(b2bInformation.getIdentity()).append("</Identity>\n");
73      cxml.append("      </Credential>\n");
74      cxml.append("      <Credential domain=\"internalsupplierid\">\n");
75      cxml.append("        <Identity>1016</Identity>\n");
76      cxml.append("      </Credential>\n");
77      cxml.append("    </To>\n");
78      cxml.append("    <Sender>\n");
79      cxml.append("      <Credential domain=\"TOPSNetworkUserId\">\n");
80      cxml.append("        <Identity>").append(user.getPrincipalName().toUpperCase()).append("</Identity>\n");
81      cxml.append("        <SharedSecret>").append(b2bInformation.getPassword()).append("</SharedSecret>\n");
82      cxml.append("      </Credential>\n");
83      cxml.append("      <UserAgent>").append(b2bInformation.getUserAgent()).append("</UserAgent>\n");
84      cxml.append("    </Sender>\n");
85      cxml.append("  </Header>\n");
86      cxml.append("  <Request deploymentMode=\"").append(b2bInformation.getEnvironment()).append("\">\n");
87      cxml.append("    <PunchOutSetupRequest operation=\"create\">\n");
88      cxml.append("      <BuyerCookie>").append(user.getPrincipalName().toUpperCase()).append("</BuyerCookie>\n");
89      //cxml.append(" <Extrinsic
90      // name=\"UserEmail\">jdoe@TOPS.com</Extrinsic>\n"); // we can't reliably
91      // get the e-mail address, so we're leaving it out
92      cxml.append("      <Extrinsic name=\"UniqueName\">").append(user.getPrincipalName().toUpperCase()).append("</Extrinsic>\n");
93      cxml.append("      <Extrinsic name=\"Department\">IU").append(user.getCampusCode()).append(user.getPrimaryDepartmentCode()).append("</Extrinsic>\n");
94      cxml.append("      <Extrinsic name=\"Campus\">").append(user.getCampusCode()).append("</Extrinsic>\n");
95      cxml.append("      <BrowserFormPost>\n");
96      cxml.append("        <URL>").append(b2bInformation.getPunchbackURL()).append("</URL>\n");
97      cxml.append("      </BrowserFormPost>\n");
98      cxml.append("      <Contact role=\"endUser\">\n");
99      cxml.append("        <Name xml:lang=\"en\">").append(user.getName()).append("</Name>\n");
100     //cxml.append(" <Email>jdoe@TOPS.com</Email>\n"); // again, we can't
101     // reliably get this, so we're leaving it out
102     cxml.append("      </Contact>\n");
103     cxml.append("      <SupplierSetup>\n");
104     cxml.append("        <URL>").append(b2bInformation.getPunchoutURL()).append("</URL>\n");
105     cxml.append("      </SupplierSetup>\n");
106     cxml.append("    </PunchOutSetupRequest>\n");
107     cxml.append("  </Request>\n");
108     cxml.append("</cXML>\n");
109 
110     return cxml.toString();
111   }

113 }
