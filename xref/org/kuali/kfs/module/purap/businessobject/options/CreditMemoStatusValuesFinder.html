29  /**
30   * Value Finder for Credit Memo Statuses.
31   */
32  public class CreditMemoStatusValuesFinder extends PurApStatusKeyValuesBase {
33  
34      /**
35       * Overide this method to sort the PO statuses for proper display. 
36       * 
37       * @see org.kuali.kfs.module.purap.businessobject.options.PurApStatusKeyValuesBase#getKeyValues()
38       */
39      public List getKeyValues() {
40          // get all PO statuses
41          KeyValuesService boService = SpringContext.getBean(KeyValuesService.class);
42          Collection<Status> statuses = boService.findAll(getStatusClass());
43          
44          // sort the statuses according to their codes alphabetically
45          int ns = statuses.size();
46          Status[] sortStatuses = new Status[ns];
47          if ( ns > 0 ) {
48              sortStatuses = (Status[])statuses.toArray(sortStatuses);
49              Arrays.sort(sortStatuses, sortStatuses[0]);       
50          }        
51          
52          // generate output
53          List labels = new ArrayList();
54          for (Status status : sortStatuses) {
55              labels.add(new KeyLabelPair(status.getStatusCode(), status.getStatusDescription()));
56          }
57          return labels;
58      }
59      
60      /**
61       * @see org.kuali.kfs.module.purap.businessobject.options.PurApStatusKeyValuesBase#getStatusClass()
62       */
63      @Override
64      public Class getStatusClass() {
65          return CreditMemoStatus.class;
66      }
