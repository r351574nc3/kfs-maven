40  public class PaymentRequestInvoiceImageAttachmentValidation extends GenericValidation {
41  
42      public boolean validate(AttributedDocumentEvent event) {
43          boolean valid = true;
44          PaymentRequestDocument document = (PaymentRequestDocument)event.getDocument();
45          GlobalVariables.getMessageMap().clearErrorPath();
46          
47          if(isDocumentStoppedInRouteNode(document, "ImageAttachment")){
48              //assume false if we're in the correct node
49              valid = false;
50              
51              //loop through notes looking for a invoice image
52              List boNotes = document.getBoNotes();
53              if (ObjectUtils.isNotNull(boNotes)) {
54                  for (Object obj : boNotes) {
55                      Note note = (Note) obj;
56                      
57                      note.refreshReferenceObject("attachment");
58                      if (ObjectUtils.isNotNull(note.getAttachment()) && PurapConstants.AttachmentTypeCodes.ATTACHMENT_TYPE_INVOICE_IMAGE.equals(note.getAttachment().getAttachmentTypeCode())) {
59                          valid = true;
60                          break;
61                      }
62                  }
63              }
64              
65              if(valid == false){
66                  GlobalVariables.getMessageMap().putError(KNSConstants.NEW_DOCUMENT_NOTE_PROPERTY_NAME, PurapKeyConstants.ERROR_PAYMENT_REQUEST_INVOICE_REQUIRED);
67              }
68          }
69          
70          GlobalVariables.getMessageMap().clearErrorPath();
71                  
72          return valid;
73      }
74  
75      protected boolean isDocumentStoppedInRouteNode(PaymentRequestDocument document, String nodeName) {
76          List<String> currentRouteLevels = new ArrayList<String>();
77          try {
78              KualiWorkflowDocument workflowDoc = document.getDocumentHeader().getWorkflowDocument();
79              currentRouteLevels = Arrays.asList(document.getDocumentHeader().getWorkflowDocument().getNodeNames());
80              if (currentRouteLevels.contains(nodeName) && workflowDoc.isApprovalRequested()) {
81                  return true;
82              }
83              return false;
84          }
85          catch (WorkflowException e) {
86              throw new RuntimeException(e);
87          }
88      }
