32  public class PurchasingPaymentInfoValidation extends GenericValidation {
33  
34      DateTimeService dateTimeService;
35      UniversityDateService universityDateService;
36      
37      public boolean validate(AttributedDocumentEvent event) {
38          boolean valid = true;
39          PurchasingDocument purDocument = (PurchasingDocument)event.getDocument();
40          
41          GlobalVariables.getMessageMap().addToErrorPath(PurapConstants.PAYMENT_INFO_ERRORS);
42          valid &= checkBeginDateBeforeEndDate(purDocument);
43          if (valid && (ObjectUtils.isNotNull(purDocument.getPurchaseOrderBeginDate()) || ObjectUtils.isNotNull(purDocument.getPurchaseOrderEndDate()))) {
44              if (ObjectUtils.isNotNull(purDocument.getPurchaseOrderBeginDate()) && ObjectUtils.isNull(purDocument.getPurchaseOrderEndDate())) {
45                  GlobalVariables.getMessageMap().putError(PurapPropertyConstants.PURCHASE_ORDER_END_DATE, PurapKeyConstants.ERROR_PURCHASE_ORDER_BEGIN_DATE_NO_END_DATE);
46                  valid &= false;
47              }
48              else {
49                  if (ObjectUtils.isNull(purDocument.getPurchaseOrderBeginDate()) && ObjectUtils.isNotNull(purDocument.getPurchaseOrderEndDate())) {
50                      GlobalVariables.getMessageMap().putError(PurapPropertyConstants.PURCHASE_ORDER_BEGIN_DATE, PurapKeyConstants.ERROR_PURCHASE_ORDER_END_DATE_NO_BEGIN_DATE);
51                      valid &= false;
52                  }
53              }
54          }
55          if (valid && ObjectUtils.isNotNull(purDocument.getPurchaseOrderBeginDate()) && ObjectUtils.isNotNull(purDocument.getPurchaseOrderEndDate())) {
56              if (ObjectUtils.isNull(purDocument.getRecurringPaymentTypeCode())) {
57                  GlobalVariables.getMessageMap().putError(PurapPropertyConstants.RECURRING_PAYMENT_TYPE_CODE, PurapKeyConstants.ERROR_RECURRING_DATE_NO_TYPE);
58  
59                  valid &= false;
60              }
61          }
62          else if (valid && ObjectUtils.isNotNull(purDocument.getRecurringPaymentTypeCode())) {
63              GlobalVariables.getMessageMap().putError(PurapPropertyConstants.PURCHASE_ORDER_BEGIN_DATE, PurapKeyConstants.ERROR_RECURRING_TYPE_NO_DATE);
64              valid &= false;
65          }
66  
67          //checks for when FY is set to next FY
68          if (purDocument.isPostingYearNext()) {
69              Integer currentFY = universityDateService.getCurrentFiscalYear();
70              Date closingDate = universityDateService.getLastDateOfFiscalYear(currentFY);
71  
72              //if recurring payment begin dates entered, begin date must be > closing date
73              if (ObjectUtils.isNotNull(purDocument.getPurchaseOrderBeginDate()) &&
74                     (purDocument.getPurchaseOrderBeginDate().before(closingDate) ||
75                      purDocument.getPurchaseOrderBeginDate().equals(closingDate))) {
76                  GlobalVariables.getMessageMap().putError(PurapPropertyConstants.PURCHASE_ORDER_BEGIN_DATE, PurapKeyConstants.ERROR_NEXT_FY_BEGIN_DATE_INVALID);
77                  valid &= false;
78              }
79          }
80  
81          GlobalVariables.getMessageMap().removeFromErrorPath(PurapConstants.PAYMENT_INFO_ERRORS);
82          return valid;
83      }
84  
85      /**
86       * Implementation of the rule that if a document has a recurring payment begin date and end date, the begin date should come
87       * before the end date. We needed to play around with this order if the fiscal year is the next fiscal year, since we
88       * were dealing just with month and day, but we don't need to do that here; we're dealing with the whole Date object.
89       * 
90       * @param purDocument the purchasing document to be validated
91       * @return boolean false if the begin date is not before the end date.
92       */
93      protected boolean checkBeginDateBeforeEndDate(PurchasingDocument purDocument) {
94          boolean valid = true;
95  
96          Date beginDate = purDocument.getPurchaseOrderBeginDate();
97          Date endDate = purDocument.getPurchaseOrderEndDate();
98          if (ObjectUtils.isNotNull(beginDate) && ObjectUtils.isNotNull(endDate)) {
99              if (dateTimeService.dateDiff(beginDate, endDate, false) <= 0) {
100                 valid &= false;
101                 GlobalVariables.getMessageMap().putError(PurapPropertyConstants.PURCHASE_ORDER_END_DATE, PurapKeyConstants.ERROR_PURCHASE_ORDER_BEGIN_DATE_AFTER_END);
102             }
103         }
104 
105         return valid;
106     }
