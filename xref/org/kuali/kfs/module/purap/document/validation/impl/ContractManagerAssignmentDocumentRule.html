67       * 
68       * @param document Contract Manager Assignment document

71      protected boolean processValidation(ContractManagerAssignmentDocument document) {

75      /**
76       * Review the list of ContractManagerAssignmentDetails where the user has entered ContractManagerCodes,
77       * validates that each entered code is valid;
78       * on the other hand, validate that at least one row has a valid CM code assigned.
79       * 
80       * @param contractManagerAssignmentDetails A list containing the code to be validated.
81       * @return Boolean indicating if validation succeeded
82       */
83      public boolean validateContractManagerCodes(List contractManagerAssignmentDetails) {
84          LOG.debug("validateContractManagerCodes(): entered method.");
85          boolean isValid = true;
86          int count = 0;
87          
88          for (Iterator iter = contractManagerAssignmentDetails.iterator(); iter.hasNext();) {
89              ContractManagerAssignmentDetail detail = (ContractManagerAssignmentDetail) iter.next();
90  
91              // Look for the contractManagerCode in the table. If not there the code is invalid.
92              if (ObjectUtils.isNotNull(detail.getContractManagerCode())) {
93                  Map fieldValues = new HashMap();
94                  fieldValues.put(PurapPropertyConstants.CONTRACT_MANAGER_CODE, detail.getContractManagerCode());
95                  fieldValues.put(KNSPropertyConstants.ACTIVE, true);
96                  if (SpringContext.getBean(BusinessObjectService.class).countMatching(ContractManager.class, fieldValues) != 1) {
97                      GlobalVariables.getMessageMap().putError(PurapConstants.ASSIGN_CONTRACT_MANAGER_TAB_ERRORS, PurapKeyConstants.INVALID_CONTRACT_MANAGER_CODE, detail.getContractManagerCode().toString());
98                      isValid = false;
99                  }
100                 else count++;
101                 if (detail.getContractManagerCode().equals(PurapConstants.APO_CONTRACT_MANAGER)) {
102                     GlobalVariables.getMessageMap().putError(PurapConstants.ASSIGN_CONTRACT_MANAGER_TAB_ERRORS, PurapKeyConstants.ERROR_APO_CONTRACT_MANAGER_CODE_CHOSEN, detail.getContractManagerCode().toString());
103                     isValid = false;
104                 }
105             }
106         }
107         
108         // check if at least one row has a valid CM code assigned
109         if (count < 1) {
110             GlobalVariables.getMessageMap().putError(PurapConstants.ASSIGN_CONTRACT_MANAGER_TAB_ERRORS, PurapKeyConstants.NO_CONTRACT_MANAGER_ASSIGNED);
111             isValid = false;
112         }
113         
114         LOG.debug("validateContractManagerCodes(): leaving method.");
115         return isValid;
116     }
