52      /**
53       * General Ledger pending entries are not created on save for this document. They are created when the document has been finally
54       * processed. Overriding this method so that entries are not created yet.
55       * 
56       * @see org.kuali.kfs.module.purap.document.PurchaseOrderDocument#prepareForSave(org.kuali.rice.kns.rule.event.KualiDocumentEvent)
57       */
58      @Override
59      public void prepareForSave(KualiDocumentEvent event) {
60          LOG.info("prepareForSave(KualiDocumentEvent) do not create gl entries");
61          setSourceAccountingLines(new ArrayList());
62          setGeneralLedgerPendingEntries(new ArrayList());
63      }
64  
65      @Override
66      public List<Long> getWorkflowEngineDocumentIdsToLock() {
67          return super.getWorkflowEngineDocumentIdsToLock();
68      }

70      /**
71       * When Purchase Order Reopen document has been processed through Workflow, the general ledger entries are created and the PO
72       * status changes to "OPEN".
73       * 
74       * @see org.kuali.kfs.module.purap.document.PurchaseOrderDocument#doRouteStatusChange()
75       */
76      @Override
77      public void doRouteStatusChange(DocumentRouteStatusChangeDTO statusChangeEvent) {
78          super.doRouteStatusChange(statusChangeEvent);
79  
80          // DOCUMENT PROCESSED
81          if (getDocumentHeader().getWorkflowDocument().stateIsProcessed()) {
82              // generate GL entries
83              SpringContext.getBean(PurapGeneralLedgerService.class).generateEntriesReopenPurchaseOrder(this);
84  
85              // update indicators
86              SpringContext.getBean(PurchaseOrderService.class).setCurrentAndPendingIndicatorsForApprovedPODocuments(this);
87  
88              // set purap status
89              SpringContext.getBean(PurapService.class).updateStatus(this, PurapConstants.PurchaseOrderStatuses.OPEN);
90          }
91          // DOCUMENT DISAPPROVED
92          else if (getDocumentHeader().getWorkflowDocument().stateIsDisapproved()) {
93              SpringContext.getBean(PurchaseOrderService.class).setCurrentAndPendingIndicatorsForDisapprovedReopenPODocuments(this);
94          }
95          // DOCUMENT CANCELED
96          else if (getDocumentHeader().getWorkflowDocument().stateIsCanceled()) {
97              SpringContext.getBean(PurchaseOrderService.class).setCurrentAndPendingIndicatorsForCancelledReopenPODocuments(this);
98          }
99  
100     }

122     @Override

124         List<GeneralLedgerPendingEntrySourceDetail> accountingLines = new ArrayList<GeneralLedgerPendingEntrySourceDetail>();
125         if (getGlOnlySourceAccountingLines() != null) {

127             while (iter.hasNext()) {

130         }
