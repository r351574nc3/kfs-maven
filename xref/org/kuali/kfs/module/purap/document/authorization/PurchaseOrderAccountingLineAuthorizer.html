30  /**
31   * Accounting line authorizer for Requisition document which allows adding accounting lines at specified nodes
32   */
33  public class PurchaseOrderAccountingLineAuthorizer extends PurapAccountingLineAuthorizer {
34      private static final String NEW_UNORDERED_ITEMS_NODE = "NewUnorderedItems";
35  
36      /**
37       * Allow new lines to be rendered at NewUnorderedItems node
38       * @see org.kuali.kfs.sys.document.authorization.AccountingLineAuthorizerBase#renderNewLine(org.kuali.kfs.sys.document.AccountingDocument, java.lang.String)
39       */
40      @Override
41      public boolean renderNewLine(AccountingDocument accountingDocument, String accountingGroupProperty) {
42          KualiWorkflowDocument workflowDoc = accountingDocument.getDocumentHeader().getWorkflowDocument();
43          String currentRouteNodeName = workflowDoc.getCurrentRouteNodeNames();
44          
45          //  if its in the NEW_UNORDERED_ITEMS node, then allow the new line to be drawn
46          if (PurchaseOrderAccountingLineAuthorizer.NEW_UNORDERED_ITEMS_NODE.equals(currentRouteNodeName)) {
47              return true;
48          }
49          
50          if (PurapConstants.PurchaseOrderDocTypes.PURCHASE_ORDER_AMENDMENT_DOCUMENT.equals(workflowDoc.getDocumentType()) && StringUtils.isNotBlank(accountingGroupProperty) && accountingGroupProperty.contains(PurapPropertyConstants.ITEM)) {
51              int itemNumber = determineItemNumberFromGroupProperty(accountingGroupProperty);
52              PurchaseOrderAmendmentDocument poaDoc = (PurchaseOrderAmendmentDocument) accountingDocument;
53              PurchaseOrderItem item = (PurchaseOrderItem) poaDoc.getItem(itemNumber);
54              return item.isNewItemForAmendment() || item.getSourceAccountingLines().size() == 0;
55          }
56          
57          return super.renderNewLine(accountingDocument, accountingGroupProperty);
58      }
59      
60      private int determineItemNumberFromGroupProperty(String accountingGroupProperty) {
61          int openBracketPos = accountingGroupProperty.indexOf("[");
62          int closeBracketPos = accountingGroupProperty.indexOf("]");
63          String itemNumberString = accountingGroupProperty.substring(openBracketPos + 1, closeBracketPos);
64          int itemNumber = new Integer(itemNumberString).intValue();
65          return itemNumber;
66      }
67      
68      @Override
69      protected boolean allowAccountingLinesAreEditable(AccountingDocument accountingDocument,
70              AccountingLine accountingLine){
71          PurApAccountingLine purapAccount = (PurApAccountingLine)accountingLine;
72          PurchaseOrderItem poItem = (PurchaseOrderItem)purapAccount.getPurapItem();
73          PurchaseOrderDocument po = (PurchaseOrderDocument)accountingDocument;
74          
75          
76          if (poItem != null && !poItem.getItemType().isAdditionalChargeIndicator()) {
77              if (!poItem.isItemActiveIndicator()) {
78                  return false;
79              }
80  
81              // if total amount has a value and is non-zero
82              if (poItem.getItemInvoicedTotalAmount() != null && poItem.getItemInvoicedTotalAmount().compareTo(new KualiDecimal(0)) != 0) {
83                  return false;
84              }
85              
86              if (po.getContainsUnpaidPaymentRequestsOrCreditMemos() && !poItem.isNewItemForAmendment()) {
87                  return false;
88              }
89              
90          }
91          return super.allowAccountingLinesAreEditable(accountingDocument, accountingLine);
92      }
93  
94      @Override
95      public boolean determineEditPermissionOnLine(AccountingDocument accountingDocument, AccountingLine accountingLine, String accountingLineCollectionProperty, boolean currentUserIsDocumentInitiator, boolean pageIsEditable) {
96       // the fields in a new line should be always editable
97          if (accountingLine.getSequenceNumber() == null) {
98              return true;
99          }
100         
101         // check the initiation permission on the document if it is in the state of preroute, but only if
102         // the PO status is not In Process.
103         KualiWorkflowDocument workflowDocument = accountingDocument.getDocumentHeader().getWorkflowDocument();
104         PurchaseOrderDocument poDocument = (PurchaseOrderDocument)accountingDocument;
105         if (!poDocument.getStatusCode().equals(PurapConstants.PurchaseOrderStatuses.IN_PROCESS) && (workflowDocument.stateIsInitiated() || workflowDocument.stateIsSaved())) {
106             if (PurapConstants.PurchaseOrderDocTypes.PURCHASE_ORDER_AMENDMENT_DOCUMENT.equals(workflowDocument.getDocumentType())) {
107                 PurApAccountingLine purapAccount = (PurApAccountingLine)accountingLine;
108                 PurchaseOrderItem item = (PurchaseOrderItem)purapAccount.getPurapItem();
109                 return item.isNewItemForAmendment() || item.getSourceAccountingLines().size() == 0;
110             }
111             else {
112                 return currentUserIsDocumentInitiator;
113             }
114         }
115         else {
116             return true;
117         }
118     }
