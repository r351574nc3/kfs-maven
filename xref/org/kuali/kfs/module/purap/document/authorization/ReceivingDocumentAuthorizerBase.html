33  public class ReceivingDocumentAuthorizerBase extends TransactionalDocumentAuthorizerBase {
34  
35      @Override
36      protected void addRoleQualification(BusinessObject businessObject, Map<String, String> attributes) {
37          super.addRoleQualification(businessObject, attributes);
38          attributes.put(PurapKimAttributes.DOCUMENT_SENSITIVE, "false");
39          ReceivingDocument purapDoc = (ReceivingDocument) businessObject;
40          if (purapDoc.getAccountsPayablePurchasingDocumentLinkIdentifier() != null) {
41              List<SensitiveData> sensitiveDataList = SpringContext.getBean(SensitiveDataService.class).getSensitiveDatasAssignedByRelatedDocId(purapDoc.getAccountsPayablePurchasingDocumentLinkIdentifier());
42              StringBuffer sensitiveDataCodes = new StringBuffer();
43              for (SensitiveData sensitiveData : sensitiveDataList) {
44                  sensitiveDataCodes.append(sensitiveData.getSensitiveDataCode()).append(";");
45              }
46              if (sensitiveDataCodes.length() > 0) {
47                  attributes.put(PurapKimAttributes.DOCUMENT_SENSITIVE, "true");
48                  attributes.put(PurapKimAttributes.SENSITIVE_DATA_CODE, sensitiveDataCodes.toString().substring(0, sensitiveDataCodes.length() - 1));
49                  attributes.put(PurapKimAttributes.ACCOUNTS_PAYABLE_PURCHASING_DOCUMENT_LINK_IDENTIFIER, purapDoc.getAccountsPayablePurchasingDocumentLinkIdentifier().toString());
50              }
51          }
52      }
53      
54      @Override
55      public boolean canEditDocumentOverview(Document document, Person user) {
56          return isAuthorizedByTemplate(document,
57                  KNSConstants.KNS_NAMESPACE,
58                  KimConstants.PermissionTemplateNames.EDIT_DOCUMENT,
59                  user.getPrincipalId());
60      }
