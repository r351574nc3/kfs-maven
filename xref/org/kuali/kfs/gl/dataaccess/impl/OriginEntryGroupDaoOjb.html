51      /**
52       * Given an origin entry group source type (defined in OriginEntrySource)
53       * 
54       * @param sourceCode the source code of the groups to find
55       * @return a OriginEntryGroup with the given source code and max ORIGIN_ENTRY_GRP_ID
56       * @see org.kuali.kfs.gl.businessobject.OriginEntrySource
57       * @see org.kuali.kfs.gl.dataaccess.OriginEntryGroupDao#getGroupWithMaxIdFromSource(java.lang.String)
58       */
59      public OriginEntryGroup getGroupWithMaxIdFromSource(String sourceCode) {
60          LOG.debug("getGroupWithMaxIdFromSource() started");
61          
62          Criteria crit = new Criteria();
63          
64          Criteria subCrit = new Criteria();
65          subCrit.addEqualTo(SOURCE_CODE, sourceCode);
66          ReportQueryByCriteria subQuery = new ReportQueryByCriteria(OriginEntryGroup.class, subCrit);
67          subQuery.setAttributes(new String[]{MAX_ORIGIN_ENTRY_GRP_ID});
68          
69          crit.addGreaterOrEqualThan(ORIGIN_ENTRY_GRP_ID, subQuery);
70          
71          QueryByCriteria qbc = QueryFactory.newQuery(OriginEntryGroup.class, crit);
72          
73          return (OriginEntryGroup) getPersistenceBrokerTemplate().getObjectByQuery(qbc);
74      }

92      /**
93       * Delete all the groups in the list.  Note...it doesn't delete the entries within them, you need
94       * OriginEntryDao.deleteGroups for that
95       * 
96       * @params groups a Collection of origin entry groups to delete
97       * @see org.kuali.kfs.gl.dataaccess.OriginEntryGroupDao#deleteGroups(java.util.Collection)
98       */
99      public void deleteGroups(Collection<OriginEntryGroup> groups) {
100         LOG.debug("deleteGroups() started");
101 
102         List ids = new ArrayList();
103         for (Iterator iter = groups.iterator(); iter.hasNext();) {
104             OriginEntryGroup element = (OriginEntryGroup) iter.next();
105             ids.add(element.getId());
106         }
107         Criteria criteria = new Criteria();
108         criteria.addIn(ID, ids);
109 
110         getPersistenceBrokerTemplate().deleteByQuery(QueryFactory.newQuery(OriginEntryGroup.class, criteria));
111         getPersistenceBrokerTemplate().clearCache();
112     }
