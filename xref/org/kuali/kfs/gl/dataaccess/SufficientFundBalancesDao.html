34      public Collection getByObjectCode(Integer universityFiscalYear, String chartOfAccountsCode, String financialObjectCode);

43       */
44      public int deleteByAccountNumber(Integer universityFiscalYear, String chartOfAccountsCode, String accountNumber);

46      /**
47       * Returns a sufficient fund balance by its primary key values

49       * @param universityFiscalYear the university fiscal year of the sufficient funds balance to return
