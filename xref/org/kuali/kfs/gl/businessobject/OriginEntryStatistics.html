28  
29      public void addDebit(KualiDecimal d) {

31              debitTotalAmount = debitTotalAmount.add(d);
32          }
33      }
34

36          if (c != null) {
37              creditTotalAmount = creditTotalAmount.add(c);
38          }
39      }
40  
41      public void addBudget(KualiDecimal b) {
42          if (b != null) {

44          }
45      }
46
