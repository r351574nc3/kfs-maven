35      public List compareEntryHistory(Class entryClass, Class historyClass, int pastYears) {
36          List<Map<String, Object>> data = null;
37  
38          String entryTable = persistenceStructureService.getTableName(entryClass);
39          String historyTable = persistenceStructureService.getTableName(historyClass);
40          
41          //String[] pks = persistenceService.getPrimaryKeys(arg0)
42          int fiscalYear = universityDateService.getCurrentFiscalYear()-pastYears;
43          
44          StringBuilder queryBuilder = new StringBuilder();
45  
46          queryBuilder.append("select eh.* ");
47          queryBuilder.append("from "+historyTable+" eh ");
48          queryBuilder.append("left join ");
49          queryBuilder.append("( ");
50          queryBuilder.append("SELECT UNIV_FISCAL_YR, FIN_COA_CD, FIN_OBJECT_CD, FIN_BALANCE_TYP_CD, UNIV_FISCAL_PRD_CD, TRN_DEBIT_CRDT_CD, ");
51          queryBuilder.append("count(*) as entry_row_cnt, sum(TRN_LDGR_ENTR_AMT) as entry_amt ");
52          queryBuilder.append("FROM "+entryTable+" ");
53          queryBuilder.append("GROUP BY UNIV_FISCAL_YR, FIN_COA_CD, FIN_OBJECT_CD, FIN_BALANCE_TYP_CD, UNIV_FISCAL_PRD_CD, TRN_DEBIT_CRDT_CD ");
54          queryBuilder.append(") e ");
55          queryBuilder.append("on eh.univ_fiscal_yr = e.univ_fiscal_yr and eh.fin_coa_cd = e.fin_coa_cd and eh.fin_object_cd = e.fin_object_cd and ");
56          queryBuilder.append("eh.fin_balance_typ_cd = e.fin_balance_typ_cd and eh.univ_fiscal_prd_cd = e.univ_fiscal_prd_cd and eh.trn_debit_crdt_cd = e.trn_debit_crdt_cd ");
57          queryBuilder.append("where e.univ_fiscal_yr >= "+fiscalYear+" and (eh.row_cnt <> e.entry_row_cnt or eh.trn_ldgr_entr_amt <> e.entry_amt or e.entry_row_cnt is null) ");
58  
59          data = getSimpleJdbcTemplate().queryForList(queryBuilder.toString());
60  
61          return data;
