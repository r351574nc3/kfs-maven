31  /**
32   * This step runs the balance forward year end process.
33   */
34  public class BalanceForwardStep extends AbstractWrappedBatchStep {
35      private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(BalanceForwardStep.class);
36  
37      private YearEndService yearEndService;
38  
39      public static final String TRANSACTION_DATE_FORMAT_STRING = "yyyy-MM-dd";
40  
41      /**
42       * @see org.kuali.kfs.sys.batch.AbstractWrappedBatchStep#getCustomBatchExecutor()
43       */
44      @Override
45      protected CustomBatchExecutor getCustomBatchExecutor() {
46          return new CustomBatchExecutor() {
47              /**
48               * This step runs the balance forward service, specifically finding the parameters the job needs, creating the origin entry
49               * groups for the output origin entries, and creating the process's reports.
50               * @return that the job finished successfully
51               * @see org.kuali.kfs.sys.batch.Step#execute(String, java.util.Date)
52               */
53              public boolean execute() {
54                  StopWatch stopWatch = new StopWatch();
55                  stopWatch.start("Balance Forward Step");
56  
57                  Date varTransactionDate;
58                  try {
59                      DateFormat transactionDateFormat = new SimpleDateFormat(TRANSACTION_DATE_FORMAT_STRING);
60                      varTransactionDate = new Date(transactionDateFormat.parse(getParameterService().getParameterValue(KfsParameterConstants.GENERAL_LEDGER_BATCH.class, GeneralLedgerConstants.ANNUAL_CLOSING_TRANSACTION_DATE_PARM)).getTime());
61                  }
62                  catch (ParseException e) {
63                      LOG.error("forwardBalances() Unable to parse transaction date", e);
64                      throw new IllegalArgumentException("Unable to parse transaction date");
65                  }
66  
67                  Integer varFiscalYear = new Integer(getParameterService().getParameterValue(KfsParameterConstants.GENERAL_LEDGER_BATCH.class, GeneralLedgerConstants.ANNUAL_CLOSING_FISCAL_YEAR_PARM));
68  
69                  yearEndService.logAllMissingPriorYearAccounts(varFiscalYear);
70                  yearEndService.logAllMissingSubFundGroups(varFiscalYear);
71  
72                  String balanceForwardsUnclosedFileName = GeneralLedgerConstants.BatchFileSystem.BALANCE_FORWARDS_FILE + GeneralLedgerConstants.BatchFileSystem.EXTENSION; 
73                  String balanceForwardsclosedFileName = GeneralLedgerConstants.BatchFileSystem.BALANCE_FORWARDS_CLOSED_FILE + GeneralLedgerConstants.BatchFileSystem.EXTENSION;
74                  
75                  BalanceForwardRuleHelper balanceForwardRuleHelper = new BalanceForwardRuleHelper(varFiscalYear, varTransactionDate, balanceForwardsclosedFileName, balanceForwardsUnclosedFileName);
76  
77                  yearEndService.forwardBalances(balanceForwardsUnclosedFileName, balanceForwardsclosedFileName, balanceForwardRuleHelper);
78  
79                  stopWatch.stop();
80                  LOG.info("Balance Forward Step took " + (stopWatch.getTotalTimeSeconds() / 60.0) + " minutes to complete");
81  
82                  return true;
83              }
84          };
85      }
86  
87      /**
88       * Sets the yearEndService attribute, allowing injection of an implementation of the service
89       * 
90       * @param yearEndService an implementation of the yearEndService
91       * @see org.kuali.module.gl.service.yearEndService
92       */
93      public void setYearEndService(YearEndService yearEndService) {
94          this.yearEndService = yearEndService;
95      }
96  }
