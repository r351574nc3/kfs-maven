74      /**
75       * Returns the file extension depending on the file type
76       * 
77       * @param fileType the file type (returned in {@link #getFileTypes()})
78       * @return the file extension
79       */
80      protected String getFileExtension(String fileType) {
81          if (KFSConstants.DATA_FILE_TYPE.equals(fileType)) {
82              return EnterpriseFeederService.DATA_FILE_SUFFIX;
83          }
84          if (KFSConstants.RECON_FILE_TYPE.equals(fileType)) {
85              return EnterpriseFeederService.RECON_FILE_SUFFIX;
86          }
87          throw new IllegalArgumentException("Unknown file type found: " + fileType);
88      }

104      * Return the file name based on information from user and file user identifier

109      * @see org.kuali.kfs.sys.batch.BatchInputFileSetType#getFileName(java.lang.String, org.kuali.rice.kim.bo.Person,

112     public String getFileName(String fileType, String principalName, String fileUserIdentifer, Date creationDate) {
113         DateTimeService dateTimeService = SpringContext.getBean(DateTimeService.class);
114         StringBuilder buf = new StringBuilder();
115         fileUserIdentifer = StringUtils.deleteWhitespace(fileUserIdentifer);
116         fileUserIdentifer = StringUtils.remove(fileUserIdentifer, FILE_NAME_PART_DELIMITER);
117         buf.append(FILE_NAME_PREFIX).append(FILE_NAME_PART_DELIMITER).append(principalName)
118                 .append(FILE_NAME_PART_DELIMITER).append(fileUserIdentifer)
119                 .append(FILE_NAME_PART_DELIMITER).append(dateTimeService.toDateTimeStringForFilename(creationDate))
120                 .append(getFileExtension(fileType));
121         return buf.toString();
122     }

194     /**
195      * Return set of file user identifiers from a list of files
196      * 
197      * @param user user who uploaded or will upload file
198      * @param files list of files objects
199      * @return Set containing all user identifiers from list of files
200      * @see org.kuali.kfs.sys.batch.BatchInputFileSetType#extractFileUserIdentifiers(org.kuali.rice.kim.bo.Person, java.util.List)
201      */
202     public Set<String> extractFileUserIdentifiers(Person user, List<File> files) {
203         Set<String> extractedFileUserIdentifiers = new TreeSet<String>();
204 
205         StringBuilder buf = new StringBuilder();
206         buf.append(FILE_NAME_PREFIX).append(FILE_NAME_PART_DELIMITER).append(user.getPrincipalName()).append(FILE_NAME_PART_DELIMITER);
207         String prefixString = buf.toString();
208         IOFileFilter prefixFilter = new PrefixFileFilter(prefixString);
209 
210         IOFileFilter suffixFilter = new OrFileFilter(new SuffixFileFilter(EnterpriseFeederService.DATA_FILE_SUFFIX), new SuffixFileFilter(EnterpriseFeederService.RECON_FILE_SUFFIX));
211 
212         IOFileFilter combinedFilter = new AndFileFilter(prefixFilter, suffixFilter);
213 
214         for (File file : files) {
215             if (combinedFilter.accept(file)) {
216                 String fileName = file.getName();
217                 if (fileName.endsWith(EnterpriseFeederService.DATA_FILE_SUFFIX)) {
218                     extractedFileUserIdentifiers.add(StringUtils.substringBetween(fileName, prefixString, EnterpriseFeederService.DATA_FILE_SUFFIX));
219                 }
220                 else if (fileName.endsWith(EnterpriseFeederService.RECON_FILE_SUFFIX)) {
221                     extractedFileUserIdentifiers.add(StringUtils.substringBetween(fileName, prefixString, EnterpriseFeederService.RECON_FILE_SUFFIX));
222                 }
223                 else {
224                     LOG.error("Unable to determine file user identifier for file name: " + fileName);
225                     throw new RuntimeException("Unable to determine file user identifier for file name: " + fileName);
226                 }
227             }
228         }
229 
230         return extractedFileUserIdentifiers;
231     }
