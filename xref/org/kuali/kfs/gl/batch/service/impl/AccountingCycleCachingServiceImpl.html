88      public boolean isCurrentActiveAccountingDocumentType(String documentTypeCode) {
89          Boolean documentTypeValid = documentTypeValidCache.get(documentTypeCode);
90          if (documentTypeValid == null) {
91              documentTypeValid = new Boolean(financialSystemDocumentTypeService.isCurrentActiveAccountingDocumentType(documentTypeCode));
92              documentTypeValidCache.put(documentTypeCode, documentTypeValid);
93          }
94          return documentTypeValid.booleanValue();
95      }
