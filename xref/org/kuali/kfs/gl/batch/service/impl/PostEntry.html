37  /**
38   * An implementation of PostTransaction that posts the actual entry
39   */
40  @Transactional
41  public class PostEntry implements PostTransaction {
42      private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(PostEntry.class);
43  
44      private AccountingCycleCachingService accountingCycleCachingService;
45      private PersistenceStructureService persistenceStructureService;
46  
47      /**
48       * Constructs an instance of PostEntry
49       */
50      public PostEntry() {
51          super();
52      }
53  
54      /**
55       * Saves the transaction as a new GL Entry
56       * 
57       * @param t the transaction which is being posted
58       * @param mode the mode the poster is currently running in
59       * @param postDate the date this transaction should post to
60       * @param posterReportWriterService the writer service where the poster is writing its report
61       * @return the accomplished post type
62       * @see org.kuali.kfs.gl.batch.service.PostTransaction#post(org.kuali.kfs.gl.businessobject.Transaction, int, java.util.Date)
63       */
64      public String post(Transaction t, int mode, Date postDate, ReportWriterService posterReportWriterService) {
65          LOG.debug("post() started");
66  
67          Entry e = new Entry(t, postDate);
68  
69          if (mode == PosterService.MODE_REVERSAL) {
70              e.setFinancialDocumentReversalDate(null);
71          }
72          
73          accountingCycleCachingService.insertEntry(e);
74  
75          return GeneralLedgerConstants.INSERT_CODE;
76      }
77  
78      /**
79       * @see org.kuali.kfs.gl.batch.service.PostTransaction#getDestinationName()
80       */
81      public String getDestinationName() {
82          return persistenceStructureService.getTableName(Entry.class);
83      }
84  
85      public void setAccountingCycleCachingService(AccountingCycleCachingService accountingCycleCachingService) {
86          this.accountingCycleCachingService = accountingCycleCachingService;
87      }
88  
89      public void setPersistenceStructureService(PersistenceStructureService persistenceStructureService) {
90          this.persistenceStructureService = persistenceStructureService;
91      }
92  
93  }
