16  package org.kuali.kfs.gl.batch.service;
17  
18  import java.util.Map;
19  
20  import org.kuali.kfs.gl.batch.BalanceForwardRuleHelper;
21  import org.kuali.kfs.gl.businessobject.OriginEntryGroup;
22  
23  
24  /**
25   * An interface declaring the methods that steps running year end services would need to use to close out activity of an ending
26   * fiscal year (though, note: OrganizationReversion, the other year end job, is not defined here).
27   */
28  public interface YearEndService {
29  
30  
31      /**
32       * The method responsible for creating origin entries that will forward qualifying encumbrances into the next fiscal year.
33       * 
34       * @param originEntryGroup the origin entry group where generated origin entries should be saved
35       * @param jobParameters the parameters necessary to run this job: the fiscal year to close and the university date the job was
36       *        run
37       * @param forwardEncumbrancesCounts the statistical counts generated by this job
38       */
39      public void forwardEncumbrances(String encumbranceForwardFileName, Map jobParameters, Map<String, Integer> counts);
40      
41      /**
42       * Implementations of this method are responsible for generating origin entries to forward the balances of qualifying balances
43       * from the previous fiscal year
44       * 
45       * @param balanceForwardsUnclosedPriorYearAccountGroup the origin entry group to save balance forwarding origin entries with
46       *        open accounts
47       * @param balanceForwardsClosedPriorYearAccountGroup the origin entry group to save balance forwarding origin entries with
48       *        closed accounts
49       * @param balanceForwardRuleHelper the BalanceForwardRuleHelper which holds the important state - the job parameters and
50       *        statistics - for the job to run
51       */
52      public void forwardBalances(String balanceForwardsUnclosedFileName, String balanceForwardsclosedFileName, BalanceForwardRuleHelper balanceForwardRuleHelper);
53      
54      /**
55       * This method is considered responsible for generating all of the origin entries that will close out nominal activity for a
56       * given fiscal year.
57       * 
58       * @param nominalClosingOriginEntryGroup the origin entry group that nominal activity closing origin entries
59       * @param nominalClosingJobParameters the parameters needed by the job to run correctly: the current university date and the
60       *        fiscal year to close
61       * @param nominalClosingCounts counts the Map of statistical counts generated by the process
62       */
63      public void closeNominalActivity(String nominalClosingFileName, Map nominalClosingJobParameters);
64      
65      /**
66       * Logs all of the missing prior year accounts that balances and encumbrances processed by year end jobs would attempt to call
67       * on
68       * 
69       * @param balanceFiscalYear the fiscal year to find balances encumbrances for
70       */
71      public void logAllMissingPriorYearAccounts(Integer fiscalYear);
72  
73      /**
74       * Logs all of the missing sub fund groups that balances and encumbrances processed by the year end job would attempt to call on
75       * 
76       * @param balanceFiscalYear the fiscal year to find balances and encumbrances for
77       */
78      public void logAllMissingSubFundGroups(Integer fiscalYear);
79  }
