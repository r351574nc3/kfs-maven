16  package org.kuali.kfs.fp.identity;
17  
18  import java.util.ArrayList;
19  import java.util.Collections;
20  import java.util.List;
21  
22  import org.kuali.kfs.sys.context.SpringContext;
23  import org.kuali.rice.kim.bo.types.dto.AttributeSet;
24  import org.kuali.rice.kim.service.RoleManagementService;
25  import org.kuali.rice.kim.service.support.impl.KimDerivedRoleTypeServiceBase;
26  
27  /**
28   * A role which determines membership in role for people who can initiate Cash Receipts.  Members are those 
29   * users which belong to KFS-SYS User but NOT KFS-FP Cash Manager.
30   */
31  public class CashReceiptInitiatorDerivedRoleTypeServiceImpl extends KimDerivedRoleTypeServiceBase {
32      private RoleManagementService roleManagementService;
33      
34      private static final String SYS_USER_ROLE_NAMESPACE = "KFS-SYS";
35      private static final String SYS_USER_ROLE_NAME = "User";
36      private static final String CASH_MANAGER_ROLE_NAMESPACE = "KFS-FP";
37      private static final String CASH_MANAGER_ROLE_NAME = "Cash Manager";
38      
39      /**
40       * Overridden to check principal id is in KFS-SYS User and not in KFS-FP Cash Manager
41       * @see org.kuali.rice.kim.service.support.impl.KimRoleTypeServiceBase#hasApplicationRole(java.lang.String, java.util.List, java.lang.String, java.lang.String, org.kuali.rice.kim.bo.types.dto.AttributeSet)
42       */
43      @Override
44      public boolean hasApplicationRole(String principalId, List<String> groupIds, String namespaceCode, String roleName, AttributeSet qualification) {
45          if (principalMemberOfSysUsers(principalId, qualification)) {
46              if (!principalMemberOfCashManagers(principalId, qualification)) {
47                  return true;
48              }
49          }
50          return false;
51      }
52  
53      /**
54       * Determines if the given principal with the given qualification is a member of KFS-SYS User
55       * @param principalId the principal id of the shmoe trying to initiate a cash receipt
56       * @param qualification the qualification of said shmoe
57       * @return true if the said shmoe is indeed a member of KFS-SYS User; false otherwise
58       */
59      protected boolean principalMemberOfSysUsers(String principalId, AttributeSet qualification) {
60          return hasRoleMembership(principalId, qualification, SYS_USER_ROLE_NAMESPACE, SYS_USER_ROLE_NAME);
61      }
62      
63      /**
64       * Determines if the given principal with the given qualification is a member of KFS-FP Cash Manager
65       * @param principalId the principal idea of the user attempting to initiate a cash receipt
66       * @param qualification the qualification of said principal
67       * @return true if principal is a member of KFS-FP Cash Manager, false otherwise
68       */
69      protected boolean principalMemberOfCashManagers(String principalId, AttributeSet qualification) {
70          return hasRoleMembership(principalId, qualification, CASH_MANAGER_ROLE_NAMESPACE, CASH_MANAGER_ROLE_NAME);
71      }
72      
73      /**
74       * Determines if a principal has a given role
75       * @param principalId the id of the principal
76       * @param qualification their qualifications from the permission/responsibility call
77       * @param namespaceCode the namespace code of the role
78       * @param roleName the name of the role
79       * @return true if the principal is a member of the role, false otherwise
80       */
81      protected boolean hasRoleMembership(String principalId, AttributeSet qualification, String namespaceCode, String roleName) {
82          String roleId = getRoleManagementService().getRoleIdByName(namespaceCode, roleName);        
83          return getRoleManagementService().principalHasRole(principalId, Collections.singletonList(roleId), qualification);
84      }
85  
86      /**
87       * Gets the roleManagementService attribute. 
88       * @return Returns the roleManagementService.
89       */
90      protected RoleManagementService getRoleManagementService() {
91          if ( roleManagementService == null) {
92              roleManagementService = SpringContext.getBean(RoleManagementService.class);
93          }
94          return roleManagementService;
95      }
96  }
