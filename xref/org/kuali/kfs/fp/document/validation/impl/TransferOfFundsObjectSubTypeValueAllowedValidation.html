16  package org.kuali.kfs.fp.document.validation.impl;
17  
18  import org.apache.commons.lang.StringUtils;
19  import org.kuali.kfs.fp.document.service.TransferOfFundsService;
20  import org.kuali.kfs.sys.KFSKeyConstants;
21  import org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent;
22  import org.kuali.kfs.sys.document.validation.impl.AccountingLineValueAllowedValidation;
23  import org.kuali.rice.kns.util.GlobalVariables;
24  
25  /**
26   * Override of standard accounting line value allowed definition to make sure that any object sub type on a transfer of funds is either mandatory transfer or non-mandatory transfer.
27   */
28  public class TransferOfFundsObjectSubTypeValueAllowedValidation extends AccountingLineValueAllowedValidation {
29      private TransferOfFundsService transferOfFundsService;
30  
31      /**
32       * Overrides the parent to make sure that the chosen object code's object sub-type code is either Mandatory Transfer or
33       * Non-Mandatory Transfer. 
34       * @see org.kuali.kfs.sys.document.validation.impl.AccountingLineValueAllowedValidation#validate(org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent)
35       */
36      @Override
37      public boolean validate(AttributedDocumentEvent event) {
38          getAccountingLineForValidation().refreshReferenceObject("objectCode");
39          String objectSubTypeCode = getAccountingLineForValidation().getObjectCode().getFinancialObjectSubTypeCode();
40  
41          // make sure a object sub type code exists for this object code
42          if (StringUtils.isBlank(objectSubTypeCode)) {
43              GlobalVariables.getMessageMap().putError("financialObjectCode", KFSKeyConstants.ERROR_DOCUMENT_TOF_OBJECT_SUB_TYPE_IS_NULL, getAccountingLineForValidation().getFinancialObjectCode());
44              return false;
45          }
46  
47          if (!transferOfFundsService.isMandatoryTransfersSubType(objectSubTypeCode) && !transferOfFundsService.isNonMandatoryTransfersSubType(objectSubTypeCode)) {
48              GlobalVariables.getMessageMap().putError("financialObjectCode", KFSKeyConstants.ERROR_DOCUMENT_TOF_OBJECT_SUB_TYPE_NOT_MANDATORY_OR_NON_MANDATORY_TRANSFER, new String[] { getAccountingLineForValidation().getObjectCode().getFinancialObjectSubType().getFinancialObjectSubTypeName(), getAccountingLineForValidation().getFinancialObjectCode() });
49              return false;
50          }
51  
52          return true;
53      }
54  
55      /**
56       * Gets the transferOfFundService attribute. 
57       * @return Returns the transferOfFundService.
58       */
59      public TransferOfFundsService getTransferOfFundsService() {
60          return transferOfFundsService;
61      }
62  
63      /**
64       * Sets the transferOfFundService attribute value.
65       * @param transferOfFundService The transferOfFundService to set.
66       */
67      public void setTransferOfFundsService(TransferOfFundsService transferOfFundsService) {
68          this.transferOfFundsService = transferOfFundsService;
69      }
70  }
