27  /**
28   * This is the business object that represents the ServiceBillingDocument in Kuali. See
29   * {@link org.kuali.kfs.fp.document.validation.impl.ServiceBillingDocumentRule} for details on how it differs from
30   * {@link InternalBillingDocument}.
31   */
32  public class ServiceBillingDocument extends InternalBillingDocument implements CapitalAssetEditable {
33  
34      /**
35       * This method further restricts the valid accounting line types exclusively to those with income or expense object type codes
36       * only. This is done by calling isIncome() and isExpense() passing the accounting line.
37       * 
38       * @param financialDocument The document used to determine if the accounting line is a debit line.
39       * @param accountingLine The accounting line to be analyzed.
40       * @return True if the accounting line passed in is an expense or income accounting line and meets the rules defined by
41       *         super.isDebit() method.
42       * @see org.kuali.kfs.fp.document.validation.impl.InternalBillingDocumentRule#isDebit(org.kuali.rice.kns.document.FinancialDocument,
43       *      org.kuali.rice.kns.bo.AccountingLine)
44       */
45      @Override
46      public boolean isDebit(GeneralLedgerPendingEntrySourceDetail postable) {
47          AccountingLine accountingLine = (AccountingLine) postable;
48          DebitDeterminerService isDebitUtils = SpringContext.getBean(DebitDeterminerService.class);
49          if (!isDebitUtils.isIncome(accountingLine) && !isDebitUtils.isExpense(accountingLine)) {
50              throw new IllegalStateException(isDebitUtils.getDebitCalculationIllegalStateExceptionMessage());
51          }
52  
53          return super.isDebit(postable);
54      }
55  
56      /**
57       * This method sets extra accounting line fields in explicit general ledger pending entries. Internal billing transactions don't
58       * have this field.
59       * 
60       * @param financialDocument The accounting document containing the general ledger pending entries being customized.
61       * @param accountingLine The accounting line the explicit general ledger pending entry was generated from.
62       * @param explicitEntry The explicit general ledger pending entry to be customized.
63       * @see FinancialDocumentRuleBase#customizeExplicitGeneralLedgerPendingEntry(FinancialDocument, AccountingLine,
64       *      GeneralLedgerPendingEntry)
65       */
66      @Override
67      public void customizeExplicitGeneralLedgerPendingEntry(GeneralLedgerPendingEntrySourceDetail postable, GeneralLedgerPendingEntry explicitEntry) {
68          String description = postable.getFinancialDocumentLineDescription();
69          if (StringUtils.isNotBlank(description)) {
70              explicitEntry.setTransactionLedgerEntryDescription(description);
71          }
72      }
