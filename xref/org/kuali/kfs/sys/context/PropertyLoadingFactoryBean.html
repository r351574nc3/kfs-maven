34  public class PropertyLoadingFactoryBean implements FactoryBean {
35      private static final String PROPERTY_FILE_NAMES_KEY = "property.files";
36      private static final String PROPERTY_TEST_FILE_NAMES_KEY = "property.test.files";
37      private static final String SECURITY_PROPERTY_FILE_NAME_KEY = "security.property.file";
38      private static final String CONFIGURATION_FILE_NAME = "configuration";
39      private static final Properties BASE_PROPERTIES = new Properties();
40      private static final String HTTP_URL_PROPERTY_NAME = "http.url";
41      private static final String KSB_REMOTING_URL_PROPERTY_NAME = "ksb.remoting.url";
42      private static final String REMOTING_URL_SUFFIX = "/remoting";
43      private Properties props = new Properties();
44      private boolean testMode;
45      private boolean secureMode;
46  
47      public Object getObject() throws Exception {
48          loadBaseProperties();
49          props.putAll(BASE_PROPERTIES);
50          if (secureMode) {
51              loadPropertyList(props,SECURITY_PROPERTY_FILE_NAME_KEY);
52          } else {
53              loadPropertyList(props,PROPERTY_FILE_NAMES_KEY);
54              if (testMode) {
55                  loadPropertyList(props,PROPERTY_TEST_FILE_NAMES_KEY);
56              }            
57          }
58          if (StringUtils.isBlank(System.getProperty(HTTP_URL_PROPERTY_NAME))) {
59              props.put(KSB_REMOTING_URL_PROPERTY_NAME, props.getProperty(KFSConstants.APPLICATION_URL_KEY) + REMOTING_URL_SUFFIX);
60          }
61          else {
62              props.put(KSB_REMOTING_URL_PROPERTY_NAME, new StringBuffer("http://").append(System.getProperty(HTTP_URL_PROPERTY_NAME)).append("/kfs-").append(props.getProperty(KFSConstants.ENVIRONMENT_KEY)).append(REMOTING_URL_SUFFIX).toString());
63          }
64          System.out.println(KSB_REMOTING_URL_PROPERTY_NAME + " set to " + props.getProperty(KSB_REMOTING_URL_PROPERTY_NAME));
65          return props;
66      }
67  
68      public Class getObjectType() {
69          return Properties.class;
70      }

72      public boolean isSingleton() {
73          return true;
74      }

76      private static void loadPropertyList(Properties props, String listPropertyName) {
77          for (String propertyFileName : getBaseListProperty(listPropertyName)) {
78              loadProperties(props,propertyFileName);
79          }
80      }
81  
82      private static void loadProperties( Properties props, String propertyFileName) {
83          InputStream propertyFileInputStream = null;
84          try {
85              try {
86                  propertyFileInputStream = new DefaultResourceLoader(ClassLoaderUtils.getDefaultClassLoader()).getResource(propertyFileName).getInputStream();
87                  props.load(propertyFileInputStream);
88              }
89              finally {
90                  if (propertyFileInputStream != null) {
91                      propertyFileInputStream.close();
92                  }
93              }
94          }
95          catch (IOException e) {
96              throw new RuntimeException("PropertyLoadingFactoryBean unable to load property file: " + propertyFileName);
97          }
98      }

100     public static String getBaseProperty(String propertyName) {
101         loadBaseProperties();
102         return BASE_PROPERTIES.getProperty(propertyName);
103     }

105     protected static List<String> getBaseListProperty(String propertyName) {
106         loadBaseProperties();
107         return Arrays.asList(BASE_PROPERTIES.getProperty(propertyName).split(","));
108     }

110     protected static void loadBaseProperties() {
111         if (BASE_PROPERTIES.isEmpty()) {
112             List<String> riceXmlConfigurations = new ArrayList<String>();
113             riceXmlConfigurations.add("classpath:META-INF/common-config-defaults.xml");
114             JAXBConfigImpl riceXmlConfigurer = new JAXBConfigImpl(riceXmlConfigurations);
115             try {
116                 riceXmlConfigurer.parseConfig();
117                 BASE_PROPERTIES.putAll(riceXmlConfigurer.getProperties());
118             }
119             catch (Exception e) {
120                 // Couldn't load the rice configs
121             }
122 
123             loadProperties(BASE_PROPERTIES, new StringBuffer("classpath:").append(CONFIGURATION_FILE_NAME).append(".properties").toString());
124         }
125 
126         final String additionalProps = BASE_PROPERTIES.getProperty("additional.config.locations");
127         final JAXBConfigImpl additionalConfigurer = new JAXBConfigImpl(java.util.Arrays.asList(additionalProps));
128         try {
129             additionalConfigurer.parseConfig();
130             System.out.println("Adding props from " + additionalProps);
131             BASE_PROPERTIES.putAll(additionalConfigurer.getProperties());                    
132         }
133         catch (Exception e) {
134             // Unable to load additional configs
135             e.printStackTrace();
136         }
137     }
