40  public class BatchStepRunner {
41      private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(BatchStepRunner.class);
42  
43      public static void main(String[] args) {
44          if (args.length < 1) {
45              System.err.println("ERROR: You must pass the name of the step to run on the command line.");
46              System.exit(8);
47          }
48          try {
49              Log4jConfigurer.configureLogging(false);
50              SpringContext.initializeBatchApplicationContext();
51              String[] stepNames;
52              if (args[0].indexOf(",") > 0) {
53                  stepNames = StringUtils.split(args[0], ",");
54              }
55              else {
56                  stepNames = new String[] { args[0] };
57              }
58              ParameterService parameterService = SpringContext.getBean(ParameterService.class);
59              DateTimeService dateTimeService = SpringContext.getBean(DateTimeService.class);
60              
61              String jobName = args.length >= 2 ? args[1] : KFSConstants.BATCH_STEP_RUNNER_JOB_NAME;
62              Date jobRunDate = dateTimeService.getCurrentDate();
63              LOG.info("Executing job: " + jobName + " steps: " + Arrays.toString(stepNames));
64              for (int i = 0; i < stepNames.length; ++i) {
65                  Step step = BatchSpringContext.getStep(stepNames[i]);
66                  if ( step != null ) {
67                      Step unProxiedStep = (Step) ProxyUtils.getTargetIfProxied(step);
68                      Class stepClass = unProxiedStep.getClass();
69                      
70                      ModuleService module = SpringContext.getBean(KualiModuleService.class).getResponsibleModuleService( stepClass );                    
71                      Appender ndcAppender = null; 
72                      String nestedDiagnosticContext = 
73                              StringUtils.substringAfter( module.getModuleConfiguration().getNamespaceCode(), "-").toLowerCase()
74                              + File.separator + step.getName()
75                              + "-" + dateTimeService.toDateTimeStringForFilename(dateTimeService.getCurrentDate());
76                      boolean ndcSet = false;;
77                      try {
78                          ndcAppender = new FileAppender(Logger.getRootLogger().getAppender("LogFile").getLayout(), getLogFileName(nestedDiagnosticContext));
79                          ndcAppender.addFilter(new NDCFilter(nestedDiagnosticContext));
80                          Logger.getRootLogger().addAppender(ndcAppender);
81                          NDC.push(nestedDiagnosticContext);
82                          ndcSet = true;
83                      } catch (Exception ex) {
84                          LOG.warn("Could not initialize custom logging for step: " + step.getName(), ex);
85                      }
86                      try {
87                          if (!Job.runStep(parameterService, jobName, i, step, jobRunDate) ) {
88                              System.exit(4);
89                          }
90                      } catch ( RuntimeException ex ) {                        
91                          throw ex;
92                      } finally {
93                          if ( ndcSet ) {
94                              ndcAppender.close();
95                              Logger.getRootLogger().removeAppender(ndcAppender);
96                              NDC.pop();
97                          }
98                      }
99                  } else {
100                     throw new IllegalArgumentException( "Unable to find step: " + stepNames[i] );
101                 }
102             }
103             LOG.info("Finished executing job: " + jobName + " steps: " + Arrays.toString(stepNames));
104             System.exit(0);
105         }
106         catch (Throwable t) {
107             System.err.println("ERROR: Exception caught: ");
108             t.printStackTrace(System.err);
109             System.exit(8);
110         }
111     }
112     
113     protected static String getLogFileName(String nestedDiagnosticContext) {
114         return SpringContext.getBean( KualiConfigurationService.class ).getPropertyString(KFSConstants.REPORTS_DIRECTORY_KEY) 
115                 + File.separator 
116                 + nestedDiagnosticContext + ".log";
117     }
