16  package org.kuali.kfs.sys.service.impl;

18  import java.util.Map;
19  import java.util.Properties;
20  
21  import org.apache.struts.action.ActionForward;
22  import org.apache.struts.action.ActionMapping;
23  import org.kuali.kfs.sys.businessobject.ElectronicPaymentClaim;
24  import org.kuali.kfs.sys.service.ElectronicFundTransferActionHelper;
25  import org.kuali.kfs.sys.service.ElectronicPaymentClaimingService;
26  import org.kuali.kfs.sys.web.struts.ElectronicFundTransferForm;
27  import org.kuali.rice.kns.util.GlobalVariables;
28  import org.kuali.rice.kns.util.KNSConstants;
29  import org.kuali.rice.kns.util.UrlFactory;
30  
31  /**
32   * An action for Electronic Fund Transfer that simply redirects to either the claiming or non-claiming lookup.
33   */
34  public class ElectronicFundTransferStartActionHelper implements ElectronicFundTransferActionHelper {
35      private ElectronicPaymentClaimingService electronicPaymentClaimingService;
36  
37      /**
38       * @see org.kuali.kfs.sys.service.ElectronicFundTransferActionHelper#performAction(org.kuali.kfs.sys.web.struts.ElectronicFundTransferForm, org.apache.struts.action.ActionMapping, java.util.Map)
39       */
40      public ActionForward performAction(ElectronicFundTransferForm form, ActionMapping mapping, Map parameterMap, String basePath) {
41          return new ActionForward((form.hasAvailableClaimingDocumentStrategies() ? getClaimingLookupUrl(form, basePath) : getNonClaimingLookupUrl(form, basePath) ), true);
42      }
43      
44      /**
45       * @return URL for non-claiming EFT search
46       */
47      protected String getNonClaimingLookupUrl(ElectronicFundTransferForm form, String basePath) {        
48          Properties props = getCommonLookupProperties(form);
49          props.put(KNSConstants.HIDE_LOOKUP_RETURN_LINK, Boolean.toString(true));
50          props.put(KNSConstants.RETURN_LOCATION_PARAMETER, basePath + "/" + getNonClaimingReturnLocation());
51          props.put(KNSConstants.BACK_LOCATION, basePath + "/" + getNonClaimingReturnLocation());
52          return UrlFactory.parameterizeUrl(basePath + "/kr/" + KNSConstants.LOOKUP_ACTION, props);
53      }
54      
55      /**
56       * @return URL for claiming EFT search
57       */
58      protected String getClaimingLookupUrl(ElectronicFundTransferForm form, String basePath) {
59          Properties props = getCommonLookupProperties(form);
60          props.put(KNSConstants.MULTIPLE_VALUE, Boolean.toString(true));
61          props.put(KNSConstants.LOOKUP_ANCHOR, KNSConstants.ANCHOR_TOP_OF_FORM);
62          props.put(KNSConstants.LOOKED_UP_COLLECTION_NAME, "claims");
63          props.put(KNSConstants.RETURN_LOCATION_PARAMETER, basePath + "/" + getClaimingReturnLocation());
64          props.put(KNSConstants.BACK_LOCATION, basePath + "/" + getClaimingReturnLocation());
65          return UrlFactory.parameterizeUrl(basePath + "/kr/" + KNSConstants.MULTIPLE_VALUE_LOOKUP_ACTION, props);
66      }
67      
68      /**
69       * @return a set of Properties common to both claiming and non-claiming lookup
70       */
71      protected Properties getCommonLookupProperties(ElectronicFundTransferForm form) {
72          Properties props = new Properties();
73          props.put(KNSConstants.SUPPRESS_ACTIONS, Boolean.toString(true));
74          props.put(KNSConstants.DOC_FORM_KEY, "88888888");
75          props.put(KNSConstants.BUSINESS_OBJECT_CLASS_ATTRIBUTE, ElectronicPaymentClaim.class.getName());
76          props.put(KNSConstants.METHOD_TO_CALL_ATTRIBUTE, "start");
77          return props;
78      }
79      
80      /**
81       * @return the location where the search should return to for claiming
82       */
83      protected String getClaimingReturnLocation() {
84          return "electronicFundTransfer.do";
85      }
86      
87      /**
88       * @return the location where the search should return to for non-claiming - ie, the portal!
89       */
90      protected String getNonClaimingReturnLocation() {
91          return "portal.do";
92      }
93      
94      /**
95       * Sets the electronicPaymentClaimingService attribute value.
96       * @param electronicPaymentClaimingService The electronicPaymentClaimingService to set.
97       */
98      public void setElectronicPaymentClaimingService(ElectronicPaymentClaimingService electronicPaymentClaimingService) {
99          this.electronicPaymentClaimingService = electronicPaymentClaimingService;
100     }
101     
102 }
