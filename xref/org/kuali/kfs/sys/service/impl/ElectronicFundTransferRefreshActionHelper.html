36  /**
37   * Represents a web action that occurs when a user returns a bunch of selected claims and redirects to the electronic funds transfer "claimng" page
38   */
39  public class ElectronicFundTransferRefreshActionHelper implements ElectronicFundTransferActionHelper {
40      private ElectronicPaymentClaimingService electronicPaymentClaimingService;
41      private DataDictionaryService ddService;
42      
43      protected static final String BASIC_FORWARD = "basic";
44      protected static final String ACTION_NAME = "claim";
45      protected static final String PORTAL_FORWARD = "portal";
46  
47      /**
48       * @see org.kuali.kfs.sys.service.ElectronicFundTransferActionHelper#performAction(org.kuali.rice.kns.web.struts.form.KualiForm, org.apache.struts.action.ActionMapping)
49       */
50      public ActionForward performAction(ElectronicFundTransferForm form, ActionMapping mapping, Map params, String basePath) {
51          // is the current user able to claim electronic funds?
52          Person currentUser = GlobalVariables.getUserSession().getPerson();
53          if (!form.hasAvailableClaimingDocumentStrategies()) {
54              throw new AuthorizationException(currentUser.getPrincipalName(), ElectronicFundTransferRefreshActionHelper.ACTION_NAME, ddService.getDataDictionary().getBusinessObjectEntry(ElectronicPaymentClaim.class.getName()).getObjectLabel());
55          }
56          // does the current user have claimed funds waiting for them?
57          String lookupResultsSequenceNumber = null;
58          if (params.get("lookupResultsSequenceNumber") != null) {
59              lookupResultsSequenceNumber = ((String[])params.get("lookupResultsSequenceNumber"))[0];
60          }
61          if (StringUtils.isBlank(lookupResultsSequenceNumber)) {
62              return mapping.findForward(PORTAL_FORWARD);
63          }
64          List<ElectronicPaymentClaim> claims = getClaimedPayments(currentUser, lookupResultsSequenceNumber);
65          if (claims.size() == 0) {
66              return mapping.findForward(PORTAL_FORWARD);
67          }
68          // if so, load their currently claimed electronic funds to the form
69          form.setClaims(claims);
70          // and redirect
71          return mapping.findForward(BASIC_FORWARD);
72      }
73      
74      /**
75       * Gets the selected electronic payment claim records from the LookupResults service
76       * @param currentUser the claiming user
77       * @param lookupResultsSequenceNumber the parameter for the lookup results sequence number
78       * @return a list of claims
79       */
80      protected List<ElectronicPaymentClaim> getClaimedPayments(Person currentUser, String lookupResultsSequenceNumber) {
81          List<ElectronicPaymentClaim> claims = new ArrayList<ElectronicPaymentClaim>();
82          try {
83              Collection selectedClaims = KNSServiceLocator.getLookupResultsService().retrieveSelectedResultBOs(lookupResultsSequenceNumber, ElectronicPaymentClaim.class, currentUser.getPrincipalId());
84              for (Object claimAsObj : selectedClaims) {
85                  ElectronicPaymentClaim claim = (ElectronicPaymentClaim) claimAsObj;
86                  if (!claim.getPaymentClaimStatusCode().equals(ElectronicPaymentClaim.ClaimStatusCodes.CLAIMED) && StringUtils.isBlank(claim.getReferenceFinancialDocumentNumber())) {
87                      claims.add(claim);
88                  }
89              }
90          }
91          catch (Exception e) {
92              throw new RuntimeException(e);
93          }        
94          return claims;
95      }
