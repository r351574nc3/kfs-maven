36  public class ReportAggregatorServiceTextImpl implements ReportAggregatorService {
37      protected String newLineCharacter;
38  
39      public void aggregateReports(File outputFile, List<File> files) {
40          PrintWriter aggregateReportWriter = null;
41          int pageNumber = 1;
42          try {
43              aggregateReportWriter = new PrintWriter(new FileWriter(outputFile));
44          }
45          catch (IOException e) {
46              throw new RuntimeException("Error opening output file.", e);
47          }
48          for (int i = 0; i < files.size(); i++) {
49              File file = files.get(i);
50              if (i != 0) {
51                  prepareForNewFile(aggregateReportWriter);
52              }
53              pageNumber = dumpFileContents(aggregateReportWriter, file, pageNumber);
54              file.delete();
55          }
56          aggregateReportWriter.close();
57      }
58  
59      protected int dumpFileContents(Writer outputWriter, File file, int currentPageNumber) {
60          try {
61              BufferedReader reader = new BufferedReader(new FileReader(file));
62              while (reader.ready()) {
63                  String line = reader.readLine();
64                  while (line.contains(KFSConstants.REPORT_WRITER_SERVICE_PAGE_NUMBER_PLACEHOLDER)) {
65                      line = StringUtils.replaceOnce(line, KFSConstants.REPORT_WRITER_SERVICE_PAGE_NUMBER_PLACEHOLDER, String.valueOf(currentPageNumber));
66                      currentPageNumber++;
67                  }
68                  outputWriter.write(line);
69                  outputWriter.write(newLineCharacter);
70              }
71              reader.close();
72              return currentPageNumber;
73          }
74          catch (IOException e) {
75              throw new RuntimeException("Error reading or writing file", e);
76          }
77      }
78      
79      protected void prepareForNewFile(PrintWriter outputWriter) {
80          // 12 represents the ASCII Form Feed character
81          outputWriter.printf("%c" + newLineCharacter, 12);
82      }

86       * @param newLineCharacter The newLineCharacter to set.
87       */
88      public void setNewLineCharacter(String newLineCharacter) {
89          this.newLineCharacter = newLineCharacter;
90      }
