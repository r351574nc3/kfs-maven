16  package org.kuali.kfs.sys;
17  
18  import java.io.File;
19  import java.io.FilenameFilter;
20  
21  /**
22   * This class provides a set of facilities that can be used to work with files
23   */
24  public class FileUtil {
25      private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(FileUtil.class);
26  
27      /**
28       * In directory looks for a pattern matching filenameFilter and returns the filename with the highest lastModified()
29       * @param directory
30       * @param filenameFilter to filter filenames in batchFileDirectoryName for
31       * @return File with highest lastModified()
32       */
33      public static File getNewestFile(File directory, FilenameFilter filenameFilter) {
34          File newestFile = null;
35          
36          File[] directoryListing = directory.listFiles(filenameFilter);
37          if (directoryListing == null || directoryListing.length == 0) {
38              return null;
39          } else {
40              for (int i = 0; i < directoryListing.length; i++) {
41                  File file = directoryListing[i];
42                  if (newestFile == null) {
43                      newestFile = file;
44                  } else {
45                      if (newestFile.lastModified() < file.lastModified()){
46                          newestFile = file;                        
47                      }
48                  }
49              }
50          }
51          
52          return newestFile;
53      }
54  }
