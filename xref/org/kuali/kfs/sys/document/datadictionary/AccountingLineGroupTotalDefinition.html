67      /**

70       */
71      @Override
72      public Renderer getTotalRenderer() {
73          GroupTotalRenderer renderer = new GroupTotalRenderer();
74          
75          renderer.setTotalLabelProperty(totalLabelProperty);
76          renderer.setRepresentedCellPropertyName(representedProperty);
77          
78          final String actualTotalProperty = this.getActualPropertyName(containingPropertyName, totalProperty);
79          renderer.setTotalProperty(actualTotalProperty);
80          
81          return renderer;
82      }
