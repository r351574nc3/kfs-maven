24  import org.kuali.kfs.sys.document.web.AccountingLineViewLines;

26  import org.kuali.rice.kns.datadictionary.DataDictionaryDefinitionBase;

29  /**
30   * Data dictionary definition for a group of multiple lines to render.  This also renders blocks - though each block will be rendered as a line with an embedded table
31   */
32  public class AccountingLineViewLinesDefinition extends DataDictionaryDefinitionBase implements AccountingLineViewRenderableElementDefinition {
33      private List<AccountingLineViewLineFillingDefinition> lines;
34      private String elementName;
35  
36      /**
37       * Validates that:
38       * 1) there is at least one child line
39       * @see org.kuali.rice.kns.datadictionary.DataDictionaryDefinition#completeValidation(java.lang.Class, java.lang.Class)
40       */
41      public void completeValidation(Class rootBusinessObjectClass, Class otherBusinessObjectClass) {
42          if (lines == null || lines.size() == 0) {
43              throw new AttributeValidationException("Please specify at least one child line for the lines definition");
44          }
45      }
46  
47      /**
48       * Gets the lines attribute. 
49       * @return Returns the lines.
50       */
51      public List<AccountingLineViewLineFillingDefinition> getLines() {
52          return lines;
53      }
54  
55      /**
56       * Sets the lines attribute value.
57       * @param lines The lines to set.
58       */
59      public void setLines(List<AccountingLineViewLineFillingDefinition> lines) {
60          this.lines = lines;
61      }
62  
63      /**
64       * Gets the elementName attribute. 
65       * @return Returns the elementName.
66       */
67      public String getElementName() {
68          return elementName;
69      }
70  
71      /**
72       * Sets the elementName attribute value.
73       * @param elementName The elementName to set.
74       */
75      public void setElementName(String elementName) {
76          this.elementName = elementName;
77      }
78  
79      /**
80       * @see org.kuali.kfs.sys.document.datadictionary.AccountingLineViewRenderableElementDefinition#createLayoutElement()
81       */
82      public TableJoining createLayoutElement(Class<? extends AccountingLine> accountingLineClass) {
83          AccountingLineViewLines layoutElement = new AccountingLineViewLines();
84          layoutElement.setDefinition(this);
85          layoutElement.setElements(getLayoutElementsForLines(accountingLineClass));
86          return layoutElement;
87      }
88      
89      /**
90       * Generates layout elements for all the child lines of this lines definition
91       * @return a List with the line elements for all child lines of this element definition 
92       */
93      protected List<AccountingLineViewLineFillingElement> getLayoutElementsForLines(Class<? extends AccountingLine> accountingLineClass) {
94          List<AccountingLineViewLineFillingElement> elements = new ArrayList<AccountingLineViewLineFillingElement>();
95          for (AccountingLineViewLineFillingDefinition line : lines) {
96              elements.add(line.createLineFillingLayoutElement(accountingLineClass));
97          }
98          return elements;
99      }
