16  package org.kuali.kfs.sys.document.web.renderers;
17  
18  import java.io.IOException;
19  
20  import javax.servlet.jsp.JspException;
21  import javax.servlet.jsp.JspWriter;
22  import javax.servlet.jsp.PageContext;
23  import javax.servlet.jsp.tagext.Tag;
24  
25  import org.apache.commons.lang.StringUtils;
26  import org.apache.struts.taglib.html.HiddenTag;
27  import org.springframework.web.util.HtmlUtils;
28  
29  /**
30   * Render which displays a field that can't be input directly but could be changed when other fields change.
31   * An example of such is the chartOfAccountsCode in accounting lines, when accounts can't cross charts and
32   * chart code is set automatically by account number.   
33   */
34  public class DynamicReadOnlyRender extends ReadOnlyRenderer {
35      // A hidden input field to store a copy of the field value so that the field will appear read-only to users
36      // but could be set by the system and read into the data object.
37      private HiddenTag shadowInputTag = new HiddenTag();
38      
39      /**
40       * @see org.kuali.kfs.sys.document.web.renderers.Renderer#render(javax.servlet.jsp.PageContext, javax.servlet.jsp.tagext.Tag)
41       */
42      public void render(PageContext pageContext, Tag parentTag) throws JspException {
43          JspWriter out = pageContext.getOut();
44  
45          try {
46              String value = discoverRenderValue();            
47              out.write(buildBeginSpan());
48              
49              if (!StringUtils.isEmpty(value)) {
50                  if (shouldRenderInquiryLink()) {
51                      out.write(buildBeginInquiryLink());
52                  }                
53                  out.write(value);                
54                  if (shouldRenderInquiryLink()) {
55                      out.write(buildEndInquiryLink());
56                  }                                
57              } else {
58                  out.write(buildNonBreakingSpace());
59              }
60              
61              out.write(buildEndSpan());
62              renderShadowInputTag(pageContext, parentTag);
63          }
64          catch (IOException ioe) {
65              throw new JspException("Difficulty rendering read only field", ioe);
66          }
67      }
68  
69      /**
70       * Generates the HTML for the opening span tag to wrap the displayed value
71       * @param propertyPrefix the property path from the form the business object being rendered
72       * @return the HTML for the opening span 
73       */
74      protected String buildBeginSpan() {
75          StringBuilder beginSpan = new StringBuilder();        
76          beginSpan.append("<span id=\"");
77          beginSpan.append(getFieldName());
78          beginSpan.append(".div\">");        
79          return beginSpan.toString();
80      }    
81  
82      /**
83       * Renders the value of the field in the hidden input tag so it can be read into the data object.
84       * @param pageContext the page context to render to
85       * @param parentTag the tag requesting all this rendering
86       */
87      protected void renderShadowInputTag(PageContext pageContext, Tag parentTag) throws JspException {
88          shadowInputTag.setPageContext(pageContext);
89          shadowInputTag.setParent(parentTag);
90          shadowInputTag.setProperty(getFieldName());
91          shadowInputTag.setValue(getField().getPropertyValue());        
92          shadowInputTag.doStartTag();
93          shadowInputTag.doEndTag();
94      }
