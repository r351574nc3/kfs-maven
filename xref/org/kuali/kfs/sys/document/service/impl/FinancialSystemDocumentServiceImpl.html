16  package org.kuali.kfs.sys.document.service.impl;

18  import java.util.ArrayList;
19  import java.util.Collection;
20  import java.util.Iterator;
21  
22  import org.kuali.kfs.sys.document.dataaccess.FinancialSystemDocumentDao;
23  import org.kuali.kfs.sys.document.service.FinancialSystemDocumentService;
24  import org.kuali.rice.kew.exception.WorkflowException;
25  import org.kuali.rice.kns.document.Document;
26  import org.kuali.rice.kns.service.DocumentService;
27  import org.springframework.transaction.annotation.Transactional;

29  /**
30   * This class is a Financial System specific Document Service class to allow for the {@link #findByDocumentHeaderStatusCode(Class, String)} method.
31   */
32  @Transactional
33  public class FinancialSystemDocumentServiceImpl implements FinancialSystemDocumentService {
34      private FinancialSystemDocumentDao financialSystemDocumentDao;
35      private DocumentService documentService;
36      
37      /**
38       * @see org.kuali.kfs.sys.document.service.FinancialSystemDocumentService#findByDocumentHeaderStatusCode(java.lang.Class, java.lang.String)
39       */
40      public <T extends Document> Collection<T> findByDocumentHeaderStatusCode(Class<T> clazz, String statusCode) throws WorkflowException {
41          Collection<T> foundDocuments = getFinancialSystemDocumentDao().findByDocumentHeaderStatusCode(clazz, statusCode);
42          Collection<T> returnDocuments = new ArrayList<T>();
43          for (Iterator<T> iter = foundDocuments.iterator(); iter.hasNext();) {
44              Document doc = (Document) iter.next();
45              returnDocuments.add((T)getDocumentService().getByDocumentHeaderId(doc.getDocumentNumber()));
46          }
47          return returnDocuments;
48      }
49  
50      public FinancialSystemDocumentDao getFinancialSystemDocumentDao() {
51          return financialSystemDocumentDao;
52      }
53  
54      public void setFinancialSystemDocumentDao(FinancialSystemDocumentDao financialSystemDocumentDao) {
55          this.financialSystemDocumentDao = financialSystemDocumentDao;
56      }
57  
58      public DocumentService getDocumentService() {
59          return documentService;
60      }
61  
62      public void setDocumentService(DocumentService documentService) {
63          this.documentService = documentService;
64      }

67  }
