20  /**
21   * Runs the fiscal year maker process which creates rows for the next fiscal year based on records for the current fiscal year for
22   * configured tables. The base year for the process is given by the system parameter
23   * <code>KFS-COA FiscalYearMakerStep SOURCE_FISCAL_YEAR</code>. In addition, the process has an option for how it should handle
24   * records that already exist in the target year. If the system parameter
25   * <code>KFS-COA FiscalYearMakerStep OVERRIDE_TARGET_YEAR_DATA_IND</code> is set to 'Y', target year records will be overwritten;
26   * Otherwise, target year records that already exist will be left alone. Each table that is setup for the fiscal year maker process
27   * has a bean configured in Spring. This bean points to an implementation of <code>FiscalYearMaker</code> (either the default
28   * implementation or custom) that sets up the <code>Criteria</code> for selecting records to copy, and performs the process of
29   * changing the records for the new year. New implementations can be created and configured in Spring to customize the behavior for
30   * a table.
31   */
32  public interface FiscalYearMakerService {
33  
34      /**
35       * Runs the fiscal year maker process for the configured base year
36       */
37      public void runProcess();
38  }
