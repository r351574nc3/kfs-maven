35      /**
36       * Overriding the return URL to create a new Tax Region document.  If CREATE_TAX_REGION_FROM_LOOKUP_INDICATOR
37       * exists, that means that the lookup is coming from the "create new" button on the tax region lookup.
38       * If so, create customer link to tax region maint doc. Otherwise, just use the normal returnUrl.
39       * 
40       * @see org.kuali.rice.kns.lookup.AbstractLookupableHelperServiceImpl#getReturnUrl(org.kuali.rice.kns.bo.BusinessObject,
41       *      java.util.Map, java.lang.String, java.util.List)
42       */
43      @SuppressWarnings("rawtypes")
44      // KFSMI-5158
45      @Override
46      public HtmlData getReturnUrl(BusinessObject businessObject, LookupForm lookupForm, List returnKeys, BusinessObjectRestrictions restrictions) {
47  
48          final String docTypeName = getTaxRegionDocumentTypeName();
49          if (docTypeName.equals(lookupForm.getFormKey())) { 
50              Properties parameters = getParameters(businessObject, lookupForm.getFieldConversions(), lookupForm.getLookupableImplServiceName(), returnKeys);
51              parameters.put(KFSConstants.DISPATCH_REQUEST_PARAMETER, KFSConstants.MAINTENANCE_NEWWITHEXISTING_ACTION);
52              parameters.put(KFSConstants.BUSINESS_OBJECT_CLASS_ATTRIBUTE, TaxRegion.class.getName());
53              parameters.put(KFSConstants.OVERRIDE_KEYS, KFSConstants.TaxRegionConstants.TAX_REGION_TYPE_CODE);
54              parameters.put(KFSConstants.REFRESH_CALLER, KFSConstants.TaxRegionConstants.TAX_REGION_TYPE_CODE + "::" + ((TaxRegionType) businessObject).getTaxRegionTypeCode());
55              
56              final String href = UrlFactory.parameterizeUrl(KFSConstants.MAINTENANCE_ACTION, parameters);
57              final String returnUrlAnchorLabel =
58                  KNSServiceLocator.getKualiConfigurationService().getPropertyString(TITLE_RETURN_URL_PREPENDTEXT_PROPERTY);
59              AnchorHtmlData anchor = new AnchorHtmlData(href, HtmlData.getTitleText(returnUrlAnchorLabel, businessObject, returnKeys, restrictions));
60              anchor.setDisplayText(returnUrlAnchorLabel);
61             return anchor;
62         }
63         else {
64             return super.getReturnUrl(businessObject, lookupForm, returnKeys, restrictions);
65         }
66      }

68      @SuppressWarnings("rawtypes")
69      @Override
70      public List<HtmlData> getCustomActionUrls(BusinessObject businessObject, List pkNames) {
71          List<HtmlData> htmlDataList = super.getCustomActionUrls(businessObject, pkNames);
72          
73          Properties parameters = new Properties();
74          parameters.put(KFSConstants.DISPATCH_REQUEST_PARAMETER, KFSConstants.MAINTENANCE_NEWWITHEXISTING_ACTION);
75          parameters.put(KFSConstants.BUSINESS_OBJECT_CLASS_ATTRIBUTE, TaxRegion.class.getName());
76          parameters.put(KFSConstants.OVERRIDE_KEYS, KFSConstants.TaxRegionConstants.TAX_REGION_TYPE_CODE);
77          parameters.put(KFSConstants.REFRESH_CALLER, KFSConstants.TaxRegionConstants.TAX_REGION_TYPE_CODE + "::" + ((TaxRegionType) businessObject).getTaxRegionTypeCode());
78          parameters.put(KFSConstants.TaxRegionConstants.TAX_REGION_TYPE_CODE, ((TaxRegionType) businessObject).getTaxRegionTypeCode());
79          String href = UrlFactory.parameterizeUrl(KFSConstants.MAINTENANCE_ACTION, parameters);
80          
81          AnchorHtmlData anchorHtmlData = new AnchorHtmlData(href, "start", "create tax region");
82          htmlDataList.add(anchorHtmlData);
83          
84          return htmlDataList;
85      }
86      
87      private String getTaxRegionDocumentTypeName() {
88          return getDataDictionaryService().getDataDictionary().getMaintenanceDocumentEntryForBusinessObjectClass(TaxRegion.class).getDocumentTypeName();
89      }
