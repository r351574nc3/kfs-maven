40      public Iterator getCanceledChecks() {
41          LOG.debug("getCanceledChecks() started");
42  
43          Collection codes = new ArrayList();
44          codes.add(PdpConstants.PaymentChangeCodes.CANCEL_DISBURSEMENT);
45          codes.add(PdpConstants.PaymentChangeCodes.CANCEL_REISSUE_DISBURSEMENT);
46  
47          Criteria crit = new Criteria();
48          crit.addIn(PdpPropertyConstants.PAYMENT_CHANGE_CODE, codes);
49          crit.addIsNull(PdpPropertyConstants.PaymentGroupHistory.PMT_CANCEL_EXTRACT_DATE);
50  
51          Criteria o1 = new Criteria();
52          o1.addNotEqualTo(PdpPropertyConstants.DISBURSEMENT_TYPE_CODE, PdpConstants.DisbursementTypeCodes.ACH);
53          Criteria o1a = new Criteria();
54          o1a.addIsNull(PdpPropertyConstants.DISBURSEMENT_TYPE_CODE);
55          o1.addOrCriteria(o1a);
56  
57          Criteria o2 = new Criteria();
58          o2.addEqualTo(PdpPropertyConstants.DISBURSEMENT_TYPE_CODE, PdpConstants.DisbursementTypeCodes.CHECK);
59          Criteria o2a = new Criteria();
60          o2a.addEqualTo(PdpPropertyConstants.PAYMENT_GROUP + "." + PdpPropertyConstants.DISBURSEMENT_TYPE_CODE, PdpConstants.DisbursementTypeCodes.CHECK);
61          o2.addOrCriteria(o2a);
62  
63          crit.addAndCriteria(o1);
64          crit.addAndCriteria(o2);
65  
66          return getPersistenceBrokerTemplate().getIteratorByQuery(new QueryByCriteria(PaymentGroupHistory.class, crit));
67      }
