51  /**
52   * This class is a special implementation of Maintainable specifically for Account Delegates. It was created to correctly update the
53   * default Start Date on edits and copies, ala JIRA #KULRNE-62.
54   */
55  public class CustomerProfileMaintenanceDocumentMaintainableImpl extends FinancialSystemMaintainable {
56      private static final org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(CustomerProfileMaintenanceDocumentMaintainableImpl.class);
57  
58      /**
59       * This method will reset AccountDelegate's Start Date to the current timestamp on edits and copies
60       * 
61       * @see org.kuali.rice.kns.maintenance.KualiMaintainableImpl#processAfterRetrieve()
62       */
63      @Override
64      public void processAfterCopy( MaintenanceDocument document, Map<String,String[]> parameters ) {
65          
66          super.processAfterCopy( document, parameters );
67          CustomerProfile customerProfile = (CustomerProfile) document.getNewMaintainableObject().getBusinessObject();
68          customerProfile.setChartCode(null);
69          customerProfile.setUnitCode(null); 
70          customerProfile.setSubUnitCode(null);
71      }
72      
73      
74      public List getSections(MaintenanceDocument document, Maintainable oldMaintainable) {
75          List<Section> sections = super.getSections(document, oldMaintainable);
76          //If oldMaintainable is null, it means we are trying to get sections for the old part
77          //If oldMaintainable is not null, it means we are trying to get sections for the new part
78          //Refer to KualiMaintenanceForm lines 288-294
79          CustomerProfile customerProfile = (CustomerProfile) document.getNewMaintainableObject().getBusinessObject();
80          if(oldMaintainable==null) return sections;
81          if (shouldReviewTypesFieldBeReadOnly(document) == false) return sections;
82  
83          for (Section section : sections) {
84              for (Row row : section.getRows()) {
85                  for (Field field : row.getFields()) {
86                      if (PdpPropertyConstants.CustomerProfile.CUSTOMER_PROFILE_CHART_CODE.equals(field.getPropertyName())) {
87                          field.setReadOnly(true);
88                       }
89                      if (PdpPropertyConstants.CustomerProfile.CUSTOMER_PROFILE_UNIT_CODE.equals(field.getPropertyName())) {
90                           field.setReadOnly(true);
91                      }
92                      if (PdpPropertyConstants.CustomerProfile.CUSTOMER_PROFILE_SUB_UNIT_CODE.equals(field.getPropertyName())) {
93                          field.setReadOnly(true);
94                      }
95                  }
96              }
97          }
98          return sections;
99      }
100

102     protected boolean shouldReviewTypesFieldBeReadOnly(MaintenanceDocument document){
103         CustomerProfile  customerProfile = (CustomerProfile)document.getNewMaintainableObject().getBusinessObject();
104         if(StringUtils.isEmpty(customerProfile.getChartCode())) return false;
105         if(StringUtils.isEmpty(customerProfile.getSubUnitCode())) return false;
106         if(StringUtils.isEmpty(customerProfile.getUnitCode())) return false;
107         return true;
108     }
