31  /**
32   * Defines methods for sending PDP emails.
33   */
34  public interface PdpEmailService {
35  
36      /**
37       * Sends email for a payment load has failed. Errors encountered will be printed out in message
38       * 
39       * @param paymentFile parsed payment file object (might not be populated completely due to errors)
40       * @param errors <code>ErrorMap</code> containing <code>ErrorMessage</code> entries
41       */
42      public void sendErrorEmail(PaymentFileLoad paymentFile, MessageMap errors);
43  
44      /**
45       * Sends email for a successful payment load. Warnings encountered will be printed out in message
46       * 
47       * @param paymentFile parsed payment file object
48       * @param warnings <code>List</code> of <code>String</code> messages
49       */
50      public void sendLoadEmail(PaymentFileLoad paymentFile, List<String> warnings);
51  
52      /**
53       * Sends email for a payment load that was held due to tax reasons
54       * 
55       * @param paymentFile parsed payment file object
56       */
57      public void sendTaxEmail(PaymentFileLoad paymentFile);
58  
59      /**
60       * Sends email for a load done internally
61       * 
62       * @param batch <code>Batch</code> created by load
63       */
64      public void sendLoadEmail(Batch batch);
65  
66      /**
67       * Sends email for a purap bundle that exceeds the maximum number of notes allowed
68       * 
69       * @param creditMemos list of credit memo documents in bundle
70       * @param paymentRequests list of payment request documents in bundle
71       * @param lineTotal total number of lines for bundle
72       * @param maxNoteLines maximum number of lines allowed
73       */
74      public void sendExceedsMaxNotesWarningEmail(List<String> creditMemos, List<String> paymentRequests, int lineTotal, int maxNoteLines);
75  
76      /**
77       * Sends summary email for an ACH extract
78       * 
79       * @param unitCounts Map containing payment counts for each unit
80       * @param unitTotals Map containing total payment amount for each unit
81       * @param extractDate date of ACH extraction
82       */
83      public void sendAchSummaryEmail(Map<String, Integer> unitCounts, Map<String, KualiDecimal> unitTotals, Date extractDate);
84  
85      /**
86       * Sends advice notification email to the payee receiving an ACH payment
87       * 
88       * @param paymentGroup ACH payment group to send notification for
89       * @param paymentDetail Payment Detail containing payment amounts
90       * @param customer Pdp Customer profile for payment
91       */
92      public void sendAchAdviceEmail(PaymentGroup paymentGroup, PaymentDetail paymentDetail, CustomerProfile customer);
93      
94      /**
95       * Sends Payment Cancellation Email
96       * 
97       * @param paymentGroup
98       * @param note
99       * @param user
100      */
101     public void sendCancelEmail(PaymentGroup paymentGroup, String note, Person user);
102     
103     /**
104      * Reads system parameter indicating whether to status emails should be sent
105      * 
106      * @return true if email should be sent, false otherwise
107      */
108     public boolean isPaymentEmailEnabled() ;
109 }
