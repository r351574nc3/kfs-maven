38  /**
39   * Checks for security access exception and forwards to security access error page
40   */
41  public class SecurityExceptionIncidentHandler extends StrutsExceptionIncidentHandler {
42  
43      /**
44       * @see org.kuali.rice.kns.web.struts.pojo.StrutsExceptionIncidentHandler#execute(java.lang.Exception, org.apache.struts.config.ExceptionConfig,
45       *      org.apache.struts.action.ActionMapping, org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
46       */
47      @Override
48      public ActionForward execute(Exception exception, ExceptionConfig exceptionConfig, ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) {
49          AccessSecurityRestrictionInfo restrictionInfo = (AccessSecurityRestrictionInfo) GlobalVariables.getUserSession().retrieveObject(SecConstants.OPEN_DOCUMENT_SECURITY_ACCESS_DENIED_ERROR_KEY);
50          if (restrictionInfo != null) {
51              String accessMessage = SpringContext.getBean(KualiConfigurationService.class).getPropertyString(SecKeyConstants.MESSAGE_OPEN_DOCUMENT_RESTRICTED);
52              accessMessage = StringUtils.replace(accessMessage, "{0}", GlobalVariables.getUserSession().getPrincipalName());
53              accessMessage = StringUtils.replace(accessMessage, "{1}", restrictionInfo.getDocumentNumber());
54              accessMessage = StringUtils.replace(accessMessage, "{2}", restrictionInfo.getPropertyLabel());
55              accessMessage = StringUtils.replace(accessMessage, "{3}", restrictionInfo.getRetrictedValue());
56              request.setAttribute(SecConstants.ACCESS_ERROR_STRING_REQUEST_KEY, accessMessage);
57  
58              GlobalVariables.getUserSession().removeObject(SecConstants.OPEN_DOCUMENT_SECURITY_ACCESS_DENIED_ERROR_KEY);
59  
60              return mapping.findForward(SecConstants.ACCESS_DENIED_ERROR_FORWARD);
61          }
62  
63          return super.execute(exception, exceptionConfig, mapping, form, request, response);
64      }
