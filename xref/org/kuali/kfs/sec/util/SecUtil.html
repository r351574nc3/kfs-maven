35      /**
36       * Compares the size of the given list against the given previous size and if different adds an info message
37       * 
38       * @param previousListSize int giving previous size of list to compare to
39       * @param results List to get size for and compare
40       * @param messageKey String key of message that should be added
41       */
42      @SuppressWarnings("rawtypes")
43      public static void compareListSizeAndAddMessageIfChanged(int previousListSize, List results, String messageKey) {
44          int currentListSize = results.size();
45  
46          if (previousListSize != currentListSize) {
47              GlobalVariables.getMessageMap().putInfo(KFSConstants.GLOBAL_MESSAGES, messageKey, (String) null);
48          }
49      }
50  
51      /**
52       * Calls access security service to check view access on given GLPE for current user. Access to view the GLPE on the document should be related to the view permissions for an
53       * accounting line with the same account attributes. Called from generalLedgerPendingEntries.tag
54       * 
55       * @param pendingEntry GeneralLedgerPendingEntry to check access for
56       * @return boolean true if current user has view permission, false otherwise
57       */
58      public static boolean canViewGLPE(Document document, GeneralLedgerPendingEntry pendingEntry) {
59          boolean canView = true;
60  
61          // If the module has not been loaded, then just skip any further checks as the services will not be defined
62          if ( SpringContext.getBean(KualiConfigurationService.class).getPropertyAsBoolean(SecConstants.ACCESS_SECURITY_MODULE_ENABLED_PROPERTY_NAME) ) {   
63              if (document instanceof AccountingDocument) {
64                  AccountingLine line = new SourceAccountingLine();
65      
66                  line.setPostingYear(pendingEntry.getUniversityFiscalYear());
67                  line.setChartOfAccountsCode(pendingEntry.getChartOfAccountsCode());
68                  line.setAccountNumber(pendingEntry.getAccountNumber());
69                  line.setSubAccountNumber(pendingEntry.getSubAccountNumber());
70                  line.setFinancialObjectCode(pendingEntry.getFinancialObjectCode());
71                  line.setFinancialSubObjectCode(pendingEntry.getFinancialSubObjectCode());
72                  line.setProjectCode(pendingEntry.getProjectCode());
73      
74                  line.refreshNonUpdateableReferences();
75      
76                  canView = SpringContext.getBean(AccessSecurityService.class).canViewDocumentAccountingLine((AccountingDocument) document, line, GlobalVariables.getUserSession().getPerson());
77              }
78          }
79  
80          return canView;
81      }
