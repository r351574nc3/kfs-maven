52  /**
53   * Default implementation of the
54   * {@link org.springframework.beans.factory.ListableBeanFactory} and
55   * {@link BeanDefinitionRegistry} interfaces: a full-fledged bean factory
56   * based on bean definition objects.
57   *
58   * <p>Typical usage is registering all bean definitions first (possibly read
59   * from a bean definition file), before accessing beans. Bean definition lookup
60   * is therefore an inexpensive operation in a local bean definition table,
61   * operating on pre-built bean definition metadata objects.
62   *
63   * <p>Can be used as a standalone bean factory, or as a superclass for custom
64   * bean factories. Note that readers for specific bean definition formats are
65   * typically implemented separately rather than as bean factory subclasses:
66   * see for example {@link PropertiesBeanDefinitionReader} and
67   * {@link org.springframework.beans.factory.xml.XmlBeanDefinitionReader}.
68   *
69   * <p>For an alternative implementation of the
70   * {@link org.springframework.beans.factory.ListableBeanFactory} interface,
71   * have a look at {@link StaticListableBeanFactory}, which manages existing
72   * bean instances rather than creating new ones based on bean definitions.
73   *
74   * @author Rod Johnson
75   * @author Juergen Hoeller
76   * @author Sam Brannen
77   * @since 16 April 2001
78   * @see StaticListableBeanFactory
79   * @see PropertiesBeanDefinitionReader
80   * @see org.springframework.beans.factory.xml.XmlBeanDefinitionReader
81   */
82  public class DefaultListableBeanFactory extends AbstractAutowireCapableBeanFactory
83      implements ConfigurableListableBeanFactory, BeanDefinitionRegistry {
84

207     public String[] getBeanNamesForType(Class type, boolean includeNonSingletons, boolean allowEagerInit) {
208         List result = new ArrayList();
209 
210         // Check all bean definitions.
211         String[] beanDefinitionNames = getBeanDefinitionNames();
212         for (int i = 0; i < beanDefinitionNames.length; i++) {
213             String beanName = beanDefinitionNames[i];
214             // Only consider bean as eligible if the bean name
215             // is not defined as alias for some other bean.
216             if (!isAlias(beanName)) {
217                 try {
218                     RootBeanDefinition mbd = getMergedLocalBeanDefinition(beanName);
219                     // Only check bean definition if it is complete.
220                     if (!mbd.isAbstract() &&
221                             (allowEagerInit || ((mbd.hasBeanClass() || !mbd.isLazyInit() || this.allowEagerClassLoading)) &&
222                                     !requiresEagerInitForType(mbd.getFactoryBeanName()))) {
223                         // In case of FactoryBean, match object created by FactoryBean.
224                         boolean isFactoryBean = isFactoryBean(beanName, mbd);
225                         boolean matchFound =
226                                 (allowEagerInit || !isFactoryBean || containsSingleton(beanName)) &&
227                                 (includeNonSingletons || isSingleton(beanName)) && isTypeMatch(beanName, type);
228                         if (!matchFound && isFactoryBean) {
229                             // In case of FactoryBean, try to match FactoryBean instance itself next.
230                             beanName = FACTORY_BEAN_PREFIX + beanName;
231                             matchFound = (includeNonSingletons || mbd.isSingleton()) && isTypeMatch(beanName, type);
232                         }
233                         if (matchFound) {
234                             result.add(beanName);
235                         }
236                     }
237                 }
238                 catch (CannotLoadBeanClassException ex) {
239                     if (allowEagerInit) {
240                         throw ex;
241                     }
242                     // Probably contains a placeholder: let's ignore it for type matching purposes.
243                     if (this.logger.isDebugEnabled()) {
244                         this.logger.debug("Ignoring bean class loading failure for bean '" + beanName + "'", ex);
245                     }
246                     onSuppressedException(ex);
247                 }
248                 catch (BeanDefinitionStoreException ex) {
249                     if (allowEagerInit) {
250                         throw ex;
251                     }
252                     // Probably contains a placeholder: let's ignore it for type matching purposes.
253                     if (this.logger.isDebugEnabled()) {
254                         this.logger.debug("Ignoring unresolvable metadata in bean definition '" + beanName + "'", ex);
255                     }
256                     onSuppressedException(ex);
257                 }
258             }
259         }
260 
261         // Check singletons too, to catch manually registered singletons.
262         String[] singletonNames = getSingletonNames();
263         for (int i = 0; i < singletonNames.length; i++) {
264             String beanName = singletonNames[i];
265             // Only check if manually registered.
266             if (!containsBeanDefinition(beanName)) {
267                 // In case of FactoryBean, match object created by FactoryBean.
268                 if (isFactoryBean(beanName)) {
269                     if ((includeNonSingletons || isSingleton(beanName)) && isTypeMatch(beanName, type)) {
270                         result.add(beanName);
271                         // Match found for this bean: do not match FactoryBean itself anymore.
272                         continue;
273                     }
274                     // In case of FactoryBean, try to match FactoryBean itself next.
275                     beanName = FACTORY_BEAN_PREFIX + beanName;
276                 }
277                 // Match raw bean instance (might be raw FactoryBean).
278                 if (isTypeMatch(beanName, type)) {
279                     result.add(beanName);
280                 }
281             }
282         }
283 
284         return StringUtils.toStringArray(result);
285     }

302     public Map getBeansOfType(Class type, boolean includeNonSingletons, boolean allowEagerInit)
303             throws BeansException {
304 
305         String[] beanNames = getBeanNamesForType(type, includeNonSingletons, allowEagerInit);
306         Map result = new LinkedHashMap(beanNames.length);
307         for (int i = 0; i < beanNames.length; i++) {
308             String beanName = beanNames[i];
309             try {
310                 result.put(beanName, getBean(beanName));
311             }
312             catch (BeanCreationException ex) {
313                 Throwable rootCause = ex.getMostSpecificCause();
314                 if (rootCause instanceof BeanCurrentlyInCreationException) {
315                     BeanCreationException bce = (BeanCreationException) rootCause;
316                     if (isCurrentlyInCreation(bce.getBeanName())) {
317                         if (this.logger.isDebugEnabled()) {
318                             this.logger.debug("Ignoring match to currently created bean '" + beanName + "': " + ex.getMessage());
319                         }
320                         onSuppressedException(ex);
321                         // Ignore: indicates a circular reference when autowiring constructors.
322                         // We want to find matches other than the currently created bean itself.
323                         continue;
324                     }
325                 }
326                 throw ex;
327             }
328         }
329         return result;
330     }

338         Assert.notNull(dependencyType, "Type must not be null");
339         if (autowiredValue != null) {

342             this.resolvableDependencies.put(dependencyType, autowiredValue);
343         }

346     public boolean isAutowireCandidate(String beanName, DependencyDescriptor descriptor)
347             throws NoSuchBeanDefinitionException {
348 
349         // Consider FactoryBeans as autowiring candidates.
350         boolean isFactoryBean = (descriptor != null && descriptor.getDependencyType() != null &&
351                 FactoryBean.class.isAssignableFrom(descriptor.getDependencyType()));
352         if (isFactoryBean) {
353             beanName = BeanFactoryUtils.transformedBeanName(beanName);
354         }
355 
356         if (!containsBeanDefinition(beanName)) {
357             if (containsSingleton(beanName)) {
358                 return true;
359             }
360             else if (getParentBeanFactory() instanceof ConfigurableListableBeanFactory) {
361                 // No bean definition found in this factory -> delegate to parent.
362                 return ((ConfigurableListableBeanFactory) getParentBeanFactory()).isAutowireCandidate(beanName, descriptor);
363             }
364         }
365 
366         return isAutowireCandidate(beanName, getMergedLocalBeanDefinition(beanName), descriptor);
367     }

369     /**
370      * Determine whether the specified bean definition qualifies as an autowire candidate,
371      * to be injected into other beans which declare a dependency of matching type.
372      * @param beanName the name of the bean definition to check
373      * @param mbd the merged bean definition to check
374      * @param descriptor the descriptor of the dependency to resolve
375      * @return whether the bean should be considered as autowire candidate
376      */
377     protected boolean isAutowireCandidate(String beanName, RootBeanDefinition mbd, DependencyDescriptor descriptor) {
378         resolveBeanClass(mbd, beanName);
379         return getAutowireCandidateResolver().isAutowireCandidate(
380                 new BeanDefinitionHolder(mbd, beanName, getAliases(beanName)), descriptor);
381     }

414     public void preInstantiateSingletons() throws BeansException {
415         if (this.logger.isInfoEnabled()) {
416             this.logger.info("Pre-instantiating singletons in " + this);
417         }
418 
419         synchronized (this.beanDefinitionMap) {
420             for (Iterator it = this.beanDefinitionNames.iterator(); it.hasNext();) {
421                 String beanName = (String) it.next();
422                 RootBeanDefinition bd = getMergedLocalBeanDefinition(beanName);
423                 if (!bd.isAbstract() && bd.isSingleton() && !bd.isLazyInit()) {
424                     if (isFactoryBean(beanName)) {
425                         FactoryBean factory = (FactoryBean) getBean(FACTORY_BEAN_PREFIX + beanName);
426                         if (factory instanceof SmartFactoryBean && ((SmartFactoryBean) factory).isEagerInit()) {
427                             getBean(beanName);
428                         }
429                     }
430                     else {
431                         getBean(beanName);
432                     }
433                 }
434             }
435         }
436     }

445         defs.add(beanName);

452     public void registerBeanDefinition(String beanName, BeanDefinition beanDefinition)
453             throws BeanDefinitionStoreException {
454 
455         Assert.hasText(beanName, "'beanName' must not be empty");
456         Assert.notNull(beanDefinition, "BeanDefinition must not be null");
457 
458         if (beanDefinition instanceof AbstractBeanDefinition) {
459             try {
460                 ((AbstractBeanDefinition) beanDefinition).validate();
461             }
462             catch (BeanDefinitionValidationException ex) {
463                 throw new BeanDefinitionStoreException(beanDefinition.getResourceDescription(), beanName,
464                         "Validation of bean definition failed", ex);
465             }
466         }
467 
468         synchronized (this.beanDefinitionMap) {
469             Object oldBeanDefinition = this.beanDefinitionMap.get(beanName);
470             if (oldBeanDefinition != null) {
471                 if (!this.allowBeanDefinitionOverriding) {
472                     throw new BeanDefinitionStoreException(beanDefinition.getResourceDescription(), beanName,
473                             "Cannot register bean definition [" + beanDefinition + "] for bean '" + beanName +
474                             "': There is already [" + oldBeanDefinition + "] bound.");
475                 }
476                 else {
477                     if (this.logger.isInfoEnabled()) {
478                         this.logger.info("Overriding bean definition for bean '" + beanName +
479                                 "': replacing [" + oldBeanDefinition + "] with [" + beanDefinition + "]");
480                     }
481                 }
482             }
483             else {
484                 this.beanDefinitionNames.add(beanName);
485                 this.frozenBeanDefinitionNames = null;
486             }
487             this.beanDefinitionMap.put(beanName, beanDefinition);
488             if ( StringUtils.hasText( beanDefinition.getParentName() ) ) {
489                 addBeanToParentMap(beanDefinition.getParentName(), beanName);
490             }
491 
492             resetBeanDefinition(beanName);
493         }
494     }
495 
496     public void removeBeanDefinition(String beanName) throws NoSuchBeanDefinitionException {
497         Assert.hasText(beanName, "'beanName' must not be empty");
498 
499         synchronized (this.beanDefinitionMap) {
500             BeanDefinition bd = (BeanDefinition) this.beanDefinitionMap.remove(beanName);
501             if (bd == null) {
502                 if (this.logger.isTraceEnabled()) {
503                     this.logger.trace("No bean named '" + beanName + "' found in " + this);
504                 }
505                 throw new NoSuchBeanDefinitionException(beanName);
506             }
507             this.beanDefinitionNames.remove(beanName);
508             this.frozenBeanDefinitionNames = null;
509             // Pull it out of the parent map
510             if ( StringUtils.hasText( bd.getParentName() ) ) {
511                 if ( beanDefinitionsByParent.get(bd.getParentName()) != null ) {
512                     beanDefinitionsByParent.get(bd.getParentName()).remove(beanName);
513                 }
514             }
515 
516             resetBeanDefinition(beanName);
517         }
518     }

520     /**
521      * Reset all bean definition caches for the given bean,
522      * including the caches of beans that are derived from it.
523      * @param beanName the name of the bean to reset
524      */
525     protected void resetBeanDefinition(String beanName) {
526         // Remove the merged bean definition for the given bean, if already created.
527         clearMergedBeanDefinition(beanName);
528 
529         // Remove corresponding bean from singleton cache, if any. Shouldn't usually
530         // be necessary, rather just meant for overriding a context's default beans
531         // (e.g. the default StaticMessageSource in a StaticApplicationContext).
532         synchronized (getSingletonMutex()) {
533             destroySingleton(beanName);
534         }
535 
536         // Reset all bean definitions that have the given bean as parent
537         // (recursively).
538         Set<String> childBeans = beanDefinitionsByParent.get(beanName);
539         if ( childBeans != null ) {
540             for ( String childBeanName : childBeans ) {
541                 resetBeanDefinition(childBeanName);
542             }
543         }
544     }

558     public Object resolveDependency(DependencyDescriptor descriptor, String beanName,
559             Set autowiredBeanNames, TypeConverter typeConverter) throws BeansException  {
560 
561         Class type = descriptor.getDependencyType();
562         if (type.isArray()) {
563             Class componentType = type.getComponentType();
564             Map matchingBeans = findAutowireCandidates(beanName, componentType, descriptor);
565             if (matchingBeans.isEmpty()) {
566                 if (descriptor.isRequired()) {
567                     raiseNoSuchBeanDefinitionException(componentType, "array of " + componentType.getName(), descriptor);
568                 }
569                 return null;
570             }
571             if (autowiredBeanNames != null) {
572                 autowiredBeanNames.addAll(matchingBeans.keySet());
573             }
574             TypeConverter converter = (typeConverter != null ? typeConverter : getTypeConverter());
575             return converter.convertIfNecessary(matchingBeans.values(), type);
576         }
577         else if (Collection.class.isAssignableFrom(type) && type.isInterface()) {
578             Class elementType = descriptor.getCollectionType();
579             if (elementType == null) {
580                 if (descriptor.isRequired()) {
581                     throw new FatalBeanException("No element type declared for collection [" + type.getName() + "]");
582                 }
583                 return null;
584             }
585             Map matchingBeans = findAutowireCandidates(beanName, elementType, descriptor);
586             if (matchingBeans.isEmpty()) {
587                 if (descriptor.isRequired()) {
588                     raiseNoSuchBeanDefinitionException(elementType, "collection of " + elementType.getName(), descriptor);
589                 }
590                 return null;
591             }
592             if (autowiredBeanNames != null) {
593                 autowiredBeanNames.addAll(matchingBeans.keySet());
594             }
595             TypeConverter converter = (typeConverter != null ? typeConverter : getTypeConverter());
596             return converter.convertIfNecessary(matchingBeans.values(), type);
597         }
598         else if (Map.class.isAssignableFrom(type) && type.isInterface()) {
599             Class keyType = descriptor.getMapKeyType();
600             if (keyType == null || !String.class.isAssignableFrom(keyType)) {
601                 if (descriptor.isRequired()) {
602                     throw new FatalBeanException("Key type [" + keyType + "] of map [" + type.getName() +
603                             "] must be assignable to [java.lang.String]");
604                 }
605                 return null;
606             }
607             Class valueType = descriptor.getMapValueType();
608             if (valueType == null) {
609                 if (descriptor.isRequired()) {
610                     throw new FatalBeanException("No value type declared for map [" + type.getName() + "]");
611                 }
612                 return null;
613             }
614             Map matchingBeans = findAutowireCandidates(beanName, valueType, descriptor);
615             if (matchingBeans.isEmpty()) {
616                 if (descriptor.isRequired()) {
617                     raiseNoSuchBeanDefinitionException(valueType, "map with value type " + valueType.getName(), descriptor);
618                 }
619                 return null;
620             }
621             if (autowiredBeanNames != null) {
622                 autowiredBeanNames.addAll(matchingBeans.keySet());
623             }
624             return matchingBeans;
625         }
626         else {
627             Map matchingBeans = findAutowireCandidates(beanName, type, descriptor);
628             if (matchingBeans.isEmpty()) {
629                 if (descriptor.isRequired()) {
630                     throw new NoSuchBeanDefinitionException(type,
631                             "Unsatisfied dependency of type [" + type + "]: expected at least 1 matching bean");
632                 }
633                 return null;
634             }
635             if (matchingBeans.size() > 1) {
636                 String primaryBeanName = determinePrimaryCandidate(matchingBeans, type);
637                 if (primaryBeanName == null) {
638                     throw new NoSuchBeanDefinitionException(type,
639                             "expected single matching bean but found " + matchingBeans.size() + ": " + matchingBeans.keySet());
640                 }
641                 if (autowiredBeanNames != null) {
642                     autowiredBeanNames.add(primaryBeanName);
643                 }
644                 return matchingBeans.get(primaryBeanName);
645             }
646             // We have exactly one match.
647             Map.Entry entry = (Map.Entry) matchingBeans.entrySet().iterator().next();
648             if (autowiredBeanNames != null) {
649                 autowiredBeanNames.add(entry.getKey());
650             }
651             return entry.getValue();
652         }
653     }

655     /**
656      * Find bean instances that match the required type.
657      * Called during autowiring for the specified bean.
658      * @param beanName the name of the bean that is about to be wired
659      * @param requiredType the actual type of bean to look for
660      * (may be an array component type or collection element type)
661      * @param descriptor the descriptor of the dependency to resolve
662      * @return a Map of candidate names and candidate instances that match
663      * the required type (never <code>null</code>)
664      * @throws BeansException in case of errors
665      * @see #autowireByType
666      * @see #autowireConstructor
667      */
668     protected Map findAutowireCandidates(String beanName, Class requiredType, DependencyDescriptor descriptor) {
669         String[] candidateNames = BeanFactoryUtils.beanNamesForTypeIncludingAncestors(
670                 this, requiredType, true, descriptor.isEager());
671         Map result = new LinkedHashMap(candidateNames.length);
672         for (Iterator it = this.resolvableDependencies.keySet().iterator(); it.hasNext();) {
673             Class autowiringType = (Class) it.next();
674             if (autowiringType.isAssignableFrom(requiredType)) {
675                 Object autowiringValue = this.resolvableDependencies.get(autowiringType);
676                 if (autowiringValue instanceof ObjectFactory && !requiredType.isInstance(autowiringValue)) {
677                     autowiringValue = ((ObjectFactory) autowiringValue).getObject();
678                 }
679                 if (requiredType.isInstance(autowiringValue)) {
680                     result.put(ObjectUtils.identityToString(autowiringValue), autowiringValue);
681                     break;
682                 }
683             }
684         }
685         for (int i = 0; i < candidateNames.length; i++) {
686             String candidateName = candidateNames[i];
687             if (!candidateName.equals(beanName) && isAutowireCandidate(candidateName, descriptor)) {
688                 result.put(candidateName, getBean(candidateName));
689             }
690         }
691         return result;
692     }

694     /**
695      * Determine the primary autowire candidate in the given set of beans.
696      * @param candidateBeans a Map of candidate names and candidate instances
697      * that match the required type, as returned by {@link #findAutowireCandidates}
698      * @param type the required type
699      * @return the name of the primary candidate, or <code>null</code> if none found
700      */
701     protected String determinePrimaryCandidate(Map candidateBeans, Class type) {
702         String primaryBeanName = null;
703         for (Iterator it = candidateBeans.entrySet().iterator(); it.hasNext();) {
704             Map.Entry entry = (Map.Entry) it.next();
705             String candidateBeanName = (String) entry.getKey();
706             if (isPrimary(candidateBeanName, entry.getValue())) {
707                 if (primaryBeanName != null) {
708                     throw new NoSuchBeanDefinitionException(type,
709                             "more than one 'primary' bean found among candidates: " + candidateBeans.keySet());
710                 }
711                 primaryBeanName = candidateBeanName;
712             }
713         }
714         return primaryBeanName;
715     }

717     /**
718      * Return whether the bean definition for the given bean name has been
719      * marked as a primary bean.
720      * @param beanName the name of the bean
721      * @param beanInstance the corresponding bean instance
722      * @return whether the given bean qualifies as primary
723      */
724     protected boolean isPrimary(String beanName, Object beanInstance) {
725         if (containsBeanDefinition(beanName)) {
726             return getMergedLocalBeanDefinition(beanName).isPrimary();
727         }
728         if (this.resolvableDependencies.values().contains(beanInstance)) {
729             return true;
730         }
731         BeanFactory parentFactory = getParentBeanFactory();
732         return (parentFactory instanceof DefaultListableBeanFactory &&
733                 ((DefaultListableBeanFactory) parentFactory).isPrimary(beanName, beanInstance));
734     }
