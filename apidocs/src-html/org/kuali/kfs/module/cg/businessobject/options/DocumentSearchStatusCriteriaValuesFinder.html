016    package org.kuali.kfs.module.cg.businessobject.options;
017    
018    import java.util.ArrayList;
019    import java.util.Collection;
020    import java.util.List;
021    
022    import org.kuali.kfs.module.cg.businessobject.AwardStatus;
023    import org.kuali.kfs.module.cg.businessobject.ProposalStatus;
024    import org.kuali.kfs.sys.context.SpringContext;
025    import org.kuali.rice.kns.lookup.keyvalues.KeyValuesBase;
026    import org.kuali.rice.kns.service.KeyValuesService;
027    import org.kuali.rice.core.util.KeyLabelPair;
028    
029    /**
030     * Allows some filtering of document search criteria.
031     */
032    public class DocumentSearchStatusCriteriaValuesFinder extends KeyValuesBase {
033    
034        private static final String STATUS_SPACE_PREFIX = "&nbsp;&nbsp;";
035    
036        /**
037         * @see org.kuali.keyvalues.KeyValuesFinder#getKeyValues()
038         */
039        public List getKeyValues() {
040            List<KeyLabelPair> labels = new ArrayList<KeyLabelPair>();
041            labels.add(new KeyLabelPair("", ""));
042    
043            labels.add(new KeyLabelPair(" ", "AWARD STATUSES"));
044            Collection<AwardStatus> awardCodes = SpringContext.getBean(KeyValuesService.class).findAll(AwardStatus.class);
045            for (AwardStatus awardStatus : awardCodes) {
046                if (awardStatus.isActive()) {
047                    labels.add(new KeyLabelPair(awardStatus.getAwardStatusCode(), STATUS_SPACE_PREFIX + awardStatus.getAwardStatusCode() + "-" + awardStatus.getAwardStatusDescription()));
048                }
049            }
050    
051            labels.add(new KeyLabelPair(" ", "PROPOSAL STATUSES"));
052            Collection<ProposalStatus> proposalCodes = SpringContext.getBean(KeyValuesService.class).findAll(ProposalStatus.class);
053            for (ProposalStatus proposalStatus : proposalCodes) {
054                if (proposalStatus.isActive()) {
055                    labels.add(new KeyLabelPair(proposalStatus.getProposalStatusCode(), STATUS_SPACE_PREFIX + proposalStatus.getProposalStatusCode() + "-" + proposalStatus.getProposalStatusDescription()));
056                }
057            }
058    
059            return labels;
060        }
061    
062    }
