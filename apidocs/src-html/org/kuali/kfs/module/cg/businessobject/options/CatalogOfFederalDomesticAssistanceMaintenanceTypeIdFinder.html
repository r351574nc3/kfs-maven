016    package org.kuali.kfs.module.cg.businessobject.options;
017    
018    import java.util.ArrayList;
019    import java.util.Collection;
020    import java.util.Iterator;
021    import java.util.List;
022    
023    import org.kuali.kfs.module.cg.businessobject.CFDA;
024    import org.kuali.kfs.sys.context.SpringContext;
025    import org.kuali.rice.kns.lookup.keyvalues.KeyValuesBase;
026    import org.kuali.rice.kns.service.KeyValuesService;
027    import org.kuali.rice.core.util.KeyLabelPair;
028    
029    /**
030     * Allows some information about persisted {@link Cfda} instances to be looked up.
031     */
032    public class CatalogOfFederalDomesticAssistanceMaintenanceTypeIdFinder extends KeyValuesBase {
033    
034        /**
035         * Retrieves the list of possible CFDA Maintenance Type IDs and generates a collection with all the possible values.
036         * 
037         * @see org.kuali.keyvalues.KeyValuesFinder#getKeyValues()
038         */
039        public List getKeyValues() {
040    
041            KeyValuesService boService = SpringContext.getBean(KeyValuesService.class);
042            Collection codes = boService.findAll(CFDA.class);
043    
044            List<KeyLabelPair> labels = new ArrayList<KeyLabelPair>();
045            labels.add(new KeyLabelPair("", ""));
046    
047            for (Iterator iter = codes.iterator(); iter.hasNext();) {
048                CFDA cfdaReference = (CFDA) iter.next();
049    
050                if (!isDuplicateValue(labels, cfdaReference.getCfdaMaintenanceTypeId())) {
051                    labels.add(new KeyLabelPair(cfdaReference.getCfdaMaintenanceTypeId(), cfdaReference.getCfdaMaintenanceTypeId()));
052                }
053            }
054    
055            return labels;
056        }
057    
058        /**
059         * This method determines if a value already exists in the collection.
060         * 
061         * @param collection The collection to be examined.
062         * @param value The value to be added to the collection if it does not already exist within it.
063         * @return True if the value passed in already exists in the collection, false otherwise.
064         */
065        private boolean isDuplicateValue(List<KeyLabelPair> collection, String value) {
066            boolean duplicate = false;
067    
068            for (KeyLabelPair klp : collection) {
069                String klpLabel = klp.getLabel();
070                if (klpLabel != null) {
071                    duplicate |= klpLabel.trim().equalsIgnoreCase(value.trim());
072                }
073            }
074    
075            return duplicate;
076        }
077    
078    }
