016    package org.kuali.kfs.module.cg.businessobject.options;
017    
018    import java.util.ArrayList;
019    import java.util.List;
020    
021    import org.kuali.kfs.module.cg.businessobject.ResearchRiskType;
022    import org.kuali.rice.kns.lookup.keyvalues.KeyValuesBase;
023    import org.kuali.rice.core.util.KeyLabelPair;
024    
025    public class ResearchRiskTypeNotificationValuesFinder extends KeyValuesBase {
026    
027        /**
028         * Constructs a ResearchRiskTypeNotificationValuesFinder.java.
029         */
030        public ResearchRiskTypeNotificationValuesFinder() {
031            super();
032        }
033    
034        /**
035         * Builds a collection of possible values to be selected from. These values are used to build out a drop down list for user
036         * selection.
037         * 
038         * @return A list of KeyLabelPair objects.
039         * @see org.kuali.rice.kns.lookup.keyvalues.KeyValuesFinder#getKeyValues()
040         */
041        public List getKeyValues() {
042    
043            List<KeyLabelPair> notificationValuesPairList = new ArrayList(4);
044    
045            notificationValuesPairList.add(new KeyLabelPair(ResearchRiskType.YES, ResearchRiskType.YES + " (Yes)"));
046            notificationValuesPairList.add(new KeyLabelPair(ResearchRiskType.NO, ResearchRiskType.NO + " (No)"));
047            notificationValuesPairList.add(new KeyLabelPair(ResearchRiskType.ALL, ResearchRiskType.ALL + " (All)"));
048            notificationValuesPairList.add(new KeyLabelPair(ResearchRiskType.NEVER, ResearchRiskType.NEVER + " (Never)"));
049    
050            return notificationValuesPairList;
051        }
052    
053    }
