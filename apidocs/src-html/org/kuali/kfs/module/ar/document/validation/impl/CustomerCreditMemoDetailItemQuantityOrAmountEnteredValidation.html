016    package org.kuali.kfs.module.ar.document.validation.impl;

018    import java.math.BigDecimal;
019    
020    import org.kuali.kfs.module.ar.ArConstants;
021    import org.kuali.kfs.module.ar.ArKeyConstants;
022    import org.kuali.kfs.module.ar.ArPropertyConstants;
023    import org.kuali.kfs.module.ar.businessobject.CustomerCreditMemoDetail;
024    import org.kuali.kfs.sys.document.validation.GenericValidation;
025    import org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent;
026    import org.kuali.rice.kns.util.GlobalVariables;
027    import org.kuali.rice.kns.util.KualiDecimal;
028    import org.kuali.rice.kns.util.ObjectUtils;
029    
030    public class CustomerCreditMemoDetailItemQuantityOrAmountEnteredValidation extends GenericValidation {
031    
032        private CustomerCreditMemoDetail customerCreditMemoDetail;
033        
034        public boolean validate(AttributedDocumentEvent event) {
035            BigDecimal quantity = customerCreditMemoDetail.getCreditMemoItemQuantity();
036            KualiDecimal amount = customerCreditMemoDetail.getCreditMemoItemTotalAmount();
037            boolean validValue;
038    
039            // don't generate error message but fail the business rules.
040            if (ObjectUtils.isNull(quantity) && ObjectUtils.isNull(amount)) {
041                return false;
042            }
043            return true;
044        }
045    
046        public CustomerCreditMemoDetail getCustomerCreditMemoDetail() {
047            return customerCreditMemoDetail;
048        }
049    
050        public void setCustomerCreditMemoDetail(CustomerCreditMemoDetail customerCreditMemoDetail) {
051            this.customerCreditMemoDetail = customerCreditMemoDetail;
052        }
053    
054    }
