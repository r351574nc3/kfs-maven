016    package org.kuali.kfs.module.ar.document.validation.impl;
017    
018    import static org.kuali.kfs.sys.document.validation.impl.AccountingDocumentRuleBaseConstants.ERROR_PATH.DOCUMENT_ERROR_PREFIX;
019    
020    import org.apache.commons.lang.StringUtils;
021    import org.kuali.kfs.module.ar.ArConstants;
022    import org.kuali.kfs.module.ar.ArKeyConstants;
023    import org.kuali.kfs.module.ar.ArPropertyConstants;
024    import org.kuali.kfs.module.ar.businessobject.CustomerInvoiceDetail;
025    import org.kuali.kfs.module.ar.document.CustomerInvoiceWriteoffDocument;
026    import org.kuali.kfs.sys.context.SpringContext;
027    import org.kuali.kfs.sys.document.validation.GenericValidation;
028    import org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent;
029    import org.kuali.rice.kns.service.ParameterService;
030    import org.kuali.rice.kns.util.GlobalVariables;
031    import org.kuali.rice.kns.util.ObjectUtils;
032    
033    public class CustomerInvoiceWriteoffChartCodeHasCorrespondingWriteoffObjectCodeValidation extends GenericValidation {
034    
035        private CustomerInvoiceDetail customerInvoiceDetail;
036        private CustomerInvoiceWriteoffDocument customerInvoiceWriteoffDocument;
037        
038        public boolean validate(AttributedDocumentEvent event) {
039            
040            String writeoffObjectCode = SpringContext.getBean(ParameterService.class).getParameterValue(CustomerInvoiceWriteoffDocument.class, ArConstants.GLPE_WRITEOFF_OBJECT_CODE_BY_CHART, customerInvoiceDetail.getChartOfAccountsCode());
041            if (StringUtils.isEmpty(writeoffObjectCode)) {
042                GlobalVariables.getMessageMap().putError(DOCUMENT_ERROR_PREFIX + ArPropertyConstants.CustomerInvoiceWriteoffDocumentFields.CUSTOMER_INVOICE_DETAILS_FOR_WRITEOFF, ArKeyConstants.ERROR_CUSTOMER_INVOICE_WRITEOFF_CHART_WRITEOFF_OBJECT_DOESNT_EXIST,customerInvoiceDetail.getChartOfAccountsCode());
043                return false;
044            }
045            return true;
046        }
047    
048        public CustomerInvoiceWriteoffDocument getCustomerInvoiceWriteoffDocument() {
049            return customerInvoiceWriteoffDocument;
050        }
051    
052        public void setCustomerInvoiceWriteoffDocument(CustomerInvoiceWriteoffDocument customerInvoiceWriteoffDocument) {
053            this.customerInvoiceWriteoffDocument = customerInvoiceWriteoffDocument;
054        }
055    
056        public CustomerInvoiceDetail getCustomerInvoiceDetail() {
057            return customerInvoiceDetail;
058        }
059    
060        public void setCustomerInvoiceDetail(CustomerInvoiceDetail customerInvoiceDetail) {
061            this.customerInvoiceDetail = customerInvoiceDetail;
062        }
063        
064    }
