016    package org.kuali.kfs.module.ar.document.validation.impl;
017    
018    import org.kuali.kfs.module.ar.ArConstants;
019    import org.kuali.kfs.module.ar.ArKeyConstants;
020    import org.kuali.kfs.module.ar.ArPropertyConstants;
021    import org.kuali.kfs.module.ar.businessobject.CustomerInvoiceDetail;
022    import org.kuali.kfs.module.ar.document.CustomerInvoiceDocument;
023    import org.kuali.kfs.module.ar.document.service.CustomerInvoiceDetailService;
024    import org.kuali.kfs.sys.document.validation.GenericValidation;
025    import org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent;
026    import org.kuali.rice.kns.util.GlobalVariables;
027    import org.kuali.rice.kns.util.ObjectUtils;
028    
029    public class CustomerInvoiceDetailDiscountGreaterThanParentValidation extends GenericValidation {
030    
031        private CustomerInvoiceDetail discountCustomerInvoiceDetail;
032        private CustomerInvoiceDocument customerInvoiceDocument;
033        private CustomerInvoiceDetailService customerInvoiceDetailService;
034    
035        public boolean validate(AttributedDocumentEvent event) {
036            boolean success = true;
037    
038            CustomerInvoiceDetail parentCustomerInvoiceDetail = discountCustomerInvoiceDetail.getParentDiscountCustomerInvoiceDetail();
039    
040            if (ObjectUtils.isNotNull(parentCustomerInvoiceDetail)) {
041    
042                // make a copy to not mess up the existing parent reference
043                CustomerInvoiceDetail copyOfParentCustomerInvoiceDetail = new CustomerInvoiceDetail();
044                copyOfParentCustomerInvoiceDetail.setInvoiceItemQuantity(parentCustomerInvoiceDetail.getInvoiceItemQuantity());
045                copyOfParentCustomerInvoiceDetail.setInvoiceItemUnitPrice(parentCustomerInvoiceDetail.getInvoiceItemUnitPrice());
046    
047                // update amounts
048                customerInvoiceDetailService.recalculateCustomerInvoiceDetail(customerInvoiceDocument, copyOfParentCustomerInvoiceDetail);
049                customerInvoiceDetailService.recalculateCustomerInvoiceDetail(customerInvoiceDocument, discountCustomerInvoiceDetail);
050    
051                // return true if abs(discount line amount) IS NOT greater than parent line
052                if (discountCustomerInvoiceDetail.getAmount().abs().isGreaterThan(copyOfParentCustomerInvoiceDetail.getAmount().abs())) {
053                    GlobalVariables.getMessageMap().putError(ArPropertyConstants.CustomerInvoiceDocumentFields.INVOICE_ITEM_UNIT_PRICE, ArKeyConstants.ERROR_CUSTOMER_INVOICE_DETAIL_DISCOUNT_AMOUNT_GREATER_THAN_PARENT_AMOUNT);
054                    success = false;
055                }
056            }
057    
058            return success;
059        }
060    
061        public CustomerInvoiceDetailService getCustomerInvoiceDetailService() {
062            return customerInvoiceDetailService;
063        }
064    
065    
066        public void setCustomerInvoiceDetailService(CustomerInvoiceDetailService customerInvoiceDetailService) {
067            this.customerInvoiceDetailService = customerInvoiceDetailService;
068        }
069    
070        public CustomerInvoiceDetail getDiscountCustomerInvoiceDetail() {
071            return discountCustomerInvoiceDetail;
072        }
073    
074    
075        public void setDiscountCustomerInvoiceDetail(CustomerInvoiceDetail discountCustomerInvoiceDetail) {
076            this.discountCustomerInvoiceDetail = discountCustomerInvoiceDetail;
077        }
078    
079    
080        public CustomerInvoiceDocument getCustomerInvoiceDocument() {
081            return customerInvoiceDocument;
082        }
083    
084        public void setCustomerInvoiceDocument(CustomerInvoiceDocument customerInvoiceDocument) {
085            this.customerInvoiceDocument = customerInvoiceDocument;
086        }
087    
088    }
