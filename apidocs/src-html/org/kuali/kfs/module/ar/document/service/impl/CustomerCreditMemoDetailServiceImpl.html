016    package org.kuali.kfs.module.ar.document.service.impl;
017    
018    import java.math.BigDecimal;
019    
020    import org.kuali.kfs.module.ar.businessobject.CustomerCreditMemoDetail;
021    import org.kuali.kfs.module.ar.document.CustomerCreditMemoDocument;
022    import org.kuali.kfs.module.ar.document.service.CustomerCreditMemoDetailService;
023    import org.kuali.kfs.module.ar.document.service.CustomerInvoiceDetailService;
024    import org.kuali.kfs.sys.context.SpringContext;
025    import org.kuali.rice.kns.util.KualiDecimal;
026    import org.kuali.rice.kns.util.ObjectUtils;
027    
028    public class CustomerCreditMemoDetailServiceImpl implements CustomerCreditMemoDetailService {
029    
030        public void recalculateCustomerCreditMemoDetail(CustomerCreditMemoDetail customerCreditMemoDetail, CustomerCreditMemoDocument customerCreditMemoDocument) {
031            CustomerInvoiceDetailService service = SpringContext.getBean(CustomerInvoiceDetailService.class);
032            
033            String invDocumentNumber = customerCreditMemoDocument.getFinancialDocumentReferenceInvoiceNumber();
034            Integer lineNumber = customerCreditMemoDetail.getReferenceInvoiceItemNumber();
035            
036            BigDecimal itemQuantity = customerCreditMemoDetail.getCreditMemoItemQuantity();
037            KualiDecimal itemAmount = customerCreditMemoDetail.getCreditMemoItemTotalAmount();
038            
039            // if line amount was entered, it takes precedence, if not, use the item quantity to re-calc amount
040            if (ObjectUtils.isNotNull(itemAmount)) {
041                customerCreditMemoDetail.recalculateBasedOnEnteredItemAmount(customerCreditMemoDocument);
042            }
043            // if item amount was entered
044            else {
045                customerCreditMemoDetail.recalculateBasedOnEnteredItemQty(customerCreditMemoDocument);
046            }
047            
048            customerCreditMemoDocument.recalculateTotalsBasedOnChangedItemAmount(customerCreditMemoDetail);                                                                                                                                                           
049        }
050    }
