030        public String generateOCRLine(KualiDecimal amountDue, String customerNumber, String docNumber) {
031           StringBuilder builder = new StringBuilder(70);
032           String amount = StringUtils.remove(amountDue.toString(), '.');
033           builder.append(StringUtils.leftPad(amount, 12, '0'));
034           builder.append(" ");
035           if (docNumber != null) {
036               builder.append(StringUtils.leftPad(docNumber, 11, '0'));
037               builder.append(" ");
038           }
039           builder.append(StringUtils.leftPad(customerNumber, 9, '0'));
040           builder.append(" ");
041           builder.append(calculateCheckDigit(amount, customerNumber, docNumber));
042           
043            return builder.toString();
044        }
