016    package org.kuali.kfs.module.cam.document.service.impl;
017    
018    import java.sql.Timestamp;
019    import java.util.List;
020    
021    import org.apache.commons.lang.StringUtils;
022    import org.kuali.kfs.module.cam.CamsConstants;
023    import org.kuali.kfs.module.cam.CamsPropertyConstants;
024    import org.kuali.kfs.module.cam.businessobject.Asset;
025    import org.kuali.kfs.module.cam.businessobject.AssetLocation;
026    import org.kuali.kfs.module.cam.document.EquipmentLoanOrReturnDocument;
027    import org.kuali.kfs.module.cam.document.service.EquipmentLoanOrReturnService;
028    import org.kuali.kfs.sys.context.SpringContext;
029    import org.kuali.rice.kns.bo.PersistableBusinessObject;
030    import org.kuali.rice.kns.service.BusinessObjectService;
031    import org.kuali.rice.kns.service.DateTimeService;
032    import org.kuali.rice.kns.util.ObjectUtils;
033    
034    public class EquipmentLoanOrReturnServiceImpl implements EquipmentLoanOrReturnService {
035        private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(EquipmentLoanOrReturnServiceImpl.class);
036    
037        private BusinessObjectService businessObjectService;
038    
039        /**
040         * @see org.kuali.kfs.module.cam.document.service.EquipmentLoanOrReturnService#processApprovedEquipmentLoanOrReturn(org.kuali.module.cams.document.EquipmentLoanOrReturn)
041         *      This method is called when the work flow document is reached its final approval
042         *      <ol>
043         *      <li>Gets the latest equipmentLoanOrReturn details from DB</li>
044         *      <li>Save asset data changes</li>
045         *      <li>Save borrower's location changes </li>
046         *      <li>Save store at location changes</li>
047         *      </ol>
048         */
049        public void processApprovedEquipmentLoanOrReturn(EquipmentLoanOrReturnDocument document) {
050            Asset updateAsset = new Asset();
051            updateAsset.setCapitalAssetNumber(document.getCapitalAssetNumber());
052            updateAsset = (Asset) getBusinessObjectService().retrieve(updateAsset);
053            updateAsset.setExpectedReturnDate(document.getExpectedReturnDate());
054            updateAsset.setLoanDate(document.getLoanDate());
055            updateAsset.setLoanReturnDate(document.getLoanReturnDate());
056            updateBorrowerLocation(document, updateAsset);
057            updateStoreAtLocation(document, updateAsset);
058            updateAsset.setLastInventoryDate(new Timestamp(SpringContext.getBean(DateTimeService.class).getCurrentSqlDate().getTime()));
059    
060            getBusinessObjectService().save((PersistableBusinessObject) updateAsset);
061        }
062    
063        /**
064         * @see org.kuali.kfs.module.cam.document.service.EquipmentLoanOrReturnService#setEquipmentLoanInfo(org.kuali.kfs.module.cam.businessobject.Asset)
065         */
066        public void setEquipmentLoanInfo(Asset asset) {
067    
068            if (asset.getExpectedReturnDate() != null && asset.getLoanReturnDate() == null) {
069                // find borrower and storage address and set those values
070                asset.refreshReferenceObject(CamsPropertyConstants.Asset.ASSET_LOCATIONS);
071                List<AssetLocation> assetLocations = asset.getAssetLocations();
072                if (ObjectUtils.isNotNull(assetLocations)) {
073                    for (AssetLocation assetLocation : assetLocations) {
074                        if (CamsConstants.AssetLocationTypeCode.BORROWER.equals(assetLocation.getAssetLocationTypeCode())) {
075                            asset.setBorrowerLocation(assetLocation);
076                        }
077                        if (CamsConstants.AssetLocationTypeCode.BORROWER_STORAGE.equals(assetLocation.getAssetLocationTypeCode())) {
078                            asset.setBorrowerStorageLocation(assetLocation);
079                        }
080                    }
081                }
082            }
083        }
084    
085        /**
086         * @see org.kuali.kfs.module.cam.document.service.EquipmentLoanOrReturnService#updateBorrowerLocation(org.kuali.module.cams.document.EquipmentLoanOrReturn)
087         */
088        protected void updateBorrowerLocation(EquipmentLoanOrReturnDocument document, Asset updateAsset) {
089            AssetLocation borrowerLocation = new AssetLocation();
090            borrowerLocation.setCapitalAssetNumber(updateAsset.getCapitalAssetNumber());
091            borrowerLocation.setAssetLocationTypeCode(CamsConstants.AssetLocationTypeCode.BORROWER);
092            borrowerLocation = (AssetLocation) getBusinessObjectService().retrieve(borrowerLocation);
093    
094            if (ObjectUtils.isNull(document.getLoanReturnDate())) {
095                if (borrowerLocation == null) {
096                    borrowerLocation = new AssetLocation();
097                    borrowerLocation.setCapitalAssetNumber(updateAsset.getCapitalAssetNumber());
098                    borrowerLocation.setAssetLocationTypeCode(CamsConstants.AssetLocationTypeCode.BORROWER);
099                    updateAsset.getAssetLocations().add(borrowerLocation);
100                }
101                borrowerLocation.setAssetLocationContactName(document.getBorrowerPerson().getPrincipalName());
102                borrowerLocation.setAssetLocationContactIdentifier(document.getBorrowerUniversalIdentifier());
103                borrowerLocation.setAssetLocationInstitutionName(document.getBorrowerPerson().getPrimaryDepartmentCode());
104                borrowerLocation.setAssetLocationPhoneNumber(document.getBorrowerPhoneNumber());
105                borrowerLocation.setAssetLocationStreetAddress(document.getBorrowerAddress());
106                borrowerLocation.setAssetLocationCityName(document.getBorrowerCityName());
107                borrowerLocation.setAssetLocationStateCode(document.getBorrowerStateCode());
108                borrowerLocation.setAssetLocationCountryCode(document.getBorrowerCountryCode());
109                borrowerLocation.setAssetLocationZipCode(document.getBorrowerZipCode());
110                getBusinessObjectService().save(borrowerLocation);
111            }
112            else {
113                if (borrowerLocation != null) {
114                    updateAsset.getAssetLocations().remove(borrowerLocation);
115                    getBusinessObjectService().delete(borrowerLocation);
116                }
117            }
118        }
119    
120        /**
121         * @see org.kuali.kfs.module.cam.document.service.EquipmentLoanOrReturnService#updateStoreAtLocation(org.kuali.module.cams.document.EquipmentLoanOrReturn)
122         */
123        protected void updateStoreAtLocation(EquipmentLoanOrReturnDocument document, Asset updateAsset) {
124            AssetLocation storeAtLocation = new AssetLocation();
125            storeAtLocation.setCapitalAssetNumber(updateAsset.getCapitalAssetNumber());
126            storeAtLocation.setAssetLocationTypeCode(CamsConstants.AssetLocationTypeCode.BORROWER_STORAGE);
127            storeAtLocation = (AssetLocation) getBusinessObjectService().retrieve(storeAtLocation);
128    
129            if (ObjectUtils.isNull(document.getLoanReturnDate()) && StringUtils.isNotBlank(document.getBorrowerStorageAddress())) {
130                if (storeAtLocation == null) {
131                    storeAtLocation = new AssetLocation();
132                    storeAtLocation.setCapitalAssetNumber(updateAsset.getCapitalAssetNumber());
133                    storeAtLocation.setAssetLocationTypeCode(CamsConstants.AssetLocationTypeCode.BORROWER_STORAGE);
134                    updateAsset.getAssetLocations().add(storeAtLocation);
135                }
136                storeAtLocation.setAssetLocationContactName(document.getBorrowerPerson().getPrincipalName());
137                storeAtLocation.setAssetLocationContactIdentifier(document.getBorrowerUniversalIdentifier());
138                storeAtLocation.setAssetLocationInstitutionName(document.getBorrowerPerson().getPrimaryDepartmentCode());
139                storeAtLocation.setAssetLocationPhoneNumber(document.getBorrowerStoragePhoneNumber());
140                storeAtLocation.setAssetLocationStreetAddress(document.getBorrowerStorageAddress());
141                storeAtLocation.setAssetLocationCityName(document.getBorrowerStorageCityName());
142                storeAtLocation.setAssetLocationStateCode(document.getBorrowerStorageStateCode());
143                storeAtLocation.setAssetLocationCountryCode(document.getBorrowerStorageCountryCode());
144                storeAtLocation.setAssetLocationZipCode(document.getBorrowerStorageZipCode());
145                getBusinessObjectService().save((PersistableBusinessObject) storeAtLocation);
146            }
147            else {
148                if (storeAtLocation != null) {
149                    updateAsset.getAssetLocations().remove(storeAtLocation);
150                    getBusinessObjectService().delete((PersistableBusinessObject) storeAtLocation);
151                }
152            }
153        }
154    
155        public BusinessObjectService getBusinessObjectService() {
156            return businessObjectService;
157        }
158    
159        public void setBusinessObjectService(BusinessObjectService businessObjectService) {
160            this.businessObjectService = businessObjectService;
161        }
