016    package org.kuali.kfs.module.cam.document.service.impl;
017    
018    import java.util.HashMap;
019    import java.util.Map;
020    
021    import org.apache.commons.lang.StringUtils;
022    import org.kuali.kfs.module.cam.CamsPropertyConstants;
023    import org.kuali.kfs.module.cam.businessobject.AssetAcquisitionType;
024    import org.kuali.kfs.module.cam.document.service.AssetAcquisitionTypeService;
025    import org.kuali.rice.kns.service.BusinessObjectService;
026    import org.springframework.transaction.annotation.Transactional;
027    
028    @Transactional
029    public class AssetAcquisitionTypeServiceImpl implements AssetAcquisitionTypeService {
030        BusinessObjectService businessObjectService;
031    
032        /**
033         * @see org.kuali.kfs.module.cam.document.service.AssetAcquisitionTypeService#doesAcquisitionTypeHaveIncomeAssetObjectCode(java.lang.String)
034         */
035        public boolean hasIncomeAssetObjectCode(String acquisitionTypeCode) {
036            Map<String, Object> pkKeys = new HashMap<String, Object>();
037            pkKeys.put(CamsPropertyConstants.AssetAcquisitionType.ACQUISITION_TYPE_CODE, acquisitionTypeCode);
038            AssetAcquisitionType assetAcquisitionType = (AssetAcquisitionType) getBusinessObjectService().findByPrimaryKey(AssetAcquisitionType.class, pkKeys);
039    
040            return StringUtils.isNotBlank(assetAcquisitionType.getIncomeAssetObjectCode());
041        }
042    
043        public BusinessObjectService getBusinessObjectService() {
044            return businessObjectService;
045        }
046    
047        public void setBusinessObjectService(BusinessObjectService businessObjectService) {
048            this.businessObjectService = businessObjectService;
049        }
050    }
