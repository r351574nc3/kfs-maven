070        /**
071         * Invokes several methods that validates each barcode error record
072         * 
073         * @param barcodeInventoryErrorDetails
074         * @return boolean
075         */
076        public boolean validateBarcodeInventoryErrorDetail(BarcodeInventoryErrorDocument document, boolean updateStatus) {
077            String errorPath = "";
078            boolean valid = true;
079            List<BarcodeInventoryErrorDetail> barcodeInventoryErrorDetails = document.getBarcodeInventoryErrorDetail();
080            List<BarcodeInventoryErrorDetail> inventory = new ArrayList<BarcodeInventoryErrorDetail>();
081    
082            Long lineNumber = new Long(0);
083            for (BarcodeInventoryErrorDetail barcodeInventoryErrorDetail : barcodeInventoryErrorDetails) {
084                barcodeInventoryErrorDetail.setErrorDescription("");
085                if (barcodeInventoryErrorDetail.getErrorCorrectionStatusCode().equals(CamsConstants.BarCodeInventoryError.STATUS_CODE_ERROR)) {
086                    valid = true;
087                    errorPath = CamsConstants.DOCUMENT_PATH + "." + CamsPropertyConstants.BarcodeInventory.BARCODE_INVENTORY_DETAIL + "[" + (lineNumber.intValue()) + "]";
088                    GlobalVariables.getMessageMap().addToErrorPath(errorPath);
089    
090                    valid &= this.validateTagNumber(barcodeInventoryErrorDetail.getAssetTagNumber());
091                    valid &= this.validateBuildingCode(barcodeInventoryErrorDetail.getBuildingCode(), barcodeInventoryErrorDetail);
092                    valid &= this.validateBuildingRoomNumber(barcodeInventoryErrorDetail.getBuildingRoomNumber(), barcodeInventoryErrorDetail);
093                    valid &= this.validateCampusCode(barcodeInventoryErrorDetail.getCampusCode(), barcodeInventoryErrorDetail);
094                    valid &= this.validateConditionCode(barcodeInventoryErrorDetail.getAssetConditionCode(), barcodeInventoryErrorDetail);
095                    valid &= this.validateInventoryDate(barcodeInventoryErrorDetail.getUploadScanTimestamp());
096                    valid &= this.validateTaggingLock(barcodeInventoryErrorDetail.getAssetTagNumber(), document.getDocumentNumber());
097    
098                    if (!valid) {
099                        barcodeInventoryErrorDetail.setErrorCorrectionStatusCode(CamsConstants.BarCodeInventoryError.STATUS_CODE_ERROR);
100    
101                        // Getting the errors from GlobalVariables.
102                        barcodeInventoryErrorDetail.setErrorDescription(getErrorMessages(errorPath));
103                    }
104                    else {
105                        if (updateStatus)
106                            barcodeInventoryErrorDetail.setErrorCorrectionStatusCode(CamsConstants.BarCodeInventoryError.STATUS_CODE_CORRECTED);
107    
108                        barcodeInventoryErrorDetail.setErrorDescription("NONE");
109                    }
110                    GlobalVariables.getMessageMap().removeFromErrorPath(errorPath);
111                }
112                lineNumber++;
113            }
114    
115            /*
116             * Since this document displays the asset lock error messages on the error description field, we don't want to display the
117             * same message at the top of the document. Therefore, deleteLockErrorMessages method deletes such errors from the
118             * GlobalVariables object.
119             */
120            deleteLockErrorMessages();
121    
122            return true;
123        }

126        /**
127         * validates the asset tag number exists in only one active asset.
128         * 
129         * @param tagNumber
130         * @return boolean
131         */
132        protected boolean validateTagNumber(String tagNumber) {
133            boolean result = true;
134            // Getting a list of active assets.
135            Collection<Asset> assets = getAssetService().findAssetsMatchingTagNumber(tagNumber);
136    
137            if (ObjectUtils.isNull(assets) || assets.isEmpty()) {
138                GlobalVariables.getMessageMap().putError(CamsPropertyConstants.BarcodeInventory.ASSET_TAG_NUMBER, CamsKeyConstants.BarcodeInventory.ERROR_CAPITAL_ASSET_DOESNT_EXIST);
139                result = false;
140            }
141            else {
142                int activeAssets = assets.size();
143                for (Asset asset : assets) {
144                    if (getAssetService().isAssetRetired(asset))
145                        activeAssets--;
146                }
147                if (activeAssets == 0) {
148                    GlobalVariables.getMessageMap().putError(CamsPropertyConstants.BarcodeInventory.ASSET_TAG_NUMBER, CamsKeyConstants.BarcodeInventory.ERROR_CAPITAL_ASSET_IS_RETIRED);
149                    result = false;
150                }
151                else if (activeAssets > 1) {
152                    GlobalVariables.getMessageMap().putError(CamsPropertyConstants.BarcodeInventory.ASSET_TAG_NUMBER, CamsKeyConstants.BarcodeInventory.ERROR_DUPLICATED_TAG_NUMBER);
153                    result = false;
154                }
155            }
156            return result;
157        }

179         * 
180         * @param campusCode
181         * @param detail
182         * @return boolean
183         */
184        protected boolean validateCampusCode(String campusCode, BarcodeInventoryErrorDetail detail) {
185            boolean result = true;
186            String label = SpringContext.getBean(DataDictionaryService.class).getDataDictionary().getBusinessObjectEntry(BarcodeInventoryErrorDetail.class.getName()).getAttributeDefinition(CamsPropertyConstants.BarcodeInventory.CAMPUS_CODE).getLabel();

188            Campus campus;
189            HashMap<String, Object> fields = new HashMap<String, Object>();
190            fields.put(KFSPropertyConstants.CAMPUS_CODE, detail.getCampusCode());
191            campus = (Campus) SpringContext.getBean(KualiModuleService.class).getResponsibleModuleService(Campus.class).getExternalizableBusinessObject(Campus.class, fields);

199                result &= false;
200            }
201            return result;
202        }

293        /**
294         * Validates the Asset doesn't have any tagging locks as result of an existing document in process where the same asset being
295         * affected
296         * 
297         * @param tagNumber
298         * @return boolean
299         */
300        protected boolean validateTaggingLock(String tagNumber, String documentNumber) {
301            boolean result = true;
302            boolean isAssetLocked = false;
303            String skipAssetLockValidation;
304    
305            // Getting system parameter in order to determine whether or not the asset locks will be ignored.
306            if (getParameterService().parameterExists(BarcodeInventoryErrorDocument.class, CamsConstants.Parameters.BAR_CODE_ERROR_DOCUMENT_IGNORES_LOCKS)) {
307                skipAssetLockValidation = getParameterService().getParameterValue(BarcodeInventoryErrorDocument.class, CamsConstants.Parameters.BAR_CODE_ERROR_DOCUMENT_IGNORES_LOCKS);
308            }
309            else {
310                LOG.warn("CAMS Parameter '" + CamsConstants.Parameters.BAR_CODE_ERROR_DOCUMENT_IGNORES_LOCKS + "' not found! - Setting default value to 'N' ");
311                skipAssetLockValidation = CamsConstants.BarCodeInventoryError.BAR_CODE_ERROR_DOCUMENT_IGNORES_LOCKS_NO;
312            }
313    
314            if (skipAssetLockValidation == null || StringUtils.isEmpty(skipAssetLockValidation) || StringUtils.equals(skipAssetLockValidation, CamsConstants.BarCodeInventoryError.BAR_CODE_ERROR_DOCUMENT_IGNORES_LOCKS_NO)) {
315                // Getting a list of active assets.
316                List<Asset> assets = getAssetService().findActiveAssetsMatchingTagNumber(tagNumber);
317                if (assets.size() > 1) {
318                    GlobalVariables.getMessageMap().putError(CamsPropertyConstants.BarcodeInventory.ASSET_TAG_NUMBER, CamsKeyConstants.BarcodeInventory.ERROR_DUPLICATED_TAG_NUMBER);
319                    result = false;
320                }
321                else if ((assets.size() > 0)) {
322                    // isAssetLocked = getAssetService().isAssetLocked("", assets.get(0).getCapitalAssetNumber());
323                    List<Long> assetNumbers = new ArrayList<Long>();
324                    assetNumbers.add(assets.get(0).getCapitalAssetNumber());
325                    List<String> lockingDocNumbers = SpringContext.getBean(AssetLockService.class).getAssetLockingDocuments(assetNumbers, CamsConstants.DocumentTypeName.ASSET_BARCODE_INVENTORY_ERROR, documentNumber);
326                    if (lockingDocNumbers != null && !lockingDocNumbers.isEmpty()) {
327                        for (String lockingDocNumber : lockingDocNumbers) {
328                            // String lockingDocumentId = getAssetService().getLockingDocumentId(documentNumber,
329                            // assets.get(0).getCapitalAssetNumber());
330                            GlobalVariables.getMessageMap().putError(CamsPropertyConstants.BarcodeInventory.ASSET_TAG_NUMBER, CamsKeyConstants.BarcodeInventory.ERROR_ASSET_LOCKED, lockingDocNumber);
331                        }
332                        result = false;
333                    }
334                }
335            }
336            else {
337                result = true;
338            }
339            return result;
340        }

343        /**
344         * Iterates over the list of errors each records might have and returns a single string with all the errors for each asset
345         * 
346         * @param errorPath
347         * @return String
348         */
349        protected String getErrorMessages(String errorPath) {
350            String message = "";
351            String[] fields = { CamsPropertyConstants.BarcodeInventory.ASSET_TAG_NUMBER, CamsPropertyConstants.BarcodeInventory.INVENTORY_DATE, CamsPropertyConstants.BarcodeInventory.CAMPUS_CODE, CamsPropertyConstants.BarcodeInventory.BUILDING_CODE, CamsPropertyConstants.BarcodeInventory.BUILDING_ROOM_NUMBER, CamsPropertyConstants.BarcodeInventory.ASSET_CONDITION_CODE };
352    
353            for (int i = 0; i < fields.length; i++) {
354                String propertyName = errorPath + "." + fields[i];
355                if (GlobalVariables.getMessageMap().containsKey(propertyName)) {
356                    for (Object errorMessage : GlobalVariables.getMessageMap().getMessages(propertyName)) {
357                        String errorMsg = getKualiConfigurationService().getPropertyString(((ErrorMessage) errorMessage).getErrorKey());
358                        message += ", " + MessageFormat.format(errorMsg, (Object[]) ((ErrorMessage) errorMessage).getMessageParameters());
359                    }
360                }
361            }
362            return (StringUtils.isEmpty(message) ? message : message.substring(2));
363        }

366        /**
367         * Deletes the asset locking error messages from the GlobalVariables.
368         * 
369         * @return none
370         */
371        protected void deleteLockErrorMessages() {
372            // Finding locking error messages
373            List<ErrorMessage> el = new ArrayList<ErrorMessage>();
374    
375            if (GlobalVariables.getMessageMap().getMessages(KFSConstants.GLOBAL_ERRORS) == null) {
376                return;
377            }
378    
379            for (Iterator<ErrorMessage> iterator = GlobalVariables.getMessageMap().getMessages(KFSConstants.GLOBAL_ERRORS).iterator(); iterator.hasNext();) {
380                ErrorMessage errorMessage = (ErrorMessage) iterator.next();
381                if (errorMessage.getErrorKey().equals(KFSKeyConstants.ERROR_MAINTENANCE_LOCKED)) {
382                    el.add(errorMessage);
383                }
384            }
385    
386            // Deleting asset locked error messages from global variable.
387            for (ErrorMessage em : el) {
388                GlobalVariables.getMessageMap().getMessages(KFSConstants.GLOBAL_ERRORS).remove(em);
389            }
390        }
