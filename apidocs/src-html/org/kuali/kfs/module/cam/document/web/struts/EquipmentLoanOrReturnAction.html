041    public class EquipmentLoanOrReturnAction extends KualiTransactionalDocumentActionBase {
042        protected static final org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(EquipmentLoanOrReturnAction.class);
043    
044        /**
045         * This method had to override because equipmentLoanOrReturn information has to be refreshed before display
046         * 
047         * @see org.kuali.rice.kns.web.struts.action.KualiDocumentActionBase#docHandler(org.apache.struts.action.ActionMapping,
048         *      org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
049         */
050        @Override
051        public ActionForward docHandler(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
052            ActionForward docHandlerForward = super.docHandler(mapping, form, request, response);
053            EquipmentLoanOrReturnForm equipmentLoanOrReturnForm = (EquipmentLoanOrReturnForm) form;
054            EquipmentLoanOrReturnDocument equipmentLoanOrReturnDocument = (EquipmentLoanOrReturnDocument) equipmentLoanOrReturnForm.getDocument();
055            BusinessObjectService service = SpringContext.getBean(BusinessObjectService.class);
056            Asset asset = equipmentLoanOrReturnDocument.getAsset();
057            asset = handleRequestFromLookup(request, equipmentLoanOrReturnForm, equipmentLoanOrReturnDocument, service, asset);
058    
059            asset = handleRequestFromWorkflow(equipmentLoanOrReturnForm, equipmentLoanOrReturnDocument, service);
060            asset = equipmentLoanOrReturnDocument.getAsset();
061    
062            asset.refreshReferenceObject(CamsPropertyConstants.Asset.ASSET_LOCATIONS);
063            asset.refreshReferenceObject(CamsPropertyConstants.Asset.ASSET_PAYMENTS);
064            SpringContext.getBean(AssetLocationService.class).setOffCampusLocation(asset);
065            SpringContext.getBean(PaymentSummaryService.class).calculateAndSetPaymentSummary(asset);
066    
067            return docHandlerForward;
068        }
069    
070        /**
071         * This method handles when request is from a work flow document search
072         * 
073         * @param equipmentLoanOrReturnForm Form
074         * @param equipmentLoanOrReturnDocument Document
075         * @param service BusinessObjectService
076         * @return Asset
077         */
078        protected Asset handleRequestFromWorkflow(EquipmentLoanOrReturnForm equipmentLoanOrReturnForm, EquipmentLoanOrReturnDocument equipmentLoanOrReturnDocument, BusinessObjectService businessObjectService) {
079            Asset newAsset = new Asset();
080            if (equipmentLoanOrReturnForm.getDocId() != null) {
081                newAsset.setCapitalAssetNumber(equipmentLoanOrReturnDocument.getCapitalAssetNumber());
082                newAsset = (Asset) businessObjectService.retrieve(newAsset);
083                equipmentLoanOrReturnDocument.setAsset(newAsset);
084            }
085            return newAsset;
086        }
087    
088        /**
089         * This method handles the request coming from asset lookup screen
090         * 
091         * @param request Request
092         * @param equipmentLoanOrReturnForm Current form
093         * @param equipmentLoanOrReturnDocument Document
094         * @param service Business Object Service
095         * @param asset Asset
096         * @return Asset
097         */
098        protected Asset handleRequestFromLookup(HttpServletRequest request, EquipmentLoanOrReturnForm equipmentLoanOrReturnForm, EquipmentLoanOrReturnDocument equipmentLoanOrReturnDocument, BusinessObjectService businessObjectService, Asset asset) {
099            Asset newAsset = asset;
100            if (equipmentLoanOrReturnForm.getDocId() == null && asset == null) {
101                newAsset = new Asset();
102                HashMap<String, Object> keys = new HashMap<String, Object>();
103                String capitalAssetNumber = request.getParameter(CAPITAL_ASSET_NUMBER);
104                keys.put(CAPITAL_ASSET_NUMBER, capitalAssetNumber);
105                newAsset = (Asset) businessObjectService.findByPrimaryKey(Asset.class, keys);
106                
107                // set document status
108                equipmentLoanOrReturnDocument.setNewLoan(true);
109                equipmentLoanOrReturnDocument.setReturnLoan(false);
110                
111                if (newAsset != null) {
112                    // populate equipmentLoanOrReturn info when loan type is renew or return loan
113                    if (!request.getParameter(CamsConstants.AssetActions.LOAN_TYPE).equals(CamsConstants.AssetActions.LOAN)) {
114                        populateEquipmentLoanOrReturnDocument(equipmentLoanOrReturnDocument, newAsset);
115                        equipmentLoanOrReturnDocument.setNewLoan(false);
116                    }
117                    // populate loan return date when loan type is return loan
118                    if (request.getParameter(CamsConstants.AssetActions.LOAN_TYPE).equals(CamsConstants.AssetActions.LOAN_RETURN)) {
119                        equipmentLoanOrReturnDocument.setLoanReturnDate(SpringContext.getBean(DateTimeService.class).getCurrentSqlDate());
120                        equipmentLoanOrReturnDocument.setReturnLoan(true);
121                    }
122                    // reset loan date and expect return date for renew loan
123                    if (request.getParameter(CamsConstants.AssetActions.LOAN_TYPE).equals(CamsConstants.AssetActions.LOAN_RENEW)) {
124                        equipmentLoanOrReturnDocument.setLoanDate(SpringContext.getBean(DateTimeService.class).getCurrentSqlDate());
125                        equipmentLoanOrReturnDocument.setExpectedReturnDate(null);
126                    }
127                    equipmentLoanOrReturnDocument.setCapitalAssetNumber(newAsset.getCapitalAssetNumber());
128                    equipmentLoanOrReturnDocument.setAsset(newAsset);
129                }
130            }
131            return newAsset;
132        }
133    
134        /**
135         * This method populate equipmentloanOrReturn document from asset and asset location
136         * 
137         * @param equipmentLoanOrReturnDocument EquipmentLoanOrReturnDocument
138         * @param newAsset Asset
139         * @return
140         */
141        protected void populateEquipmentLoanOrReturnDocument(EquipmentLoanOrReturnDocument equipmentLoanOrReturnDocument, Asset newAsset) {
142            equipmentLoanOrReturnDocument.setLoanDate(newAsset.getLoanDate());
143            equipmentLoanOrReturnDocument.setLoanReturnDate(newAsset.getLoanReturnDate());
144            equipmentLoanOrReturnDocument.setExpectedReturnDate(newAsset.getExpectedReturnDate());
145            // populate borrower address
146            AssetLocation borrowerLocation = new AssetLocation();
147            borrowerLocation.setCapitalAssetNumber(newAsset.getCapitalAssetNumber());
148            borrowerLocation.setAssetLocationTypeCode(CamsConstants.AssetLocationTypeCode.BORROWER);
149            borrowerLocation = (AssetLocation) SpringContext.getBean(BusinessObjectService.class).retrieve(borrowerLocation);
150            if (borrowerLocation != null) {
151                equipmentLoanOrReturnDocument.setBorrowerUniversalIdentifier(borrowerLocation.getAssetLocationContactIdentifier());
152                equipmentLoanOrReturnDocument.setBorrowerAddress(borrowerLocation.getAssetLocationStreetAddress());
153                equipmentLoanOrReturnDocument.setBorrowerCityName(borrowerLocation.getAssetLocationCityName());
154                equipmentLoanOrReturnDocument.setBorrowerStateCode(borrowerLocation.getAssetLocationStateCode());
155                equipmentLoanOrReturnDocument.setBorrowerZipCode(borrowerLocation.getAssetLocationZipCode());
156                equipmentLoanOrReturnDocument.setBorrowerCountryCode(borrowerLocation.getAssetLocationCountryCode());
157                equipmentLoanOrReturnDocument.setBorrowerPhoneNumber(borrowerLocation.getAssetLocationPhoneNumber());
158            }
159            // populate stored at address
160            AssetLocation storeAtLocation = new AssetLocation();
161            storeAtLocation.setCapitalAssetNumber(newAsset.getCapitalAssetNumber());
162            storeAtLocation.setAssetLocationTypeCode(CamsConstants.AssetLocationTypeCode.BORROWER_STORAGE);
163            storeAtLocation = (AssetLocation) SpringContext.getBean(BusinessObjectService.class).retrieve(storeAtLocation);
164            if (storeAtLocation != null) {
165                equipmentLoanOrReturnDocument.setBorrowerStorageAddress(storeAtLocation.getAssetLocationStreetAddress());
166                equipmentLoanOrReturnDocument.setBorrowerStorageCityName(storeAtLocation.getAssetLocationCityName());
167                equipmentLoanOrReturnDocument.setBorrowerStorageStateCode(storeAtLocation.getAssetLocationStateCode());
168                equipmentLoanOrReturnDocument.setBorrowerStorageZipCode(storeAtLocation.getAssetLocationZipCode());
169                equipmentLoanOrReturnDocument.setBorrowerStorageCountryCode(storeAtLocation.getAssetLocationCountryCode());
170                equipmentLoanOrReturnDocument.setBorrowerStoragePhoneNumber(storeAtLocation.getAssetLocationPhoneNumber());
171            }
172        }
