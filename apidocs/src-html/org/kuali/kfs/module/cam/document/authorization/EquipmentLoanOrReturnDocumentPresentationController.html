016    package org.kuali.kfs.module.cam.document.authorization;
017    
018    import java.util.List;
019    import java.util.Set;
020    
021    import org.kuali.kfs.module.cam.CamsConstants;
022    import org.kuali.kfs.module.cam.document.EquipmentLoanOrReturnDocument;
023    import org.kuali.kfs.module.cam.document.service.AssetService;
024    import org.kuali.kfs.sys.context.SpringContext;
025    import org.kuali.kfs.sys.document.authorization.FinancialSystemTransactionalDocumentPresentationControllerBase;
026    import org.kuali.rice.kns.document.Document;
027    import org.kuali.rice.kns.workflow.service.KualiWorkflowDocument;
028    
029    /**
030     * Presentation Controller for Equipment Loan Or Return Documents
031     */
032    public class EquipmentLoanOrReturnDocumentPresentationController extends FinancialSystemTransactionalDocumentPresentationControllerBase {
033        
034        @Override
035        public Set<String> getEditModes(Document document) {
036            Set<String> editModes = super.getEditModes(document);
037            
038            EquipmentLoanOrReturnDocument equipmentLoanOrReturnDocument = (EquipmentLoanOrReturnDocument) document;
039            
040            // setup new loan document
041            if (equipmentLoanOrReturnDocument.isNewLoan()) {
042                editModes.add(CamsConstants.EquipmentLoanOrReturnEditMode.DISPLAY_NEW_LOAN_TAB);
043            }
044            
045            // setup return loan document
046            if (equipmentLoanOrReturnDocument.isReturnLoan()) {
047                editModes.add(CamsConstants.EquipmentLoanOrReturnEditMode.DISPLAY_RETURN_LOAN_FIELDS_READ_ONLY);
048            }
049            
050            return editModes;
051        }
052        
053        @Override
054        protected boolean canEdit(Document document) {
055            KualiWorkflowDocument workflowDocument = (KualiWorkflowDocument) document.getDocumentHeader().getWorkflowDocument();
056    
057            if (workflowDocument.stateIsEnroute()) {
058                List<String> nodeNames = SpringContext.getBean(AssetService.class).getCurrentRouteLevels(workflowDocument);
059    
060                if (nodeNames.contains(CamsConstants.RouteLevelNames.BORROWER)) {
061                    return false;
062                }
063            }
064            return super.canEdit(document);
065        }
066    }
