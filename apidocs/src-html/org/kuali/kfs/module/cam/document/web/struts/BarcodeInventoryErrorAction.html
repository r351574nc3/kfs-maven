069         * @see org.kuali.rice.kns.web.struts.action.KualiDocumentActionBase#approve(org.apache.struts.action.ActionMapping,
070         *      org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
071         */
072        @Override
073        public ActionForward approve(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
074            BarcodeInventoryErrorForm bcieForm = (BarcodeInventoryErrorForm) form;
075            BarcodeInventoryErrorDocument document = bcieForm.getBarcodeInventoryErrorDocument();
076            //prevent approval before save if barcode inventory item list not validated
077            if(!document.isDocumentCorrected()){
078                GlobalVariables.getMessageMap().putError(CamsPropertyConstants.BarcodeInventory.DOCUMENT_NUMBER, CamsKeyConstants.BarcodeInventory.ERROR_VALIDATE_ITEMS_BEFORE_APPROVE, document.getDocumentNumber());
079                return mapping.findForward(KFSConstants.MAPPING_BASIC);
080            }
081            
082            getBusinessObjectService().save(document.getBarcodeInventoryErrorDetail());
083    
084            if (this.getAssetBarcodeInventoryLoadService().isCurrentUserInitiator(document)) {
085                this.invokeRules(document, false);
086            }
087    
088            return super.approve(mapping, form, request, response);
089        }

121        /**
122         * Searches and replaces BCIE document data on the document
123         * 
124         * @param mapping
125         * @param form
126         * @param request
127         * @param response
128         * @return ActionForward
129         * @throws Exception
130         */
131        public ActionForward searchAndReplace(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
132            BarcodeInventoryErrorForm barcodeInventoryErrorForm = (BarcodeInventoryErrorForm) form;
133            BarcodeInventoryErrorDocument document = barcodeInventoryErrorForm.getBarcodeInventoryErrorDocument();
134            List<BarcodeInventoryErrorDetail> barcodeInventoryErrorDetails = document.getBarcodeInventoryErrorDetail();
135    
136            // Validating search criteria
137            if (validateGlobalReplaceFields(document)) {
138                BarcodeInventoryErrorDetailPredicate predicatedClosure = new BarcodeInventoryErrorDetailPredicate(barcodeInventoryErrorForm.getBarcodeInventoryErrorDocument());
139    
140                // searches and replaces
141                CollectionUtils.forAllDo(barcodeInventoryErrorDetails, predicatedClosure);
142    
143                document.setBarcodeInventoryErrorDetail(barcodeInventoryErrorDetails);
144                barcodeInventoryErrorForm.setDocument(document);
145            }
146            // Validating.
147            this.invokeRules(document, false);
148    
149            // Search fields initialization.
150            barcodeInventoryErrorForm.getBarcodeInventoryErrorDocument().resetSearchFields();
151    
152            // Reseting the checkboxes
153            barcodeInventoryErrorForm.resetCheckBoxes();
154    
155            // Displaying JSP
156            return mapping.findForward(KFSConstants.MAPPING_BASIC);
157        }

160        /**
161         * Validates all the selected records and saves them
162         * 
163         * @param mapping
164         * @param form
165         * @param request
166         * @param response
167         * @return
168         * @throws Exception
169         */
170        public ActionForward validateLines(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
171            BarcodeInventoryErrorForm barcodeInventoryErrorForm = (BarcodeInventoryErrorForm) form;
172            BarcodeInventoryErrorDocument document = barcodeInventoryErrorForm.getBarcodeInventoryErrorDocument();
173            List<BarcodeInventoryErrorDetail> barcodeInventoryErrorDetails = document.getBarcodeInventoryErrorDetail();
174    
175            String currentUserID = GlobalVariables.getUserSession().getPerson().getPrincipalId();
176    
177            List selectedRows = new ArrayList();
178            int selectedCheckboxes[] = barcodeInventoryErrorForm.getRowCheckbox();
179    
180            // If rows were selected.....
181            if (selectedCheckboxes != null) {
182                for(int i=0;i < selectedCheckboxes.length;i++) {
183                    selectedRows.add(selectedCheckboxes[i]);
184                }
185    
186                // Validate
187                this.invokeRules(document, true);
188    
189                barcodeInventoryErrorDetails = document.getBarcodeInventoryErrorDetail();
190                for (BarcodeInventoryErrorDetail detail : barcodeInventoryErrorDetails) {
191                    int uploadRowNumber = detail.getUploadRowNumber().intValue();
192                    if (selectedRows.contains(uploadRowNumber)) {
193                        if (detail.getErrorCorrectionStatusCode().equals(CamsConstants.BarCodeInventoryError.STATUS_CODE_CORRECTED)) {
194                            detail.setInventoryCorrectionTimestamp(getDateTimeService().getCurrentTimestamp());
195                            detail.setCorrectorUniversalIdentifier(currentUserID);
196    
197                            getAssetBarcodeInventoryLoadService().updateAssetInformation(detail, false);
198                        }
199                    } else {
200                        //Reseting the status back to error for those rows that were not selected by the user to be validated.
201                        //detail.getCorrectorUniversalIdentifier() = null means record has not been saved as corrected
202                        // if (detail.getCorrectorUniversalIdentifier() == null) {
203                        // detail.getInventoryCorrectionTimestamp() used to determine error_status instead of detail.getCorrectorUniversalIdentifier()
204                         if (detail.getInventoryCorrectionTimestamp() == null) {
205                            detail.setErrorCorrectionStatusCode(CamsConstants.BarCodeInventoryError.STATUS_CODE_ERROR);
206                        }
207                    }
208                }
209    
210            }
211            else {
212                GlobalVariables.getMessageMap().putErrorForSectionId(CamsPropertyConstants.COMMON_ERROR_SECTION_ID, CamsKeyConstants.BarcodeInventory.ERROR_CHECKBOX_MUST_BE_CHECKED);
213            }
214    
215            // Saving data
216            getBusinessObjectService().save(document.getBarcodeInventoryErrorDetail());
217    
218            // Loading changes on page
219            this.loadDocument((KualiDocumentFormBase) form);
220    
221            // resetting the checkboxes
222            barcodeInventoryErrorForm.resetCheckBoxes();
223    
224            return mapping.findForward(KFSConstants.MAPPING_BASIC);
225        }

228        /**
229         * Deletes selected lines from the document
230         * 
231         * @param mapping
232         * @param form
233         * @param request
234         * @param response
235         * @return
236         * @throws Exception
237         */
238        public ActionForward deleteLines(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
239            BarcodeInventoryErrorForm barcodeInventoryErrorForm = (BarcodeInventoryErrorForm) form;
240            BarcodeInventoryErrorDocument document = barcodeInventoryErrorForm.getBarcodeInventoryErrorDocument();
241            List<BarcodeInventoryErrorDetail> barcodeInventoryErrorDetails = document.getBarcodeInventoryErrorDetail();
242            BarcodeInventoryErrorDetail barcodeInventoryErrorDetail;
243    
244            String currentUserID = GlobalVariables.getUserSession().getPerson().getPrincipalId();
245    
246            int selectedCheckboxes[] = barcodeInventoryErrorForm.getRowCheckbox();
247    
248            // If rows were selected
249            if (!(selectedCheckboxes == null)) {
250                for (int i = 0; i < selectedCheckboxes.length; i++) {
251                    for (BarcodeInventoryErrorDetail detail : barcodeInventoryErrorDetails) {
252                        if (detail.getUploadRowNumber().compareTo(new Long(selectedCheckboxes[i])) == 0) {
253                            detail.setErrorCorrectionStatusCode(CamsConstants.BarCodeInventoryError.STATUS_CODE_DELETED);
254                            detail.setInventoryCorrectionTimestamp(getDateTimeService().getCurrentTimestamp());
255                            detail.setCorrectorUniversalIdentifier(currentUserID);
256                        }
257                    }
258                }
259            }
260            else {
261                GlobalVariables.getMessageMap().putErrorForSectionId(CamsPropertyConstants.COMMON_ERROR_SECTION_ID, CamsKeyConstants.BarcodeInventory.ERROR_CHECKBOX_MUST_BE_CHECKED);
262            }
263    
264            // Saving data
265            getBusinessObjectService().save(document.getBarcodeInventoryErrorDetail());
266    
267            // Clearing the checkboxes
268            barcodeInventoryErrorForm.resetCheckBoxes();
269    
270            this.loadDocument((KualiDocumentFormBase) form);
271            return mapping.findForward(KFSConstants.MAPPING_BASIC);
272        }

275        protected boolean validateGlobalReplaceFields(BarcodeInventoryErrorDocument document) {
276            if (StringUtils.isBlank(document.getCurrentScanCode()) && StringUtils.isBlank(document.getCurrentCampusCode()) && StringUtils.isBlank(document.getCurrentBuildingNumber()) && StringUtils.isBlank(document.getCurrentRoom()) && StringUtils.isBlank(document.getCurrentSubroom()) && StringUtils.isBlank(document.getCurrentConditionCode()) && StringUtils.isBlank(document.getCurrentTagNumber())) {
277    
278                GlobalVariables.getMessageMap().putErrorForSectionId(CamsPropertyConstants.BCIE_GLOBAL_REPLACE_ERROR_SECTION_ID, CamsKeyConstants.BarcodeInventory.ERROR_GLOBAL_REPLACE_SEARCH_CRITERIA);
279                return false;
280            }
281            return true;
282        }

292        }
