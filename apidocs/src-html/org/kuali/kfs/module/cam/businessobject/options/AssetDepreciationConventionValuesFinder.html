016    package org.kuali.kfs.module.cam.businessobject.options;
017    
018    
019    import java.util.ArrayList;
020    import java.util.List;
021    
022    import org.kuali.rice.kns.lookup.keyvalues.KeyValuesBase;
023    import org.kuali.rice.core.util.KeyLabelPair;
024    
025    
026    public class AssetDepreciationConventionValuesFinder extends KeyValuesBase {
027    
028        /**
029        * Constructs a AssetDepreciationConventionValuesFinder.java.
030        */
031        public AssetDepreciationConventionValuesFinder() {
032            super();
033        }
034    
035        /**
036        * Builds a collection of possible values to be selected from. These values are used to build out a drop down list for user
037        * selection.
038        * 
039        * @return A list of KeyLabelPair objects.
040        * @see org.kuali.rice.kns.lookup.keyvalues.KeyValuesFinder#getKeyValues()
041        */
042        public List getKeyValues() {
043    
044            List<KeyLabelPair> notificationValuesPairList = new ArrayList(4);
045    
046            notificationValuesPairList.add(new KeyLabelPair("FY", " Full Year"));
047            notificationValuesPairList.add(new KeyLabelPair("HY", " Half Year"));
048    
049            return notificationValuesPairList;
050        }
051    
052    }
