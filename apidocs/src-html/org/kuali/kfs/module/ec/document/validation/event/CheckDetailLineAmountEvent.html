016    package org.kuali.kfs.module.ec.document.validation.event;
017    
018    import org.kuali.kfs.module.ec.businessobject.EffortCertificationDetail;
019    import org.kuali.kfs.module.ec.document.EffortCertificationDocument;
020    import org.kuali.kfs.module.ec.document.validation.CheckDetailLineAmountRule;
021    import org.kuali.rice.kns.rule.BusinessRule;
022    import org.kuali.rice.kns.rule.event.KualiDocumentEventBase;
023    
024    public class CheckDetailLineAmountEvent extends KualiDocumentEventBase {
025        private EffortCertificationDocument effortCertificationDocument;
026        private EffortCertificationDetail effortCertificationDetail;
027    
028        /**
029         * Constructs a AddDetailLineEvent.java.
030         * 
031         * @param description
032         * @param errorPathPrefix
033         * @param document
034         */
035        public CheckDetailLineAmountEvent(String description, String errorPathPrefix, EffortCertificationDocument effortCertificationDocument, EffortCertificationDetail effortCertificationDetail) {
036            super(description, errorPathPrefix, effortCertificationDocument);
037            this.document = effortCertificationDocument;
038            this.effortCertificationDocument = effortCertificationDocument;
039            this.effortCertificationDetail = effortCertificationDetail;
040        }
041    
042        /**
043         * @see org.kuali.rice.kns.rule.event.KualiDocumentEvent#getRuleInterfaceClass()
044         */
045        @SuppressWarnings("unchecked")
046        public Class getRuleInterfaceClass() {
047            return CheckDetailLineAmountRule.class;
048        }
049    
050        /**
051         * @see org.kuali.rice.kns.rule.event.KualiDocumentEvent#invokeRuleMethod(org.kuali.rice.kns.rule.BusinessRule)
052         */
053        public boolean invokeRuleMethod(BusinessRule rule) {
054            return ((CheckDetailLineAmountRule<EffortCertificationDocument, EffortCertificationDetail>) rule).processCheckDetailLineAmountRules(effortCertificationDocument, effortCertificationDetail);
055        }
056    }
