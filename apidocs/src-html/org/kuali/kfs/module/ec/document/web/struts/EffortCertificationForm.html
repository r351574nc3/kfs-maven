140        /**
141         * get the relationship metadata for the detail line fields
142         * 
143         * @return the relationship metadata for the detail line fields
144         */
145        public Map<String, BusinessObjectRelationship> getRelationshipMetadata() {
146            LOG.info("getRelationshipMetadata() start");
147    
148            PersistenceStructureService persistenceStructureService = SpringContext.getBean(PersistenceStructureService.class);
149    
150            Map<String, BusinessObjectRelationship> relationshipMetadata = new HashMap<String, BusinessObjectRelationship>();
151            for (String attributeName : this.getInquirableFieldNames()) {
152                Map<String, Class<? extends BusinessObject>> primitiveReference = LookupUtils.getPrimitiveReference(newDetailLine, attributeName);
153    
154                if (primitiveReference != null && !primitiveReference.isEmpty()) {
155                    BusinessObjectRelationship primitiveRelationship = this.getPrimitiveBusinessObjectRelationship(persistenceStructureService.getRelationshipMetadata(newDetailLine.getClass(), attributeName));
156                    relationshipMetadata.put(attributeName, primitiveRelationship);
157                }
158            }
159    
160            return relationshipMetadata;
161        }

185        /**
186         * pick up the primitive relationship for an attribute from a set of relationships. Generally, the primitive relationship is
187         * that has the minimum number of primary keys.
188         * 
189         * @param relationshipMetadata the relationship metadata that contains the primitive relationship
190         * @return the primitive relationship for an attribute from a set of relationships.
191         */
192        protected BusinessObjectRelationship getPrimitiveBusinessObjectRelationship(Map<String, BusinessObjectRelationship> relationshipMetadata) {
193            int minCountOfKeys = Integer.MAX_VALUE;
194            BusinessObjectRelationship primitiveRelationship = null;
195    
196            for (String attribute : relationshipMetadata.keySet()) {
197                BusinessObjectRelationship currentRelationship = relationshipMetadata.get(attribute);
198    
199                Map<String, String> parentToChildReferences = currentRelationship.getParentToChildReferences();
200                if (parentToChildReferences.size() < minCountOfKeys) {
201                    minCountOfKeys = parentToChildReferences.size();
202                    primitiveRelationship = currentRelationship;
203                }
204            }
205            return primitiveRelationship;
206        }

208        /**
209         * Gets the inquiryUrl attribute.
210         * 
211         * @return Returns the inquiryUrl for the detail lines in the document.
212         */
213        protected List<Map<String, HtmlData>> getDetailLineFieldInquiryUrl(List<EffortCertificationDetail> detailLines) {
214            LOG.info("getDetailLineFieldInquiryUrl(List<EffortCertificationDetail>) start");
215    
216            Map<String, String> noninquirableFieldValues = this.getNoninquirableFieldValues();
217            Inquirable inquirable = this.getInquirable();
218    
219            List<Map<String, HtmlData>> inquiryURL = new ArrayList<Map<String, HtmlData>>();
220            for (EffortCertificationDetail detailLine : detailLines) {
221                detailLine.refreshNonUpdateableReferences();
222                
223                Map<String, HtmlData> inquiryURLForAttribute = new HashMap<String, HtmlData>();            
224                for (String attributeName : this.getInquirableFieldNames()) {                
225                    // exclude the non inquirable field values
226                    Object attributeValue = ObjectUtils.getPropertyValue(detailLine, attributeName);
227                    String noninquirableFieldValue = noninquirableFieldValues.get(attributeName);
228                    if(noninquirableFieldValue!=null && noninquirableFieldValue.equals(attributeValue)) {
229                        continue;
230                    }
231    
232                    HtmlData inquiryHref;
233                    if (this.getCustomizedInquirableFieldNames().contains(attributeName)) {
234                        inquiryHref = this.getCustomizedInquiryUrl(detailLine, attributeName);
235                    }
236                    else {
237                        inquiryHref = inquirable.getInquiryUrl(detailLine, attributeName, false);
238                    }
239    
240                    inquiryURLForAttribute.put(attributeName, inquiryHref);
241                }
242    
243                inquiryURL.add(inquiryURLForAttribute);
244            }
245    
246            return inquiryURL;
247        }

249        /**
250         * get the inquiry URL for the specified attribute
251         * 
252         * @param detailLine the detail line containing the given attribute
253         * @param attributeName the specified attribute name
254         * @return the inquiry URL for the specified attribute
255         */
256        protected HtmlData getCustomizedInquiryUrl(EffortCertificationDetail detailLine, String attributeName) {
257            if (StringUtils.equals(attributeName, KFSPropertyConstants.POSITION_NUMBER)) {
258                AnchorHtmlData inquiryHref = (AnchorHtmlData) getEffortPositionDataDetailsInquirableImpl().getInquiryUrl(detailLine, attributeName);
259                inquiryHref.setHref(this.getCompleteURL(inquiryHref.getHref()));
260                return inquiryHref;
261            }
262            
263            AnchorHtmlData inquiryHref = (AnchorHtmlData) getInquirable().getInquiryUrl(detailLine, attributeName, false);
264            inquiryHref.setHref(this.getCompleteURL(inquiryHref.getHref()));
265    
266            return inquiryHref;
267        }

325        /**
326         * append the extract query string into the given base URL
327         * 
328         * @param baseURL the given base URL. If the parameter is blank, the base URL won't be changed
329         * @return the complete URL built from the given base URL and extra query strings
330         */
331        protected String getCompleteURL(String baseURL) {
332            if (StringUtils.isBlank(baseURL)) {
333                return baseURL;
334            }
335    
336            String completeURL = baseURL;
337            EffortCertificationDocument document = (EffortCertificationDocument) this.getDocument();
338            Properties properties = new Properties();
339    
340            properties.put(KFSPropertyConstants.EMPLID, document.getEmplid());
341            properties.put(EffortPropertyConstants.EFFORT_CERTIFICATION_REPORT_NUMBER, document.getEffortCertificationReportNumber());
342            properties.put(KFSPropertyConstants.UNIVERSITY_FISCAL_YEAR, document.getUniversityFiscalYear());
343    
344            StringBuilder queryString = new StringBuilder();
345            for (Object key : properties.keySet()) {
346                queryString.append("&").append(key).append("=").append(properties.get(key));
347            }
348    
349            return completeURL.concat(queryString.toString());
350        }

352        /**
353         * Gets the fieldInfo attribute.
354         * 
355         * @return Returns the fieldInfo.
356         */
357        protected List<Map<String, String>> getFieldInfo(List<EffortCertificationDetail> detailLines) {
358            LOG.info("getFieldInfo(List<EffortCertificationDetail>) start");
359    
360            List<Map<String, String>> fieldInfo = new ArrayList<Map<String, String>>();
361            EffortCertificationDocument document = (EffortCertificationDocument) this.getDocument();
362            KualiDecimal totalOriginalPayrollAmount = document.getTotalOriginalPayrollAmount();
363    
364            for (EffortCertificationDetail detailLine : detailLines) {
365                detailLine.refreshNonUpdateableReferences();
366    
367                Map<String, String> fieldInfoForAttribute = new HashMap<String, String>();
368    
369                fieldInfoForAttribute.put(KFSPropertyConstants.CHART_OF_ACCOUNTS_CODE, detailLine.getChartOfAccounts().getFinChartOfAccountDescription());
370    
371                String accountInfo = buildAccountInfo(detailLine.getAccount());
372                fieldInfoForAttribute.put(KFSPropertyConstants.ACCOUNT_NUMBER, accountInfo);
373    
374                SubAccount subAccount = detailLine.getSubAccount();
375                if (ObjectUtils.isNotNull(subAccount)) {
376                    fieldInfoForAttribute.put(KFSPropertyConstants.SUB_ACCOUNT_NUMBER, subAccount.getSubAccountName());
377                }
378    
379                ObjectCode objectCode = detailLine.getFinancialObject();
380                if (ObjectUtils.isNotNull(objectCode)) {
381                    fieldInfoForAttribute.put(KFSPropertyConstants.FINANCIAL_OBJECT_CODE, objectCode.getFinancialObjectCodeName());
382                }
383    
384                Account sourceAccount = detailLine.getSourceAccount();
385                if (ObjectUtils.isNotNull(sourceAccount)) {
386                    fieldInfoForAttribute.put(EffortPropertyConstants.SOURCE_ACCOUNT_NUMBER, sourceAccount.getAccountName());
387                }
388    
389                Chart sourceChart = detailLine.getSourceChartOfAccounts();
390                if (ObjectUtils.isNotNull(sourceChart)) {
391                    fieldInfoForAttribute.put(EffortPropertyConstants.SOURCE_CHART_OF_ACCOUNTS_CODE, sourceChart.getFinChartOfAccountDescription());
392                }
393    
394                KualiDecimal originalPayrollAmount = detailLine.getEffortCertificationOriginalPayrollAmount();
395                String actualOriginalPercent = PayrollAmountHolder.recalculateEffortPercentAsString(totalOriginalPayrollAmount, originalPayrollAmount);
396                fieldInfoForAttribute.put(EffortPropertyConstants.EFFORT_CERTIFICATION_CALCULATED_OVERALL_PERCENT, actualOriginalPercent);
397    
398                fieldInfo.add(fieldInfoForAttribute);
399            }
400    
401            return fieldInfo;
402        }

422        /**
423         * build the descriptive information of the given account. The information includes account name and project director's name if
424         * any
425         * 
426         * @param chartOfAccountsCode the given chart of accounts code
427         * @param accountNumber the given account number
428         * @return the descriptive information of the given account
429         */
430        public static String buildAccountInfo(Account account) {
431            if (ObjectUtils.isNull(account)) {
432                return KFSConstants.EMPTY_STRING;
433            }
434    
435            String projectDirectorName = KFSConstants.EMPTY_STRING;
436    
437            try {
438                ContractsAndGrantsModuleService contractsAndGrantsModuleService = SpringContext.getBean(ContractsAndGrantsModuleService.class);
439                Person projectDirector = contractsAndGrantsModuleService.getProjectDirectorForAccount(account);
440    
441                projectDirectorName = projectDirector != null ? MessageFormat.format("  ({0})", projectDirector.getName()) : KFSConstants.EMPTY_STRING;
442            }
443            catch (Exception e) {
444                LOG.error("Cannot find a project director for the account:" + account);
445            }
446    
447            return MessageFormat.format("{0}{1}", account.getAccountName(), projectDirectorName);
448        }

450        /**
451         * load the descriptive information of the given account. This method is used by DWR.
452         * 
453         * @param chartOfAccountsCode the given chart of accounts code
454         * @param accountNumber the given account number
455         * @return the descriptive information of the given account
456         */
457        public static String loadAccountInfo(String chartOfAccountsCode, String accountNumber) {
458            Account account = SpringContext.getBean(AccountService.class).getByPrimaryIdWithCaching(chartOfAccountsCode, accountNumber);
459    
460            return buildAccountInfo(account);
461        }
