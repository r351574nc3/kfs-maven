016    package org.kuali.kfs.module.ec.document.service.impl;
017    
018    import java.util.List;
019    
020    import org.apache.commons.lang.StringUtils;
021    import org.kuali.kfs.module.ec.businessobject.EffortCertificationReportDefinition;
022    import org.kuali.kfs.module.ec.dataaccess.EffortCertificationReportDefinitionDao;
023    import org.kuali.kfs.module.ec.document.service.EffortCertificationAutomaticReportPeriodUpdateService;
024    import org.springframework.transaction.annotation.Transactional;
025    
026    /**
027     * @see org.rg.kuali.module.effort.service.EffortCertificationAutomaticReportPeriodUpdateService
028     */
029    @Transactional
030    public class EffortCertificationAutomaticReportPeriodUpdateServiceImpl implements EffortCertificationAutomaticReportPeriodUpdateService {
031    
032        private EffortCertificationReportDefinitionDao reportDefinitionDao;
033    
034        /**
035         * @see org.kuali.kfs.module.ec.document.service.EffortCertificationAutomaticReportPeriodUpdateService#getAllReportDefinitions()
036         */
037        public List<EffortCertificationReportDefinition> getAllReportDefinitions() {
038            return this.reportDefinitionDao.getAll();
039        }
040    
041        /**
042         * @see org.kuali.kfs.module.ec.document.service.EffortCertificationAutomaticReportPeriodUpdateService#isAnOverlappingReportDefinition(org.kuali.kfs.module.ec.businessobject.EffortCertificationReportDefinition)
043         */
044        public boolean isAnOverlappingReportDefinition(EffortCertificationReportDefinition reportDefinition) {
045            List<EffortCertificationReportDefinition> potentialOverlappingRecords = reportDefinitionDao.getAllOtherActiveByType(reportDefinition);
046    
047            for (EffortCertificationReportDefinition potentialOverlappingRecord : potentialOverlappingRecords) {
048                if (isOverlapping(potentialOverlappingRecord, reportDefinition)) {
049                    return true;
050                }
051            }
052    
053            return false;
054        }
055    
056        /**
057         * get's spring managed effortCertificationReportDefinitionDao
058         * 
059         * @return
060         */
061        public EffortCertificationReportDefinitionDao getEffortCertificationReportDefinitionDao() {
062            return reportDefinitionDao;
063        }
064    
065        /**
066         * set's spring managed effortCertificationReportDefinitionDao
067         * 
068         * @param effortCertificationReportDefinitionDao
069         */
070        public void setEffortCertificationReportDefinitionDao(EffortCertificationReportDefinitionDao effortCertificationReportDefinitionDao) {
071            this.reportDefinitionDao = effortCertificationReportDefinitionDao;
072        }
073    
074        /**
075         * Compares existingRecord and newRecord to see if they are overlapping (years and periods).
076         * 
077         * @param existingRecord
078         * @param newRecord
079         * @return boolean representing whether or not the two report defintions overlap.
080         */
081        protected boolean isOverlapping(EffortCertificationReportDefinition existingReportDefinition, EffortCertificationReportDefinition newReportDefiniton) {
082            Integer existingStartYear = existingReportDefinition.getEffortCertificationReportBeginFiscalYear();
083            String existingStartPeriod = existingReportDefinition.getEffortCertificationReportBeginPeriodCode();
084    
085            Integer existingEndYear = existingReportDefinition.getEffortCertificationReportEndFiscalYear();
086            String existingEndPeriod = existingReportDefinition.getEffortCertificationReportEndPeriodCode();
087    
088            if (existingStartYear == null || existingStartPeriod == null || existingEndYear == null || existingEndPeriod == null) {
089                return false;
090            }
091    
092            Integer newStartYear = newReportDefiniton.getEffortCertificationReportBeginFiscalYear();
093            String newStartPeriod = newReportDefiniton.getEffortCertificationReportBeginPeriodCode();
094    
095            Integer newEndYear = newReportDefiniton.getEffortCertificationReportEndFiscalYear();
096            String newEndPeriod = newReportDefiniton.getEffortCertificationReportEndPeriodCode();
097    
098            boolean isNewStartPeriodWithin = isPeriodWithinRange(existingStartYear, existingStartPeriod, existingEndYear, existingEndPeriod, newStartYear, newStartPeriod);
099            if (isNewStartPeriodWithin) {
100                return true;
101            }
102    
103            boolean isNewEndPeriodWithin = isPeriodWithinRange(existingStartYear, existingStartPeriod, existingEndYear, existingEndPeriod, newEndYear, newEndPeriod);
104            if (isNewEndPeriodWithin) {
105                return true;
106            }
107    
108            boolean isExistingStartPeriodWithin = isPeriodWithinRange(newStartYear, newStartPeriod, newEndYear, newEndPeriod, existingStartYear, existingStartPeriod);
109            if (isExistingStartPeriodWithin) {
110                return true;
111            }
112    
113            boolean isExistingEndPeriodWithin = isPeriodWithinRange(newStartYear, newStartPeriod, newEndYear, newEndPeriod, existingEndYear, existingEndPeriod);
114            if (isExistingEndPeriodWithin) {
115                return true;
116            }
117    
118            return false;
119        }
120    
121        protected boolean isPeriodWithinRange(Integer startYear, String startPeriod, Integer endYear, String endPeriod, Integer year, String period) {
122            return comparePeriod(startYear, startPeriod, year, period) <= 0 && comparePeriod(endYear, endPeriod, year, period) >= 0;
123        }
124    
125        protected int comparePeriod(Integer year, String periodCode, Integer anotherYear, String anotherPeriodCode) {
126            String period = year + periodCode;
127            String anotherPeriod = anotherYear + anotherPeriodCode;
128    
129            return period.compareTo(anotherPeriod);
130        }
131    }
