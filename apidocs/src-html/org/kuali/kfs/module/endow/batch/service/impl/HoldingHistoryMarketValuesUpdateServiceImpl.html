062        /**
063         * @see org.kuali.kfs.module.endow.batch.service.HoldingHistoryMarketValuesUpdateService#updateHoldingHistoryMarketValues()
064         */
065        public boolean updateHoldingHistoryMarketValues() {
066            boolean success = true ;
067            
068            LOG.debug("updateHoldingHistoryMarketValues() started");
069            
070            //writes the exception report header
071            holdingHistoryMarketValuesExceptionReportWriterService.writeNewLines(1);
072            holdingHistoryMarketValuesExceptionReportWriterService.writeTableHeader(holdingHistoryMarketValueExceptionReportHeader);
073            
074            //update the market values for the holding history value adjustment documents.
075            success = processUpdateHoldingHistoryMarketValues();
076            
077            return success;
078        }
079        
080        /**
081         * This method will use documentsearchcriteriaDTO to search for the documents that are in Final status
082         * @return true if update is successful else false
083         */
084        protected boolean processUpdateHoldingHistoryMarketValues() {
085            boolean success = true;
086            
087            Collection<HoldingHistoryValueAdjustmentDocument> holdingHistoryValueAdjustmentDocuments = holdingHistoryValueAdjustmentDocumentService.getHoldingHistoryValueAdjustmentDocument(EndowConstants.HoldingHistoryValueAdjustmentDocument.TRANSACTION_POSTED_NO);
088            
089            if (holdingHistoryValueAdjustmentDocuments.isEmpty()) {
090                setExceptionReportTableRowReason("There are no records in Holding History Value Adjustment Documents table.  The market values updates process is not executed");
091                holdingHistoryMarketValuesExceptionReportWriterService.writeTableRow(holdingHistoryMarketValueExceptionRowReason);
092                return success;
093            }
094            
095            for (HoldingHistoryValueAdjustmentDocument holdingHistoryValueAdjustmentDocument : holdingHistoryValueAdjustmentDocuments) {
096                String documentHeaderId = holdingHistoryValueAdjustmentDocument.getDocumentNumber();
097    
098                Document document = findDocumentForMarketValueUpdate(documentHeaderId);
099                
100                if (document != null) {
101                    if (document.getDocumentHeader().getWorkflowDocument().stateIsFinal()) {
102                        // the state of the document is final.. so processing the document.
103                        if (updateHoldingHistoryRecords(holdingHistoryValueAdjustmentDocument)) {
104                            //update the HoldingHistoryValueAdjustmentDocument's transactionPosted column to Y
105                            holdingHistoryValueAdjustmentDocument.setTransactionPosted(true);
106                            if (!holdingHistoryValueAdjustmentDocumentService.saveHoldingHistory(holdingHistoryValueAdjustmentDocument)) {
107                                setUnwantedReportColumnsToBlank();                                
108                                writeTableRowAndTableReason(holdingHistoryValueAdjustmentDocument, "Unable to set Transaction Posted flag in Holding History Value Adjustment Document.");
109                            }
110                        }
111                    }
112                }
113                else {
114                    //include document number....
115                    LOG.error("Document with document number: " + documentHeaderId + " is NULL.  It should never have been null");
116                    setUnwantedReportColumnsToBlank();                
117                    writeTableRowAndTableReason(holdingHistoryValueAdjustmentDocument, "Document with document number: " + documentHeaderId + " The document does not exist in the workflow.");                
118                }
119            }
120            
121            return success;
122        }

135        /**
136         * writes out the table row values for document type, secuityId, kemId and then writes the reason row and inserts a blank line
137         * @param ehva the holding history value adjustment document
138         * @param reasonMessage the reason message
139         */
140        protected void writeTableRowAndTableReason(HoldingHistoryValueAdjustmentDocument ehva, String reasonMessage) {
141            getExceptionReportTableRowValues(ehva);
142            
143            holdingHistoryMarketValuesExceptionReportWriterService.writeTableRow(holdingHistoryMarketValueExceptionRowValues);            
144            setExceptionReportTableRowReason(reasonMessage);
145            
146            holdingHistoryMarketValuesExceptionReportWriterService.writeTableRow(holdingHistoryMarketValueExceptionRowReason);            
147            holdingHistoryMarketValuesExceptionReportWriterService.writeNewLines(1);
148        }

150        /**
151         * gets the values for document type, securityId, and kemdId that will be written to the exception report
152         * @param ehva
153         */
154        protected void getExceptionReportTableRowValues(HoldingHistoryValueAdjustmentDocument ehva) {
155            String documentTypeName = ehva.getDocumentHeader().getWorkflowDocument().getDocumentType();
156            
157            holdingHistoryMarketValueExceptionRowValues.setColumnHeading1(documentTypeName);
158            holdingHistoryMarketValueExceptionRowValues.setColumnHeading2(ehva.getSecurityId());
159            holdingHistoryMarketValueExceptionRowValues.setColumnHeading3(holdingHistoryService.getKemIdFromHoldingHistory(ehva.getSecurityId()));
160        }

163         * sets the exception message with the passed in value.
164         * @param reasonForException The reason that will be set in the exception report
165         */
166        protected void setExceptionReportTableRowReason(String reasonForException) {
167            
168            holdingHistoryMarketValueExceptionRowReason.setColumnHeading1("Reason: " + reasonForException);
169            holdingHistoryMarketValueExceptionRowReason.setColumnHeading2("");
170            holdingHistoryMarketValueExceptionRowReason.setColumnHeading3("");
171            setUnwantedReportColumnsToBlank();
172        }

174        /**
175         * This method updates the holding history records.
176         * The monthEndDateId is retrieved from END_ME_DT_T using the monthEndDate.  The holding history records 
177         * are fetched for matching securityId and monthEndDateId.  The market value is updated on each holding history
178         * record and then the record will be saved.
179         * @param ehva
180         * @return true if the market value is updated else false
181         */
182        protected boolean updateHoldingHistoryRecords(HoldingHistoryValueAdjustmentDocument ehva) {
183            LOG.info("Updating HoldingHistoryValueAdjustment document with Security Id: " + ehva.getSecurityId());
184    
185            boolean success = true;
186            
187            getExceptionReportTableRowValues(ehva);
188            
189            Collection<HoldingHistory> holdingHistoryRecords = holdingHistoryService.getHoldingHistoryBySecuritIdAndMonthEndId(ehva.getSecurityId(), ehva.getHoldingMonthEndDate());
190    
191            if (holdingHistoryRecords.isEmpty()) {
192                setUnwantedReportColumnsToBlank();            
193                writeTableRowAndTableReason(ehva, "There are no Holding History records to match the Holding History Value Adjustment document's security id and monthEndDateID.");
194                return false;
195            }
196            // process the holding history records
197            for (HoldingHistory holdingHistoryRecord : holdingHistoryRecords) {
198                holdingHistoryRecord.setSecurityUnitVal(ehva.getSecurityUnitValue());
199    
200                BigDecimal marketValue = getMarketValue(ehva, holdingHistoryRecord.getUnits());
201                if (ObjectUtils.isNull(marketValue)) {
202                    return false;
203                }
204                
205                holdingHistoryRecord.setMarketValue(marketValue);
206                
207                if (!holdingHistoryService.saveHoldingHistory(holdingHistoryRecord)) {
208                    holdingHistoryMarketValueExceptionRowValues.setColumnHeading3(holdingHistoryRecord.getKemid());
209                    holdingHistoryMarketValueExceptionRowValues.setColumnHeading4(holdingHistoryRecord.getRegistrationCode());
210                    holdingHistoryMarketValueExceptionRowValues.setColumnHeading5(holdingHistoryRecord.getMonthEndDateId().toString());
211                    holdingHistoryMarketValueExceptionRowValues.setColumnHeading6(holdingHistoryRecord.getIncomePrincipal().getName());
212                    holdingHistoryMarketValueExceptionRowValues.setColumnHeading7(holdingHistoryRecord.getLotNumber().toString());                
213                    writeTableRowAndTableReason(ehva, "Unable to save Holding History record.");
214                    return false;
215                }
216            }
217            
218            return success;
219        }
220        
221        /**
222         * This method calculates the market value.  If the class type code = B (Bonds) then market value = [Units X Unit Value]/100
223         * else Units x Unit Value
224         * @param ehva
225         * @return marketValue The calculated market value
226         */
227        protected BigDecimal getMarketValue(HoldingHistoryValueAdjustmentDocument ehva, BigDecimal units) {
228            BigDecimal marketValue = BigDecimal.ZERO;
229            try {
230                BigDecimal tmpValue = KEMCalculationRoundingHelper.multiply(ehva.getSecurityUnitValue(), units, EndowConstants.Scale.SECURITY_MARKET_VALUE);
231                
232                if (ehva.getSecurity().getClassCode().getClassCodeType().equalsIgnoreCase(EndowConstants.ClassCodeTypes.BOND)) {
233                    tmpValue = KEMCalculationRoundingHelper.divide(tmpValue, BigDecimal.valueOf(100), EndowConstants.Scale.SECURITY_MARKET_VALUE);
234                }
235                
236                marketValue = marketValue.add(tmpValue);  
237            }
238            catch (Exception ex) {
239                setUnwantedReportColumnsToBlank();            
240                writeTableRowAndTableReason(ehva, "Reason: " + ex.getMessage() + " - Unable to update the market value.");
241                return null;
242            }
243            
244            return marketValue;        
245        }

247        /**
248         * This method finds the document for the given document header id
249         * @param documentHeaderId
250         * @return document The document in the workflow that matches the document header id.
251         */
252        protected Document findDocumentForMarketValueUpdate(String documentHeaderId) {
253            Document document = null;
254            
255            try {
256                document = documentService.getByDocumentHeaderId(documentHeaderId);
257            }
258            catch (WorkflowException wfe) {
259                LOG.error("Exception encountered on finding the document: " + documentHeaderId + " - " + wfe.getMessage());
260            }
261            
262            return document;
263        }
