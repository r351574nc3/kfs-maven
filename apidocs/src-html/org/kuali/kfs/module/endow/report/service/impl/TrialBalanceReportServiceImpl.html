040        /**

044        public List<TrialBalanceReportDataHolder> getTrialBalanceReportForAllKemids(String endownmentOption, String closedIndicator) {
045            return getTrialBalanceReportsByKemidByIds(null, endownmentOption, closedIndicator);
046        }
047        
048        /**
049         * 
050         * @see org.kuali.kfs.module.endow.report.service.TrialBalanceReportService#getTrialBalanceReportsByKemidByIds(java.util.List, java.lang.String)
051         */
052        public List<TrialBalanceReportDataHolder> getTrialBalanceReportsByKemidByIds(List<String> kemids, String endownmentOption, String closedIndicator) {
053            
054            List<TrialBalanceReportDataHolder> trialBalanceReportList = new ArrayList<TrialBalanceReportDataHolder>();
055            List<KEMID> kemidRecords = kemidDao.getKemidRecordsByIds(kemids, endownmentOption, closedIndicator);
056            if (kemidRecords == null || kemidRecords.isEmpty()) {
057                return null;
058            }
059            
060            for (KEMID kemidObj : kemidRecords) {
061                
062                // get the kemid and the title
063                TrialBalanceReportDataHolder trialBalanceReport = new TrialBalanceReportDataHolder(); 
064                trialBalanceReport.setKemid(kemidObj.getKemid());
065                trialBalanceReport.setKemidName(kemidObj.getShortTitle());
066    
067                Map<String, Object> primaryKeys = new HashMap<String, Object>();
068                primaryKeys.put(EndowPropertyConstants.KEMID, kemidObj.getKemid());
069                
070                // get income cash balance, principal cash balance            
071                KemidCurrentCash kemidCurrentCash = (KemidCurrentCash) businessObjectService.findByPrimaryKey(KemidCurrentCash.class, primaryKeys);
072                BigDecimal subTotalKemidTotalMarketValue = BigDecimal.ZERO;  
073                if (ObjectUtils.isNotNull(kemidCurrentCash)) {
074                    trialBalanceReport.setInocmeCashBalance(kemidCurrentCash.getCurrentIncomeCash());
075                    trialBalanceReport.setPrincipalcashBalance(kemidCurrentCash.getCurrentPrincipalCash());
076                    // add the cash amount to market value
077                    subTotalKemidTotalMarketValue = subTotalKemidTotalMarketValue.add(kemidCurrentCash.getCurrentIncomeCash().bigDecimalValue()).add(kemidCurrentCash.getCurrentPrincipalCash().bigDecimalValue());
078                } else {
079                    trialBalanceReport.setInocmeCashBalance(KualiDecimal.ZERO);
080                    trialBalanceReport.setPrincipalcashBalance(KualiDecimal.ZERO);  
081                }
082                
083                // get available expendable funds
084                KEMIDCurrentAvailableBalance kemidCurrentAvailableBalance = (KEMIDCurrentAvailableBalance) businessObjectService.findByPrimaryKey(KEMIDCurrentAvailableBalance.class, primaryKeys);
085                if (ObjectUtils.isNotNull(kemidCurrentAvailableBalance)) {
086                    trialBalanceReport.setAvailableExpendableFunds(kemidCurrentAvailableBalance.getAvailableTotalCash());
087                } else {
088                    trialBalanceReport.setAvailableExpendableFunds(BigDecimal.ZERO);
089                }
090                
091                // get sub total market value, FY remainder estimated income                       
092                BigDecimal subTotalRemainderEstimatedIncome = BigDecimal.ZERO;            
093                List<CurrentTaxLotBalance> CurrentTaxLotBalanceRecords = (List<CurrentTaxLotBalance>) businessObjectService.findMatching(CurrentTaxLotBalance.class, primaryKeys);
094                if (ObjectUtils.isNotNull(CurrentTaxLotBalanceRecords)) {
095                    for (CurrentTaxLotBalance currentTaxLotBalance : CurrentTaxLotBalanceRecords) {
096                        subTotalKemidTotalMarketValue = subTotalKemidTotalMarketValue.add(currentTaxLotBalance.getHoldingMarketValue());
097                        subTotalRemainderEstimatedIncome = subTotalRemainderEstimatedIncome.add(currentTaxLotBalance.getRemainderOfFYEstimatedIncome());
098                        trialBalanceReport.setKemidTotalMarketValue(subTotalKemidTotalMarketValue);
099                        trialBalanceReport.setFyRemainderEstimatedIncome(subTotalRemainderEstimatedIncome);
100                    }                
101                }
102                
103                // add this new one
104                trialBalanceReportList.add(trialBalanceReport);
105            }
106            
107            return trialBalanceReportList;
108        }

110        /**
111         * 
112         * @see org.kuali.kfs.module.endow.report.service.TrialBalanceReportService#getTrialBalanceReportsByOtherCriteria(java.util.List, java.util.List, java.util.List, java.util.List, java.util.List, java.util.List, java.lang.String)
113         */
114        public List<TrialBalanceReportDataHolder> getTrialBalanceReportsByOtherCriteria(List<String> benefittingOrganziationCampusCodes, List<String> benefittingOrganziationChartCodes,
115                List<String> benefittingOrganziationCodes, List<String> typeCodes, List<String> purposeCodes, List<String> combineGroupCodes, String endowmnetOption, String closedIndicator) {
116    
117            List<String> kemids = getKemidsByOtherCriteria(benefittingOrganziationCampusCodes, benefittingOrganziationChartCodes,
118                    benefittingOrganziationCodes, typeCodes, purposeCodes, combineGroupCodes);
119            
120            // Now that we have all the kemids to be reported by the use of the other criteria, go to get the report data 
121            if (kemids.size() == 0) {
122                return null;
123            } else {            
124                return getTrialBalanceReportsByKemidByIds(kemids, endowmnetOption, closedIndicator);
125            }
126        }
