163        /**
164         *     
165         * @see org.kuali.kfs.module.endow.report.service.EndowmentReportService#createReportHeaderSheetData(java.util.List, java.util.List, java.util.List, java.util.List, java.util.List, java.util.List, java.util.List, java.lang.String, java.lang.String)
166         */
167        public EndowmentReportHeaderDataHolder createReportHeaderSheetData(
168                        List<String> kemidsSelected, 
169                        List<String> benefittingOrganziationCampuses,
170                        List<String> benefittingOrganziationCharts,
171                        List<String> benefittingOrganziations,
172                        List<String> typeCodes,
173                        List<String> purposeCodes,
174                        List<String> combineGroupCodes,
175                        String reportName, 
176                        String endowmnetOption,
177                        String reportOption) {
178                    
179            EndowmentReportHeaderDataHolder reportRequestHeaderDataHolder = new EndowmentReportHeaderDataHolder();
180            
181            // get request report
182            reportRequestHeaderDataHolder.setInstitutionName(getInstitutionName());
183            reportRequestHeaderDataHolder.setReportRequested(reportName);
184            reportRequestHeaderDataHolder.setRequestedBy(getReportRequestor());
185            String endowmentOptionDesc = "";
186            if (EndowConstants.EndowmentReport.BOTH.equalsIgnoreCase(endowmnetOption)) {
187                endowmentOptionDesc = EndowConstants.EndowmentReport.BOTH_ENDOWMENT_OPTION;
188            } else {
189                endowmentOptionDesc = EndowConstants.YES.equalsIgnoreCase(endowmnetOption) ? EndowConstants.EndowmentReport.ENDOWMENT : EndowConstants.EndowmentReport.NON_ENDOWED;
190            }
191            reportRequestHeaderDataHolder.setEndowmentOption(endowmentOptionDesc);
192            String reportOptionDesc = "";
193            if (reportOption != null) {
194                if (EndowConstants.EndowmentReport.BOTH.equalsIgnoreCase(reportOption)) {
195                    reportOptionDesc = EndowConstants.EndowmentReport.BOTH_REPORT_OPTION;
196                } else if (EndowConstants.EndowmentReport.DETAIL.equalsIgnoreCase(reportOption)) {
197                    reportOptionDesc = EndowConstants.EndowmentReport.DETAIL_REPORT; 
198                } else if (EndowConstants.EndowmentReport.TOTAL.equalsIgnoreCase(reportOption)) {
199                    reportOptionDesc = EndowConstants.EndowmentReport.TOTAL_REPORT;
200                }
201            }
202            reportRequestHeaderDataHolder.setReportOption(reportOptionDesc);
203            
204            // get criteria
205            reportRequestHeaderDataHolder.setBenefittingCampus(getBenefittingCampuses(benefittingOrganziationCampuses));
206            reportRequestHeaderDataHolder.setBenefittingChart(getBenefittingCharts(benefittingOrganziationCharts));
207            reportRequestHeaderDataHolder.setBenefittingOrganization(getBenefittingOrganizations(benefittingOrganziations));
208            reportRequestHeaderDataHolder.setKemidTypeCode(getKemidTypeCodes(typeCodes));
209            reportRequestHeaderDataHolder.setKemidPurposeCode(getKemidPurposeCodes(purposeCodes));
210            reportRequestHeaderDataHolder.setCombineGroupCode(getCombineGroupCodes(combineGroupCodes));
211            
212            // get kemids selected
213            reportRequestHeaderDataHolder.setKemidsSelected(kemidsSelected);
214    
215            // get kemdis with multiple benefittting organizations
216            reportRequestHeaderDataHolder.setKemidsWithMultipleBenefittingOrganizationsDataHolders(getKemidsWithMultipleBenefittingOrganizations(kemidsSelected));
217            
218            return reportRequestHeaderDataHolder;
219        }

221        /**
222         * Gets the kemids that meet the given conditions
223         * 
224         * @param benefittingOrganziationCampusCodes
225         * @param benefittingOrganziationChartCodes
226         * @param benefittingOrganziationCodes
227         * @param typeCodes
228         * @param purposeCodes
229         * @param combineGroupCodes
230         * @return
231         */
232        public List<String> getKemidsByOtherCriteria(List<String> benefittingOrganziationCampusCodes, List<String> benefittingOrganziationChartCodes,
233                List<String> benefittingOrganziationCodes, List<String> typeCodes, List<String> purposeCodes, List<String> combineGroupCodes) {
234          
235            // 4.1.9 - If any of the non-required criteria are left blank, ignore it as a criterion
236            // The criteria should be the AND combination.
237            
238            Set<String> kemids = new HashSet<String>();
239            
240            if (ObjectUtils.isNotNull(benefittingOrganziationCampusCodes) && !benefittingOrganziationCampusCodes.isEmpty()) {
241                retainCommonKemids(kemids, kemidBenefittingOrganizationDao.getKemidsByCampusCode(benefittingOrganziationCampusCodes));
242            }
243            if (ObjectUtils.isNotNull(benefittingOrganziationChartCodes) && !benefittingOrganziationChartCodes.isEmpty()) {
244                retainCommonKemids(kemids, kemidBenefittingOrganizationDao.getKemidsByAttribute(EndowPropertyConstants.KEMID_BENE_CHRT_CD, benefittingOrganziationChartCodes));
245            }
246            if (ObjectUtils.isNotNull(benefittingOrganziationCodes) && !benefittingOrganziationCodes.isEmpty()) {
247                retainCommonKemids(kemids, kemidBenefittingOrganizationDao.getKemidsByAttribute(EndowPropertyConstants.KEMID_BENE_ORG_CD, benefittingOrganziationCodes));
248            }
249            if (ObjectUtils.isNotNull(typeCodes) && !typeCodes.isEmpty()) {        
250                retainCommonKemids(kemids, kemidDao.getKemidsByAttribute(EndowPropertyConstants.KEMID_TYPE_CODE, typeCodes));
251            }
252            if (ObjectUtils.isNotNull(purposeCodes) && !purposeCodes.isEmpty()) {
253                retainCommonKemids(kemids, kemidDao.getKemidsByAttribute(EndowPropertyConstants.KEMID_PRPS_CD, purposeCodes));
254            }
255            if (ObjectUtils.isNotNull(combineGroupCodes) && !combineGroupCodes.isEmpty()) {        
256                retainCommonKemids(kemids, kemidReportGroupDao.getKemidsByAttribute(EndowPropertyConstants.KEMID_REPORT_GRP_CD, combineGroupCodes));
257            }        
258            
259            return new ArrayList<String>(kemids);
260        }

262        /** 
263         * Retains the kemids that are in common.  
264         * 
265         * @param kemids
266         * @param list
267         */
268        public void retainCommonKemids(Set<String> kemids, List<String> list) {
269            if (list != null && !list.isEmpty()) {
270                if (kemids.isEmpty()) {
271                    kemids.addAll(list);
272                } else {
273                    kemids.retainAll(list);
274                }
275            }
276        }

278        /**
279         * method to pickup all kemids based on user selection of endowmentOption and closed indicator
280         * 
281         * @param kemids
282         * @param endowmentOption
283         * @param endingDate
284         * @return List<String> newKemids
285         */
286        public List<String> getKemidsBasedOnUserSelection(List<String> kemids, String endowmentOption, String closedIndicator) {
287            List<String> newKemids = new ArrayList<String>();
288            
289            //get KEMID records for the given kemid and endowmentOption and closedIndicator..
290            List<KEMID> kemidRecords = kemidDao.getKemidRecordsByIds(kemids, endowmentOption, closedIndicator);
291            
292            for (KEMID kemidRecord : kemidRecords) {
293                if (!newKemids.contains(kemidRecord.getKemid())) {
294                    newKemids.add(kemidRecord.getKemid());
295                }
296            }
297            
298            return newKemids;
299        }

301        /**
302         * method to pickup all kemids from the list of kemids where for each kemid, if
303         * there is record in END_HIST_CSH_T table
304         * 
305         * @param kemids
306         * @param beginningDate
307         * @param endingDate
308         * @return List<String> newKemids
309         */
310        public List<String> getKemidsInHistoryCash(List<String> kemids, String beginningDate, String endingDate) {
311            List<String> newKemids = new ArrayList<String>();
312            List<KemidHistoricalCash> kemidHistoryCash = new ArrayList<KemidHistoricalCash>();
313            
314            MonthEndDate beginningMED = getPreviousMonthEndDate(convertStringToDate(beginningDate));
315            MonthEndDate endingMED = getMonthEndDate(convertStringToDate(endingDate));
316            
317            kemidHistoryCash = kemidHistoricalCashDao.getHistoricalCashRecords(kemids, beginningMED.getMonthEndDateId(), endingMED.getMonthEndDateId());
318    
319            for (KemidHistoricalCash historyCash : kemidHistoryCash) {
320                if (!newKemids.contains(historyCash.getKemid())) {
321                    newKemids.add(historyCash.getKemid());
322                }
323            }
324            
325            return newKemids;
326        }

438        /**
439         * Creates footer data
440         * 
441         * @param kemidOjb
442         * @return
443         */
444        protected EndowmentReportFooterDataHolder createFooterData(KEMID kemidOjb) {
445            
446            List<KemidBenefittingOrganization> benefittingOrganizationList = getKemidBenefittingOrganization(kemidOjb.getKemid());
447            EndowmentReportFooterDataHolder footerDataHolder = new EndowmentReportFooterDataHolder();
448            
449            footerDataHolder.setReference(kemidOjb.getKemid());
450            footerDataHolder.setEstablishedDate(getDateTimeService().toDateString(kemidOjb.getDateEstablished()));
451            footerDataHolder.setKemidType(kemidOjb.getType().getName());
452            footerDataHolder.setKemidPurpose(kemidOjb.getPurpose().getName());
453            footerDataHolder.setReportRunDate(getDateTimeService().toDateString(kemService.getCurrentDate()));
454            
455            if (benefittingOrganizationList != null) {
456                for (KemidBenefittingOrganization benefittingOrganization : benefittingOrganizationList) {
457                    BenefittingForFooter benefitting = footerDataHolder.createBenefittingForFooter();
458                    benefitting.setChartName(benefittingOrganization.getChart().getName());
459                    benefitting.setBenefittingPercent(benefittingOrganization.getBenefitPrecent().toString());
460                    Organization organization = getOrganization(benefittingOrganization.getBenefittingChartCode(), benefittingOrganization.getBenefittingOrgCode());
461                    // some organizations and campuses may not exist 
462                    if (ObjectUtils.isNotNull(organization)) {
463                        benefitting.setOrganizationName(organization.getOrganizationName());
464                        CampusImpl campus = getCampus(organization.getOrganizationPhysicalCampusCode());
465                        if (ObjectUtils.isNotNull(campus)) {
466                            benefitting.setCampusName(campus.getCampusName());
467                        } else {
468                            benefitting.setCampusName("");
469                        }
470                    } else {
471                        benefitting.setOrganizationName("");
472                    }                
473                }
474            }
475            
476            return footerDataHolder;
477        }
