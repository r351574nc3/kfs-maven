056        /**
057         * 
058         * @see org.kuali.kfs.module.endow.report.service.AssetStatementReportService#getAssetStatementReportsByKemidByIds(java.util.List, java.lang.String)
059         */
060        public List<AssetStatementReportDataHolder> getAssetStatementReportsByKemidByIds(List<String> kemids, String monthEndDate, String endowmentOption, String reportOption, String closedIndicator) {
061            
062            List<AssetStatementReportDataHolder> assetStatementReportList = null;
063    
064            // we need a valid month end date
065            Date date = convertStringToDate(monthEndDate);
066            if (ObjectUtils.isNull(date)) {
067                return null;
068            }
069    
070            // get valid kemids
071            List<String> kemidsSelected = kemidDao.getKemidsByAttributeWithEndowmentOption(EndowPropertyConstants.KEMID, kemids, endowmentOption, closedIndicator);
072            if (kemidsSelected == null || kemidsSelected.isEmpty()) {
073                return null;
074            }
075            // remove kemids that do not exist in historical cash
076            kemidsSelected = getKemidsInHistoryCash(kemidsSelected, monthEndDate);
077            
078            // get historical cash records
079            MonthEndDate endingMED = getMonthEndDate(convertStringToDate(monthEndDate));
080            KualiInteger medId = endingMED.getMonthEndDateId();
081            List<KemidHistoricalCash> historyCashRecords = getKemidHistoricalCashRecords(kemidsSelected, medId);
082            
083            if (KFSConstants.ParameterValues.YES.equalsIgnoreCase(endowmentOption)) {
084                // create a data holder for endowment
085                assetStatementReportList = createDataHoldersForEndowment(historyCashRecords, monthEndDate, medId, reportOption);            
086            } else if (KFSConstants.ParameterValues.NO.equalsIgnoreCase(endowmentOption)) {
087                // create a data holder for non-endowed
088                assetStatementReportList = createDataHoldersForNonEndowed(historyCashRecords, monthEndDate, medId, reportOption);
089            }    
090            return assetStatementReportList;      
091        }

093        /**
094         * Creates a collection of data holder for endowment
095         * 
096         * @param historyCashRecords
097         * @param monthEndDate
098         * @param medId
099         * @param reportOption
100         * @return
101         */
102        protected List<AssetStatementReportDataHolder> createDataHoldersForEndowment(List<KemidHistoricalCash> historyCashRecords, String monthEndDate, KualiInteger medId, String reportOption) {
103            
104            List<AssetStatementReportDataHolder> assetStatementReportList = new ArrayList<AssetStatementReportDataHolder>();
105            
106            for (KemidHistoricalCash historyCash : historyCashRecords) {
107                
108                KEMID kemidOjb = getKemid(historyCash.getKemid());
109                List<HoldingHistory> holdingHistoryRecordsForIncome = holdingHistoryDao.getHoldingHistoryByKemidIdAndMonthEndIdAndIpInd(historyCash.getKemid(), medId, IncomePrincipalIndicator.INCOME);
110                List<HoldingHistory> holdingHistoryRecordsForPrincipal = holdingHistoryDao.getHoldingHistoryByKemidIdAndMonthEndIdAndIpInd(historyCash.getKemid(), medId, IncomePrincipalIndicator.PRINCIPAL);
111                
112                // create a data holder
113                if ((holdingHistoryRecordsForIncome != null && !holdingHistoryRecordsForIncome.isEmpty()) || (holdingHistoryRecordsForPrincipal != null && !holdingHistoryRecordsForPrincipal.isEmpty())) {
114                    AssetStatementReportDataHolder dataHolder = new AssetStatementReportDataHolder();
115                    dataHolder.setInstitution(getInstitutionName());
116                    dataHolder.setKemid(historyCash.getKemid());
117                    dataHolder.setKemidLongTitle(kemidOjb.getLongTitle());
118                    dataHolder.setMonthEndDate(monthEndDate);
119                    
120                    dataHolder.setHistoryIncomeCash(historyCash.getHistoricalIncomeCash().bigDecimalValue());
121                    dataHolder.setHistoryPrincipalCash(historyCash.getHistoricalPrincipalCash().bigDecimalValue());
122                    
123                    // for income
124                    if (holdingHistoryRecordsForIncome != null) {
125                        for (HoldingHistory holdingHistory : holdingHistoryRecordsForIncome) {
126                            Security security = getSecurityById(holdingHistory.getSecurityId());
127                            SecurityReportingGroup securityReportingGroup = getSecurityReportingGroupBySecurityId(holdingHistory.getSecurityId());
128                            ReportGroupData rgIncome = dataHolder.createReportGroupData(securityReportingGroup, security, IncomePrincipalIndicator.INCOME);
129        
130                            rgIncome.addSumOfUnits(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_UNITS, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.INCOME));
131                            rgIncome.addSumOfMarketValue(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_MARKET_VALUE, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.INCOME));
132                            rgIncome.addSumOfEstimatedIncome(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.INCOME));
133                            rgIncome.addSumOfRemainderOfFYEstimated(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_REMAIDER_OF_FY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.INCOME));
134                            rgIncome.addSumOfNextFYEstimatedIncome(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_NEXT_FY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.INCOME));
135                        }
136                    }
137                    
138                    // for principal
139                    if (holdingHistoryRecordsForPrincipal != null) {
140                        for (HoldingHistory holdingHistory : holdingHistoryRecordsForPrincipal) {
141                            Security security = getSecurityById(holdingHistory.getSecurityId());
142                            SecurityReportingGroup securityReportingGroup = getSecurityReportingGroupBySecurityId(holdingHistory.getSecurityId());
143                            ReportGroupData rgPrincipal = dataHolder.createReportGroupData(securityReportingGroup, security, IncomePrincipalIndicator.PRINCIPAL);
144                        
145                            rgPrincipal.addSumOfUnits(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_UNITS, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.PRINCIPAL));
146                            rgPrincipal.addSumOfMarketValue(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_MARKET_VALUE, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.PRINCIPAL));
147                            rgPrincipal.addSumOfEstimatedIncome(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.PRINCIPAL));
148                            rgPrincipal.addSumOfRemainderOfFYEstimated(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_REMAIDER_OF_FY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.PRINCIPAL));
149                            rgPrincipal.addSumOfNextFYEstimatedIncome(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_NEXT_FY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), IncomePrincipalIndicator.PRINCIPAL));                
150                        }
151                    }
152                    
153                    // add footer data
154                    if (!reportOption.equalsIgnoreCase(EndowConstants.EndowmentReport.TOTAL)) {     
155                        dataHolder.setFooter(createFooterData(kemidOjb));
156                    }
157                    
158                    // add this new one
159                    assetStatementReportList.add(dataHolder);
160                }
161            }
162            
163            return assetStatementReportList;
164        }

166        /**
167         * Creates a collection of data holder for non-endowed
168         * 
169         * @param historyCashRecords
170         * @param monthEndDate
171         * @param medId
172         * @param reportOption
173         * @return
174         */
175        protected List<AssetStatementReportDataHolder> createDataHoldersForNonEndowed(List<KemidHistoricalCash> historyCashRecords, String monthEndDate, KualiInteger medId, String reportOption) {
176            
177            List<AssetStatementReportDataHolder> assetStatementReportList = new ArrayList<AssetStatementReportDataHolder>();
178            for (KemidHistoricalCash historyCash : historyCashRecords) {
179                
180                // get related objects
181                KEMID kemidOjb = getKemid(historyCash.getKemid());
182                List<HoldingHistory> holdingHistoryRecords = holdingHistoryDao.getHoldingHistoryByKemidIdAndMonthEndIdAndIpInd(historyCash.getKemid(), medId, "");
183                
184                // create a data holder
185                if (holdingHistoryRecords != null && !holdingHistoryRecords.isEmpty()) {
186                    // create a data holder
187                    AssetStatementReportDataHolder dataHolder = new AssetStatementReportDataHolder();
188                    dataHolder.setInstitution(getInstitutionName());
189                    dataHolder.setKemid(historyCash.getKemid());
190                    dataHolder.setKemidLongTitle(kemidOjb.getLongTitle());
191                    dataHolder.setMonthEndDate(monthEndDate);
192                    
193                    dataHolder.setHistoryIncomeCash(historyCash.getHistoricalIncomeCash().bigDecimalValue());
194                    dataHolder.setHistoryPrincipalCash(historyCash.getHistoricalPrincipalCash().bigDecimalValue());
195                    
196                    // use income for non-endowed
197                    for (HoldingHistory holdingHistory : holdingHistoryRecords) {
198                        Security security = getSecurityById(holdingHistory.getSecurityId());
199                        SecurityReportingGroup securityReportingGroup = getSecurityReportingGroupBySecurityId(holdingHistory.getSecurityId());
200                        ReportGroupData rgIncome = dataHolder.createReportGroupData(securityReportingGroup, security, IncomePrincipalIndicator.INCOME);
201    
202                        rgIncome.addSumOfUnits(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_UNITS, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), ""));
203                        rgIncome.addSumOfMarketValue(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_MARKET_VALUE, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), ""));
204                        rgIncome.addSumOfEstimatedIncome(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), ""));
205                        rgIncome.addSumOfRemainderOfFYEstimated(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_REMAIDER_OF_FY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), ""));
206                        rgIncome.addSumOfNextFYEstimatedIncome(holdingHistoryDao.getSumOfHoldginHistoryAttribute(EndowPropertyConstants.HOLDING_HISTORY_NEXT_FY_ESTIMATED_INCOME, kemidOjb.getKemid(), medId, holdingHistory.getSecurityId(), ""));
207                    }
208                    
209                    // add footer data
210                    if (!reportOption.equalsIgnoreCase(EndowConstants.EndowmentReport.TOTAL)) {
211                        dataHolder.setFooter(createFooterData(kemidOjb));
212                    }
213                    
214                    // add this new one
215                    assetStatementReportList.add(dataHolder);
216                }
217            }    
218            
219            return assetStatementReportList;
220        }

222        /**
223         * 
224         * @see org.kuali.kfs.module.endow.report.service.AssetStatementReportService#getAssetStatementReportsByOtherCriteria(java.util.List, java.util.List, java.util.List, java.util.List, java.util.List, java.util.List, java.lang.String)
225         */
226        public List<AssetStatementReportDataHolder> getAssetStatementReportsByOtherCriteria(List<String> benefittingOrganziationCampusCodes, List<String> benefittingOrganziationChartCodes,
227                List<String> benefittingOrganziationCodes, List<String> typeCodes, List<String> purposeCodes, List<String> combineGroupCodes, String monthEndDate, String endowmnetOption, 
228                String reportOption, String closedIndicator) {
229            
230            List<String> kemids = getKemidsByOtherCriteria(benefittingOrganziationCampusCodes, benefittingOrganziationChartCodes,
231                    benefittingOrganziationCodes, typeCodes, purposeCodes, combineGroupCodes);         
232            
233            // Now that we have all the kemids to be reported by the use of the other criteria, go to get the report data 
234            if (kemids.size() == 0) {
235                return null;
236            } else {            
237                return getAssetStatementReportsByKemidByIds(kemids, monthEndDate, endowmnetOption, reportOption, closedIndicator);
238            }
239        }

243         * 
244         * @param securityId
245         * @return
246         */
247        protected SecurityReportingGroup getSecurityReportingGroupBySecurityId(String securityId) {
248            // get Security
249            Security security = getSecurityById(securityId);
250            // get class code
251            ClassCode classCode = getClassCodeById(security.getSecurityClassCode());
252            // get security report group
253            SecurityReportingGroup securityReportingGroup = getSecurityReportingGroup(classCode.getSecurityReportingGrp());
254            
255            return securityReportingGroup;
