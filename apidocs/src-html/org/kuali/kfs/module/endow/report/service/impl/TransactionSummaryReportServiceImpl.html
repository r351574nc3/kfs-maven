061        /**
062         * 
063         * @see org.kuali.kfs.module.endow.report.service.TransactionStatementReportService#getTransactionStatementReportsByKemidByIds(java.util.List, java.lang.String)
064         */
065        public List<TransactionSummaryReportDataHolder> getTransactionSummaryReportsByKemidByIds(List<String> kemids, String beginningDate, String endingDate, String endowmentOption, String closedIndicator, String reportOption) {
066            
067            List<TransactionSummaryReportDataHolder> transactionSummaryReportList = new ArrayList<TransactionSummaryReportDataHolder>();
068    
069            Date beginDate = convertStringToDate(beginningDate);
070            Date endDate = convertStringToDate(endingDate);        
071            if (ObjectUtils.isNull(beginDate) || ObjectUtils.isNull(beginDate)) {
072                return null;
073            }
074            //gather kemids based on the user selection of endowmentOption and closed indicators...
075            kemids = getKemidsBasedOnUserSelection(kemids, endowmentOption, closedIndicator);
076            
077            if (kemids.size() == 0) {
078                return null;
079            }
080            
081            //eliminate kemids that are not in 
082            kemids = getKemidsInHistoryCash(kemids, beginningDate, endingDate);
083            
084            if (kemids.size() == 0) {
085                return null;
086            }
087            
088            MonthEndDate beginningMED = getPreviousMonthEndDate(convertStringToDate(beginningDate));
089            MonthEndDate endingMED = getMonthEndDate(convertStringToDate(endingDate));
090            
091            for (String kemid : kemids) {
092                TransactionSummaryReportDataHolder transactionSummaryReportDataHolder = new TransactionSummaryReportDataHolder();
093                
094                List<HoldingHistory> holdingHistoryList = holdingHistoryDao.getHoldingHistoryByKemid(kemid);
095    
096                getHistoryCashAmounts(transactionSummaryReportDataHolder, kemid, beginningMED.getMonthEndDateId(), endingMED.getMonthEndDateId());
097                
098                for (HoldingHistory holdingHistory : holdingHistoryList) {
099                    //if month end beginning date
100                    if (holdingHistory.getMonthEndDateId().equals(beginningMED.getMonthEndDateId())) {
101                        if (holdingHistory.getIncomePrincipalIndicator().equalsIgnoreCase(EndowConstants.IncomePrincipalIndicator.INCOME)) {
102                            transactionSummaryReportDataHolder.setIncomeBeginningMarketValue(transactionSummaryReportDataHolder.getIncomeBeginningMarketValue().add(holdingHistory.getMarketValue()));
103                        }
104                        if (holdingHistory.getIncomePrincipalIndicator().equalsIgnoreCase(EndowConstants.IncomePrincipalIndicator.PRINCIPAL)) {
105                            transactionSummaryReportDataHolder.setPrincipalBeginningMarketValue(transactionSummaryReportDataHolder.getPrincipalBeginningMarketValue().add(holdingHistory.getMarketValue()));
106                        }
107                    }
108                    // if month end ending date ..
109                    if (holdingHistory.getMonthEndDateId().equals(endingMED.getMonthEndDateId())) {
110                        if (holdingHistory.getIncomePrincipalIndicator().equalsIgnoreCase(EndowConstants.IncomePrincipalIndicator.INCOME)) {
111                            transactionSummaryReportDataHolder.setIncomeEndingMarketValue(transactionSummaryReportDataHolder.getIncomeEndingMarketValue().add(holdingHistory.getMarketValue()));
112                        }
113                        if (holdingHistory.getIncomePrincipalIndicator().equalsIgnoreCase(EndowConstants.IncomePrincipalIndicator.PRINCIPAL)) {
114                            transactionSummaryReportDataHolder.setPrincipalEndingMarketValue(transactionSummaryReportDataHolder.getPrincipalEndingMarketValue().add(holdingHistory.getMarketValue()));
115                        }
116                    }
117                    
118                    //start calculating change in market value. It will also be calculated during transaction archive records calculations
119                    transactionSummaryReportDataHolder.setIncomeChangeInMarketValue(transactionSummaryReportDataHolder.getIncomeEndingMarketValue().subtract(transactionSummaryReportDataHolder.getIncomeBeginningMarketValue()));
120                    transactionSummaryReportDataHolder.setPrincipalChangeInMarketValue(transactionSummaryReportDataHolder.getPrincipalEndingMarketValue().subtract(transactionSummaryReportDataHolder.getPrincipalBeginningMarketValue()));
121                    
122                    transactionSummaryReportDataHolder.setNext12MonthsEstimatedIncome(transactionSummaryReportDataHolder.getNext12MonthsEstimatedIncome().add(holdingHistory.getEstimatedIncome()));
123                    transactionSummaryReportDataHolder.setRemainderOfFYEstimatedIncome(transactionSummaryReportDataHolder.getRemainderOfFYEstimatedIncome().add(holdingHistory.getRemainderOfFYEstimatedIncome()));
124                    transactionSummaryReportDataHolder.setNextFYEstimatedIncome(transactionSummaryReportDataHolder.getNextFYEstimatedIncome().add(holdingHistory.getNextFYEstimatedIncome()));
125                }
126                
127                //get the transaction archive records now....
128                List<TransactionArchive> transactionArchives = transactionArchiveDao.getTransactionArchivesByKemid(kemid, beginDate, endDate);
129    
130                for (TransactionArchive transactionArchive : transactionArchives) {
131                    transactionArchive.refreshNonUpdateableReferences();
132                    
133                    if (EndowConstants.DocumentTypeNames.ENDOWMENT_CASH_TRANSFER.equalsIgnoreCase(transactionArchive.getTypeCode()) ||
134                            EndowConstants.DocumentTypeNames.ENDOWMENT_SECURITY_TRANSFER.equalsIgnoreCase(transactionArchive.getTypeCode())) {
135                        //cash transfers...
136                        if (EndowConstants.DocumentTypeNames.ENDOWMENT_CASH_TRANSFER.equalsIgnoreCase(transactionArchive.getTypeCode())) {
137                            getTransactionArchiveTotalsForCashTransfers(transactionSummaryReportDataHolder, transactionArchive);
138                        }
139                        //security tranfers....
140                        if (EndowConstants.DocumentTypeNames.ENDOWMENT_SECURITY_TRANSFER.equalsIgnoreCase(transactionArchive.getTypeCode())) {
141                            getTransactionArchiveTotalsForSecurityTransfers(transactionSummaryReportDataHolder, transactionArchive);
142                        }
143                    } 
144                    else {
145                        //income type code...
146                        if (transactionArchive.getEtranObj().getEndowmentTransactionTypeCode().equalsIgnoreCase(EndowmentTransactionTypeCodes.INCOME_TYPE_CODE)) {
147                            getTransactionArchiveTotalsForIncomeTypeCode(transactionSummaryReportDataHolder, transactionArchive);
148                        }    
149                        //expense type code...
150                        if (transactionArchive.getEtranObj().getEndowmentTransactionTypeCode().equalsIgnoreCase(EndowmentTransactionTypeCodes.EXPENSE_TYPE_CODE)) {
151                            getTransactionArchiveTotalsForExpenseTypeCode(transactionSummaryReportDataHolder, transactionArchive);
152                        }
153                    }
154                }
155                
156                //combine the similar etrancode records in the list for
157                //contributions, expenses, cash transfers, and security transfers....
158                if (transactionSummaryReportDataHolder.getReportGroupsForContributions().size() > 0) {
159                    combineContributionsData(transactionSummaryReportDataHolder);
160                }
161                
162                if (transactionSummaryReportDataHolder.getReportGroupsForExpenses().size() > 0) {
163                    combineExpensesData(transactionSummaryReportDataHolder);
164                }
165                
166                if (transactionSummaryReportDataHolder.getReportGroupsForCashTransfers().size() > 0) {
167                    combineCashTransfersData(transactionSummaryReportDataHolder);
168                }
169                
170                if (transactionSummaryReportDataHolder.getReportGroupsForSecurityTransfers().size() > 0) {
171                    combineSecurityTransfersData(transactionSummaryReportDataHolder);
172                }
173                
174                //setup the report header information....
175                transactionSummaryReportDataHolder.setInstitution(getInstitutionName());
176                transactionSummaryReportDataHolder.setKemid(kemid);
177                
178                KEMID kemidObj = getKemid(kemid);
179                
180                transactionSummaryReportDataHolder.setKemidLongTitle(kemidObj.getLongTitle());
181                transactionSummaryReportDataHolder.setBeginningDate(beginningDate);
182                transactionSummaryReportDataHolder.setEndingDate(endingDate);
183                
184                //setup the report footer information...Only for Detail report option
185                if (!reportOption.equalsIgnoreCase(EndowConstants.EndowmentReport.TOTAL)) {
186                    transactionSummaryReportDataHolder.setFooter(createFooterData(kemidObj));
187                }
188    
189                // add this new one
190                transactionSummaryReportList.add(transactionSummaryReportDataHolder);
191            }
192            
193            return transactionSummaryReportList;
194        }

196        /**
197         * Method to retrieve the records from END_HIST_CSH_T table for the given
198         * kemid and retrieve the the income and principal cash amounts.
199         * @param kemid
200         * @param historyIncomeCash
201         * @param historyPrincipalCash
202         * @param beginningMED
203         * @param endingMED
204         */
205        protected void getHistoryCashAmounts(TransactionSummaryReportDataHolder transactionSummaryReportDataHolder, String kemid, KualiInteger beginningMed, KualiInteger endingMed) {
206            List<String> kemids = new ArrayList<String>();
207            kemids.add(kemid);
208            
209            List<KemidHistoricalCash> kemidHistoryCash = kemidHistoricalCashDao.getHistoricalCashRecords(kemids, beginningMed, endingMed);
210            
211            for (KemidHistoricalCash historyCash : kemidHistoryCash) {
212                if (historyCash.getMonthEndDateId().equals(beginningMed)) {
213                    transactionSummaryReportDataHolder.setIncomeBeginningMarketValue(historyCash.getHistoricalIncomeCash().bigDecimalValue());
214                    transactionSummaryReportDataHolder.setPrincipalBeginningMarketValue(historyCash.getHistoricalPrincipalCash().bigDecimalValue());
215                }
216                if (historyCash.getMonthEndDateId().equals(endingMed)) {
217                    transactionSummaryReportDataHolder.setIncomeEndingMarketValue(historyCash.getHistoricalIncomeCash().bigDecimalValue());
218                    transactionSummaryReportDataHolder.setPrincipalEndingMarketValue(historyCash.getHistoricalPrincipalCash().bigDecimalValue());
219                }
220            }
221        }

241        /**
242         * Method to look into each transaction archive record and gather income or principal amounts.
243         * 
244         * @param transactionSummaryReportDataHolder
245         * @param transactionArchive
246         */
247        protected void getTransactionArchiveTotalsForIncomeTypeCode(TransactionSummaryReportDataHolder transactionSummaryReportDataHolder, TransactionArchive transactionArchive) {
248            transactionArchive.refreshNonUpdateableReferences();
249            
250            BigDecimal transactionSecurityCostIncome = BigDecimal.ZERO;
251            BigDecimal transactionSecurityCostPrincipal = BigDecimal.ZERO;
252            
253            //get the transaction archive data totals....
254            List<TransactionArchiveSecurity> archiveSecurities = transactionArchive.getArchiveSecurities();
255            for (TransactionArchiveSecurity archiveSecurity : archiveSecurities) {
256                if (transactionArchive.getIncomePrincipalIndicatorCode().equalsIgnoreCase(EndowConstants.IncomePrincipalIndicator.INCOME)) {
257                    transactionSecurityCostIncome = transactionSecurityCostIncome.add(archiveSecurity.getHoldingCost());
258                }
259                if (transactionArchive.getIncomePrincipalIndicatorCode().equalsIgnoreCase(EndowConstants.IncomePrincipalIndicator.PRINCIPAL)) {
260                    transactionSecurityCostPrincipal = transactionSecurityCostPrincipal.add(archiveSecurity.getHoldingCost());
261                }
262            }
263            
264            ContributionsDataHolder contributionsData = transactionSummaryReportDataHolder.new ContributionsDataHolder();
265            contributionsData.setContributionsDescription(transactionArchive.getEtranObj().getName());
266            
267            //get the transaction archive data totals....
268         //   List<TransactionArchiveSecurity> archiveSecurities = transactionArchive.getArchiveSecurities();
269         //   for (TransactionArchiveSecurity archiveSecurity : archiveSecurities) {
270         //       transactionSecurityCost = transactionSecurityCost.add(archiveSecurity.getHoldingCost());
271         //   }
272            //income type code...
273            if (transactionArchive.getEtranObj().getEndowmentTransactionTypeCode().equalsIgnoreCase(EndowmentTransactionTypeCodes.INCOME_TYPE_CODE)) {
274                if (transactionArchive.getIncomePrincipalIndicatorCode().equalsIgnoreCase(EndowConstants.IncomePrincipalIndicator.INCOME)) {
275                    contributionsData.setIncomeContributions(transactionSecurityCostIncome.add(contributionsData.getIncomeContributions().add(transactionArchive.getIncomeCashAmount())));
276                    transactionSummaryReportDataHolder.setIncomeChangeInMarketValue(transactionSummaryReportDataHolder.getIncomeChangeInMarketValue().subtract(contributionsData.getIncomeContributions()));
277                }
278                
279                if (transactionArchive.getIncomePrincipalIndicatorCode().equalsIgnoreCase(EndowConstants.IncomePrincipalIndicator.PRINCIPAL)) {
280                    contributionsData.setPrincipalContributions(transactionSecurityCostPrincipal.add(contributionsData.getPrincipalContributions().add(transactionArchive.getPrincipalCashAmount())));
281                    transactionSummaryReportDataHolder.setPrincipalChangeInMarketValue(transactionSummaryReportDataHolder.getPrincipalChangeInMarketValue().subtract(contributionsData.getPrincipalContributions()));
282                }
283            }
284            
285            transactionSummaryReportDataHolder.getReportGroupsForContributions().add(contributionsData);
286        }

391        /**
392         * Helper method to first retrieve all the contributions records and go over all of them
393         * to combine based on same description of the contributions.  The combined records are then
394         * added to the data holder.
395         * 
396         * @param transactionSummaryReportDataHolder
397         */
398        protected void combineContributionsData(TransactionSummaryReportDataHolder transactionSummaryReportDataHolder) {
399            String contributionsDescription = null;
400            BigDecimal incomeContributionsAmount = BigDecimal.ZERO;
401            BigDecimal PrincipalContributionsAmount = BigDecimal.ZERO;
402            
403            List<ContributionsDataHolder> contributionsDataList = new ArrayList<ContributionsDataHolder>();
404            
405            ContributionsDataHolder combinedContributionsData = transactionSummaryReportDataHolder.new ContributionsDataHolder();
406            
407            List<ContributionsDataHolder> contributionsData = transactionSummaryReportDataHolder.getReportGroupsForContributions();
408    
409            for (ContributionsDataHolder contributionData : contributionsData) {
410                if (contributionsDescription == null) {
411                    contributionsDescription = contributionData.getContributionsDescription();
412                }
413                //same description == same etran code so combine the totals.....
414                if (contributionsDescription.equals(contributionData.getContributionsDescription())) {
415                    incomeContributionsAmount = incomeContributionsAmount.add(contributionData.getIncomeContributions());
416                    PrincipalContributionsAmount = PrincipalContributionsAmount.add(contributionData.getPrincipalContributions());
417                }
418                else { //write out the record....
419                    combinedContributionsData.setContributionsDescription(contributionsDescription);
420                    combinedContributionsData.setIncomeContributions(incomeContributionsAmount);
421                    combinedContributionsData.setPrincipalContributions(PrincipalContributionsAmount);
422                    transactionSummaryReportDataHolder.setIncomeChangeInMarketValue(transactionSummaryReportDataHolder.getIncomeChangeInMarketValue().subtract(incomeContributionsAmount));
423                    transactionSummaryReportDataHolder.setPrincipalChangeInMarketValue(transactionSummaryReportDataHolder.getPrincipalChangeInMarketValue().subtract(PrincipalContributionsAmount));
424                    contributionsDataList.add(combinedContributionsData);
425                    
426                    //create a new holder...
427                    incomeContributionsAmount = BigDecimal.ZERO;
428                    PrincipalContributionsAmount = BigDecimal.ZERO;
429                    combinedContributionsData = transactionSummaryReportDataHolder.new ContributionsDataHolder();                
430                    contributionsDescription = contributionData.getContributionsDescription();
431                    incomeContributionsAmount = incomeContributionsAmount.add(contributionData.getIncomeContributions());
432                    PrincipalContributionsAmount = PrincipalContributionsAmount.add(contributionData.getPrincipalContributions());
433                }
434            }
435            
436            //add the last data holder....
437            combinedContributionsData.setContributionsDescription(contributionsDescription);
438            combinedContributionsData.setIncomeContributions(incomeContributionsAmount);
439            combinedContributionsData.setPrincipalContributions(PrincipalContributionsAmount);
440            transactionSummaryReportDataHolder.setIncomeChangeInMarketValue(transactionSummaryReportDataHolder.getIncomeChangeInMarketValue().subtract(incomeContributionsAmount));
441            transactionSummaryReportDataHolder.setPrincipalChangeInMarketValue(transactionSummaryReportDataHolder.getPrincipalChangeInMarketValue().subtract(PrincipalContributionsAmount));
442            contributionsDataList.add(combinedContributionsData);
443    
444            //now remove the current list of contributions and add the newly created combined list.
445            transactionSummaryReportDataHolder.getReportGroupsForContributions().removeAll(contributionsData);
446            transactionSummaryReportDataHolder.getReportGroupsForContributions().addAll(contributionsDataList);
447        }

467            for (ExpensesDataHolder expenseData : expensesData) {

469                    expensesDescription = expenseData.getExpensesDescription();
470                }
471                //same description == same etran code so combine the totals.....
472                if (expensesDescription.equals(expenseData.getExpensesDescription())) {
473                    incomeExpensesAmount = incomeExpensesAmount.add(expenseData.getIncomeExpenses());
474                    PrincipalExpensesAmount = PrincipalExpensesAmount.add(expenseData.getPrincipalExpenses());
475                }
476                else { //write out the record....
477                    combinedExpensesData.setExpensesDescription(expensesDescription);
478                    combinedExpensesData.setIncomeExpenses(incomeExpensesAmount);
479                    combinedExpensesData.setPrincipalExpenses(PrincipalExpensesAmount);
480                    transactionSummaryReportDataHolder.setIncomeChangeInMarketValue(transactionSummaryReportDataHolder.getIncomeChangeInMarketValue().subtract(incomeExpensesAmount));
481                    transactionSummaryReportDataHolder.setPrincipalChangeInMarketValue(transactionSummaryReportDataHolder.getPrincipalChangeInMarketValue().subtract(PrincipalExpensesAmount));
482                    expensesDataList.add(combinedExpensesData);
483                    
484                    //create a new holder...
485                    incomeExpensesAmount = BigDecimal.ZERO;
486                    PrincipalExpensesAmount = BigDecimal.ZERO;
487                    combinedExpensesData = transactionSummaryReportDataHolder.new ExpensesDataHolder();                
488                    expensesDescription = expenseData.getExpensesDescription();
489                    incomeExpensesAmount = incomeExpensesAmount.add(expenseData.getIncomeExpenses());
490                    PrincipalExpensesAmount = PrincipalExpensesAmount.add(expenseData.getPrincipalExpenses());
491                }
492            }
493            
494            //add the last data holder....
495            combinedExpensesData.setExpensesDescription(expensesDescription);
496            combinedExpensesData.setIncomeExpenses(incomeExpensesAmount);
497            combinedExpensesData.setPrincipalExpenses(PrincipalExpensesAmount);
498            transactionSummaryReportDataHolder.setIncomeChangeInMarketValue(transactionSummaryReportDataHolder.getIncomeChangeInMarketValue().subtract(incomeExpensesAmount));
499            transactionSummaryReportDataHolder.setPrincipalChangeInMarketValue(transactionSummaryReportDataHolder.getPrincipalChangeInMarketValue().subtract(PrincipalExpensesAmount));
500            expensesDataList.add(combinedExpensesData);
501    
502            //now remove the current list of contributions and add the newly created combined list.
503            transactionSummaryReportDataHolder.getReportGroupsForExpenses().removeAll(expensesData);
504            transactionSummaryReportDataHolder.getReportGroupsForExpenses().addAll(expensesDataList);
505        }
