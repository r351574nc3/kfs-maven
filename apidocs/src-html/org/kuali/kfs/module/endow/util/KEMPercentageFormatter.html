016    package org.kuali.kfs.module.endow.util;
017    
018    import java.math.BigDecimal;
019    import java.text.DecimalFormat;
020    import java.text.NumberFormat;
021    import java.text.ParseException;
022    
023    import org.kuali.rice.kns.util.KualiDecimal;
024    import org.kuali.rice.kns.util.KualiPercent;
025    import org.kuali.rice.kns.util.RiceKeyConstants;
026    import org.kuali.rice.kns.web.format.Formatter;
027    import org.kuali.rice.kns.web.format.FormatException;
028    
029    /**
030     * in DB, the data of the percent is stored in the format x.xxxx
031     * which is always not bigger than 1.0000 and not less than 0.0000
032     * For the display, we'd like to have xxx.xx%
033     * 
034     */
035    public class KEMPercentageFormatter extends Formatter {
036     // begin Kuali Foundation modification
037     // ToDo: Borrow the code from PercentageFormatter, shall I change
038     // the value of serialVersionUID?
039     private static final long serialVersionUID = 1323889942436009589L;
040     // end Kuali Foundation modification
041    
042     /**
043      * The default scale for percentage values
044      */
045     public final static int PERCENTAGE_SCALE = 2;
046    
047     /**
048      * The default format for percentage values
049      */
050     public final static String PERCENTAGE_FORMAT = "0.####%";
051    
052     // begin Kuali Foundation modification
053     // removed PARSE_MSG
054     // end Kuali Foundation modification
055     
056     /**
057      * Unformats its argument and returns a BigDecimal instance initialized with the resulting string value
058      * 
059      * @return a BigDecimal initialized with the provided string
060      */
061     protected Object convertToObject(String target) {
062          try {
063             DecimalFormat formatter = new DecimalFormat(PERCENTAGE_FORMAT);
064             
065             Number parsedNumber = formatter.parse(target);
066             System.out.println(">>parsedNumber ="+parsedNumber);
067             return new KualiPercent(parsedNumber.doubleValue());
068    
069         }
070         catch (NumberFormatException e) {
071             throw new FormatException("parsing", RiceKeyConstants.ERROR_PERCENTAGE, target, e);
072         }
073         catch (ParseException e) {
074             throw new FormatException("parsing", RiceKeyConstants.ERROR_PERCENTAGE, target, e);
075         }
076    
077     }
078    
079     /**
080      * Returns a string representation of its argument, formatted as a percentage value.
081      * 
082      * @return a formatted String
083      */
084     public Object format(Object value) {
085         if (value == null)
086             return "N/A";
087    
088         if (value instanceof KualiDecimal) {
089              value = ((KualiDecimal)value).bigDecimalValue();
090              DecimalFormat df = new DecimalFormat ( PERCENTAGE_FORMAT ) ;
091              String theFormattedValue = df.format(((BigDecimal) value).doubleValue());
092              System.out.println(">>theFormattedValue="+theFormattedValue);
093              return theFormattedValue;
094         }        
095         else
096              throw new RuntimeException("Error: the date type is not KualiDecimal");     
097         
098             
099     }
100    }
