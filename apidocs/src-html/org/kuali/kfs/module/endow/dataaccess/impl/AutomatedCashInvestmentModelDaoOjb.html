016    package org.kuali.kfs.module.endow.dataaccess.impl;
017    
018    import java.sql.Date;
019    import java.util.List;
020    
021    import org.apache.ojb.broker.query.Criteria;
022    import org.apache.ojb.broker.query.QueryFactory;
023    import org.kuali.kfs.module.endow.EndowPropertyConstants;
024    import org.kuali.kfs.module.endow.businessobject.AutomatedCashInvestmentModel;
025    import org.kuali.kfs.module.endow.dataaccess.AutomatedCashInvestmentModelDao;
026    import org.kuali.rice.kns.dao.impl.PlatformAwareDaoBaseOjb;
027    
028    
029    public class AutomatedCashInvestmentModelDaoOjb extends PlatformAwareDaoBaseOjb implements AutomatedCashInvestmentModelDao {
030            
031        /**
032         * @see org.kuali.kfs.module.endow.dataaccess.AutomatedCashInvestmentModelDao#getAutomatedCashInvestmentModelWithNextPayDateEqualToCurrentDate()
033         */
034        public List<AutomatedCashInvestmentModel> getAutomatedCashInvestmentModelWithNextPayDateEqualToCurrentDate(Date currentDate) {
035            Criteria criteria = new Criteria();
036            criteria.addEqualTo(EndowPropertyConstants.ACI_MODEL_NEXT_DUE_DATE, currentDate);
037            criteria.addNotNull(EndowPropertyConstants.ACI_MODEL_FREQUENCY_CDOE);
038            criteria.addEqualTo(EndowPropertyConstants.ACI_MODEL_ACTIVE_INDICATOR, "Y");
039            return (List<AutomatedCashInvestmentModel>) getPersistenceBrokerTemplate().getCollectionByQuery(QueryFactory.newQuery(AutomatedCashInvestmentModel.class, criteria));
040        }
041    
042    }
