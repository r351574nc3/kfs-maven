024    public interface CurrentTaxLotBalanceDao {

026        /**
027         * Gets all records for the Security in END_CURR_TAX_LOT_BAL_T.

041        /**
042         * Calculates the total Holding market value based on FEE_BAL_TYP_CD = CMV
043         * @param feeMethod feeMethod object
044         * @return totalHoldingMarketValue
045         */
046        public BigDecimal getCurrentTaxLotBalanceTotalHoldingMarketValue(FeeMethod feeMethod);
047        
048    }
