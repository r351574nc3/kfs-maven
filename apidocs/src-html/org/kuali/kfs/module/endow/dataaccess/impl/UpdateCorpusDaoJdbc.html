060        /**
061         * 
062         * @see org.kuali.kfs.module.endow.dataaccess.UpdateCorpusDao#updateCorpusAmountsFromTransactionArchive()
063         */
064        public void updateCorpusAmountsFromTransactionArchive(Date currentDate){
065    
066            LOG.info("updateCorpusAmountsFromTransactionArchive() started");
067            
068            Map<String, Object> currentCorpus = null;
069            BigDecimal currentCorpusAmount = null;
070            Date tomrrowsDate = getNextDay(currentDate);
071    
072            try {
073                String selectTransactionArchiveSql = "SELECT TRAN_KEMID, SUM(TRAN_CORPUS_AMT) TRAN_CORPUS_AMT FROM END_TRAN_ARCHV_T WHERE TRAN_CORPUS_IND = ? AND (TRAN_PSTD_DT >= ? AND TRAN_PSTD_DT < ?) GROUP BY TRAN_KEMID";
074                
075                String selectCurrentCorpusSql = "SELECT CRNT_CORPUS_VAL FROM END_CRNT_ENDOW_CORPUS_T WHERE KEMID = ?";
076                String updateCurrentCorpusSql = "UPDATE END_CRNT_ENDOW_CORPUS_T SET CRNT_CORPUS_VAL = ? WHERE KEMID = ?";
077                String insertCurrentCorpusSql = "INSERT INTO END_CRNT_ENDOW_CORPUS_T (KEMID, CRNT_CORPUS_VAL, CRNT_PRIN_MVAL, PRIOR_FY_CORPUS_VAL, PRIOR_FY_PRIN_MVAL, VER_NBR, OBJ_ID) VALUES (?, ?, 0, 0, 0, 1, ?)";
078                            
079                String updateKemIdCorpusSql = "UPDATE END_KEMID_CORPUS_VAL_T SET CRNT_CORPUS_VAL = ? WHERE KEMID = ?";            
080                
081                //Summarize transaction archive records by KemID on current date for CorpusAmount.
082                SqlRowSet pendingEntryRowSet = getJdbcTemplate().queryForRowSet(selectTransactionArchiveSql, new Object[] { EndowConstants.YES, currentDate, tomrrowsDate});
083    
084                int updateCount = 0;
085                int insertCount = 0;
086                
087                //Loop over transaction archive
088                while (pendingEntryRowSet.next()) {
089                    
090                    //store kem id and transaction corpus amount
091                    String kemId = pendingEntryRowSet.getString(EndowPropertyConstants.ColumnNames.UpdateCorpus.TRANSACTION_ARCHIVE_KEMID);
092                    BigDecimal corpusAmount = pendingEntryRowSet.getBigDecimal(EndowPropertyConstants.ColumnNames.UpdateCorpus.TRANSACTION_CORPUS_AMT);
093                                    
094                    try {
095                        //Try and get an Endow Current Transaction
096                        currentCorpus = getSimpleJdbcTemplate().queryForMap(selectCurrentCorpusSql, kemId);
097                    }
098                    catch (IncorrectResultSizeDataAccessException ex) {
099                        if (ex.getActualSize() != 0) {
100                            LOG.error("current corpus sql returned more than one row, aborting", ex);
101                            throw ex;
102                        }
103                        // no rows returned - that's ok
104                    }
105    
106                    if (currentCorpus != null) {
107                        //if exists update Current Corpus (trans amt + current amt)
108                        updateCount++;
109    
110                        currentCorpusAmount = (BigDecimal) currentCorpus.get(EndowPropertyConstants.ColumnNames.UpdateCorpus.CRNT_ENDOW_CORPUS_VAL);
111                        currentCorpusAmount = currentCorpusAmount.add(corpusAmount);
112                        
113                        // A current corpus exists, so we need to update it
114                        getSimpleJdbcTemplate().update(updateCurrentCorpusSql, currentCorpusAmount, kemId);
115                        
116                    } else {
117                         //if doesn't exist insert Current Corpus(trans amt)
118                        insertCount++;
119    
120                        currentCorpusAmount = new BigDecimal("0");
121                        currentCorpusAmount = currentCorpusAmount.add(corpusAmount);
122    
123                        // No current corpus exists, so we need to insert one
124                        getSimpleJdbcTemplate().update(insertCurrentCorpusSql, kemId, currentCorpusAmount, new Guid().toString());
125                    }
126                    
127                    // Now update KEM ID Corpus table
128                    getSimpleJdbcTemplate().update(updateKemIdCorpusSql, currentCorpusAmount, kemId);
129    
130                }
131                
132                LOG.info ("updateCorpusAmountsFromTransactionArchive has inserted Corpus Amount values for " + insertCount + " records in the END_CRNT_ENDOW_CORPUS_T table.");
133                LOG.info ("updateCorpusAmountsFromTransactionArchive has updated Corpus Amount values for " + updateCount + " records in the END_CRNT_ENDOW_CORPUS_T table.");            
134                LOG.info ("updateCorpusAmountsFromTransactionArchive has updated Corpus Amount values for " + (updateCount + insertCount) + " records in the END_KEMID_CORPUS_VAL_T table.");
135    
136            }
137            catch (Exception e) {
138                LOG.error("updateCorpusAmountsFromTransactionArchive() Exception running sql", e);
139                throw new RuntimeException("Unable to execute: " + e.getMessage(), e);
140            }        
141            
142            LOG.info("updateCorpusAmountsFromTransactionArchive() completed");
143        }

145        /**
146         * 
147         * @see org.kuali.kfs.module.endow.dataaccess.UpdateCorpusDao#updateKEMIDCorpusPrincipalMarketValue()
148         */
149        public void updateKemIdCorpusPrincipalMarketValue(){
150         
151            LOG.info("updateKemIdCorpusPrincipalMarketValue() started");
152            
153            try{
154                String selectCurrentBalanceSql = "select t2.KEMID, t2.PRIN_AT_MARKET from END_KEMID_CORPUS_VAL_T t1, END_KEMID_CRNT_BAL_V t2 where t1.KEMID = t2.KEMID";
155                String updateCurrentCorpusSql = "UPDATE END_KEMID_CORPUS_VAL_T SET CRNT_PRIN_MVAL = ? WHERE KEMID = ?";
156                
157                SqlRowSet pendingEntryRowSet = getJdbcTemplate().queryForRowSet(selectCurrentBalanceSql);
158    
159                int count = 0;
160                String kemId = null; 
161                BigDecimal principalAtMarket = null;
162                
163                while (pendingEntryRowSet.next()) {
164                    
165                    //store kem id and prin market
166                    kemId = pendingEntryRowSet.getString(EndowPropertyConstants.ColumnNames.UpdateCorpus.CRNT_BALANCE_KEMID);
167                    principalAtMarket = pendingEntryRowSet.getBigDecimal(EndowPropertyConstants.ColumnNames.UpdateCorpus.CRNT_BALANCE_PRINCIPAL_AT_MARKET);
168    
169                    if(ObjectUtils.isNull(principalAtMarket)){
170                        principalAtMarket = new BigDecimal("0");
171                    }
172                    
173                    // Update current corpus with new prin market value
174                    getSimpleJdbcTemplate().update(updateCurrentCorpusSql, principalAtMarket, kemId);
175                    
176                    count++;
177                }
178                
179                LOG.info ("updateKemIdCorpusPrincipalMarketValue has updated Principal Market Value for " + count + " records in the END_KEMID_CORPUS_VAL_T table.");
180            }
181            catch (Exception e) {
182                LOG.error("updateKemIdCorpusPrincipalMarketValue() Exception running sql", e);
183                throw new RuntimeException("Unable to execute: " + e.getMessage(), e);
184            }        
185                
186            LOG.info("updateKemIdCorpusPrincipalMarketValue() completed");
187        }

215        /**
216         * Returns the next day
217         * 
218         * @param currentDate
219         * @return
220         */
221        protected Date getNextDay(Date currentDate){
222            
223            Calendar calendar = Calendar.getInstance();
224            calendar.setTime(currentDate);             
225            calendar.add(Calendar.DAY_OF_YEAR, 1);
226            
227            return new java.sql.Date(calendar.getTime().getTime());
228        }
