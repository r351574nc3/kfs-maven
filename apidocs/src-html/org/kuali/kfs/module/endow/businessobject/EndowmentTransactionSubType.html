016    package org.kuali.kfs.module.endow.businessobject;
017    
018    import java.util.LinkedHashMap;
019    
020    import org.apache.commons.lang.StringUtils;
021    import org.kuali.kfs.sys.KFSConstants;
022    import org.kuali.rice.kns.bo.KualiCodeBase;
023    
024    /**
025     * Business Object for the Endowment Transaction Sub Type table.
026     */
027    public class EndowmentTransactionSubType extends KualiCodeBase {
028    
029        /**
030         * @see org.kuali.rice.kns.bo.BusinessObjectBase#toStringMapper()
031         */
032        @Override
033        protected LinkedHashMap toStringMapper() {
034            LinkedHashMap m = new LinkedHashMap();
035            m.put(KFSConstants.GENERIC_CODE_PROPERTY_NAME, this.code);
036            return m;
037        }
038    
039        /**
040         * @see org.kuali.rice.kns.bo.KualiCodeBase#getCodeAndDescription()
041         */
042        @Override
043        public String getCodeAndDescription() {
044            if (StringUtils.isEmpty(code)) {
045                return KFSConstants.EMPTY_STRING;
046            }
047            return super.getCodeAndDescription();
048        }
049    }
