016    package org.kuali.kfs.module.endow.businessobject.options;
017    
018    import java.util.ArrayList;
019    import java.util.Collection;
020    import java.util.Iterator;
021    import java.util.List;
022    
023    import org.kuali.kfs.module.endow.businessobject.FeeRateDefinitionCode;
024    import org.kuali.kfs.sys.context.SpringContext;
025    import org.kuali.rice.core.util.KeyLabelPair;
026    import org.kuali.rice.kns.lookup.keyvalues.KeyValuesBase;
027    import org.kuali.rice.kns.service.KeyValuesService;
028    
029    public class FeeRateDefinitionCodeValuesFinder extends KeyValuesBase {
030        
031        /**
032         * @see org.kuali.rice.kns.lookup.keyvalues.KeyValuesFinder#getKeyValues()
033         */
034        public List<KeyLabelPair> getKeyValues() {
035    
036            KeyValuesService boService = SpringContext.getBean(KeyValuesService.class);
037            Collection<FeeRateDefinitionCode> codes = boService.findAll(FeeRateDefinitionCode.class);
038            List<KeyLabelPair> labels = new ArrayList<KeyLabelPair>();
039            
040            labels.add(new KeyLabelPair("", ""));
041            for (Iterator<FeeRateDefinitionCode> iter = codes.iterator(); iter.hasNext();) {
042                FeeRateDefinitionCode feeRateDefinitionCode = (FeeRateDefinitionCode) iter.next();
043                labels.add(new KeyLabelPair(feeRateDefinitionCode.getCode(), feeRateDefinitionCode.getCodeAndDescription()));
044            }
045    
046            return labels;
047        }
048    
049    }
