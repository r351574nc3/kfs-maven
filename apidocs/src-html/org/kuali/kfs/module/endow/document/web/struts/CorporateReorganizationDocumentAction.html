057         */
058        @Override
059        protected void updateTransactionLineTaxLots(boolean isUpdate, boolean isSource, EndowmentTransactionLinesDocument etlDocument, EndowmentTransactionLine transLine) {
060            CorporateReorganizationDocument corporateReorganizationDocument = (CorporateReorganizationDocument) etlDocument;
061    
062            if (transLine instanceof EndowmentSourceTransactionLine) {
063                UpdateTaxLotsBasedOnAccMethodAndTransSubtypeService taxLotsService = SpringContext.getBean(UpdateTaxLotsBasedOnAccMethodAndTransSubtypeService.class);
064                taxLotsService.updateTransactionLineTaxLots(isUpdate, corporateReorganizationDocument, transLine);
065            }
066    
067            if (transLine instanceof EndowmentTargetTransactionLine) {
068                UpdateSecurityTransferTargetTaxLotsService taxLotsService = SpringContext.getBean(UpdateSecurityTransferTargetTaxLotsService.class);
069                taxLotsService.updateTransactionLineTaxLots(corporateReorganizationDocument, transLine);
070            }
071    
072        }

078        @Override
079        public ActionForward deleteSourceTransactionLine(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
080    
081            ActionForward actionForward = super.deleteSourceTransactionLine(mapping, form, request, response);
082            EndowmentTransactionLinesDocumentFormBase etlForm = (EndowmentTransactionLinesDocumentFormBase) form;
083    
084            // clear the current TargetLine
085            ((EndowmentTransactionLinesDocumentFormBase) form).setNewTargetTransactionLine(new EndowmentTargetTransactionLine());
086            
087            // delete target transaction lines as well
088            List<EndowmentTransactionLine> targetTransactionLines = etlForm.getEndowmentTransactionLinesDocumentBase().getTargetTransactionLines();
089            if (targetTransactionLines != null && targetTransactionLines.size() > 0) {
090                for (int i = 0; i < targetTransactionLines.size(); i++) {
091                    deleteTransactionLine(false, etlForm, i);
092                }
093            }
094    
095            return actionForward;
096        }

100         * @see org.kuali.kfs.module.endow.document.web.struts.EndowmentTaxLotLinesDocumentActionBase#deleteSourceTaxLotLine(org.apache.struts.action.ActionMapping, org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
101         */
102        @Override
103        public ActionForward deleteSourceTaxLotLine(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
104    
105            ActionForward actionForward = super.deleteSourceTaxLotLine(mapping, form, request, response);
106            EndowmentTransactionLinesDocumentFormBase documentForm = (EndowmentTransactionLinesDocumentFormBase) form;
107            EndowmentTransactionLinesDocument endowmentDocument = (EndowmentTransactionLinesDocument) documentForm.getDocument();
108    
109            boolean rulePassed = true;
110    
111            // if the document has target transaction lines update the related tax lots
112            if (endowmentDocument.getTargetTransactionLines() != null && endowmentDocument.getTargetTransactionLines().size() > 0) {
113                for (int i = 0; i < endowmentDocument.getTargetTransactionLines().size(); i++) {
114    
115                    EndowmentTransactionLine transLine = endowmentDocument.getTargetTransactionLines().get(i);
116    
117                    // check any business rules
118                    rulePassed &= SpringContext.getBean(KualiRuleService.class).applyRules(new RefreshTransactionLineEvent(EXISTING_SOURCE_TRAN_LINE_PROPERTY_NAME, endowmentDocument, transLine, i));
119    
120                    if (rulePassed) {
121                        updateTransactionLineTaxLots(true, true, endowmentDocument, transLine);
122                    }
123    
124                    if (endowmentDocument instanceof AmountTotaling)
125                        ((FinancialSystemDocumentHeader) documentForm.getDocument().getDocumentHeader()).setFinancialDocumentTotalAmount(((AmountTotaling) endowmentDocument).getTotalDollarAmount());
126                }
127    
128            }
129    
130            return actionForward;
131        }

134        /**
135         * @see org.kuali.kfs.module.endow.document.web.struts.EndowmentTransactionLinesDocumentActionBase#insertSourceTransactionLine(org.apache.struts.action.ActionMapping, org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
136         */
137        @Override
138        public ActionForward insertSourceTransactionLine(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
139            
140            EndowmentTransactionLinesDocumentFormBase documentForm = (EndowmentTransactionLinesDocumentFormBase) form;
141            EndowmentTransactionLinesDocument endowmentDocument = (EndowmentTransactionLinesDocument) documentForm.getDocument();
142    
143            EndowmentSourceTransactionLine sourceTransLine = (EndowmentSourceTransactionLine) documentForm.getNewSourceTransactionLine();
144            EndowmentTargetTransactionLine targetTransLine = (EndowmentTargetTransactionLine) documentForm.getNewTargetTransactionLine();
145            
146            boolean rulePassed = true;
147    
148            // check any business rules
149            rulePassed &= SpringContext.getBean(KualiRuleService.class).applyRules(new AddTransactionLineEvent(NEW_SOURCE_TRAN_LINE_PROPERTY_NAME, endowmentDocument, sourceTransLine));
150    
151            if (rulePassed) {
152                
153                EndowmentSecurityDetailsDocumentBase securityDocument = (EndowmentSecurityDetailsDocumentBase)endowmentDocument;
154                EndowmentTransactionSecurity sourceTranSecurity = securityDocument.getSourceTransactionSecurity();
155                
156                // Get the etran from the Security BO class code.
157                Security sourceSecurity = (Security) SpringContext.getBean(SecurityService.class).getByPrimaryKey(sourceTranSecurity.getSecurityID());
158                String sourceEtran = ObjectUtils.isNotNull(sourceSecurity) ? sourceSecurity.getClassCode().getSecurityEndowmentTransactionCode() : "";
159                
160                // Add the transaction line.
161                sourceTransLine.setEtranCode(sourceEtran);
162                insertTransactionLine(true,  documentForm, sourceTransLine);
163                
164                fillInTargetTransactionLine(targetTransLine, sourceTransLine);
165                documentForm.setNewTargetTransactionLine(targetTransLine);
166                
167                // Clear the used newTargetLine
168                documentForm.setNewSourceTransactionLine(new EndowmentSourceTransactionLine());
169                
170            }
171    
172            return mapping.findForward(KFSConstants.MAPPING_BASIC);
173        }

217        /**
218         * @see org.kuali.kfs.module.endow.document.web.struts.EndowmentTaxLotLinesDocumentActionBase#refreshSourceTaxLots(org.apache.struts.action.ActionMapping, org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
219         */
220        @Override
221        public ActionForward refreshSourceTaxLots(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
222    
223            EndowmentTransactionLinesDocumentFormBase documentForm = (EndowmentTransactionLinesDocumentFormBase) form;
224            EndowmentTransactionLinesDocument endowmentDocument = (EndowmentTransactionLinesDocument) documentForm.getDocument();
225            int selectedLine = this.getSelectedLine(request);
226            
227            EndowmentTransactionLine sourceTransLine = endowmentDocument.getSourceTransactionLines().get(selectedLine);
228    
229            boolean sourceRulePassed = true;
230            
231            // Check any business rules
232            sourceRulePassed &= SpringContext.getBean(KualiRuleService.class).applyRules(new RefreshTransactionLineEvent(EXISTING_SOURCE_TRAN_LINE_PROPERTY_NAME, endowmentDocument, sourceTransLine, selectedLine));
233            if (sourceRulePassed) {
234                
235                // Verify that the source security etran code hasen't changed. 
236                // If it has changed, then update the source transaction line.
237                String securityEtranCode = getSecurityEtranCode(true, form);
238                if (!securityEtranCode.equals(sourceTransLine.getEtranCode())) {
239                    sourceTransLine.setEtranCode(securityEtranCode);
240                }
241                
242                updateTransactionLineTaxLots(false, true,  endowmentDocument, sourceTransLine);
243                
244                List<EndowmentTransactionLine> targetTransactionLines = documentForm.getEndowmentTransactionLinesDocumentBase().getTargetTransactionLines();
245                
246                // If there are no target transaction lines, then we know that nothing
247                // has been added, therefore just update the new line portion; otherwise,
248                // update the line in the list.
249                EndowmentTransactionLine targetTransLine = null;
250                if (!targetTransactionLines.isEmpty()) {
251                    targetTransLine = endowmentDocument.getTargetTransactionLines().get(selectedLine);
252                    fillInTargetTransactionLine((EndowmentTargetTransactionLine)targetTransLine, (EndowmentSourceTransactionLine)sourceTransLine);
253                    updateTransactionLineTaxLots(false, false, endowmentDocument, targetTransLine);
254                }
255                else {
256                    targetTransLine = documentForm.getNewTargetTransactionLine();
257                    fillInTargetTransactionLine((EndowmentTargetTransactionLine)targetTransLine, (EndowmentSourceTransactionLine)sourceTransLine);
258                    documentForm.setNewTargetTransactionLine(targetTransLine);
259                }
260            }
261    
262            if (endowmentDocument instanceof AmountTotaling)
263                ((FinancialSystemDocumentHeader) documentForm.getDocument().getDocumentHeader()).setFinancialDocumentTotalAmount(((AmountTotaling) endowmentDocument).getTotalDollarAmount());
264    
265            return mapping.findForward(KFSConstants.MAPPING_BASIC);
266        }

304        private void fillInTargetTransactionLine(EndowmentTargetTransactionLine targetTransLine, EndowmentSourceTransactionLine sourceTransLine) {
305            // Copy relevant source fields over to the target fields.
306            targetTransLine.setTransactionIPIndicatorCode(sourceTransLine.getTransactionIPIndicatorCode());       
307            targetTransLine.setTransactionAmount(sourceTransLine.getTransactionAmount());
308            targetTransLine.setKemid(sourceTransLine.getKemid());
309        }

311        private String getSecurityEtranCode(boolean isSource, ActionForm form) {
312            
313            String etranCode = "";
314            
315            EndowmentTransactionLinesDocumentFormBase documentForm = (EndowmentTransactionLinesDocumentFormBase) form;
316            EndowmentTransactionLinesDocument endowmentDocument = (EndowmentTransactionLinesDocument) documentForm.getDocument();
317            
318            if (isSource) {
319                EndowmentSecurityDetailsDocumentBase securityDocument = (EndowmentSecurityDetailsDocumentBase)endowmentDocument;
320                EndowmentTransactionSecurity sourceTranSecurity = securityDocument.getSourceTransactionSecurity();
321                
322                // Get the etran from the Security BO class code.
323                Security sourceSecurity = (Security) SpringContext.getBean(SecurityService.class).getByPrimaryKey(sourceTranSecurity.getSecurityID());
324                etranCode = ObjectUtils.isNotNull(sourceSecurity) ? sourceSecurity.getClassCode().getSecurityEndowmentTransactionCode() : "";
325                
326            }
327            else {
328                EndowmentSecurityDetailsDocumentBase securityDocument = (EndowmentSecurityDetailsDocumentBase)endowmentDocument;
329                EndowmentTransactionSecurity targetTranSecurity = securityDocument.getTargetTransactionSecurity();
330                
331                // Get the etran from the Security BO class code.
332                Security targetSecurity = (Security) SpringContext.getBean(SecurityService.class).getByPrimaryKey(targetTranSecurity.getSecurityID());
333                etranCode = ObjectUtils.isNotNull(targetSecurity) ? targetSecurity.getClassCode().getSecurityEndowmentTransactionCode() : "";
334            }
335            
336            return etranCode;
337        }
