053        /**
054         * @see org.kuali.kfs.module.endow.document.service.UpdateTaxLotsBasedOnAccMethodAndTransSubtypeService#updateTransactionLineTaxLots(boolean,
055         *      org.kuali.kfs.module.endow.document.EndowmentTaxLotLinesDocument,
056         *      org.kuali.kfs.module.endow.businessobject.EndowmentTransactionLine)
057         */
058        public void updateTransactionLineTaxLots(boolean isUpdate, EndowmentTaxLotLinesDocument endowmentTaxLotLinesDocument, EndowmentTransactionLine transLine) {
059    
060            EndowmentTransactionSecurity endowmentTransactionSecurity = endowmentTaxLotLinesDocument.getSourceTransactionSecurity();
061            String accountingMethod = parameterService.getParameterValue(KfsParameterConstants.ENDOWMENT_ALL.class, EndowParameterKeyConstants.TAX_LOTS_ACCOUNTING_METHOD);
062            Security security = securityService.getByPrimaryKey(endowmentTransactionSecurity.getSecurityID());
063    
064            if (ObjectUtils.isNotNull(security)) {
065                // In the Asset Decrease document, if transaction sub type is non-cash and IF the user inserts an amount in the
066                // transaction amount, that amount along with the units sill be used to generate the transaction tax lot lines. The
067                // amount will not be calculated by the addition of the transaction line. In the event the user has entered an amount in
068                // the transaction amount field, that amount along with the units will be spread proportionately among the tax lots.
069                if (endowmentTaxLotLinesDocument instanceof AssetDecreaseDocument && EndowConstants.TransactionSubTypeCode.NON_CASH.equalsIgnoreCase(endowmentTaxLotLinesDocument.getTransactionSubTypeCode()) && transLine.getTransactionAmount() != null && transLine.getTransactionAmount().bigDecimalValue().compareTo(BigDecimal.ZERO) != 0) {
070                    updateTaxLotsForSubTypeNonCashAndTransAmtNonZero(isUpdate, endowmentTransactionSecurity, transLine);
071                }
072                else {
073    
074                    if (EndowConstants.TaxLotsAccountingMethodOptions.AVERAGE_BALANCE.equalsIgnoreCase(accountingMethod) || (EndowConstants.TaxLotsAccountingMethodOptions.FIFO.equalsIgnoreCase(accountingMethod) && !security.getClassCode().isTaxLotIndicator())) {
075                        if (EndowConstants.TransactionSubTypeCode.CASH.equalsIgnoreCase(endowmentTaxLotLinesDocument.getTransactionSubTypeCode())) {
076                            updateTaxLotsForAccountingMethodAverageBalance(true, isUpdate, endowmentTransactionSecurity, transLine);
077                        }
078                        if (EndowConstants.TransactionSubTypeCode.NON_CASH.equalsIgnoreCase(endowmentTaxLotLinesDocument.getTransactionSubTypeCode())) {
079                            updateTaxLotsForAccountingMethodAverageBalance(false, isUpdate, endowmentTransactionSecurity, transLine);
080                            setTransactionLineTotal(transLine);
081                        }
082                    }
083    
084                    if ((EndowConstants.TaxLotsAccountingMethodOptions.FIFO.equalsIgnoreCase(accountingMethod) || EndowConstants.TaxLotsAccountingMethodOptions.LIFO.equalsIgnoreCase(accountingMethod)) && security.getClassCode().isTaxLotIndicator()) {
085                        boolean isFIFO = EndowConstants.TaxLotsAccountingMethodOptions.FIFO.equalsIgnoreCase(accountingMethod);
086    
087                        if (EndowConstants.TransactionSubTypeCode.CASH.equalsIgnoreCase(endowmentTaxLotLinesDocument.getTransactionSubTypeCode())) {
088                            updateTaxLotsForAccountingMethodFIFOorLIFO(true, isUpdate, isFIFO, endowmentTransactionSecurity, transLine);
089                        }
090                        if (EndowConstants.TransactionSubTypeCode.NON_CASH.equalsIgnoreCase(endowmentTaxLotLinesDocument.getTransactionSubTypeCode())) {
091                            updateTaxLotsForAccountingMethodFIFOorLIFO(false, isUpdate, isFIFO, endowmentTransactionSecurity, transLine);
092                            setTransactionLineTotal(transLine);
093                        }
094                    }
095                }
096            }

100        /**
101         * Updates the tax lots for the transaction line in the case the accounting method is Average Balance.
102         * 
103         * @param isSubTypeCash
104         * @param endowmentTransactionSecurity
105         * @param transLine
106         */
107        private void updateTaxLotsForAccountingMethodAverageBalance(boolean isSubTypeCash, boolean isUpdate, EndowmentTransactionSecurity endowmentTransactionSecurity, EndowmentTransactionLine transLine) {
108    
109            BigDecimal transactionUnits = transLine.getTransactionUnits().bigDecimalValue();
110            BigDecimal totalTaxLotsUnits = BigDecimal.ZERO;
111            BigDecimal transactionAmount = BigDecimal.ZERO;
112            BigDecimal perUnitValue = BigDecimal.ZERO;
113    
114            List<HoldingTaxLot> holdingTaxLots = new ArrayList<HoldingTaxLot>();
115            Map<Integer, HoldingTaxLot> lotsMap = new HashMap<Integer, HoldingTaxLot>();
116    
117            if (!isUpdate) {
118                transLine.getTaxLotLines().clear();
119                holdingTaxLots = taxLotService.getAllTaxLots(transLine.getKemid(), endowmentTransactionSecurity.getSecurityID(), endowmentTransactionSecurity.getRegistrationCode(), transLine.getTransactionIPIndicatorCode());
120            }
121            else {
122                List<EndowmentTransactionTaxLotLine> existingTransactionLines = transLine.getTaxLotLines();
123                for (EndowmentTransactionTaxLotLine endowmentTransactionTaxLotLine : existingTransactionLines) {
124                    HoldingTaxLot holdingTaxLot = taxLotService.getByPrimaryKey(transLine.getKemid(), endowmentTransactionSecurity.getSecurityID(), endowmentTransactionSecurity.getRegistrationCode(), endowmentTransactionTaxLotLine.getTransactionHoldingLotNumber(), transLine.getTransactionIPIndicatorCode());
125    
126                    if (ObjectUtils.isNotNull(holdingTaxLot)) {
127                        holdingTaxLots.add(holdingTaxLot);
128                    }
129                }
130    
131                transLine.getTaxLotLines().clear();
132            }
133    
134            Map<KualiInteger, EndowmentTransactionTaxLotLine> decreaseHoldingTaxLots = new HashMap<KualiInteger, EndowmentTransactionTaxLotLine>();
135            if (isSubTypeCash) {
136                transactionAmount = transLine.getTransactionAmount().bigDecimalValue();
137                // 1. Calculate per unit value
138                perUnitValue = KEMCalculationRoundingHelper.divide(transactionAmount, transactionUnits, 5);
139            }
140    
141            if (holdingTaxLots != null && holdingTaxLots.size() > 0) {
142                boolean keepIntegers = true;
143                // compute the total number of units for tax lots
144                for (HoldingTaxLot holdingTaxLot : holdingTaxLots) {
145                    totalTaxLotsUnits = totalTaxLotsUnits.add(holdingTaxLot.getUnits());
146    
147                    // 3. Calculate the number of units to be transacted in each lot
148                    // check if percentage and tax lot units are integers
149                    BigDecimal lotUnits = BigDecimal.ZERO;
150                    try {
151                        int lotUnitsInt = holdingTaxLot.getUnits().intValueExact();
152                    }
153                    catch (ArithmeticException ex) {
154                        keepIntegers = false;
155    
156                    }
157    
158                }
159                // 2. Calculate percentage each lot contains of the total units
160                BigDecimal percentage = KEMCalculationRoundingHelper.divide(transLine.getTransactionUnits().bigDecimalValue(), totalTaxLotsUnits, 5);
161    
162    
163                for (HoldingTaxLot holdingTaxLot : holdingTaxLots) {
164                    EndowmentTransactionTaxLotLine taxLotLine = new EndowmentTransactionTaxLotLine();
165                    taxLotLine.setDocumentLineNumber(transLine.getTransactionLineNumber());
166    
167                    BigDecimal lotUnits = BigDecimal.ZERO;
168    
169                    lotUnits = KEMCalculationRoundingHelper.multiply(percentage, holdingTaxLot.getUnits(), 5);
170    
171                    // IF all original units per lot are integers (no decimal values), the result is rounded to the nearest
172                    // integer and stored with the five decimals as zero. If the original units are not all integers, then the
173                    // value is rounded to five decimals and stored as the five decimal values.
174                    if (keepIntegers) {
175                        lotUnits = lotUnits.setScale(0, BigDecimal.ROUND_HALF_UP);
176                        lotUnits = lotUnits.setScale(5);
177                    }
178                    taxLotLine.setLotUnits(lotUnits);
179    
180                    // 5. Calculate original unit value for each tax lot
181                    BigDecimal originalUnitValue = KEMCalculationRoundingHelper.divide(holdingTaxLot.getCost(), holdingTaxLot.getUnits(), 5);
182    
183                    // 6. Calculate original cost
184                    BigDecimal originalCost = KEMCalculationRoundingHelper.multiply(lotUnits, originalUnitValue, 2);
185                    taxLotLine.setLotHoldingCost(originalCost);
186    
187                    if (isSubTypeCash) {
188    
189                        // 4. Calculate the value received for units sold in each tax lot
190                        BigDecimal valueReceived = KEMCalculationRoundingHelper.multiply(lotUnits, perUnitValue, 2);
191    
192                        // 7. Calculate Gain or loss
193                        calculateGainLoss(holdingTaxLot, taxLotLine, valueReceived, originalCost);
194                    }
195    
196                    // set tax lot line lot number and acquired date
197                    taxLotLine.setTransactionHoldingLotNumber(holdingTaxLot.getLotNumber().intValue());
198                    taxLotLine.setKemid(transLine.getKemid());
199                    taxLotLine.setSecurityID(holdingTaxLot.getSecurityId());
200                    taxLotLine.setRegistrationCode(holdingTaxLot.getRegistrationCode());
201                    taxLotLine.setIpIndicator(holdingTaxLot.getIncomePrincipalIndicator());
202                    taxLotLine.setLotAcquiredDate(holdingTaxLot.getAcquiredDate());
203    
204                    // set the new lot indicator
205                    taxLotLine.setNewLotIndicator(false);
206    
207                    // add the new tax lot line to the transaction line tax lots
208                    addTaxLotLine(transLine, taxLotLine);
209                    lotsMap.put(taxLotLine.getTransactionHoldingLotNumber(), holdingTaxLot);
210                }
211    
212                // Adjust the number of units if the total is different from the transaction line units
213                adjustUnitsNumberAndAmountsForAverageBalance(lotsMap, transLine, keepIntegers, isSubTypeCash, perUnitValue);
214    
215            }
216        }

218        /**
219         * Calculates Gain or Loss for the tax lot line and determines if it's long term or short term
220         * 
221         * @param holdingTaxLot
222         * @param taxLotLine
223         * @param valueReceived
224         * @param originalCost
225         */
226        private void calculateGainLoss(HoldingTaxLot holdingTaxLot, EndowmentTransactionTaxLotLine taxLotLine, BigDecimal valueReceived, BigDecimal originalCost) {
227            BigDecimal gainOrLoss = valueReceived.subtract(originalCost);
228            gainOrLoss = gainOrLoss.setScale(2, BigDecimal.ROUND_HALF_UP);
229    
230            // Determine if short or long term gain/loss
231            Date currentDate = kemService.getCurrentDate();
232            Date acquiredDate = holdingTaxLot.getAcquiredDate();
233    
234            Calendar calendarAcquiredDate = Calendar.getInstance();
235            calendarAcquiredDate.setTime(acquiredDate);
236            calendarAcquiredDate.add(Calendar.MONTH, EndowConstants.SHORT_VS_LONG_TERM_PERIOD);
237    
238            if (calendarAcquiredDate.getTime().before(currentDate)) {
239                // long term gain/loss
240                taxLotLine.setLotLongTermGainLoss(gainOrLoss);
241            }
242            // short term gain/loss
243            else {
244                taxLotLine.setLotShortTermGainLoss(gainOrLoss);
245            }
246        }

249        /**
250         * Adjusts the number of units if the total is different from the transaction line units.
251         * 
252         * @param lotsMap
253         * @param transLine
254         * @param keepIntegers
255         * @param isSubTypeCash
256         * @param perUnitValue
257         */
258        private void adjustUnitsNumberAndAmountsForAverageBalance(Map<Integer, HoldingTaxLot> lotsMap, EndowmentTransactionLine transLine, boolean keepIntegers, boolean isSubTypeCash, BigDecimal perUnitValue) {
259            // Adjust the number of units if the total is different from the transaction line units
260            BigDecimal totalComputedTaxLotUnits = BigDecimal.ZERO;
261            BigDecimal totalComputedCost = BigDecimal.ZERO;
262            EndowmentTransactionTaxLotLine oldestTaxLotLine = null;
263    
264            if (transLine.getTaxLotLines() != null && transLine.getTaxLotLines().size() > 0) {
265                for (EndowmentTransactionTaxLotLine taxLotLine : transLine.getTaxLotLines()) {
266                    BigDecimal lotUnits = taxLotLine.getLotUnits().negate();
267    
268                    // calculate the total number of units to be decreased
269                    totalComputedTaxLotUnits = totalComputedTaxLotUnits.add(lotUnits);
270                    totalComputedCost = totalComputedCost.add(taxLotLine.getLotHoldingCost().negate());
271    
272                    if (taxLotLine.getLotShortTermGainLoss() != null) {
273                        totalComputedCost = totalComputedCost.add(taxLotLine.getLotShortTermGainLoss());
274                    }
275    
276                    if (taxLotLine.getLotLongTermGainLoss() != null) {
277                        totalComputedCost = totalComputedCost.add(taxLotLine.getLotLongTermGainLoss());
278                    }
279    
280                    // keep the tax lot with the oldest acquired date so that we can adjust the units for that one in case the
281                    // number of units needs and adjustment
282                    if (oldestTaxLotLine != null) {
283                        if (oldestTaxLotLine.getLotAcquiredDate().after(taxLotLine.getLotAcquiredDate())) {
284                            oldestTaxLotLine = taxLotLine;
285                        }
286                    }
287                    else {
288                        oldestTaxLotLine = taxLotLine;
289                    }
290                }
291            }
292    
293            // compare with the negated number of units on the transaction line because the units on the tax lots have been negated
294            if (totalComputedTaxLotUnits.compareTo(transLine.getTransactionUnits().bigDecimalValue().negate()) != 0) {
295                BigDecimal difUnits = transLine.getTransactionUnits().bigDecimalValue().subtract(totalComputedTaxLotUnits);
296                oldestTaxLotLine.setLotUnits(oldestTaxLotLine.getLotUnits().add(difUnits.negate()));
297                oldestTaxLotLine.setLotUnits(oldestTaxLotLine.getLotUnits().negate());
298    
299                if (isSubTypeCash) {
300                    // update totalComputedCost
301                    totalComputedCost = totalComputedCost.subtract(oldestTaxLotLine.getLotHoldingCost().negate());
302                    if (oldestTaxLotLine.getLotShortTermGainLoss() != null) {
303                        totalComputedCost = totalComputedCost.subtract(oldestTaxLotLine.getLotShortTermGainLoss());
304                    }
305    
306                    if (oldestTaxLotLine.getLotLongTermGainLoss() != null) {
307                        totalComputedCost = totalComputedCost.subtract(oldestTaxLotLine.getLotLongTermGainLoss());
308                    }
309                }
310    
311                HoldingTaxLot holdingTaxLot = lotsMap.get(oldestTaxLotLine.getTransactionHoldingLotNumber());
312                BigDecimal originalUnitValue = KEMCalculationRoundingHelper.divide(holdingTaxLot.getCost(), holdingTaxLot.getUnits(), 5);
313                BigDecimal originalCost = KEMCalculationRoundingHelper.multiply(oldestTaxLotLine.getLotUnits(), originalUnitValue, 2);
314                oldestTaxLotLine.setLotHoldingCost(originalCost);
315    
316                if (isSubTypeCash) {
317                    // 4. Calculate the value received for units sold in each tax lot
318                    BigDecimal valueReceived = KEMCalculationRoundingHelper.multiply(oldestTaxLotLine.getLotUnits(), perUnitValue, 2);
319    
320                    // 7. Calculate Gain or loss
321                    calculateGainLoss(holdingTaxLot, oldestTaxLotLine, valueReceived, originalCost);
322                    // update totalComputedCost
323                    totalComputedCost = totalComputedCost.add(oldestTaxLotLine.getLotHoldingCost());
324                    if (oldestTaxLotLine.getLotShortTermGainLoss() != null) {
325                        totalComputedCost = totalComputedCost.add(oldestTaxLotLine.getLotShortTermGainLoss());
326                    }
327    
328                    if (oldestTaxLotLine.getLotLongTermGainLoss() != null) {
329                        totalComputedCost = totalComputedCost.add(oldestTaxLotLine.getLotLongTermGainLoss());
330                    }
331                }
332                oldestTaxLotLine.setLotHoldingCost(oldestTaxLotLine.getLotHoldingCost().negate());
333                oldestTaxLotLine.setLotUnits(oldestTaxLotLine.getLotUnits().negate());
334    
335            }
336    
337            if (isSubTypeCash) {
338                // compare total computed cost with the transaction line amount
339                if (totalComputedCost.compareTo(transLine.getTransactionUnits().bigDecimalValue()) != 0) {
340                    BigDecimal difAmount = transLine.getTransactionAmount().bigDecimalValue().subtract(totalComputedCost);
341                    oldestTaxLotLine.setLotHoldingCost(oldestTaxLotLine.getLotHoldingCost().negate().add(difAmount));
342    
343                    oldestTaxLotLine.setLotHoldingCost(oldestTaxLotLine.getLotHoldingCost().negate());
344                }
345            }
346    
347        }

350        /**
351         * Updates the tax lots for the transaction line in the case the accounting method is FIFO or LIFO.
352         * 
353         * @param isSubTypeCash
354         * @param isFIFO
355         * @param endowmentTransactionSecurity
356         * @param transLine
357         */
358        private void updateTaxLotsForAccountingMethodFIFOorLIFO(boolean isSubTypeCash, boolean isUpdate, boolean isFIFO, EndowmentTransactionSecurity endowmentTransactionSecurity, EndowmentTransactionLine transLine) {
359            BigDecimal transactionUnits = transLine.getTransactionUnits().bigDecimalValue();
360            BigDecimal transactionAmount = BigDecimal.ZERO;
361            BigDecimal perUnitVal = BigDecimal.ZERO;
362            List<HoldingTaxLot> holdingTaxLots = new ArrayList<HoldingTaxLot>();
363    
364            if (!isUpdate) {
365                transLine.getTaxLotLines().clear();
366                holdingTaxLots = taxLotService.getAllTaxLotsOrderByAcquiredDate(transLine.getKemid(), endowmentTransactionSecurity.getSecurityID(), endowmentTransactionSecurity.getRegistrationCode(), transLine.getTransactionIPIndicatorCode(), isFIFO);
367            }
368            else {
369                List<EndowmentTransactionTaxLotLine> existingTransactionLines = transLine.getTaxLotLines();
370                for (EndowmentTransactionTaxLotLine endowmentTransactionTaxLotLine : existingTransactionLines) {
371                    HoldingTaxLot holdingTaxLot = taxLotService.getByPrimaryKey(transLine.getKemid(), endowmentTransactionSecurity.getSecurityID(), endowmentTransactionSecurity.getRegistrationCode(), endowmentTransactionTaxLotLine.getTransactionHoldingLotNumber(), transLine.getTransactionIPIndicatorCode());
372    
373                    if (ObjectUtils.isNotNull(holdingTaxLot)) {
374                        holdingTaxLots.add(holdingTaxLot);
375                    }
376                }
377                transLine.getTaxLotLines().clear();
378            }
379    
380            Map<KualiInteger, EndowmentTransactionTaxLotLine> decreaseHoldingTaxLots = new HashMap<KualiInteger, EndowmentTransactionTaxLotLine>();
381    
382            if (isSubTypeCash) {
383                transactionAmount = transLine.getTransactionAmount().bigDecimalValue();
384                // 1. Compute per unit value
385                perUnitVal = KEMCalculationRoundingHelper.divide(transactionAmount, transactionUnits, 5);
386            }
387    
388            BigDecimal remainingUnits = transactionUnits;
389    
390            if (holdingTaxLots != null && holdingTaxLots.size() > 0) {
391                // compute the total number of units for tax lots
392                for (HoldingTaxLot holdingTaxLot : holdingTaxLots) {
393                    EndowmentTransactionTaxLotLine taxLotLine = new EndowmentTransactionTaxLotLine();
394                    taxLotLine.setDocumentLineNumber(transLine.getTransactionLineNumber());
395    
396                    // 2. Set the lot units
397    
398                    // if transaction units is greater than the holding tax lot units then take all the holding tax lot units
399                    if (remainingUnits.compareTo(holdingTaxLot.getUnits()) == 1) {
400                        taxLotLine.setLotUnits(holdingTaxLot.getUnits());
401                        remainingUnits = remainingUnits.subtract(holdingTaxLot.getUnits());
402                    }
403                    // if transaction units is less than the holding tax lot units then take just the remaining number of units
404                    else {
405                        taxLotLine.setLotUnits(remainingUnits);
406                        remainingUnits = BigDecimal.ZERO;
407                    }
408    
409                    // 4. Calculate the original unit value: cost/units
410                    BigDecimal originalUnitVal = KEMCalculationRoundingHelper.divide(holdingTaxLot.getCost(), holdingTaxLot.getUnits(), 5);
411    
412                    // 5. Calculate original cost
413                    BigDecimal originalCost = KEMCalculationRoundingHelper.multiply(originalUnitVal, taxLotLine.getLotUnits(), 2);
414    
415                    // set the tax lot holding cost
416                    taxLotLine.setLotHoldingCost(originalCost);
417    
418                    // if sub type cash then the
419                    if (isSubTypeCash) {
420                        // 3. Calculate the value received for the units sold and round to 2 decimals
421                        BigDecimal receivedValue = KEMCalculationRoundingHelper.multiply(perUnitVal, taxLotLine.getLotUnits(), 2);
422                        // 6. Calculate the gain or loss
423                        calculateGainLoss(holdingTaxLot, taxLotLine, receivedValue, originalCost);
424                    }
425    
426                    // set the lot number
427                    taxLotLine.setTransactionHoldingLotNumber(holdingTaxLot.getLotNumber().intValue());
428                    taxLotLine.setKemid(transLine.getKemid());
429                    taxLotLine.setSecurityID(holdingTaxLot.getSecurityId());
430                    taxLotLine.setRegistrationCode(holdingTaxLot.getRegistrationCode());
431                    taxLotLine.setIpIndicator(holdingTaxLot.getIncomePrincipalIndicator());
432                    // set the lot acquired date
433                    taxLotLine.setLotAcquiredDate(holdingTaxLot.getAcquiredDate());
434    
435                    addTaxLotLine(transLine, taxLotLine);
436    
437                    if (remainingUnits.compareTo(BigDecimal.ZERO) == 0) {
438                        break;
439                    }
440                }
441            }
442        }

444        /**
445         * Updates the tax lots for the transaction line in the case the transaction sub type is non-cash and the user entered the
446         * transaction amount. This method is specific to the Asset Decrease document.
447         * 
448         * @param isSubTypeCash
449         * @param endowmentTransactionSecurity
450         * @param transLine
451         */
452        private void updateTaxLotsForSubTypeNonCashAndTransAmtNonZero(boolean isUpdate, EndowmentTransactionSecurity endowmentTransactionSecurity, EndowmentTransactionLine transLine) {
453    
454            BigDecimal transactionUnits = transLine.getTransactionUnits().bigDecimalValue();
455            BigDecimal totalTaxLotsUnits = BigDecimal.ZERO;
456            BigDecimal transactionAmount = transLine.getTransactionAmount().bigDecimalValue();
457            BigDecimal perUnitValue = BigDecimal.ZERO;
458    
459            List<HoldingTaxLot> holdingTaxLots = new ArrayList<HoldingTaxLot>();
460            Map<Integer, HoldingTaxLot> lotsMap = new HashMap<Integer, HoldingTaxLot>();
461    
462            if (!isUpdate) {
463                transLine.getTaxLotLines().clear();
464                holdingTaxLots = taxLotService.getAllTaxLots(transLine.getKemid(), endowmentTransactionSecurity.getSecurityID(), endowmentTransactionSecurity.getRegistrationCode(), transLine.getTransactionIPIndicatorCode());
465            }
466            else {
467                List<EndowmentTransactionTaxLotLine> existingTransactionLines = transLine.getTaxLotLines();
468                for (EndowmentTransactionTaxLotLine endowmentTransactionTaxLotLine : existingTransactionLines) {
469                    HoldingTaxLot holdingTaxLot = taxLotService.getByPrimaryKey(transLine.getKemid(), endowmentTransactionSecurity.getSecurityID(), endowmentTransactionSecurity.getRegistrationCode(), endowmentTransactionTaxLotLine.getTransactionHoldingLotNumber(), transLine.getTransactionIPIndicatorCode());
470    
471                    if (ObjectUtils.isNotNull(holdingTaxLot)) {
472                        holdingTaxLots.add(holdingTaxLot);
473                    }
474                }
475    
476                transLine.getTaxLotLines().clear();
477            }
478    
479            Map<KualiInteger, EndowmentTransactionTaxLotLine> decreaseHoldingTaxLots = new HashMap<KualiInteger, EndowmentTransactionTaxLotLine>();
480    
481            if (holdingTaxLots != null && holdingTaxLots.size() > 0) {
482                boolean keepIntegers = true;
483                // compute the total number of units for tax lots
484                for (HoldingTaxLot holdingTaxLot : holdingTaxLots) {
485                    totalTaxLotsUnits = totalTaxLotsUnits.add(holdingTaxLot.getUnits());
486    
487                    // 3. Calculate the number of units to be transacted in each lot
488                    // check if percentage and tax lot units are integers
489                    BigDecimal lotUnits = BigDecimal.ZERO;
490                    try {
491                        int lotUnitsInt = holdingTaxLot.getUnits().intValueExact();
492                    }
493                    catch (ArithmeticException ex) {
494                        keepIntegers = false;
495                    }
496    
497                }
498    
499                for (HoldingTaxLot holdingTaxLot : holdingTaxLots) {
500                    EndowmentTransactionTaxLotLine taxLotLine = new EndowmentTransactionTaxLotLine();
501                    taxLotLine.setDocumentLineNumber(transLine.getTransactionLineNumber());
502    
503                    BigDecimal lotUnits = BigDecimal.ZERO;
504                    // 2. Calculate percentage each lot holds out of the total units
505                    BigDecimal percentage = KEMCalculationRoundingHelper.divide(holdingTaxLot.getUnits(), totalTaxLotsUnits, 5);
506    
507    
508                    lotUnits = KEMCalculationRoundingHelper.multiply(percentage, transLine.getTransactionUnits().bigDecimalValue(), 5);
509    
510                    // IF all original units per lot are integers (no decimal values), the result is rounded to the nearest
511                    // integer and stored with the five decimals as zero. If the original units are not all integers, then the
512                    // value is rounded to five decimals and stored as the five decimal values.
513                    if (keepIntegers) {
514                        lotUnits = lotUnits.setScale(0, BigDecimal.ROUND_HALF_UP);
515                        lotUnits = lotUnits.setScale(5);
516                    }
517                    taxLotLine.setLotUnits(lotUnits);
518    
519                    // 6. Calculate holding cost
520                    BigDecimal holdingCost = KEMCalculationRoundingHelper.multiply(percentage, transactionAmount, 2);
521                    taxLotLine.setLotHoldingCost(holdingCost);
522    
523    
524                    // set tax lot line lot number and acquired date
525                    taxLotLine.setTransactionHoldingLotNumber(holdingTaxLot.getLotNumber().intValue());
526                    taxLotLine.setKemid(transLine.getKemid());
527                    taxLotLine.setSecurityID(holdingTaxLot.getSecurityId());
528                    taxLotLine.setRegistrationCode(holdingTaxLot.getRegistrationCode());
529                    taxLotLine.setIpIndicator(holdingTaxLot.getIncomePrincipalIndicator());
530                    taxLotLine.setLotAcquiredDate(holdingTaxLot.getAcquiredDate());
531    
532                    // set the new lot indicator
533                    taxLotLine.setNewLotIndicator(false);
534    
535                    // add the new tax lot line to the transaction line tax lots
536                    addTaxLotLine(transLine, taxLotLine);
537                    lotsMap.put(taxLotLine.getTransactionHoldingLotNumber(), holdingTaxLot);
538                }
539                adjustUnitsAndAmountsForNonCashAndTransactionAmtNotZero(transLine);
540    
541            }
542        }
543    
544        /**
545         * Adjusts the number of units and the amounts if the total is different from the transaction line units. The correction will be
546         * done on the oldest tax lot. This method is specific to the Asset Decrease document and is used in
547         * updateTaxLotsForSubTypeNonCashAndTransAmtNonZero().
548         * 
549         * @param transLine
550         * @param keepIntegers
551         */
552        private void adjustUnitsAndAmountsForNonCashAndTransactionAmtNotZero(EndowmentTransactionLine transLine) {
553            // Adjust the number of units if the total is different from the transaction line units
554            BigDecimal totalComputedTaxLotUnits = BigDecimal.ZERO;
555            BigDecimal totalComputedCost = BigDecimal.ZERO;
556            EndowmentTransactionTaxLotLine oldestTaxLotLine = null;
557    
558            if (transLine.getTaxLotLines() != null && transLine.getTaxLotLines().size() > 0) {
559                for (EndowmentTransactionTaxLotLine taxLotLine : transLine.getTaxLotLines()) {
560                    BigDecimal lotUnits = taxLotLine.getLotUnits().negate();
561    
562                    // calculate the total number of units to be decreased
563                    totalComputedTaxLotUnits = totalComputedTaxLotUnits.add(lotUnits);
564                    totalComputedCost = totalComputedCost.add(taxLotLine.getLotHoldingCost().negate());
565    
566                    if (taxLotLine.getLotShortTermGainLoss() != null) {
567                        totalComputedCost = totalComputedCost.add(taxLotLine.getLotShortTermGainLoss());
568                    }
569    
570                    if (taxLotLine.getLotLongTermGainLoss() != null) {
571                        totalComputedCost = totalComputedCost.add(taxLotLine.getLotLongTermGainLoss());
572                    }
573    
574                    // keep the tax lot with the oldest acquired date so that we can adjust the units for that one in case the
575                    // number of units needs and adjustment
576                    if (oldestTaxLotLine != null) {
577                        if (oldestTaxLotLine.getLotAcquiredDate().after(taxLotLine.getLotAcquiredDate())) {
578                            oldestTaxLotLine = taxLotLine;
579                        }
580                    }
581                    else {
582                        oldestTaxLotLine = taxLotLine;
583                    }
584                }
585            }
586    
587            // compare with the negated number of units on the transaction line because the units on the tax lots have been negated
588            if (totalComputedTaxLotUnits.compareTo(transLine.getTransactionUnits().bigDecimalValue().negate()) != 0) {
589                BigDecimal difUnits = transLine.getTransactionUnits().bigDecimalValue().subtract(totalComputedTaxLotUnits);
590                oldestTaxLotLine.setLotUnits(oldestTaxLotLine.getLotUnits().add(difUnits.negate()));
591    
592                BigDecimal difAmount = transLine.getTransactionAmount().bigDecimalValue().subtract(totalComputedCost);
593                oldestTaxLotLine.setLotHoldingCost(oldestTaxLotLine.getLotHoldingCost().add(difAmount.negate()));
594            }
595    
596        }

598        /**
599         * Adds a tax lot line to a transaction line but first it negates the units and cost.
600         * 
601         * @param transactionLine
602         * @param taxLotLine
603         */
604        private void addTaxLotLine(EndowmentTransactionLine transactionLine, EndowmentTransactionTaxLotLine taxLotLine) {
605            // negate units and cost
606            taxLotLine.setLotUnits(taxLotLine.getLotUnits().negate());
607            taxLotLine.setLotHoldingCost(taxLotLine.getLotHoldingCost().negate());
608    
609            // add the tax lot line to the transaction line
610            transactionLine.getTaxLotLines().add(taxLotLine);
611        }

613        /**
614         * Sets the transaction line amount to be the total amount of all tax lot lines times negative 1. This is applied in case the
615         * transaction sub type is non-cash.
616         * 
617         * @param transactionLine
618         */
619        private void setTransactionLineTotal(EndowmentTransactionLine transactionLine) {
620            List<EndowmentTransactionTaxLotLine> taxLots = transactionLine.getTaxLotLines();
621            BigDecimal totalAmount = BigDecimal.ZERO;
622            if (taxLots != null && taxLots.size() > 0) {
623                for (EndowmentTransactionTaxLotLine taxLot : taxLots) {
624                    totalAmount = totalAmount.add(taxLot.getLotHoldingCost());
625                }
626            }
627    
628            totalAmount = totalAmount.negate();
629    
630            transactionLine.setTransactionAmount(new KualiDecimal(totalAmount));
631        }
