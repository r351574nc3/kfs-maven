016    package org.kuali.kfs.module.endow.document.web.struts;
017    
018    import javax.servlet.http.HttpServletRequest;
019    
020    import org.kuali.kfs.module.endow.EndowConstants;
021    import org.kuali.kfs.module.endow.document.CorpusAdjustmentDocument;
022    import org.kuali.kfs.sys.context.SpringContext;
023    import org.kuali.rice.kns.service.BusinessObjectDictionaryService;
024    import org.kuali.rice.kns.service.DataDictionaryService;
025    
026    public class CorpusAdjustmentDocumentForm extends EndowmentTransactionLinesDocumentFormBase {
027    
028        public CorpusAdjustmentDocumentForm() {
029            super();
030            setSourceGroupLabelName(EndowConstants.DECREASE_TRANSACTION_LINE_GROUP_LABEL_NAME);
031            setTargetGroupLabelName(EndowConstants.INCREASE_TRANSACTION_LINE_GROUP_LABEL_NAME);
032    
033            // don't show these values on the edoc.
034            setShowIncomeTotalAmount(false);
035            setShowIncomeTotalUnits(false);
036            setShowPrincipalTotalUnits(false);
037    
038            // set the drop-down value to P-Principal and make the field readonly
039            getNewSourceTransactionLine().setTransactionIPIndicatorCode(EndowConstants.IncomePrincipalIndicator.PRINCIPAL);
040            getNewTargetTransactionLine().setTransactionIPIndicatorCode(EndowConstants.IncomePrincipalIndicator.PRINCIPAL);
041    
042            setFeildValueToPrincipal(true);
043    
044            // do not show the etran code on the UI screen
045            setShowETranCode(false);
046        }
047    
048        @Override
049        protected String getDefaultDocumentTypeName() {
050            // use the DataDictionaryService service to get the document type name...
051            return SpringContext.getBean(DataDictionaryService.class).getDocumentTypeNameByClass(CorpusAdjustmentDocument.class);
052        }
053    
054        @Override
055        public void populate(HttpServletRequest request) {
056            super.populate(request);
057            SpringContext.getBean(BusinessObjectDictionaryService.class).performForceUppercase(this.getCorpusAdjustmentDocument());
058        }
059    
060        /**
061         * This method gets the Corpus Adjustment document
062         * 
063         * @return the CorpusAdjustmentDocument
064         */
065        public CorpusAdjustmentDocument getCorpusAdjustmentDocument() {
066            return (CorpusAdjustmentDocument) getDocument();
067        }
068    
069    
070    }
