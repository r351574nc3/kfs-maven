016    package org.kuali.kfs.module.endow.document.service.impl;
017    
018    import java.util.HashMap;
019    import java.util.Map;
020    
021    import org.apache.commons.lang.StringUtils;
022    import org.kuali.kfs.module.endow.EndowPropertyConstants;
023    import org.kuali.kfs.module.endow.businessobject.DonorLabel;
024    import org.kuali.kfs.module.endow.document.service.DonorLabelService;
025    import org.kuali.rice.kns.service.BusinessObjectService;
026    
027    public class DonorLabelServiceImpl implements DonorLabelService {
028    
029        private BusinessObjectService businessObjectService;
030    
031        /**
032         * @see org.kuali.kfs.module.endow.document.service.DonorLabelService#getByPrimaryKey(java.lang.String)
033         */
034        public DonorLabel getByPrimaryKey(String code) {
035            DonorLabel donorLabel = null;
036            if (StringUtils.isNotBlank(code)) {
037                Map<String, String> criteria = new HashMap<String, String>();
038                criteria.put(EndowPropertyConstants.KUALICODEBASE_CODE, code);
039    
040                donorLabel = (DonorLabel) businessObjectService.findByPrimaryKey(DonorLabel.class, criteria);
041            }
042            return donorLabel;
043        }
044    
045        /**
046         * This method gets the businessObjectService.
047         * 
048         * @return businessObjectService
049         */
050        public BusinessObjectService getBusinessObjectService() {
051            return businessObjectService;
052        }
053    
054        /**
055         * This method sets the businessObjectService
056         * 
057         * @param businessObjectService
058         */
059        public void setBusinessObjectService(BusinessObjectService businessObjectService) {
060            this.businessObjectService = businessObjectService;
061        }
062    
063    }
