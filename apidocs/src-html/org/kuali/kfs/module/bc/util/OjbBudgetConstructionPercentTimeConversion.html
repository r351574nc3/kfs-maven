016    package org.kuali.kfs.module.bc.util;
017    
018    import java.math.BigDecimal;
019    
020    import org.apache.ojb.broker.accesslayer.conversions.ConversionException;
021    import org.apache.ojb.broker.accesslayer.conversions.FieldConversion;
022    
023    /**
024     * Ensures that the database Percent Time value is rounded to 2 decimal places when reading. Any saved Percent Time value will
025     * automatically be forced to 2 decimals during validation.
026     */
027    public class OjbBudgetConstructionPercentTimeConversion implements FieldConversion {
028    
029        /**
030         * @see org.apache.ojb.broker.accesslayer.conversions.FieldConversion#javaToSql(java.lang.Object)
031         */
032        public Object javaToSql(Object source) throws ConversionException {
033            if (source != null && source instanceof BigDecimal) {
034                BigDecimal converted = (BigDecimal) source;
035                return converted;
036            }
037            else {
038                return null;
039            }
040        }
041    
042        /**
043         * @see org.apache.ojb.broker.accesslayer.conversions.FieldConversion#sqlToJava(java.lang.Object)
044         */
045        public Object sqlToJava(Object source) throws ConversionException {
046            // Check for null, and verify object type.
047            // Do conversion if our type is correct (BigDecimal).
048            if (source != null && source instanceof BigDecimal) {
049                BigDecimal converted = (BigDecimal) source;
050    
051                return converted.setScale(2, BigDecimal.ROUND_HALF_UP);
052            }
053            else {
054                return null;
055            }
056        }
057    
058    }
