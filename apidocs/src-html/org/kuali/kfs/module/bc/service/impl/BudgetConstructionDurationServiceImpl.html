016    package org.kuali.kfs.module.bc.service.impl;
017    
018    import java.util.HashMap;
019    import java.util.Map;
020    
021    import org.kuali.kfs.module.bc.BCPropertyConstants;
022    import org.kuali.kfs.module.bc.businessobject.BudgetConstructionDuration;
023    import org.kuali.kfs.module.bc.service.BudgetConstructionDurationService;
024    import org.kuali.rice.kns.service.BusinessObjectService;
025    
026    /**
027     * implements the service methods defined in BudgetConstructionDurationService
028     */
029    public class BudgetConstructionDurationServiceImpl implements BudgetConstructionDurationService {
030        private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(BudgetConstructionDurationServiceImpl.class);
031    
032        private BusinessObjectService businessObjectService;
033    
034        /**
035         * @see org.kuali.kfs.module.bc.service.BudgetConstructionDurationService#getByPrimaryId(java.lang.String)
036         */
037        public BudgetConstructionDuration getByPrimaryId(String durationCode) {
038            Map<String, Object> primaryKeys = new HashMap<String, Object>();
039            primaryKeys.put(BCPropertyConstants.APPOINTMENT_DURATION_CODE, durationCode);
040    
041            return (BudgetConstructionDuration) businessObjectService.findByPrimaryKey(BudgetConstructionDuration.class, primaryKeys);
042        }
043    
044        /**
045         * Sets the businessObjectService attribute value.
046         * 
047         * @param businessObjectService The businessObjectService to set.
048         */
049        public void setBusinessObjectService(BusinessObjectService businessObjectService) {
050            this.businessObjectService = businessObjectService;
051        }
052    }
