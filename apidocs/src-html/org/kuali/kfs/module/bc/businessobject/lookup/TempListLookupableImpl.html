016    package org.kuali.kfs.module.bc.businessobject.lookup;
017    
018    import org.apache.commons.lang.StringUtils;
019    import org.kuali.kfs.sys.KFSConstants;
020    import org.kuali.rice.kns.lookup.KualiLookupableImpl;
021    
022    /**
023     * Lookupable impl for lookups using the temp list action and jsp. Corrects maintenance new link.
024     */
025    public class TempListLookupableImpl extends KualiLookupableImpl {
026    
027        /**
028         * @see org.kuali.core.lookup.KualiLookupableImpl#getCreateNewUrl()
029         */
030        @Override
031        public String getCreateNewUrl() {
032            String url = super.getCreateNewUrl();
033            url = StringUtils.replace(url, KFSConstants.MAINTENANCE_ACTION, KFSConstants.RICE_PATH_PREFIX + KFSConstants.MAINTENANCE_ACTION);
034            url = StringUtils.replace(url, "images/", KFSConstants.RICE_PATH_PREFIX + "images/");
035            
036            return url;
037        }
038    
039    }
