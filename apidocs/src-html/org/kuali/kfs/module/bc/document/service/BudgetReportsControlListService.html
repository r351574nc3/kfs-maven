016    package org.kuali.kfs.module.bc.document.service;
017    
018    import java.util.Collection;
019    import java.util.List;
020    
021    import org.kuali.kfs.module.bc.BCConstants.Report.BuildMode;
022    import org.kuali.kfs.module.bc.businessobject.BudgetConstructionObjectPick;
023    import org.kuali.kfs.module.bc.businessobject.BudgetConstructionPullup;
024    import org.kuali.kfs.module.bc.businessobject.BudgetConstructionReasonCodePick;
025    import org.kuali.kfs.module.bc.businessobject.BudgetConstructionSubFundPick;
026    
027    /**
028     * Defines methods that manipulate objects used by the Organization Selection screens. Manipulated objects include
029     * BudgetConstructionPullup with methods that populate and depopulate the associated table for a specific user.
030     */
031    public interface BudgetReportsControlListService {
032    
033        /**
034         * Rebuilds the report control list for the given parameters.
035         * 
036         * @param principalId - current user who is running the report
037         * @param universityFiscalYear - budget fiscal year
038         * @param chartOfAccountsCode - point of view chart of accounts code
039         * @param organizationCode - point of view organization code
040         * @param buildMode - indicates whether the accounts should be restricted to GL pending budget, monthly budget, or bnc
041         *        appointment funding
042         */
043        public void updateReportsControlList(String principalId, Integer universityFiscalYear, String chartOfAccountsCode, String organizationCode, BuildMode buildMode);
044    
045        /**
046         * Rebuilds the subfunds available for selection based on the control list.
047         * 
048         * @param principalId - current user who is running the report
049         */
050        public void updateReportSubFundGroupSelectList(String principalId);
051    
052        /**
053         * Rebuilds the object codes available for selection based on the control list.
054         * 
055         * @param principalId - current user who is running the report
056         */
057        public void updateReportObjectCodeSelectList(String principalId);
058    
059        /**
060         * Rebuilds the reason codes available for selection based on the control list.
061         * 
062         * @param principalId - current user who is running the report
063         */
064        public void updateReportReasonCodeSelectList(String principalId);
065    
066        /**
067         * Retrieves organizations selected by given user.
068         * 
069         * @param principalId - current user who is running the report
070         * @return Collection<BudgetConstructionPullup> that are selected by the given user
071         */
072        public Collection<BudgetConstructionPullup> retrieveSelectedOrganziations(String principalId);
073    
074        /**
075         * Retrieves sub funds that are available for selection by the given user.
076         * 
077         * @param principalId - current user who is running the report
078         * @return Collection<BudgetConstructionSubFundPick> that are selected by the given user
079         */
080        public Collection<BudgetConstructionSubFundPick> retrieveSubFundList(String principalId);
081    
082        /**
083         * Retrieves object codes that are available for selection by the given user.
084         * 
085         * @param principalId - current user who is running the report
086         * @return Collection<BudgetConstructionObjectPick> that are selected by the given user
087         */
088        public Collection<BudgetConstructionObjectPick> retrieveObjectCodeList(String principalId);
089    
090        /**
091         * Retrieves reason codes that are available for selection by the given user.
092         * 
093         * @param principalId - current user who is running the report
094         * @return Collection<BudgetConstructionReasonCodePick> that are selected by the given user
095         */
096        public Collection<BudgetConstructionReasonCodePick> retrieveReasonCodeList(String principalId);
097        
098        /**
099         * Updates the select flag for each sub fund selection record.
100         * 
101         * @param subFundPickList - List containing the updated BudgetConstructionSubFundPick objects
102         */
103        public void updateSubFundSelectFlags(List<BudgetConstructionSubFundPick> subFundPickList);
104    
105        /**
106         * Updates the select flag for each object code selection record.
107         * 
108         * @param objectCodePickList - List containing the updated BudgetConstructionObjectPick objects
109         */
110        public void updateObjectCodeSelectFlags(List<BudgetConstructionObjectPick> objectCodePickList);
111    
112        /**
113         * Updates the select flag for each reason code selection record.
114         * 
115         * @param objectCodePickList - List containing the updated BudgetConstructionReasonCodePick objects
116         */
117        public void updateReasonCodeSelectFlags(List<BudgetConstructionReasonCodePick> reasonCodePickList);
118    }
