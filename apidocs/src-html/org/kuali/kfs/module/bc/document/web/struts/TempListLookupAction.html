074        /**
075         * @see org.kuali.rice.kns.web.struts.action.KualiLookupAction#execute(org.apache.struts.action.ActionMapping,
076         *      org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
077         */
078        @Override
079        public ActionForward execute(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
080    
081            TempListLookupForm tempListLookupForm = (TempListLookupForm) form;
082    
083            // check for session heart beat only if running inside BC system
084            if (tempListLookupForm.getTempListLookupMode() != BCConstants.TempListLookupMode.DEFAULT_LOOKUP_MODE) {
085                Boolean isBCHeartBeating = (Boolean) GlobalVariables.getUserSession().retrieveObject(BCConstants.BC_HEARTBEAT_SESSIONFLAG);
086                if (isBCHeartBeating == null) {
087                    if (tempListLookupForm.isMainWindow()) {
088                        Properties parameters = new Properties();
089                        parameters.put(KFSConstants.DISPATCH_REQUEST_PARAMETER, BCConstants.LOAD_EXPANSION_SCREEN_METHOD_SESSION_TIMEOUT);
090    
091                        String lookupUrl = UrlFactory.parameterizeUrl("/" + BCConstants.BC_SELECTION_ACTION, parameters);
092    
093                        return new ActionForward(lookupUrl, true);
094                    }
095                    else {
096                        GlobalVariables.getMessageList().add(BCKeyConstants.MESSAGE_BUDGET_PREVIOUS_SESSION_TIMEOUT);
097                        return mapping.findForward(BCConstants.MAPPING_LOST_SESSION_RETURNING);
098                    }
099                }
100            }
101    
102            return super.execute(mapping, form, request, response);
103        }

105        /**
106         * Does not supress actions if in position or incumbent lookup mode in the BC application, otherwise calls on
107         * KualiLookupAction.supressActions
108         * 
109         * @see org.kuali.rice.kns.web.struts.action.KualiLookupAction#supressActionsIfNeeded(org.apache.struts.action.ActionForm)
110         */
111        @Override
112        protected void supressActionsIfNeeded(ActionForm form) throws ClassNotFoundException {
113    
114            TempListLookupForm tempListLookupForm = (TempListLookupForm) form;
115            if ((tempListLookupForm.getTempListLookupMode() == BCConstants.TempListLookupMode.BUDGET_POSITION_LOOKUP) || (tempListLookupForm.getTempListLookupMode() == BCConstants.TempListLookupMode.INTENDED_INCUMBENT)) {
116    
117                // we want the actions in actions column to show when in position or incumbent lookup mode
118                // regardless of what KIM thinks
119                tempListLookupForm.setSuppressActions(false);
120            }
121            else if (tempListLookupForm.getTempListLookupMode() == BCConstants.TempListLookupMode.LOCK_MONITOR) {
122                // check if current user has permission to unlock
123                String[] rootOrg = SpringContext.getBean(OrganizationService.class).getRootOrganizationCode();
124                AttributeSet qualification = new AttributeSet();
125                qualification.put(BCPropertyConstants.ORGANIZATION_CHART_OF_ACCOUNTS_CODE, rootOrg[0]);
126                qualification.put(KfsKimAttributes.ORGANIZATION_CODE, rootOrg[1]);
127    
128                boolean canUnlock = SpringContext.getBean(IdentityManagementService.class).isAuthorized(GlobalVariables.getUserSession().getPerson().getPrincipalId(), BCConstants.BUDGET_CONSTRUCTION_NAMESPACE, BCConstants.KimConstants.UNLOCK_PERMISSION_NAME, null, qualification);
129                tempListLookupForm.setSuppressActions(!canUnlock);
130                tempListLookupForm.setSupplementalActionsEnabled(canUnlock);
131            }
132            else {
133                super.supressActionsIfNeeded(form);
134            }
135        }

137        /**
138         * TempListLookupAction can be called to build and display different lists. This method determines what the requested behavior
139         * is and either makes a build call for that list or sets up a message (if the list has already been built). If the request
140         * parameter showInitialResults is true, an initial search will be performed before display of the screen.
141         * 
142         * @see org.kuali.rice.kns.web.struts.action.KualiLookupAction#start(org.apache.struts.action.ActionMapping,
143         *      org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
144         */
145        @Override
146        public ActionForward start(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
147            TempListLookupForm tempListLookupForm = (TempListLookupForm) form;
148    
149            // determine requested lookup action
150            switch (tempListLookupForm.getTempListLookupMode()) {
151                case BCConstants.TempListLookupMode.INTENDED_INCUMBENT_SELECT:
152                    SpringContext.getBean(OrganizationSalarySettingSearchService.class).buildIntendedIncumbentSelect(tempListLookupForm.getPrincipalId(), tempListLookupForm.getUniversityFiscalYear());
153                    break;
154    
155                case BCConstants.TempListLookupMode.POSITION_SELECT:
156                    SpringContext.getBean(OrganizationSalarySettingSearchService.class).buildPositionSelect(tempListLookupForm.getPrincipalId(), tempListLookupForm.getUniversityFiscalYear());
157                    break;
158    
159                case BCConstants.TempListLookupMode.ACCOUNT_SELECT_ABOVE_POV:
160                    // check report mode for 2PLG or Sync
161                    // this case is handled after performing the search, see below
162                    break;
163    
164                case BCConstants.TempListLookupMode.ACCOUNT_SELECT_BUDGETED_DOCUMENTS:
165                    // Show Budgeted Docs (BudgetConstructionAccountSelect) in the point of view subtree for the selected org(s)
166                    // The table was already built in OrganizationSelectionTreeAction.performShowBudgetDocs
167                    GlobalVariables.getMessageList().add(KFSKeyConstants.budget.MSG_ACCOUNT_LIST);
168                    break;
169    
170                case BCConstants.TempListLookupMode.ACCOUNT_SELECT_PULLUP_DOCUMENTS:
171                    // Show Budgeted Docs (BudgetConstructionAccountSelect) for the selected org(s) below the users point of view
172                    // The table was already built in OrganizationSelectionTreeAction.performShowPullUpBudgetDocs
173                    GlobalVariables.getMessageList().add(BCKeyConstants.MSG_ACCOUNT_PULLUP_LIST);
174                    break;
175    
176                case BCConstants.TempListLookupMode.ACCOUNT_SELECT_PUSHDOWN_DOCUMENTS:
177                    // Show Budgeted Docs (BudgetConstructionAccountSelect) for the selected org(s) at the users point of view
178                    // The table was already built in OrganizationSelectionTreeAction.performShowPullUpBudgetDocs
179                    GlobalVariables.getMessageList().add(BCKeyConstants.MSG_ACCOUNT_PUSHDOWN_LIST);
180                    break;
181    
182                case BCConstants.TempListLookupMode.ACCOUNT_SELECT_MANAGER_DELEGATE:
183                    // Show Budgeted Docs (BudgetConstructionAccountSelect) where the user is a fiscal officer or delegate
184                    // The table was already built in BudgetConstructionSelectionTreeAction.performMyAccounts
185                    GlobalVariables.getMessageList().add(BCKeyConstants.MSG_ACCOUNT_MANAGER_DELEGATE_LIST);
186                    break;
187            }
188    
189            ActionForward forward = super.start(mapping, form, request, response);
190            if (tempListLookupForm.isShowInitialResults()) {
191                forward = search(mapping, form, request, response);
192            }
193    
194            // handling this mode case here instead of inside case test above
195            // since it requires knowing if rows exist
196            if (tempListLookupForm.getTempListLookupMode() == BCConstants.TempListLookupMode.ACCOUNT_SELECT_ABOVE_POV) {
197                if (tempListLookupForm.isForceToAccountListScreen()) {
198                    if (tempListLookupForm.getLookupable().getRows().isEmpty()) {
199                        GlobalVariables.getMessageList().add(KFSKeyConstants.budget.MSG_REPORT_EMPTY_ACCOUNT_LIST);
200                    }
201                    else {
202                        GlobalVariables.getMessageList().add(KFSKeyConstants.budget.MSG_REPORT_ACCOUNT_LIST);
203                    }
204                }
205                else {
206                    // Show Account above current point of view for user
207                    // The table was already built in OrganizationSelectionTreeAction.performReport
208                    GlobalVariables.getMessageList().add(KFSKeyConstants.budget.MSG_REPORT_ACCOUNT_LIST);
209                }
210            }
211            return forward;
212        }

214        /**
215         * This differs from KualiLookupAction.clearValues in that any atributes marked hidden will not be cleared. This is to support
216         * BC temp tables that use principalId to operate on the set of rows associated with the current user.
217         * 
218         * @see org.kuali.rice.kns.web.struts.action.KualiLookupAction#clearValues(org.apache.struts.action.ActionMapping,
219         *      org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
220         */
221        @Override
222        public ActionForward clearValues(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
223            LookupForm lookupForm = (LookupForm) form;
224            Lookupable kualiLookupable = lookupForm.getLookupable();
225            if (kualiLookupable == null) {
226                LOG.error("Lookupable is null.");
227                throw new RuntimeException("Lookupable is null.");
228            }
229    
230            for (Iterator iter = kualiLookupable.getRows().iterator(); iter.hasNext();) {
231                Row row = (Row) iter.next();
232                for (Iterator iterator = row.getFields().iterator(); iterator.hasNext();) {
233                    Field field = (Field) iterator.next();
234                    if (!field.getFieldType().equals(Field.RADIO) && !field.getFieldType().equals(Field.HIDDEN)) {
235                        field.setPropertyValue(field.getDefaultValue());
236                    }
237                }
238            }
239            return mapping.findForward(KFSConstants.MAPPING_BASIC);
240        }

306        /**
307         * Validates the get new action for position then calls BudgetPositionService to pull the new position record.
308         * 
309         * @see org.kuali.rice.kns.web.struts.action.KualiLookupAction#start(org.apache.struts.action.ActionMapping,
310         *      org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
311         */
312        public ActionForward getNewPosition(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
313            TempListLookupForm tempListLookupForm = (TempListLookupForm) form;
314    
315            // check bc updates are allowed and updates from payroll are turned on
316            if (!tempListLookupForm.isGetNewPositionEnabled()) {
317                LOG.error("get new position not enabled.");
318                throw new RuntimeException("get new position not enabled");
319            }
320    
321            // verify a position number to retrieve was given
322            String positionNumber = (String) tempListLookupForm.getFieldsForLookup().get(BCPropertyConstants.POSITION_NUMBER);
323            if (StringUtils.isBlank(positionNumber)) {
324                GlobalVariables.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.ERROR_REQUIRED_FOR_GET_NEW_POSITION);
325                return mapping.findForward(KFSConstants.MAPPING_BASIC);
326            }
327    
328            // call service to pull new position
329            try {
330                SpringContext.getBean(BudgetConstructionPositionService.class).pullNewPositionFromExternal(tempListLookupForm.getUniversityFiscalYear(), positionNumber);
331            }
332            catch (PositionNotFoundException e) {
333                GlobalVariables.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, e.getMessageKey(), e.getMessageParameters());
334            }
335            catch (BudgetPositionAlreadyExistsException e1) {
336                GlobalVariables.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, e1.getMessageKey(), e1.getMessageParameters());
337            }
338    
339            // perform search which should return the new budget position
340            return this.search(mapping, form, request, response);
341        }

452            TempListLookupForm tempListLookupForm = (TempListLookupForm) form;
453    
454            // populate BudgetConstructionLockSummary with the information for the record to unlock (passed on unlock button)
455            String unlockKeyString = (String) request.getAttribute(KFSConstants.METHOD_TO_CALL_ATTRIBUTE);
456            if (StringUtils.isBlank(unlockKeyString)) {
457                unlockKeyString = request.getParameter(KNSConstants.METHOD_TO_CALL_PATH);
458            }
459            BudgetConstructionLockSummary lockSummary = populateLockSummary(unlockKeyString);
460            String lockKeyMessage = buildLockKeyMessage(lockSummary.getLockType(), lockSummary.getLockUserId(), lockSummary.getChartOfAccountsCode(), lockSummary.getAccountNumber(), lockSummary.getSubAccountNumber(), lockSummary.getUniversityFiscalYear(), lockSummary.getPositionNumber());
461    
462            // confirm the unlock
463            ActionForward forward = doUnlockConfirmation(mapping, form, request, response, lockSummary.getLockType(), lockKeyMessage);
464            if (forward != null) {
465                return forward;
466            }
467    
468            // verify lock for user still exists, if not give warning message
469            boolean lockExists = SpringContext.getBean(LockService.class).checkLockExists(lockSummary);
470            if (!lockExists) {
471                GlobalVariables.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.MSG_LOCK_NOTEXIST, lockSummary.getLockType(), lockKeyMessage);
472            }
473            else {
474                // do the unlock
475                LockStatus lockStatus = SpringContext.getBean(LockService.class).doUnlock(lockSummary);
476                if (LockStatus.SUCCESS.equals(lockStatus)) {
477                    String successMessage = SpringContext.getBean(KualiConfigurationService.class).getPropertyString(BCKeyConstants.MSG_UNLOCK_SUCCESSFUL);
478                    tempListLookupForm.addMessage(MessageFormat.format(successMessage, lockSummary.getLockType(), lockKeyMessage));
479                }
480                else {
481                    GlobalVariables.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, BCKeyConstants.MSG_UNLOCK_NOTSUCCESSFUL, lockSummary.getLockType(), lockKeyMessage);
482                }
483            }
484    
485            // refresh locks before returning
486            return this.search(mapping, form, request, response);
487        }

490        /**
491         * Gives a confirmation first time called. The next time will check the confirmation result. If the returned forward is not
492         * null, that indicates we are fowarding to the question or they selected No to the confirmation and we should return to the
493         * unlock page.
494         * 
495         * @see org.kuali.rice.kns.web.struts.action.KualiAction#execute(org.apache.struts.action.ActionMapping,
496         *      org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
497         */
498        public ActionForward doUnlockConfirmation(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response, String lockType, String lockKeyMessage) throws Exception {
499            TempListLookupForm lookupForm = (TempListLookupForm) form;
500    
501            String question = request.getParameter(KFSConstants.QUESTION_INST_ATTRIBUTE_NAME);
502            if (question == null) { // question hasn't been asked
503                String message = SpringContext.getBean(KualiConfigurationService.class).getPropertyString(BCKeyConstants.MSG_UNLOCK_CONFIRMATION);
504                message = MessageFormat.format(message, lockType, lockKeyMessage);
505    
506                return this.performQuestionWithoutInput(mapping, form, request, response, BCConstants.UNLOCK_CONFIRMATION_QUESTION, message, KFSConstants.CONFIRMATION_QUESTION, BCConstants.TEMP_LIST_UNLOCK_METHOD, "");
507            }
508            else {
509                // get result of confirmation, if yes return null which will indicate the unlock can continue
510                String buttonClicked = request.getParameter(KFSConstants.QUESTION_CLICKED_BUTTON);
511                if ((BCConstants.UNLOCK_CONFIRMATION_QUESTION.equals(question)) && ConfirmationQuestion.YES.equals(buttonClicked)) {
512                    return null;
513                }
514            }
515    
516            // selected no to confirmation so return to lock screen with refreshed results
517            return this.search(mapping, form, request, response);
518        }

520        /**
521         * Parses the methodToCall parameter which contains the lock information in a known format. Populates a
522         * BudgetConstructionLockSummary that represents the record to unlock.
523         * 
524         * @param methodToCallString - request parameter containing lock information
525         * @return lockSummary populated from request parameter
526         */
527        protected BudgetConstructionLockSummary populateLockSummary(String methodToCallString) {
528            BudgetConstructionLockSummary lockSummary = new BudgetConstructionLockSummary();
529    
530            // parse lock fields from methodToCall parameter
531            String lockType = StringUtils.substringBetween(methodToCallString, KFSConstants.METHOD_TO_CALL_PARM1_LEFT_DEL, KFSConstants.METHOD_TO_CALL_PARM1_RIGHT_DEL);
532            String lockFieldsString = StringUtils.substringBetween(methodToCallString, KFSConstants.METHOD_TO_CALL_PARM9_LEFT_DEL, KFSConstants.METHOD_TO_CALL_PARM9_RIGHT_DEL);
533            String lockUser = StringUtils.substringBetween(methodToCallString, KFSConstants.METHOD_TO_CALL_PARM3_LEFT_DEL, KFSConstants.METHOD_TO_CALL_PARM3_RIGHT_DEL);
534    
535            // space was replaced by underscore for html
536            lockSummary.setLockType(StringUtils.replace(lockType, "_", " "));
537            lockSummary.setLockUserId(lockUser);
538    
539            // parse key fields
540            StrTokenizer strTokenizer = new StrTokenizer(lockFieldsString, BCConstants.LOCK_STRING_DELIMITER);
541            strTokenizer.setIgnoreEmptyTokens(false);
542            String fiscalYear = strTokenizer.nextToken();
543            if (fiscalYear != null) {
544                lockSummary.setUniversityFiscalYear(Integer.parseInt(fiscalYear));
545            }
546    
547            lockSummary.setChartOfAccountsCode(strTokenizer.nextToken());
548            lockSummary.setAccountNumber(strTokenizer.nextToken());
549            lockSummary.setSubAccountNumber(strTokenizer.nextToken());
550            lockSummary.setPositionNumber(strTokenizer.nextToken());
551    
552            return lockSummary;
553        }

555        /**
556         * Retrieves the message text for the lock key and fills in message parameters based on the lock type.
557         * 
558         * @return lockKey built from given parameters
559         */
560        protected String buildLockKeyMessage(String lockType, String lockUserId, String chartOfAccountsCode, String accountNumber, String subAccountNumber, Integer fiscalYear, String positionNumber) {
561            KualiConfigurationService kualiConfigurationService = SpringContext.getBean(KualiConfigurationService.class);
562    
563            String lockKeyMessage = "";
564            if (BCConstants.LockTypes.POSITION_LOCK.equals(lockType)) {
565                lockKeyMessage = kualiConfigurationService.getPropertyString(BCKeyConstants.MSG_LOCK_POSITIONKEY);
566                lockKeyMessage = MessageFormat.format(lockKeyMessage, lockUserId, fiscalYear.toString(), positionNumber);
567            }
568            else if (BCConstants.LockTypes.POSITION_FUNDING_LOCK.equals(lockType)) {
569                lockKeyMessage = kualiConfigurationService.getPropertyString(BCKeyConstants.MSG_LOCK_POSITIONFUNDINGKEY);
570                lockKeyMessage = MessageFormat.format(lockKeyMessage, lockUserId, fiscalYear.toString(), positionNumber, chartOfAccountsCode, accountNumber, subAccountNumber);
571            }
572            else {
573                lockKeyMessage = kualiConfigurationService.getPropertyString(BCKeyConstants.MSG_LOCK_ACCOUNTKEY);
574                lockKeyMessage = MessageFormat.format(lockKeyMessage, lockUserId, fiscalYear.toString(), chartOfAccountsCode, accountNumber, subAccountNumber);
575            }
576    
577            return lockKeyMessage;
578        }
