016    package org.kuali.kfs.module.bc.document.validation.event;
017    
018    import org.kuali.kfs.module.bc.businessobject.PendingBudgetConstructionAppointmentFunding;
019    import org.kuali.kfs.module.bc.document.BudgetConstructionDocument;
020    import org.kuali.kfs.module.bc.document.validation.SalarySettingRule;
021    import org.kuali.rice.kns.rule.BusinessRule;
022    
023    public class QuickSaveSalarySettingEvent extends SalarySettingBaseEvent {
024        PendingBudgetConstructionAppointmentFunding appointmentFunding;
025    
026        /**
027         * Constructs a SaveSalarySettingEvent.java.
028         * 
029         * @param description the given description
030         * @param errorPathPrefix the specified error path prefix
031         * @param document the specified budget construction document
032         * @param appointmentFunding the current appointment funding being varified
033         */
034        public QuickSaveSalarySettingEvent(String description, String errorPathPrefix, BudgetConstructionDocument document, PendingBudgetConstructionAppointmentFunding appointmentFunding) {
035            super(description, errorPathPrefix, document);
036            this.appointmentFunding = appointmentFunding;
037        }
038    
039        /**
040         * @see org.kuali.kfs.module.bc.document.validation.event.BudgetExpansionEvent#invokeExpansionRuleMethod(org.kuali.rice.kns.rule.BusinessRule)
041         */
042        @Override
043        public boolean invokeExpansionRuleMethod(BusinessRule rule) {
044            return ((SalarySettingRule) rule).processQuickSaveAppointmentFunding(appointmentFunding);
045        }
046    }
