016    package org.kuali.kfs.module.bc.document.web.struts;
017    
018    import org.apache.struts.upload.FormFile;
019    
020    public class PayrateImportExportForm extends BudgetExpansionForm {
021        
022        private String positionUnionCode;
023        private String csfFreezeDate;
024        private int exportCount;
025        private String fileName;
026        private int importCount;
027        private int updateCount;
028        private FormFile file;
029        private String csfFreezeDateFormattedForExportFile;
030        
031        public FormFile getFile() {
032            return file;
033        }
034    
035        public void setFile(FormFile file) {
036            this.file = file;
037        }
038    
039        public String getFileName() {
040            return fileName;
041        }
042    
043        public void setFileName(String fileName) {
044            this.fileName = fileName;
045        }
046    
047        public int getImportCount() {
048            return importCount;
049        }
050    
051        public void setImportCount(int importCount) {
052            this.importCount = importCount;
053        }
054    
055        public int getUpdateCount() {
056            return updateCount;
057        }
058    
059        public void setUpdateCount(int updateCount) {
060            this.updateCount = updateCount;
061        }
062        
063        public String getCsfFreezeDate() {
064            return csfFreezeDate;
065        }
066    
067        public void setCsfFreezeDate(String csfFreezeDate) {
068            this.csfFreezeDate = csfFreezeDate;
069        }
070    
071        public int getExportCount() {
072            return exportCount;
073        }
074    
075        public void setExportCount(int exportCount) {
076            this.exportCount = exportCount;
077        }
078    
079        public String getPositionUnionCode() {
080            return positionUnionCode;
081        }
082    
083        public void setPositionUnionCode(String positionUnionCode) {
084            if (positionUnionCode != null) this.positionUnionCode = positionUnionCode.toUpperCase();
085        }
086        
087        public String getCsfFreezeDateFormattedForExportFile() {
088            return this.csfFreezeDateFormattedForExportFile;
089        }
090        
091        public void setCsfFreezeDateFormattedForExportFile(String formattedDate) {
092            this.csfFreezeDateFormattedForExportFile = formattedDate;
093        }
094    }
