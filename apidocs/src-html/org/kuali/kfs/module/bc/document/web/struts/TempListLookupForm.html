059        /**
060         * Checks if the get new button for retrieving a position from the payroll service should be enabled.
061         * 
062         * @return true if button should be enabled, false otherwise
063         */
064        public boolean isGetNewPositionEnabled() {
065            // check if position data is maintained externally
066            boolean positionDataExternal = BudgetParameterFinder.getPayrollPositionFeedIndicator();
067            if (!positionDataExternal) {
068                return false;
069            }
070    
071            // check if budget updates and updates from HR are enabled
072            FiscalYearFunctionControlService fiscalYearFunctionControlService = SpringContext.getBean(FiscalYearFunctionControlService.class);
073            boolean budgetUpdatesAllowed = fiscalYearFunctionControlService.isBudgetUpdateAllowed(getUniversityFiscalYear());
074            boolean updatesFromHumanResourcedAllowed = fiscalYearFunctionControlService.isApplicationUpdateFromHumanResourcesAllowed(getUniversityFiscalYear());
075    
076            return budgetUpdatesAllowed && updatesFromHumanResourcedAllowed;
077        }
078    
079        /**
080         * Checks if the get new button for retrieving an incumbent from the payroll service should be enabled.
081         * 
082         * @return true if button should be enabled, false otherwise
083         */
084        public boolean isGetNewIncumbentEnabled() {
085            // check if incumbent data is maintained externally
086            boolean incumbentDataExternal = BudgetParameterFinder.getPayrollIncumbentFeedIndictor();
087            if (!incumbentDataExternal) {
088                return false;
089            }
090    
091            // check if budget updates and updates from HR are enabled
092            FiscalYearFunctionControlService fiscalYearFunctionControlService = SpringContext.getBean(FiscalYearFunctionControlService.class);
093            boolean budgetUpdatesAllowed = fiscalYearFunctionControlService.isBudgetUpdateAllowed(getUniversityFiscalYear());
094            boolean updatesFromHumanResourcedAllowed = fiscalYearFunctionControlService.isApplicationUpdateFromHumanResourcesAllowed(getUniversityFiscalYear());
095    
096            return budgetUpdatesAllowed && updatesFromHumanResourcedAllowed;
097        }

099        /**
100         * @see org.kuali.rice.kns.web.struts.form.LookupForm#populate(javax.servlet.http.HttpServletRequest)
101         */
102        @Override
103        public void populate(HttpServletRequest request) {
104            String refreshCaller = getParameter(request, KNSConstants.REFRESH_CALLER);
105            if (StringUtils.isNotBlank(refreshCaller) && KNSConstants.QUESTION_REFRESH.equals(refreshCaller) ) {
106                setMethodToCall(WebUtils.parseMethodToCall(this, request));
107                if (BCConstants.TEMP_LIST_UNLOCK_METHOD.equals(getMethodToCall())) {
108                    String lookupImplID = KNSServiceLocator.getBusinessObjectDictionaryService().getLookupableID(BudgetConstructionLockSummary.class);
109                    if (lookupImplID == null) {
110                        lookupImplID = "kualiLookupable";
111                    }
112    
113                    setLookupableImplServiceName(lookupImplID);
114                }
115            }
116            
117            super.populate(request);
118        }
