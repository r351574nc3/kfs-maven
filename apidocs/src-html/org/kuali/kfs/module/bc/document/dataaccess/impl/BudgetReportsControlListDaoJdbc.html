209        /**
210         * @see org.kuali.kfs.module.bc.document.dataaccess.BudgetReportsControlListDao#updateReportControlList(java.lang.String, java.lang.Integer,
211         *      java.lang.String, java.lang.String, org.kuali.kfs.module.bc.BCConstants.Report.BuildMode)
212         */
213        public void updateReportControlList(String principalName, Integer universityFiscalYear, String chartOfAccountsCode, String organizationCode, BuildMode buildMode) {
214            // clear out previous data for user
215            clearTempTableByUnvlId("LD_BCN_CTRL_LIST_T", "PERSON_UNVL_ID", principalName);
216    
217            Guid idForSession = new Guid();
218    
219            // build 1st temp table with list of accounts for the selected organizations
220            getSimpleJdbcTemplate().update(updateReportsControlList[0], idForSession.toString(), principalName, universityFiscalYear);
221    
222            // build 2nd temp table with list of accounts from 1 temp table that are also contained in user's point of view
223            getSimpleJdbcTemplate().update(updateReportsControlList[1], idForSession.toString(), principalName, chartOfAccountsCode, organizationCode, idForSession.toString());
224    
225            // constrain account list further based on buildMode
226            switch (buildMode) {
227                case PBGL:
228                    getSimpleJdbcTemplate().update(updateReportsControlList[2], idForSession.toString());
229                    break;
230                case MONTH:
231                    getSimpleJdbcTemplate().update(updateReportsControlList[3], idForSession.toString());
232                    break;
233                case BCAF:
234                    getSimpleJdbcTemplate().update(updateReportsControlList[4], idForSession.toString());
235                    break;
236            }
237    
238            // clear out temp tables
239            clearTempTableBySesId("LD_BCN_BUILD_CTRL_LIST01_MT", "SESID", idForSession.toString());
240            clearTempTableBySesId("LD_BCN_BUILD_CTRL_LIST02_MT", "SESID", idForSession.toString());
241        }
