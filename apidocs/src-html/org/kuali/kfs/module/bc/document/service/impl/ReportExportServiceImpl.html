060    
061            // read u_where %\
062            // (univ_fiscal_yr.ld_pnd_bcnstr_gl_t = univ_fiscal_yr.ld_bcn_acct_dump_t & %\
063            // fin_coa_cd.ld_pnd_bcnstr_gl_t = fin_coa_cd.ld_bcn_acct_dump_t & %\
064            // account_nbr.ld_pnd_bcnstr_gl_t = account_nbr.ld_bcn_acct_dump_t & %\
065            // sub_acct_nbr.ld_pnd_bcnstr_gl_t = sub_acct_nbr.ld_bcn_acct_dump_t) %\
066            // order by "fin_object_cd, fin_sub_obj_cd"
067    
068            /*
069             * Find all BudgetConstructionAccountDump objects for principalId and iterate through returned collection
070             * then, retrieve all PendingBudgetConstructionGeneralLedger objects for fiscal year, chart, account and sub-account of
071             * account dump record (order by object and sub object code). Iterate through this collection. Build up a String with the
072             * fields below and add new line character and then add built up String to StringBuilder. Finally return StringBuilder.
073             */
074    
075            //
076            //
077            // 
078            // $line$ = "%%$line$%%$dlm$%%fdoc_nbr.ld_pnd_bcnstr_gl_t%%$dlm$%%$sep$"
079            // $line$ = "%%$line$%%univ_fiscal_yr.ld_pnd_bcnstr_gl_t%%$sep$"
080            // $line$ = "%%$line$%%$dlm$%%fin_coa_cd.ld_pnd_bcnstr_gl_t%%$dlm$%%$sep$"
081            // $line$ = "%%$line$%%$dlm$%%account_nbr.ld_pnd_bcnstr_gl_t%%$dlm$%%$sep$"
082            // ;add in org code field
083            // $line$ = "%%$line$%%$dlm$%%$org_cd$%%$dlm$%%$sep$"
084            // $line$ = "%%$line$%%$dlm$%%sub_acct_nbr.ld_pnd_bcnstr_gl_t%%$dlm$%%$sep$"
085            // $line$ = "%%$line$%%$dlm$%%fin_object_cd.ld_pnd_bcnstr_gl_t%%$dlm$%%$sep$"
086            // $line$ = "%%$line$%%$dlm$%%fin_sub_obj_cd.ld_pnd_bcnstr_gl_t%%$dlm$%%$sep$"
087            // $line$ = %\
088            // "%%$line$%%$dlm$%%fin_balance_typ_cd.ld_pnd_bcnstr_gl_t%%$dlm$%%$sep$"
089            // $line$ = "%%$line$%%$dlm$%%fin_obj_typ_cd.ld_pnd_bcnstr_gl_t%%$dlm$%%$sep$"
090            // $gennum$ = fin_beg_bal_ln_amt.ld_pnd_bcnstr_gl_t
091            // $line$ = "%%$line$%%$gennum$%%$sep$"
092            // $gennum$ = acln_annl_bal_amt.ld_pnd_bcnstr_gl_t
093            // $line$ = "%%$line$%%$gennum$%%$sep$"
094            // ;rc code added 12/20/2004 - gwp
095            // $line$ = "%%$line$%%$dlm$%%$rc_cd$%%$dlm$"
096            // $line$ = "%%$line$%%^"
097    
098            /*
099             * NOTE: org code and rc code above come from BudgetConstructionAccountReports and BudgetConstructionOrganizationReports As
100             * you iterate, retrieve BudgetConstructionAccountReports based on the ld_pnd_bcnstr_gl_t chart and account, the org code is
101             * reportsToOrganizationCode then, the rc code come from budgetConstructionOrganizationReports.responsibilityCenterCode (a
102             * reference to BudgetConstructionAccountReports)
103             */
104    
105            // update account dump table
106            updateAccountDump(principalId);
107    
108            StringBuilder results = new StringBuilder();
109    
110            List<BudgetConstructionAccountDump> accountDumpRecords = getBudgetConstructionAccountDump(principalId);
111            for (BudgetConstructionAccountDump accountRecord : accountDumpRecords) {
112                List<PendingBudgetConstructionGeneralLedger> pendingEntryList = getPendingBudgetConstructionGeneralLedgerRecords(accountRecord);
113    
114                for (PendingBudgetConstructionGeneralLedger pendingEntry : pendingEntryList) {
115                    results.append(constructAccountDumpLine(pendingEntry, textDelimiter, fieldSeperator));
116                }
117            }
118            reportDumpDao.cleanAccountDump(principalId);
119    
120            return results;
121        }

414        /**
415         * Constructs a line of the Funding Dump File
416         * 
417         * @param fundingRecord
418         * @param fieldSeperator
419         * @param textDelimiter
420         * @return
421         */
422        protected String constructFundingDumpLine(PendingBudgetConstructionAppointmentFunding fundingRecord, String fieldSeperator, String textDelimiter) {
423            HashMap accountReportSearchParameters = new HashMap();
424            accountReportSearchParameters.put(KFSPropertyConstants.CHART_OF_ACCOUNTS_CODE, fundingRecord.getChartOfAccountsCode());
425            accountReportSearchParameters.put(KFSPropertyConstants.ACCOUNT_NUMBER, fundingRecord.getAccountNumber());
426            BudgetConstructionAccountReports accountReport = (BudgetConstructionAccountReports) this.businessObjectService.findByPrimaryKey(BudgetConstructionAccountReports.class, accountReportSearchParameters);
427    
428            if (!fundingRecord.getEmplid().equals(BCConstants.VACANT_EMPLID)) {
429                fundingRecord.refreshReferenceObject(BCPropertyConstants.BUDGET_CONSTRUCTION_INTENDED_INCUMBENT);
430            }
431    
432            String line = "";
433            line = line + fundingRecord.getUniversityFiscalYear() + fieldSeperator;
434            line = line + textDelimiter + fundingRecord.getChartOfAccountsCode() + textDelimiter + fieldSeperator;
435            line = line + textDelimiter + fundingRecord.getAccountNumber() + textDelimiter + fieldSeperator;
436            line = line + textDelimiter + accountReport.getReportsToOrganizationCode() + textDelimiter + fieldSeperator;
437            line = line + textDelimiter + fundingRecord.getSubAccountNumber() + textDelimiter + fieldSeperator;
438            line = line + textDelimiter + fundingRecord.getFinancialObjectCode() + textDelimiter + fieldSeperator;
439            line = line + textDelimiter + fundingRecord.getFinancialSubObjectCode() + textDelimiter + fieldSeperator;
440            line = line + textDelimiter + fundingRecord.getPositionNumber() + textDelimiter + fieldSeperator;
441            line = line + textDelimiter + fundingRecord.getBudgetConstructionPosition().getPositionDescription() + textDelimiter + fieldSeperator;
442            line = line + textDelimiter + fundingRecord.getBudgetConstructionPosition().getSetidSalary() + textDelimiter + fieldSeperator;
443            line = line + textDelimiter + fundingRecord.getBudgetConstructionPosition().getPositionSalaryPlanDefault() + textDelimiter + fieldSeperator;
444            line = line + textDelimiter + fundingRecord.getBudgetConstructionPosition().getPositionGradeDefault() + textDelimiter + fieldSeperator;
445            line = line + fundingRecord.getBudgetConstructionPosition().getIuNormalWorkMonths() + fieldSeperator;
446            line = line + fundingRecord.getBudgetConstructionPosition().getIuPayMonths() + fieldSeperator;
447            line = line + textDelimiter + fundingRecord.getEmplid() + textDelimiter + fieldSeperator;
448    
449            if (ObjectUtils.isNotNull(fundingRecord.getBudgetConstructionIntendedIncumbent())) {
450                line = line + textDelimiter + fundingRecord.getBudgetConstructionIntendedIncumbent().getName() + textDelimiter + fieldSeperator;
451                line = line + textDelimiter + fundingRecord.getBudgetConstructionIntendedIncumbent().getIuClassificationLevel() + textDelimiter + fieldSeperator;
452            }
453            else {
454                line = line + textDelimiter + textDelimiter + fieldSeperator;
455                line = line + textDelimiter + textDelimiter + fieldSeperator;
456            }
457            if (ObjectUtils.isNotNull(fundingRecord.getBudgetConstructionAdministrativePost()))
458                line = line + textDelimiter + fundingRecord.getBudgetConstructionAdministrativePost().getAdministrativePost() + textDelimiter + fieldSeperator;
459            else
460                line = line + textDelimiter + textDelimiter + fieldSeperator;
461    
462            // output blanks when no associated CSF row - relation is optional 1-1 configured in OJB as collection
463            // and typed as TypedArrayList
464            if (fundingRecord.getBcnCalculatedSalaryFoundationTracker().isEmpty()) {
465                line = line + "" + fieldSeperator;
466                line = line + "" + fieldSeperator;
467                line = line + "" + fieldSeperator;
468    
469            }
470            else {
471                line = line + (new KualiDecimal(fundingRecord.getBcnCalculatedSalaryFoundationTracker().get(0).getCsfAmount().intValue())) + fieldSeperator;
472                line = line + (new KualiDecimal(fundingRecord.getBcnCalculatedSalaryFoundationTracker().get(0).getCsfFullTimeEmploymentQuantity())) + fieldSeperator;
473                line = line + (new KualiDecimal(fundingRecord.getBcnCalculatedSalaryFoundationTracker().get(0).getCsfTimePercent())) + fieldSeperator;
474    
475            }
476            line = line + textDelimiter + fundingRecord.getAppointmentFundingDurationCode() + textDelimiter + fieldSeperator;
477            line = line + new KualiDecimal(fundingRecord.getAppointmentRequestedCsfAmount().intValue()) + fieldSeperator;
478            line = line + new KualiDecimal(fundingRecord.getAppointmentRequestedCsfFteQuantity()) + fieldSeperator;
479            line = line + new KualiDecimal(fundingRecord.getAppointmentRequestedTimePercent()) + fieldSeperator;
480            line = line + new KualiDecimal(fundingRecord.getAppointmentTotalIntendedAmount().intValue()) + fieldSeperator;
481            line = line + new KualiDecimal(fundingRecord.getAppointmentTotalIntendedFteQuantity()) + fieldSeperator;
482            line = line + new KualiDecimal(fundingRecord.getAppointmentRequestedAmount().intValue()) + fieldSeperator;
483            line = line + new KualiDecimal(fundingRecord.getAppointmentRequestedTimePercent()) + fieldSeperator;
484            line = line + new KualiDecimal(fundingRecord.getAppointmentRequestedFteQuantity()) + fieldSeperator;
485            line = line + new KualiDecimal(fundingRecord.getAppointmentRequestedPayRate()) + fieldSeperator;
486            line = line + textDelimiter + (fundingRecord.isAppointmentFundingDeleteIndicator() ? "Y" : "N") + textDelimiter + fieldSeperator;
487            line = line + fundingRecord.getAppointmentFundingMonth() + fieldSeperator;
488            
489            List<BudgetConstructionAppointmentFundingReason> appointmentFundingReasonList = fundingRecord.getBudgetConstructionAppointmentFundingReason(); 
490            if (ObjectUtils.isNotNull(appointmentFundingReasonList) && !appointmentFundingReasonList.isEmpty()){
491                line = line + textDelimiter + ((appointmentFundingReasonList.get(0).getAppointmentFundingReasonCode() == null) ? "" : appointmentFundingReasonList.get(0).getAppointmentFundingReasonCode()) + textDelimiter + fieldSeperator;
492            }
493            else {
494                line = line + textDelimiter + "" + textDelimiter + fieldSeperator;
495            }
496            line = line + textDelimiter + accountReport.getBudgetConstructionOrganizationReports().getResponsibilityCenterCode() + textDelimiter;
497            line = line + "\r\n";
498    
499            return line;
500        }
