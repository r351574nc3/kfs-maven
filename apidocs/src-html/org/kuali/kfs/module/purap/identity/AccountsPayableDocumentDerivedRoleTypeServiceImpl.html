016    package org.kuali.kfs.module.purap.identity;
017    
018    import java.util.ArrayList;
019    import java.util.Iterator;
020    import java.util.List;
021    
022    import org.kuali.kfs.module.purap.PurapConstants;
023    import org.kuali.kfs.module.purap.document.AccountsPayableDocument;
024    import org.kuali.kfs.sys.KFSConstants;
025    import org.kuali.kfs.sys.businessobject.SourceAccountingLine;
026    import org.kuali.kfs.sys.context.SpringContext;
027    import org.kuali.kfs.sys.identity.KfsKimAttributes;
028    import org.kuali.rice.kew.exception.WorkflowException;
029    import org.kuali.rice.kim.bo.impl.KimAttributes;
030    import org.kuali.rice.kim.bo.types.dto.AttributeSet;
031    import org.kuali.rice.kim.service.RoleManagementService;
032    import org.kuali.rice.kim.service.support.impl.KimDerivedRoleTypeServiceBase;
033    import org.kuali.rice.kns.service.DocumentService;
034    
035    public class AccountsPayableDocumentDerivedRoleTypeServiceImpl extends KimDerivedRoleTypeServiceBase {
036        private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(AccountsPayableDocumentDerivedRoleTypeServiceImpl.class);
037        
038        protected static final String FISCAL_OFFICER_ROLE_NAME = "Fiscal Officer";
039        protected static final String SUB_ACCOUNT_ROLE_NAME = "Sub-Account Reviewer";
040        protected static final String ACCOUNTING_REVIEWER_ROLE_NAME = "Accounting Reviewer";
041        
042        /**
043         * Overridden to check if the current user has document reviewer permission based on the data in the document.
044         * 
045         * @see org.kuali.rice.kim.service.support.impl.KimRoleTypeServiceBase#(java.lang.String, java.util.List, java.lang.String, java.lang.String, org.kuali.rice.kim.bo.types.dto.AttributeSet)
046         */
047        @Override
048        public boolean hasApplicationRole(String principalId, List<String> groupIds, String namespaceCode, String roleName, AttributeSet qualification) {
049        
050            String docId = new String(qualification.get(KimAttributes.DOCUMENT_NUMBER));
051            
052            List roleIds = new ArrayList();
053            roleIds.add(SpringContext.getBean(RoleManagementService.class).getRoleIdByName(KFSConstants.ParameterNamespaces.KFS, FISCAL_OFFICER_ROLE_NAME));
054            roleIds.add(SpringContext.getBean(RoleManagementService.class).getRoleIdByName(PurapConstants.PURAP_NAMESPACE, SUB_ACCOUNT_ROLE_NAME));
055            roleIds.add(SpringContext.getBean(RoleManagementService.class).getRoleIdByName(KFSConstants.ParameterNamespaces.KFS, ACCOUNTING_REVIEWER_ROLE_NAME    ));
056            
057            try {
058                AccountsPayableDocument apDocument = (AccountsPayableDocument)SpringContext.getBean(DocumentService.class).getByDocumentHeaderId(docId);
059                for (Iterator iter = apDocument.getSourceAccountingLines().iterator(); iter.hasNext();) {
060                    SourceAccountingLine accountingLine = (SourceAccountingLine) iter.next();
061                    
062                    AttributeSet roleQualifier = new AttributeSet();
063                    roleQualifier.put(KfsKimAttributes.DOCUMENT_TYPE_NAME, apDocument.getDocumentHeader().getWorkflowDocument().getDocumentType());
064                    roleQualifier.put(KfsKimAttributes.FINANCIAL_DOCUMENT_TOTAL_AMOUNT, apDocument.getDocumentHeader().getFinancialDocumentTotalAmount().toString());
065                    roleQualifier.put(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE, accountingLine.getChartOfAccountsCode());
066                    roleQualifier.put(KfsKimAttributes.ORGANIZATION_CODE, accountingLine.getAccount().getOrganizationCode());
067                    roleQualifier.put(KfsKimAttributes.ACCOUNT_NUMBER, accountingLine.getAccountNumber());
068                    roleQualifier.put(KfsKimAttributes.SUB_ACCOUNT_NUMBER, accountingLine.getSubAccountNumber());
069                    roleQualifier.put(KfsKimAttributes.ACCOUNTING_LINE_OVERRIDE_CODE, accountingLine.getOverrideCode());
070    
071                    if (SpringContext.getBean(RoleManagementService.class).principalHasRole(principalId, roleIds, roleQualifier)) {
072                        return true;
073                    }
074                    
075                }
076            }
077            catch (WorkflowException we) {
078                LOG.error("Exception encountered when retrieving document number " + docId + ".", we);
079                throw new RuntimeException("Exception encountered when retrieving document number " + docId + ".", we);
080            }
081            
082            return false;
083        }
084    
085        
086    }
