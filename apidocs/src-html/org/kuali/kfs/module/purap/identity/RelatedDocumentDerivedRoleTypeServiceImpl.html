017    
018    import java.util.ArrayList;
019    import java.util.List;
020    
021    import org.apache.commons.lang.StringUtils;
022    import org.kuali.kfs.module.purap.document.PurchasingAccountsPayableDocument;
023    import org.kuali.kfs.module.purap.document.service.PurapService;
024    import org.kuali.kfs.sys.context.SpringContext;
025    import org.kuali.kfs.sys.identity.KfsKimAttributes;
026    import org.kuali.rice.kew.exception.WorkflowException;
027    import org.kuali.rice.kew.role.service.impl.RouteLogDerivedRoleTypeServiceImpl;

029    import org.kuali.rice.kim.bo.role.dto.RoleMembershipInfo;
030    import org.kuali.rice.kim.bo.types.dto.AttributeSet;
031    import org.kuali.rice.kim.service.RoleManagementService;

033    import org.kuali.rice.kns.service.DocumentService;
034    import org.kuali.rice.kns.util.KNSConstants;
035    
036    public class RelatedDocumentDerivedRoleTypeServiceImpl extends KimDerivedRoleTypeServiceBase {
037        protected static final String SOURCE_DOCUMENT_ROUTER_ROLE_NAME = "Source Document Router";
038        protected static final String SENSITIVE_RELATED_DOCUMENT_INITATOR_OR_REVIEWER_ROLE_NAME = "Sensitive Related Document Initiator Or Reviewer";
039    
040        private DocumentService documentService;
041        private PurapService purapService;
042        private RoleManagementService roleManagementService;
043    
044        /**
045         * This service will accept the following attributes: Document Number Context: An fyi to the initiator - in the case of
046         * Automatic Purchase Orders (apo), the fyi is supposed to go to the requisition router. Otherwise, it should go to the PO
047         * router. Requirements: - KFS-PURAP Source Document Router - for Automated Purchase Order, Requisition router according to
048         * KR-WKFLW Router role / for normal Purchase Order, Purchase Order router according to KR-WKFLW Router
049         * 
050         * @see org.kuali.rice.kim.service.support.impl.KimRoleTypeServiceBase#getPrincipalIdsFromApplicationRole(java.lang.String,
051         *      java.lang.String, org.kuali.rice.kim.bo.types.dto.AttributeSet)
052         */
053        @Override
054        public List<RoleMembershipInfo> getRoleMembersFromApplicationRole(String namespaceCode, String roleName, AttributeSet qualification) {
055            List<RoleMembershipInfo> members = new ArrayList<RoleMembershipInfo>();
056            if(qualification!=null && !qualification.isEmpty()){
057                if (SOURCE_DOCUMENT_ROUTER_ROLE_NAME.equals(roleName)) {
058                    try {
059                        PurchasingAccountsPayableDocument document = (PurchasingAccountsPayableDocument) getDocumentService().getByDocumentHeaderId(qualification.get(PurapKimAttributes.DOCUMENT_NUMBER));
060                        if (document != null) {
061                            PurchasingAccountsPayableDocument sourceDocument = document.getPurApSourceDocumentIfPossible();
062                            if (sourceDocument != null && StringUtils.isNotBlank(sourceDocument.getDocumentHeader().getWorkflowDocument().getRoutedByPrincipalId()) ) {
063                                AttributeSet roleQualifier = new AttributeSet(1);
064                                roleQualifier.put(KfsKimAttributes.DOCUMENT_NUMBER, sourceDocument.getDocumentNumber() );
065                                members.add( new RoleMembershipInfo(null,null,sourceDocument.getDocumentHeader().getWorkflowDocument().getRoutedByPrincipalId(),Role.PRINCIPAL_MEMBER_TYPE,roleQualifier) );
066                            }
067                        }
068                    }
069                    catch (WorkflowException e) {
070                        throw new RuntimeException("Unable to load document in getPrincipalIdsFromApplicationRole", e);
071                    }
072                }
073                else if (SENSITIVE_RELATED_DOCUMENT_INITATOR_OR_REVIEWER_ROLE_NAME.equals(roleName)) {
074                    for (String documentId : getPurapService().getRelatedDocumentIds(new Integer(qualification.get(PurapKimAttributes.ACCOUNTS_PAYABLE_PURCHASING_DOCUMENT_LINK_IDENTIFIER)))) {
075                        AttributeSet tempQualification = new AttributeSet(1);
076                        tempQualification.put(PurapKimAttributes.DOCUMENT_NUMBER, documentId);
077                        for ( String principalId : getRoleManagementService().getRoleMemberPrincipalIds(KNSConstants.KUALI_RICE_WORKFLOW_NAMESPACE, RouteLogDerivedRoleTypeServiceImpl.INITIATOR_OR_REVIEWER_ROLE_NAME, tempQualification) ) {
078                            members.add( new RoleMembershipInfo(null,null,principalId,Role.PRINCIPAL_MEMBER_TYPE,tempQualification) );
079                        }
080                    }
081                }
082            }
083            return members;
084        }
085    
086        protected DocumentService getDocumentService() {
087            if (documentService == null) {
088                documentService = SpringContext.getBean(DocumentService.class);
089            }
090            return documentService;
091        }
092    
093        protected PurapService getPurapService() {
094            if (purapService == null) {
095                purapService = SpringContext.getBean(PurapService.class);
096            }
097            return purapService;
098        }
099    
100        protected RoleManagementService getRoleManagementService() {
101            if (roleManagementService == null) {
102                roleManagementService = SpringContext.getBean(RoleManagementService.class);
103            }
104            return roleManagementService;
105        }
106    }
