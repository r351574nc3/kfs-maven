016    package org.kuali.kfs.module.purap.businessobject.options;
017    
018    import java.util.ArrayList;
019    import java.util.List;
020    
021    import org.kuali.kfs.module.purap.PurapConstants;
022    import org.kuali.kfs.vnd.VendorConstants;
023    import org.kuali.rice.kns.lookup.keyvalues.KeyValuesBase;
024    import org.kuali.rice.core.util.KeyLabelPair;
025    
026    /**
027     * Value Finder for manual Purchase Order Status Changes.
028     */
029    public class StatusChangeValuesFinder extends KeyValuesBase {
030    
031        /**
032         * Returns code/description pairs of all manual Purchase Order Status Changes.
033         * 
034         * @see org.kuali.rice.kns.lookup.keyvalues.KeyValuesFinder#getKeyValues()
035         */
036        public List getKeyValues() {
037            List labels = new ArrayList();
038            labels.add(new KeyLabelPair(PurapConstants.PurchaseOrderStatuses.IN_PROCESS, VendorConstants.NONE));
039            labels.add(new KeyLabelPair(PurapConstants.PurchaseOrderStatuses.WAITING_FOR_DEPARTMENT, "Department"));
040            labels.add(new KeyLabelPair(PurapConstants.PurchaseOrderStatuses.WAITING_FOR_VENDOR, "Vendor"));
041            return labels;
042        }
043    }
