016    package org.kuali.kfs.module.purap.web.struts;
017    
018    import javax.servlet.http.HttpServletRequest;
019    
020    import org.kuali.rice.kns.util.KNSConstants;
021    import org.kuali.rice.kns.web.struts.form.KualiForm;
022    
023    /**
024     * Struts action form for the b2b shopping
025     */
026    public class B2BForm extends KualiForm {
027        private String shopUrl;
028    
029        public String getShopUrl() {
030            return shopUrl;
031        }
032    
033        public void setShopUrl(String shopUrl) {
034            this.shopUrl = shopUrl;
035        }
036    
037        /**
038         * @see org.kuali.rice.kns.web.struts.form.KualiDocumentFormBase#shouldMethodToCallParameterBeUsed(java.lang.String, java.lang.String, javax.servlet.http.HttpServletRequest)
039         */
040        @Override
041        public boolean shouldMethodToCallParameterBeUsed(String methodToCallParameterName, String methodToCallParameterValue, HttpServletRequest request) {
042            if (KNSConstants.DISPATCH_REQUEST_PARAMETER.equals(methodToCallParameterName) && 
043               ("shopCatalogs".equals(methodToCallParameterValue) || "returnFromShopping".equals(methodToCallParameterValue))) {
044                return true;
045            }
046            return super.shouldMethodToCallParameterBeUsed(methodToCallParameterName, methodToCallParameterValue, request);
047        }
048    
049    }
