016    package org.kuali.kfs.module.purap.batch;
017    
018    import java.io.File;
019    
020    import org.kuali.kfs.module.purap.PurapConstants;
021    import org.kuali.kfs.module.purap.PurapKeyConstants;
022    import org.kuali.kfs.module.purap.businessobject.ElectronicInvoice;
023    import org.kuali.kfs.sys.batch.XmlBatchInputFileTypeBase;
024    
025    /**
026     * Batch input type for the electronic invoice job.
027     */
028    public class ElectronicInvoiceInputFileType extends XmlBatchInputFileTypeBase {
029        protected String reportPath;
030        protected String reportPrefix;
031        protected String reportExtension;
032        
033        /**
034         * Returns the identifier of the electronic invoice file type
035         * 
036         * @return the electronic invoice file type identifier
037         */
038        public String getFileTypeIdentifer() {
039            return PurapConstants.ELECTRONIC_INVOICE_FILE_TYPE_INDENTIFIER;
040        }
041    
042        public boolean validate(Object parsedFileContents) {
043            return true;
044        }
045    
046        public String getTitleKey() {
047            return PurapKeyConstants.MESSAGE_BATCH_UPLOAD_TITLE_EINVOICE;
048        }
049    
050        public String getAuthorPrincipalName(File file) {
051            return null;
052        }
053    
054        public String getFileName(String principalName, 
055                                  Object parsedFileContents, 
056                                  String fileUserIdentifer) {
057            
058            if (!(parsedFileContents instanceof ElectronicInvoice)){
059                throw new RuntimeException("Invalid object type.");
060            }
061            
062            String fileName = ((ElectronicInvoice)parsedFileContents).getFileName();
063            int whereDot = fileName.lastIndexOf('.');
064            
065            return fileName.substring(0, whereDot);
066        }
067        
068        
069        /**
070         * Sets the reportPath attribute value.
071         * @param reportPath The reportPath to set.
072         */
073        public void setReportPath(String reportPath) {
074            this.reportPath = reportPath;
075        }
076    
077        /**
078         * Sets the reportExtension attribute value.
079         * @param reportExtension The reportExtension to set.
080         */
081        public void setReportExtension(String reportExtension) {
082            this.reportExtension = reportExtension;
083        }
084    
085        /**
086         * Sets the fileReportPrefix attribute value.
087         * @param fileReportPrefix The fileReportPrefix to set.
088         */
089        public void setReportPrefix(String reportPrefix) {
090            this.reportPrefix = reportPrefix;
091        }
092    
093        public String getReportPath() {
094            return reportPath;
095        }
096    
097        public String getReportPrefix() {
098            return reportPrefix;
099        }
100    
101        public String getReportExtension() {
102            return reportExtension;
103        }
104    }
