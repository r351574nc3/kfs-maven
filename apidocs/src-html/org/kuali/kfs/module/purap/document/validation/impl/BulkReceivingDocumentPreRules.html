043    public class BulkReceivingDocumentPreRules extends PromptBeforeValidationBase{
044    
045        public boolean doPrompts(Document document) {
046            
047            BulkReceivingDocument bulkReceivingDocument = (BulkReceivingDocument)document;
048            
049            HashMap<String, String> duplicateMessages = SpringContext.getBean(BulkReceivingService.class).bulkReceivingDuplicateMessages(bulkReceivingDocument);
050            
051            if (duplicateMessages != null && !duplicateMessages.isEmpty()) {
052                Iterator iterator = duplicateMessages.values().iterator();
053                StringBuffer msg = new StringBuffer();
054                while(iterator.hasNext()) {
055                    msg.append(iterator.next());
056                }
057                boolean proceed = super.askOrAnalyzeYesNoQuestion(PurapConstants.BulkReceivingDocumentStrings.DUPLICATE_BULK_RECEIVING_DOCUMENT_QUESTION, msg.toString());
058                if (!proceed) {
059                    abortRulesCheck();
060                }
061            }
062            
063            return true;
064        }
