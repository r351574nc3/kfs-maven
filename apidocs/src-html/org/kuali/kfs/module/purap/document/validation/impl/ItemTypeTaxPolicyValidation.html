016    package org.kuali.kfs.module.purap.document.validation.impl;
017    
018    import org.kuali.kfs.module.purap.PurapParameterConstants;
019    import org.kuali.kfs.module.purap.businessobject.PurApItem;
020    import org.kuali.kfs.module.purap.document.PurchasingDocument;
021    import org.kuali.kfs.module.purap.document.service.PurapService;
022    import org.kuali.kfs.sys.context.SpringContext;
023    import org.kuali.kfs.sys.document.validation.GenericValidation;
024    import org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent;
025    import org.kuali.kfs.sys.service.impl.KfsParameterConstants;
026    import org.kuali.rice.kns.service.ParameterService;
027    import org.kuali.rice.kns.util.GlobalVariables;
028    
029    public class ItemTypeTaxPolicyValidation extends GenericValidation {
030        private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(ItemTypeTaxPolicyValidation.class);
031        protected PurapService purapService;
032        
033        public static final String ERROR_INVALID_ITEM_TYPE_FOR_ACCOUNT_TAX_POLICY="error.itemTypeCode.taxPolicy";
034        public static final String FIELD_NAME_ITEM_TYPE_CODE = "itemTypeCode";    
035        
036    
037        /**
038         * @see org.kuali.kfs.sys.document.validation.Validation#validate(org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent)
039         */
040        public boolean validate(AttributedDocumentEvent event) {
041            boolean isValid = true;
042            PurchasingDocument purchasingDocument = (PurchasingDocument)event.getDocument();
043    
044            boolean salesTaxInd = SpringContext.getBean(ParameterService.class).getIndicatorParameter(KfsParameterConstants.PURCHASING_DOCUMENT.class, PurapParameterConstants.ENABLE_SALES_TAX_IND);
045            boolean useTaxIndicator = purchasingDocument.isUseTaxIndicator();
046    
047            //FIXME: KFSMI-6006 remove call code variable once this code has been fixed
048            boolean callCode = false;
049            
050            //if sales tax or use tax is enabled, attempt tax check
051            if( callCode && (salesTaxInd || useTaxIndicator) ){
052                String errorPath = "document.items";
053                GlobalVariables.getMessageMap().clearErrorPath();
054                GlobalVariables.getMessageMap().addToErrorPath(errorPath);                    
055                for ( PurApItem item : purchasingDocument.getItems() )
056                {
057                    if ( getPurapService().isItemTypeConflictWithTaxPolicy(purchasingDocument, item)) {
058                        String itemIdentifier = item.getItemIdentifierString();
059                        
060                        GlobalVariables.getMessageMap().putError(FIELD_NAME_ITEM_TYPE_CODE, ERROR_INVALID_ITEM_TYPE_FOR_ACCOUNT_TAX_POLICY, itemIdentifier );
061                        isValid = false;
062                    }
063                }
064            }
065            
066            return isValid;
067        }
068    
069    
070        public PurapService getPurapService() {
071            return purapService;
072        }
073    
074    
075        public void setPurapService(PurapService purapService) {
076            this.purapService = purapService;
077        }
078    
079    }
