041    public class ItemTypeRule extends MaintenanceDocumentRuleBase {
042    
043        protected boolean processCustomApproveDocumentBusinessRules(MaintenanceDocument document) {
044            LOG.info("processCustomApproveDocumentBusinessRules called");
045            this.setupConvenienceObjects();
046            boolean success = this.checkForSystemParametersExistence();
047            return success && super.processCustomApproveDocumentBusinessRules(document);
048        }
049    
050        protected boolean processCustomRouteDocumentBusinessRules(MaintenanceDocument document) {
051            LOG.info("processCustomRouteDocumentBusinessRules called");
052            this.setupConvenienceObjects();
053            boolean success = this.checkForSystemParametersExistence() && this.checkIndicators();
054            return success && super.processCustomRouteDocumentBusinessRules(document);
055        }
056    
057        protected boolean processCustomSaveDocumentBusinessRules(MaintenanceDocument document) {
058            LOG.info("processCustomSaveDocumentBusinessRules called");
059            this.setupConvenienceObjects();
060            boolean success = this.checkForSystemParametersExistence();
061            return success && super.processCustomSaveDocumentBusinessRules(document);
062        }
063    
064        protected boolean checkForSystemParametersExistence() {
065            LOG.info("checkForSystemParametersExistence called");
066            boolean success = true;
067            List<String> cmAdditionalCharges = SpringContext.getBean(ParameterService.class).getParameterValues(VendorCreditMemoDocument.class, PurapConstants.BELOW_THE_LINES_PARAMETER);
068            List<String> preqAdditionalCharges = SpringContext.getBean(ParameterService.class).getParameterValues(PaymentRequestDocument.class, PurapConstants.BELOW_THE_LINES_PARAMETER);
069            List<String> poAdditionalCharges = SpringContext.getBean(ParameterService.class).getParameterValues(PurchaseOrderDocument.class, PurapConstants.BELOW_THE_LINES_PARAMETER);
070            List<String> reqAdditionalCharges = SpringContext.getBean(ParameterService.class).getParameterValues(RequisitionDocument.class, PurapConstants.BELOW_THE_LINES_PARAMETER);
071            List<String> defaultNonQuantityItemTypes = SpringContext.getBean(KualiConfigurationService.class).getParameterValues(PurapConstants.PURAP_NAMESPACE, "Document", PurapParameterConstants.DEFAULT_NON_QUANTITY_ITEM_TYPE);
072            List<String> defaultQuantityItemTypes = SpringContext.getBean(KualiConfigurationService.class).getParameterValues(PurapConstants.PURAP_NAMESPACE, "Document", PurapParameterConstants.DEFAULT_QUANTITY_ITEM_TYPE);
073            List<String> cmAllowNegative = SpringContext.getBean(ParameterService.class).getParameterValues(VendorCreditMemoDocument.class, PurapConstants.ITEM_ALLOWS_NEGATIVE);
074            List<String> preqAllowNegative = SpringContext.getBean(ParameterService.class).getParameterValues(PaymentRequestDocument.class, PurapConstants.ITEM_ALLOWS_NEGATIVE);
075            List<String> poAllowNegative = SpringContext.getBean(ParameterService.class).getParameterValues(PurchaseOrderDocument.class, PurapConstants.ITEM_ALLOWS_NEGATIVE);
076            List<String> reqAllowNegative = SpringContext.getBean(ParameterService.class).getParameterValues(RequisitionDocument.class, PurapConstants.ITEM_ALLOWS_NEGATIVE);
077            List<String> cmAllowPositive = SpringContext.getBean(ParameterService.class).getParameterValues(VendorCreditMemoDocument.class, PurapConstants.ITEM_ALLOWS_POSITIVE);
078            List<String> preqAllowPositive = SpringContext.getBean(ParameterService.class).getParameterValues(PaymentRequestDocument.class, PurapConstants.ITEM_ALLOWS_POSITIVE);
079            List<String> poAllowPositive = SpringContext.getBean(ParameterService.class).getParameterValues(PurchaseOrderDocument.class, PurapConstants.ITEM_ALLOWS_POSITIVE);
080            List<String> reqAllowPositive = SpringContext.getBean(ParameterService.class).getParameterValues(RequisitionDocument.class, PurapConstants.ITEM_ALLOWS_POSITIVE);
081            List<String> cmAllowZero = SpringContext.getBean(ParameterService.class).getParameterValues(VendorCreditMemoDocument.class, PurapConstants.ITEM_ALLOWS_ZERO);
082            List<String> preqAllowZero = SpringContext.getBean(ParameterService.class).getParameterValues(PaymentRequestDocument.class, PurapConstants.ITEM_ALLOWS_ZERO);
083            List<String> poAllowZero = SpringContext.getBean(ParameterService.class).getParameterValues(PurchaseOrderDocument.class, PurapConstants.ITEM_ALLOWS_ZERO);
084            List<String> reqAllowZero = SpringContext.getBean(ParameterService.class).getParameterValues(RequisitionDocument.class, PurapConstants.ITEM_ALLOWS_ZERO);
085            List<String> cmRequireDescription = SpringContext.getBean(ParameterService.class).getParameterValues(VendorCreditMemoDocument.class, PurapConstants.ITEM_REQUIRES_USER_ENTERED_DESCRIPTION);
086            List<String> preqRequireDescription = SpringContext.getBean(ParameterService.class).getParameterValues(PaymentRequestDocument.class, PurapConstants.ITEM_REQUIRES_USER_ENTERED_DESCRIPTION);
087            List<String> poRequireDescription = SpringContext.getBean(ParameterService.class).getParameterValues(PurchaseOrderDocument.class, PurapConstants.ITEM_REQUIRES_USER_ENTERED_DESCRIPTION);
088            List<String> reqRequireDescription = SpringContext.getBean(ParameterService.class).getParameterValues(RequisitionDocument.class, PurapConstants.ITEM_REQUIRES_USER_ENTERED_DESCRIPTION);
089            List<String> preqRestrictingAccountEdit = SpringContext.getBean(ParameterService.class).getParameterValues(PaymentRequestDocument.class, PurapParameterConstants.PURAP_ITEM_TYPES_RESTRICTING_ACCOUNT_EDIT);
090            List<String> poRestrictingAccountEdit = SpringContext.getBean(ParameterService.class).getParameterValues(PurchaseOrderDocument.class, PurapParameterConstants.PURAP_ITEM_TYPES_RESTRICTING_ACCOUNT_EDIT);
091            List<String> reqRestrictingAccountEdit = SpringContext.getBean(ParameterService.class).getParameterValues(RequisitionDocument.class, PurapParameterConstants.PURAP_ITEM_TYPES_RESTRICTING_ACCOUNT_EDIT);
092            
093            ItemType newBo = (ItemType)getNewBo();
094            ItemType oldBo= (ItemType)getOldBo();
095           
096            if ((!newBo.isActive() && oldBo.isActive()) &&
097                 (cmAdditionalCharges.contains(newBo.getItemTypeCode()) || 
098                  preqAdditionalCharges.contains(newBo.getItemTypeCode()) ||
099                  poAdditionalCharges.contains(newBo.getItemTypeCode()) ||
100                  reqAdditionalCharges.contains(newBo.getItemTypeCode()) ||
101                  defaultNonQuantityItemTypes.contains(newBo.getItemTypeCode()) ||
102                  defaultQuantityItemTypes.contains(newBo.getItemTypeCode()) ||
103                  cmAllowNegative.contains(newBo.getItemTypeCode()) ||
104                  preqAllowNegative.contains(newBo.getItemTypeCode()) ||
105                  poAllowNegative.contains(newBo.getItemTypeCode()) ||
106                  reqAllowNegative.contains(newBo.getItemTypeCode()) ||
107                  cmAllowPositive.contains(newBo.getItemTypeCode()) ||
108                  preqAllowPositive.contains(newBo.getItemTypeCode()) ||
109                  poAllowPositive.contains(newBo.getItemTypeCode()) ||
110                  reqAllowPositive.contains(newBo.getItemTypeCode()) ||
111                  cmAllowZero.contains(newBo.getItemTypeCode()) ||
112                  preqAllowZero.contains(newBo.getItemTypeCode()) ||
113                  poAllowZero.contains(newBo.getItemTypeCode()) ||
114                  reqAllowZero.contains(newBo.getItemTypeCode()) ||
115                  cmRequireDescription.contains(newBo.getItemTypeCode()) ||
116                  preqRequireDescription.contains(newBo.getItemTypeCode()) ||
117                  poRequireDescription.contains(newBo.getItemTypeCode()) ||
118                  reqRequireDescription.contains(newBo.getItemTypeCode()) ||
119                  preqRestrictingAccountEdit.contains(newBo.getItemTypeCode()) ||
120                  poRestrictingAccountEdit.contains(newBo.getItemTypeCode()) ||
121                  reqRestrictingAccountEdit.contains(newBo.getItemTypeCode()) )
122                ) {
123                success = false;
124                String documentLabel = SpringContext.getBean(BusinessObjectDictionaryService.class).getMaintainableLabel(newBo.getClass()); 
125                putGlobalError(KFSKeyConstants.ERROR_CANNOT_INACTIVATE_USED_IN_SYSTEM_PARAMETERS, documentLabel);
126            }
127            
128            return success;
129        }
130        
131        protected boolean checkIndicators() {
132            
133            boolean checkResult = true;
134            ItemType newBo = (ItemType)getNewBo();
135                    
136            // Both Quantity Based General Ledger Indicator and Additional Charge Indicator cannot be Yes at the same time
137            if (newBo.isActive() && newBo.isAdditionalChargeIndicator() && newBo.isQuantityBasedGeneralLedgerIndicator()) {
138                putFieldError(PurapPropertyConstants.ITEM_TYPE_QUANTITY_BASED, PurapKeyConstants.ERROR_ITEM_TYPE_QUANTITY_BASED_NOT_ALLOWED_WITH_ADDITIONAL_CHARGE);
139                checkResult = false;
140            }
141            
142            return checkResult;
143        }
144    }
