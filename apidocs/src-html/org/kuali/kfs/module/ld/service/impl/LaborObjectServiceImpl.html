016    package org.kuali.kfs.module.ld.service.impl;
017    
018    import java.util.Collection;
019    import java.util.List;
020    import java.util.Map;
021    
022    import org.kuali.kfs.module.ld.businessobject.LaborObject;
023    import org.kuali.kfs.module.ld.dataaccess.LaborObjectDao;
024    import org.kuali.kfs.module.ld.service.LaborObjectService;
025    import org.springframework.transaction.annotation.Transactional;
026    
027    /**
028     * This interface provides its clients with access to labor object in the backend data store.
029     * 
030     * @see org.kuali.kfs.module.ld.businessobject.LaborObject
031     */
032    @Transactional
033    public class LaborObjectServiceImpl implements LaborObjectService {
034        private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(LaborObjectServiceImpl.class);
035    
036        private LaborObjectDao laborObjectDao;
037    
038        /**
039         * @see org.kuali.kfs.module.ld.service.LaborObjectService#findAllLaborObjectInPositionGroups(java.util.Map, java.util.List)
040         */
041        public Collection<LaborObject> findAllLaborObjectInPositionGroups(Map<String, Object> fieldValues, List<String> positionGroupCodes) {
042            return laborObjectDao.findAllLaborObjectInPositionGroups(fieldValues, positionGroupCodes);
043        }
044    
045        /**
046         * Sets the laborObjectDao attribute value.
047         * @param laborObjectDao The laborObjectDao to set.
048         */
049        public void setLaborObjectDao(LaborObjectDao laborObjectDao) {
050            this.laborObjectDao = laborObjectDao;
051        }
052    }
