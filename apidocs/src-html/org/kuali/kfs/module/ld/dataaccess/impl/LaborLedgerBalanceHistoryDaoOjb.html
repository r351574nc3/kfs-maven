016    package org.kuali.kfs.module.ld.dataaccess.impl;
017    
018    import java.util.ArrayList;
019    import java.util.Iterator;
020    import java.util.List;
021    
022    import org.apache.ojb.broker.query.Criteria;
023    import org.apache.ojb.broker.query.QueryFactory;
024    import org.apache.ojb.broker.query.ReportQueryByCriteria;
025    import org.kuali.kfs.gl.dataaccess.LedgerBalanceHistoryBalancingDao;
026    import org.kuali.kfs.module.ld.businessobject.LaborBalanceHistory;
027    import org.kuali.kfs.sys.KFSPropertyConstants;
028    import org.kuali.rice.kns.dao.impl.PlatformAwareDaoBaseOjb;
029    
030    /**
031     * This is the data access object for labor balance history
032     * 
033     * @see org.kuali.kfs.module.ld.businessobject.LaborBalanceHistory
034     */
035    public class LaborLedgerBalanceHistoryDaoOjb extends PlatformAwareDaoBaseOjb implements LedgerBalanceHistoryBalancingDao {
036        private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(LaborLedgerBalanceHistoryDaoOjb.class);
037    
038        /**
039         * @see org.kuali.kfs.gl.dataaccess.LedgerBalanceHistoryBalancingDao#findDistinctFiscalYears()
040         */
041        public List<Integer> findDistinctFiscalYears() {
042            Criteria crit = new Criteria();
043            ReportQueryByCriteria q = QueryFactory.newReportQuery(LaborBalanceHistory.class, crit);
044            q.setAttributes(new String[] { KFSPropertyConstants.UNIVERSITY_FISCAL_YEAR });
045            q.setDistinct(true);
046    
047            Iterator<Object[]> years = getPersistenceBrokerTemplate().getReportQueryIteratorByQuery(q);
048            List<Integer> yearList = new ArrayList<Integer>();
049            
050            while (years != null && years.hasNext()) {
051                Object[] year = years.next();
052                yearList.add(new Integer(year[0].toString()));
053            }
054            
055            return yearList;
056        }
057    }
