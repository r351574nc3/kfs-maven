053        /**
054         * Populates source a accounting line encumbrance code (D, R or null) for a given source accounting line 
055         * 
056         * @see org.kuali.kfs.fp.document.web.struts.JournalVoucherForm#populateSourceAccountingLineEncumbranceCode(org.kuali.kfs.sys.businessobject.SourceAccountingLine)
057         */
058        @Override
059        protected void populateSourceAccountingLineEncumbranceCode(SourceAccountingLine sourceLine) {
060            BalanceType selectedBalanceType = getSelectedBalanceType();
061            if (selectedBalanceType != null && StringUtils.isNotBlank(selectedBalanceType.getCode())) {
062                sourceLine.setBalanceTyp(selectedBalanceType);
063                sourceLine.setBalanceTypeCode(selectedBalanceType.getCode());
064    
065                /*
066                 * set the encumbrance update code appropriately. If balance type is an external encumbrance, update code can be D or R.
067                 * For all other encumbrance types it must be D. If the balance type is not an encumbrance type the update code is set
068                 * to null
069                 */
070                if (selectedBalanceType.isFinBalanceTypeEncumIndicator()) {
071                    if (StringUtils.isBlank(sourceLine.getEncumbranceUpdateCode()) || !KFSConstants.ENCUMB_UPDT_DOCUMENT_CD.equals(sourceLine.getEncumbranceUpdateCode())) {
072                        sourceLine.setEncumbranceUpdateCode(KFSConstants.ENCUMB_UPDT_REFERENCE_DOCUMENT_CD);
073                    }
074                }
075                else {
076                    sourceLine.setEncumbranceUpdateCode(null);
077                }
078            }
079            else {
080                // it's the first time in, the form will be empty the first time in set up default selection value
081                selectedBalanceType = SpringContext.getBean(BalanceTypeService.class).getBalanceTypeByCode(KFSConstants.BALANCE_TYPE_ACTUAL);
082                setSelectedBalanceType(selectedBalanceType);
083                setOriginalBalanceType(selectedBalanceType.getCode());
084    
085                sourceLine.setEncumbranceUpdateCode(null);
086            }
087        }
