016    package org.kuali.kfs.module.ld.document.service;
017    
018    import java.io.File;
019    import java.io.IOException;
020    import java.io.OutputStream;
021    import java.util.Iterator;
022    import java.util.List;
023    
024    import org.kuali.kfs.gl.businessobject.CorrectionChangeGroup;
025    import org.kuali.kfs.gl.document.CorrectionDocumentUtils;
026    import org.kuali.kfs.gl.document.GeneralLedgerCorrectionProcessDocument;
027    import org.kuali.kfs.gl.document.web.CorrectionDocumentEntryMetadata;
028    import org.kuali.kfs.module.ld.businessobject.LaborOriginEntry;
029    import org.kuali.kfs.module.ld.document.LaborCorrectionDocument;
030    import org.kuali.rice.kns.web.ui.Column;
031    
032    
033    /**
034     * Defines methods that must be implemented by classes providing a LaborCorrectionDocumentServiceImpl.
035     */
036    public interface LaborCorrectionDocumentService {
037        public final static String CORRECTION_TYPE_MANUAL = "M";
038        public final static String CORRECTION_TYPE_CRITERIA = "C";
039        public final static String CORRECTION_TYPE_REMOVE_GROUP_FROM_PROCESSING = "R";
040    
041        public final static String SYSTEM_DATABASE = "D";
042        public final static String SYSTEM_UPLOAD = "U";
043    
044        /**
045         * When passed into {@link #retrievePersistedInputOriginEntries(CorrectionDocument, int)} and
046         * {@link #retrievePersistedOutputOriginEntries(CorrectionDocument, int)} as the int parameter, this will signify that there is
047         * no abort threshold (i.e. the methods should return all of the persisted rows, regardless of number of rows.
048         */
049        public final static int UNLIMITED_ABORT_THRESHOLD = CorrectionDocumentUtils.RECORD_COUNT_FUNCTIONALITY_LIMIT_IS_UNLIMITED;
050    
051        /**
052         * Find and return correctionChangeGroup with document number and Group Number
053         * 
054         * @param docId, i
055         * @return
056         */
057        public CorrectionChangeGroup findByDocumentNumberAndCorrectionChangeGroupNumber(String docId, int i);
058    
059        /**
060         * Find and return correctionChange with document number and Group Number
061         * 
062         * @param docId, i
063         * @return list of correctionChange
064         */
065        public List findByDocumentHeaderIdAndCorrectionGroupNumber(String docId, int i);
066    
067        /**
068         * Find and return list of correctionCriteria with document number and Group Number
069         * 
070         * @param docId, i
071         * @return
072         */
073        public List findByDocumentNumberAndCorrectionGroupNumber(String docId, int i);
074    
075        /**
076         * Find and return laborCorrectionDocument with document number
077         * 
078         * @param docId, i
079         * @return
080         */
081        public LaborCorrectionDocument findByCorrectionDocumentHeaderId(String docId);
082    
083        /**
084         * Returns metadata to help render columns in the LLCP. Do not modify this list or the contents in this list.
085         * 
086         * @param docId
087         * @return
088         */
089        public List<Column> getTableRenderColumnMetadata(String docId);
090    
091        /**
092         * This method persists an Iterator of input origin entries for a document that is in the initiated or saved state
093         * 
094         * @param document an initiated or saved document
095         * @param entries
096         */
097        public void persistInputOriginEntriesForInitiatedOrSavedDocument(LaborCorrectionDocument document, Iterator<LaborOriginEntry> entries);
098    
099        /**
100         * @see org.kuali.kfs.module.ld.document.service.LaborCorrectionDocumentService#removePersistedInputOriginEntriesForInitiatedOrSavedDocument(org.kuali.kfs.module.ld.document.LaborCorrectionDocument)
101         */
102        public void removePersistedInputOriginEntries(LaborCorrectionDocument document);
103    
104        /**
105         * @see org.kuali.kfs.module.ld.document.service.LaborCorrectionDocumentService#removePersistedInputOriginEntriesForInitiatedOrSavedDocument(org.kuali.kfs.module.ld.document.LaborCorrectionDocument)
106         */
107        public void removePersistedInputOriginEntries(String docId);
108    
109        /**
110         * Retrieves input origin entries that have been persisted for this document
111         * 
112         * @param document the document
113         * @param abortThreshold if the file exceeds this number of rows, then null is returned. {@link UNLIMITED_ABORT_THRESHOLD}
114         *        signifies that there is no limit
115         * @return the list, or null if there are too many origin entries
116         * @throws RuntimeException several reasons, primarily relating to underlying persistence layer problems
117         */
118        public List<LaborOriginEntry> retrievePersistedInputOriginEntries(LaborCorrectionDocument document, int abortThreshold);
119    
120        /**
121         * Returns true if the system is storing input origin entries for this class. Note that this does not mean that there's at least
122         * one input origin entry record persisted for this document, but merely returns true if and only if the underlying persistence
123         * mechanism has a record of this document's origin entries. See the docs for the implementations of this method for more
124         * implementation specific details.
125         * 
126         * @param document
127         * @return
128         */
129        public boolean areInputOriginEntriesPersisted(LaborCorrectionDocument document);
130    
131        /**
132         * Writes out the persisted input origin entries in an {@link OutputStream} in a flat file format
133         * 
134         * @param document
135         * @param out an open and ready output stream
136         * @throws IOException
137         * @throws RuntimeException several reasons, including if the entries are not persisted
138         */
139        public void writePersistedInputOriginEntriesToStream(LaborCorrectionDocument document, OutputStream out) throws IOException;
140    
141        /**
142         * This method persists an Iterator of input origin entries for a document that is in the initiated or saved state
143         * 
144         * @param document an initiated or saved document
145         * @param entries
146         */
147        public void persistOutputLaborOriginEntriesForInitiatedOrSavedDocument(LaborCorrectionDocument document, Iterator<LaborOriginEntry> entries);
148    
149        /**
150         * @see org.kuali.kfs.module.ld.document.service.LaborCorrectionDocumentService#removePersistedOutputOriginEntriesForInitiatedOrSavedDocument(org.kuali.module.labor.document.CorrectionDocument)
151         */
152        public void removePersistedOutputOriginEntries(LaborCorrectionDocument document);
153    
154        /**
155         * @see org.kuali.kfs.module.ld.document.service.LaborCorrectionDocumentService#removePersistedOutputOriginEntriesForInitiatedOrSavedDocument(org.kuali.module.labor.document.CorrectionDocument)
156         */
157        public void removePersistedOutputOriginEntries(String docId);
158    
159        /**
160         * Retrieves output origin entries that have been persisted for this document
161         * 
162         * @param document the document
163         * @param abortThreshold if the file exceeds this number of rows, then null is returned. {@link UNLIMITED_ABORT_THRESHOLD}
164         *        signifies that there is no limit
165         * @return the list, or null if there are too many origin entries
166         * @throws RuntimeException several reasons, primarily relating to underlying persistence layer problems
167         */
168        public List<LaborOriginEntry> retrievePersistedOutputOriginEntries(LaborCorrectionDocument document, int abortThreshold);
169    
170        /**
171         * Retrieves input origin entries that have been persisted for this document in an iterator. Implementations of this method may
172         * choose to implement this method in a way that consumes very little memory.
173         * 
174         * @param document the document
175         * @return the iterator
176         * @throws RuntimeException several reasons, primarily relating to underlying persistence layer problems
177         */
178        public Iterator<LaborOriginEntry> retrievePersistedInputOriginEntriesAsIterator(LaborCorrectionDocument document);
179    
180        /**
181         * Retrieves output origin entries that have been persisted for this document in an iterator. Implementations of this method may
182         * choose to implement this method in a way that consumes very little memory.
183         * 
184         * @param document the document
185         * @return the iterator
186         * @throws RuntimeException several reasons, primarily relating to underlying persistence layer problems
187         */
188        public Iterator<LaborOriginEntry> retrievePersistedOutputOriginEntriesAsIterator(LaborCorrectionDocument document);
189    
190        /**
191         * Returns true if the system is storing output origin entries for this class. Note that this does not mean that there's at
192         * least one output origin entry record persisted for this document, but merely returns true if and only if the underlying
193         * persistence mechanism has a record of this document's origin entries. See the docs for the implementations of this method for
194         * more implementation specific details.
195         * 
196         * @param document
197         * @return
198         */
199        public boolean areOutputOriginEntriesPersisted(LaborCorrectionDocument document);
200    
201        /**
202         * Writes out the persisted output origin entries in an {@link OutputStream} in a flat file format\
203         * 
204         * @param document
205         * @param out axn open and ready output stream
206         * @throws IOException
207         * @throws RuntimeException several reasons, including if the entries are not persisted
208         */
209        public void writePersistedOutputOriginEntriesToStream(LaborCorrectionDocument document, OutputStream out) throws IOException;
210    
211        /**
212         * Saves the input and output origin entry groups for a document prior to saving the document
213         * 
214         * @param document
215         * @param LaborCorrectionDocumentEntryMetadata
216         */
217        public void persistOriginEntryGroupsForDocumentSave(LaborCorrectionDocument document, CorrectionDocumentEntryMetadata correctionDocumentEntryMetadata);
218        
219        public String generateOutputOriginEntryFileName(String docId);
220        
221        public String createOutputFileForProcessing(String docId, java.util.Date today);
222        
223        public String getBatchFileDirectoryName();
224        
225        public String getLlcpDirectoryName();
226        
227        /**
228         * Generate a text report for the given correction document
229         * 
230         * @param document LLCP document to report on
231         */
232        public void generateCorrectionReport(LaborCorrectionDocument document);
233        
234        public void aggregateCorrectionDocumentReports(GeneralLedgerCorrectionProcessDocument document);
235    }
