034    /**
035     * labor Document class for the Labor Ledger Correction Process.
036     */
037    public class LaborCorrectionDocument extends GeneralLedgerCorrectionProcessDocument {
038        private static final org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(LaborCorrectionDocument.class);
039    
040        /**
041         * Constructs a LaborCorrectionDocument.java.
042         */
043        public LaborCorrectionDocument() {
044            super();
045        }
046    
047        /**
048         * @param change
049         * @see org.kuali.rice.kns.document.DocumentBase#doRouteStatusChange(org.kuali.rice.kew.clientapp.vo.DocumentRouteLevelChangeDTO)
050         */
051        @Override
052        public void doRouteStatusChange(DocumentRouteStatusChangeDTO statusChangeEvent) {
053            // NOTE: DO NOT call the super implementation.  It has the GLCP processing logic which should not be run here.
054            // The code below is copied from the super.super implementation:
055            if (getDocumentHeader().getWorkflowDocument().stateIsCanceled()) {
056                getDocumentHeader().setFinancialDocumentStatusCode(KFSConstants.DocumentStatusCodes.CANCELLED);
057            } else if (getDocumentHeader().getWorkflowDocument().stateIsEnroute()) {
058                getDocumentHeader().setFinancialDocumentStatusCode(KFSConstants.DocumentStatusCodes.ENROUTE);
059            }
060            if (getDocumentHeader().getWorkflowDocument().stateIsDisapproved()) {
061                getDocumentHeader().setFinancialDocumentStatusCode(KFSConstants.DocumentStatusCodes.DISAPPROVED);
062            }
063            if (getDocumentHeader().getWorkflowDocument().stateIsProcessed()) {
064                getDocumentHeader().setFinancialDocumentStatusCode(KFSConstants.DocumentStatusCodes.APPROVED);
065            }
066            if ( LOG.isInfoEnabled() ) {
067                LOG.info("Document: " + statusChangeEvent.getRouteHeaderId() + " -- Status is: " + getDocumentHeader().getFinancialDocumentStatusCode());
068            }
069            // End of super.super code
070            if (getDocumentHeader().getWorkflowDocument().stateIsProcessed()) {
071                String docId = getDocumentNumber();
072                if ( LOG.isInfoEnabled() ) {
073                    LOG.info( "Document " + docId + " moving to Processed Status - starting final processing");
074                }
075                // this code is performed asynchronously
076                // First, save the origin entries to the origin entry table
077                LaborCorrectionDocumentService laborCorrectionDocumentService = SpringContext.getBean(LaborCorrectionDocumentService.class);
078                OriginEntryGroupService originEntryGroupService = (OriginEntryGroupService) SpringContext.getBean(LaborOriginEntryGroupService.class);
079                
080                // QUESTION - since this *is* the labor correction document - why are we loading it again?
081                LaborCorrectionDocument doc = laborCorrectionDocumentService.findByCorrectionDocumentHeaderId(docId);
082    
083                String correctionType = doc.getCorrectionTypeCode();
084                if (LaborCorrectionDocumentService.CORRECTION_TYPE_REMOVE_GROUP_FROM_PROCESSING.equals(correctionType)) {
085                    
086                    String dataFileName = doc.getCorrectionInputFileName();
087                    String doneFileName = dataFileName.replace(GeneralLedgerConstants.BatchFileSystem.EXTENSION, GeneralLedgerConstants.BatchFileSystem.DONE_FILE_EXTENSION);
088                    originEntryGroupService.deleteFile(doneFileName);
089                    if ( LOG.isInfoEnabled() ) {
090                        LOG.info( "Document " + docId + " : deleted done file to remove from processing: " + doneFileName );
091                    }
092                    
093                } else if (LaborCorrectionDocumentService.CORRECTION_TYPE_MANUAL.equals(correctionType) 
094                        || LaborCorrectionDocumentService.CORRECTION_TYPE_CRITERIA.equals(correctionType)) {
095                    
096                    synchronized ( LaborCorrectionDocumentService.class ) {
097                        if ( !checkForExistingOutputDocument( docId ) ) {
098    
099                            Date today = SpringContext.getBean(DateTimeService.class).getCurrentDate();
100    
101                            String outputFileName = "";
102                            if (!doc.getCorrectionFileDelete()) {
103                                outputFileName = laborCorrectionDocumentService.createOutputFileForProcessing(docId, today);
104                            } else {
105                                if ( LOG.isInfoEnabled() ) {
106                                    LOG.info( "Document " + docId + " : set to delete output file - no file will be created" );
107                                }
108                            }
109                            doc.setCorrectionOutputFileName(outputFileName);
110                            
111                            if ( LOG.isInfoEnabled() ) {
112                                LOG.info( "Document " + docId + " : about to run scrubber -- output file: " + outputFileName );
113                            }                        
114                            Step step = BatchSpringContext.getStep(LaborCorrectionProcessScrubberStep.STEP_NAME);                        
115                            LaborCorrectionProcessScrubberStep correctionStep = (LaborCorrectionProcessScrubberStep) ProxyUtils.getTargetIfProxied(step);
116                            correctionStep.setDocumentId(docId);
117                            try {
118                                step.execute(getClass().getName(), today);
119                            } catch (Exception e) {
120                                LOG.error("LLCP scrubber encountered error:", e);
121                                throw new RuntimeException("LLCP scrubber encountered error:", e);
122                            }
123                            correctionStep.setDocumentId(null);
124                            if ( LOG.isInfoEnabled() ) {
125                                LOG.info( "Document " + docId + " : completed scrubber run -- generating reports" );
126                            }                        
127                            
128                            laborCorrectionDocumentService.generateCorrectionReport(this);
129                            laborCorrectionDocumentService.aggregateCorrectionDocumentReports(this);
130                        } else {
131                            LOG.warn( "Attempt to re-process final LLCP operations for document: " + docId + "  File with that document number already exists." );
132                        }
133                    }
134                } else {
135                    LOG.error("LLCP doc " + docId + " has an unknown correction type code: " + correctionType);
136                }
137                if ( LOG.isInfoEnabled() ) {
138                    LOG.info( "Document " + docId + " moving to Processed Status - completed final processing");
139                }
140            }
141        }
142
