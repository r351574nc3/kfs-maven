094        @Override
095        public ActionForward execute(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
096            LOG.debug("execute() started");
097    
098            CorrectionForm correctionForm = (CorrectionForm) form;
099    
100            // Init our services once
101            if (originEntryGroupService == null) {
102                CorrectionAction.originEntryGroupService = (OriginEntryGroupService)SpringContext.getBean(LaborOriginEntryGroupService.class);;
103                CorrectionAction.originEntryService = SpringContext.getBean(OriginEntryService.class);
104                CorrectionAction.dateTimeService = SpringContext.getBean(DateTimeService.class);
105                CorrectionAction.kualiConfigurationService = SpringContext.getBean(KualiConfigurationService.class);
106            }
107    
108            LaborCorrectionForm rForm = (LaborCorrectionForm) form;
109            LOG.debug("execute() methodToCall: " + rForm.getMethodToCall());
110    
111            Collection<OriginEntryFull> persistedOriginEntries = null;
112    
113            // If we are called from the docHandler or reload, ignore the persisted origin entries because we are either creating a new
114            // document
115            // or loading an old one
116            if (!(KFSConstants.DOC_HANDLER_METHOD.equals(rForm.getMethodToCall()) || KFSConstants.RELOAD_METHOD_TO_CALL.equals(rForm.getMethodToCall()))) {
117                restoreSystemAndEditMethod(rForm);
118                restoreInputGroupSelectionForDatabaseEdits(rForm);
119                if (!rForm.isRestrictedFunctionalityMode()) {
120                    if (StringUtils.isNotBlank(rForm.getGlcpSearchResultsSequenceNumber())) {
121                        rForm.setAllEntries(SpringContext.getBean(GlCorrectionProcessOriginEntryService.class).retrieveAllEntries(rForm.getGlcpSearchResultsSequenceNumber()));
122                        if (rForm.getAllEntries() == null) {
123                            rForm.setDisplayEntries(null);
124                        }
125                        else {
126                            rForm.setDisplayEntries(new ArrayList<OriginEntryFull>(rForm.getAllEntries()));
127                        }
128    
129                        if ((!"showOutputGroup".equals(rForm.getMethodToCall())) && rForm.getShowOutputFlag()) {
130                            // reapply the any criteria to pare down the list if the match criteria only flag is checked
131                            LaborCorrectionDocument document = rForm.getLaborCorrectionDocument();
132                            List<CorrectionChangeGroup> groups = document.getCorrectionChangeGroup();
133                            //TODO:- need to change for LLCP                      
134                            updateEntriesFromCriteria(rForm, rForm.isRestrictedFunctionalityMode());
135                        }
136    
137                        if (!KFSConstants.TableRenderConstants.SORT_METHOD.equals(rForm.getMethodToCall())) {
138                            // if sorting, we'll let the action take care of the sorting
139                            KualiTableRenderFormMetadata originEntrySearchResultTableMetadata = rForm.getOriginEntrySearchResultTableMetadata();
140                            if (originEntrySearchResultTableMetadata.getPreviouslySortedColumnIndex() != -1) {
141                                List<Column> columns = SpringContext.getBean(LaborCorrectionDocumentService.class).getTableRenderColumnMetadata(rForm.getDocument().getDocumentNumber());
142    
143                                String propertyToSortName = columns.get(originEntrySearchResultTableMetadata.getPreviouslySortedColumnIndex()).getPropertyName();
144                                Comparator valueComparator = columns.get(originEntrySearchResultTableMetadata.getPreviouslySortedColumnIndex()).getValueComparator();
145                                sortList(rForm.getDisplayEntries(), propertyToSortName, valueComparator, originEntrySearchResultTableMetadata.isSortDescending());
146                            }
147                            if (rForm.getAllEntries() != null) {
148                                int maxRowsPerPage = CorrectionDocumentUtils.getRecordsPerPage();
149                                originEntrySearchResultTableMetadata.jumpToPage(originEntrySearchResultTableMetadata.getViewedPageNumber(), rForm.getDisplayEntries().size(), maxRowsPerPage);
150                                originEntrySearchResultTableMetadata.setColumnToSortIndex(originEntrySearchResultTableMetadata.getPreviouslySortedColumnIndex());
151                            }
152                        }
153                    }
154                }
155            }
156    
157            ActionForward af = super.superExecute(mapping, form, request, response);
158            return af;
159        }

167        @Override
168        public ActionForward save(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
169            LOG.debug("save() started");
170            
171            LaborCorrectionForm laborCorrectionForm = (LaborCorrectionForm) form;
172            LaborCorrectionDocument document = laborCorrectionForm.getLaborCorrectionDocument();
173    
174            // Did they pick the edit method and system?
175            if (!checkMainDropdown(laborCorrectionForm)) {
176                return mapping.findForward(KFSConstants.MAPPING_BASIC);
177            }
178            if (!checkRestrictedFunctionalityModeForManualEdit(laborCorrectionForm)) {
179                return mapping.findForward(KFSConstants.MAPPING_BASIC);
180            }
181            if (!validGroupsItemsForDocumentSave(laborCorrectionForm)) {
182                return mapping.findForward(KFSConstants.MAPPING_BASIC);
183            }
184            if (!validChangeGroups(laborCorrectionForm)) {
185                return mapping.findForward(KFSConstants.MAPPING_BASIC);
186            }
187            if (!checkInputGroupPersistedForDocumentSave(laborCorrectionForm)) {
188                return mapping.findForward(KFSConstants.MAPPING_BASIC);
189            }
190    
191            // Populate document
192            document.setCorrectionTypeCode(laborCorrectionForm.getEditMethod());
193            document.setCorrectionSelection(laborCorrectionForm.getMatchCriteriaOnly());
194            document.setCorrectionFileDelete(!laborCorrectionForm.getProcessInBatch());
195            document.setCorrectionInputFileName(laborCorrectionForm.getInputGroupId());
196            document.setCorrectionOutputFileName(null); 
197            if (laborCorrectionForm.getDataLoadedFlag() || laborCorrectionForm.isRestrictedFunctionalityMode()) {
198                document.setCorrectionInputFileName(laborCorrectionForm.getInputGroupId());
199            }
200            else {
201                document.setCorrectionInputFileName(null);
202            }
203            document.setCorrectionOutputFileName(null);
204    
205            SpringContext.getBean(LaborCorrectionDocumentService.class).persistOriginEntryGroupsForDocumentSave(document, laborCorrectionForm);
206            
207            LOG.debug("save() doc type name: " + laborCorrectionForm.getDocTypeName());
208            ActionForward af = super.superSave(mapping, form, request, response);
209            return af;
210        }

214         * Called when the document is loaded from action list or doc search or a new document is created.
215         * 
216         * @see org.kuali.rice.kns.web.struts.action.KualiDocumentActionBase#docHandler(org.apache.struts.action.ActionMapping,
217         *      org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
218         */
219        @Override
220        public ActionForward docHandler(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
221            LOG.debug("docHandler() started");
222    
223            LaborCorrectionForm laborCorrectionForm = (LaborCorrectionForm) form;
224            String command = laborCorrectionForm.getCommand();
225    
226            if (KEWConstants.INITIATE_COMMAND.equals(command)) {
227                laborCorrectionForm.clearForm();
228                createDocument(laborCorrectionForm);
229            }
230            else {
231                loadDocument(laborCorrectionForm);
232    
233                LaborCorrectionDocument laborDocument = laborCorrectionForm.getLaborCorrectionDocument();
234                laborCorrectionForm.setInputGroupIdFromLastDocumentLoad(laborDocument.getCorrectionInputFileName());
235                populateAuthorizationFields(laborCorrectionForm);
236                Map<String, String> documentActions = laborCorrectionForm.getDocumentActions();
237                if (documentActions.containsKey(KNSConstants.KUALI_ACTION_CAN_EDIT)) {
238                    // They have saved the document and they are retreiving it to be completed
239                    laborCorrectionForm.setProcessInBatch(!laborDocument.getCorrectionFileDelete());
240                    laborCorrectionForm.setMatchCriteriaOnly(laborDocument.getCorrectionSelection());
241                    laborCorrectionForm.setEditMethod(laborDocument.getCorrectionTypeCode());
242                    
243                    if (laborDocument.getCorrectionInputFileName() != null) {
244                        if (CorrectionDocumentService.CORRECTION_TYPE_CRITERIA.equals(laborDocument.getCorrectionTypeCode())) {
245                            loadPersistedInputGroup(laborCorrectionForm);
246                            laborCorrectionForm.setDeleteFileFlag(false);
247                        }
248                        else if (CorrectionDocumentService.CORRECTION_TYPE_MANUAL.equals(laborDocument.getCorrectionTypeCode())) {
249                            // for the "true" param below, when the origin entries are persisted in the CorrectionDocumentService, they
250                            // // are likely
251                            // // not to have origin entry IDs assigned to them. So, we create pseudo entry IDs that are
252                            // // unique within the allEntries list, but not necessarily within the DB. The persistence layer
253                            // // is responsible for auto-incrementing entry IDs in the DB.
254                            loadPersistedOutputGroup(laborCorrectionForm, true);
255    
256                            laborCorrectionForm.setManualEditFlag(true);
257                            laborCorrectionForm.setEditableFlag(false);
258                            laborCorrectionForm.setDeleteFileFlag(false);
259                        }
260                        else if (CorrectionDocumentService.CORRECTION_TYPE_REMOVE_GROUP_FROM_PROCESSING.equals(laborDocument.getCorrectionTypeCode())) {
261                            loadPersistedInputGroup(laborCorrectionForm);
262                            laborCorrectionForm.setDeleteFileFlag(true);
263                        }
264                        else {
265                            throw new RuntimeException("Unknown edit method " + laborDocument.getCorrectionTypeCode());
266                        }
267                        laborCorrectionForm.setDataLoadedFlag(true);
268                    }
269                    else {
270                        laborCorrectionForm.setDataLoadedFlag(false);
271                    }
272                    laborCorrectionForm.setShowOutputFlag(false);
273                    laborCorrectionForm.setInputFileName(laborDocument.getCorrectionInputFileName());
274                    if (laborDocument.getCorrectionInputFileName() != null) {
275                        laborCorrectionForm.setChooseSystem(CorrectionDocumentService.SYSTEM_UPLOAD);
276                    }
277                    else {
278                        laborCorrectionForm.setChooseSystem(CorrectionDocumentService.SYSTEM_DATABASE);
279                    }
280    
281                    laborCorrectionForm.setPreviousChooseSystem(laborCorrectionForm.getChooseSystem());
282                    laborCorrectionForm.setPreviousEditMethod(laborCorrectionForm.getEditMethod());
283                    laborCorrectionForm.setPreviousInputGroupId(laborCorrectionForm.getInputGroupId());
284                }
285                else {
286                    // They are calling this from their action list to look at it or approve it
287                    laborCorrectionForm.setProcessInBatch(!laborDocument.getCorrectionFileDelete());
288                    laborCorrectionForm.setMatchCriteriaOnly(laborDocument.getCorrectionSelection());
289    
290                    // we don't care about setting entry IDs for the records below, so the param is false below
291                    loadPersistedOutputGroup(laborCorrectionForm, false);
292                    laborCorrectionForm.setShowOutputFlag(true);
293                }
294                laborCorrectionForm.setInputGroupIdFromLastDocumentLoadIsMissing(!originEntryGroupService.getGroupExists(laborDocument.getCorrectionInputFileName()));
295            }
296    
297            return mapping.findForward(KFSConstants.MAPPING_BASIC);
298        }

302        /**
303         * This handles the action for uploading a file
304         * 
305         * @see org.kuali.kfs.gl.document.web.struts.CorrectionAction#uploadFile(org.apache.struts.action.ActionMapping,
306         *      org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
307         */
308        public ActionForward uploadFile(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws FileNotFoundException, IOException, Exception {
309            LOG.debug("uploadFile() started");
310    
311            LaborCorrectionForm laborCorrectionForm = (LaborCorrectionForm) form;
312            LaborCorrectionDocument document = laborCorrectionForm.getLaborCorrectionDocument();
313    
314            Date now = CorrectionAction.dateTimeService.getCurrentDate();
315            //creat file after all enries loaded well
316            //OriginEntryGroup newOriginEntryGroup = CorrectionAction.originEntryGroupService.createGroup(today, OriginEntrySource.LABOR_CORRECTION_PROCESS_EDOC, false, false, false);
317    
318            FormFile sourceFile = laborCorrectionForm.getSourceFile();
319            String llcpDirectory = SpringContext.getBean(LaborCorrectionDocumentService.class).getLlcpDirectoryName();
320            String fullFileName = llcpDirectory + File.separator + sourceFile.getFileName() + "-" + CorrectionAction.dateTimeService.toDateTimeStringForFilename(now);
321            BufferedReader br = new BufferedReader(new InputStreamReader(sourceFile.getInputStream()));
322            
323            //create a file
324            File uploadedFile = new File(fullFileName);
325            PrintStream uploadedFilePrintStream;
326            try {
327                uploadedFilePrintStream = new PrintStream(uploadedFile);
328            } catch (IOException e) {
329                throw new RuntimeException(e);
330            }
331    
332            //write entries to file
333            int loadedCount = 0;
334            String stringLine;
335            while ((stringLine= br.readLine()) != null){
336                try {
337                    uploadedFilePrintStream.printf("%s\n", stringLine);
338                    loadedCount++;
339                } catch (Exception e) {
340                    throw new IOException(e.toString());
341                }
342            }
343            uploadedFilePrintStream.close();
344            
345    
346            int recordCountFunctionalityLimit = CorrectionDocumentUtils.getRecordCountFunctionalityLimit();
347            if (CorrectionDocumentUtils.isRestrictedFunctionalityMode(loadedCount, recordCountFunctionalityLimit)) {
348                laborCorrectionForm.setRestrictedFunctionalityMode(true);
349                laborCorrectionForm.setDataLoadedFlag(false);
350                document.setCorrectionInputFileName(fullFileName);
351                laborCorrectionForm.setInputFileName(fullFileName);
352    
353                if (CorrectionDocumentService.CORRECTION_TYPE_MANUAL.equals(laborCorrectionForm.getEditMethod())) {
354                    // the group size is not suitable for manual editing because it is too large
355                    if (recordCountFunctionalityLimit == CorrectionDocumentUtils.RECORD_COUNT_FUNCTIONALITY_LIMIT_IS_NONE) {
356                        GlobalVariables.getMessageMap().putError(SYSTEM_AND_EDIT_METHOD_ERROR_KEY, KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_UNABLE_TO_MANUAL_EDIT_ANY_GROUP);
357                    }
358                    else {
359                        GlobalVariables.getMessageMap().putError(SYSTEM_AND_EDIT_METHOD_ERROR_KEY, KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_UNABLE_TO_MANUAL_EDIT_LARGE_GROUP, String.valueOf(recordCountFunctionalityLimit));
360                    }
361                }
362            }
363            else {
364                laborCorrectionForm.setRestrictedFunctionalityMode(false);
365                if (loadedCount > 0) {
366                    //now we can load all data from file
367                    List<LaborOriginEntry> originEntryList = new ArrayList();
368                    Map loadErrorMap = laborOriginEntryService.getEntriesByGroupIdWithPath(uploadedFile.getAbsolutePath(), originEntryList);
369                    //put errors on GlobalVariables
370                    if (loadErrorMap.size() > 0){
371                        Iterator iter = loadErrorMap.keySet().iterator();
372                        while(iter.hasNext()){
373                            Integer lineNumber = (Integer) iter.next();
374                            List<Message> messageList = (List<Message>) loadErrorMap.get(lineNumber);
375                            if (messageList.size() > 0){
376                                for (Message errorMmessage : messageList){
377                                    GlobalVariables.getMessageMap().putError("fileUpload", KFSKeyConstants.ERROR_INVALID_FORMAT_ORIGIN_ENTRY_FROM_TEXT_FILE, new String[] {lineNumber.toString(), errorMmessage.toString()});
378                                }    
379                            }
380                        }
381                        return mapping.findForward(KFSConstants.MAPPING_BASIC);
382                    }
383                    // Set all the data that we know                
384                    laborCorrectionForm.setDataLoadedFlag(true);
385                    laborCorrectionForm.setInputFileName(fullFileName);
386                    document.setCorrectionInputFileName(fullFileName);
387                    List<OriginEntryFull> originEntryFullList = new ArrayList();
388                    originEntryFullList.addAll(originEntryList);
389                    loadAllEntries(originEntryFullList, laborCorrectionForm);
390                    
391                    if (CorrectionDocumentService.CORRECTION_TYPE_MANUAL.equals(laborCorrectionForm.getEditMethod())) {
392                        laborCorrectionForm.setEditableFlag(false);
393                        laborCorrectionForm.setManualEditFlag(true);
394                    }
395                }
396                else {
397                    GlobalVariables.getMessageMap().putError("fileUpload", KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_NO_RECORDS);
398                }
399            }
400    
401            if (document.getCorrectionChangeGroup().isEmpty()) {
402                document.addCorrectionChangeGroup(new CorrectionChangeGroup());
403            }
404    
405            return mapping.findForward(KFSConstants.MAPPING_BASIC);
406        }

482         * @see org.kuali.kfs.gl.document.web.struts.CorrectionAction#saveManualEntry(org.apache.struts.action.ActionMapping, org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)

484        public ActionForward saveManualEntry(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
485            LOG.debug("saveManualEdit() started");
486    
487            LaborCorrectionForm laborCorrectionForm = (LaborCorrectionForm) form;
488            LaborCorrectionDocument document = laborCorrectionForm.getLaborCorrectionDocument();
489    
490            if (validLaborOriginEntry(laborCorrectionForm)) {
491                int entryId = laborCorrectionForm.getLaborEntryForManualEdit().getEntryId();
492    
493                // Find it and replace it with the one from the edit spot
494                for (Iterator<OriginEntryFull> iter = laborCorrectionForm.getAllEntries().iterator(); iter.hasNext();) {
495                    OriginEntryFull element = iter.next();
496                    if (element.getEntryId() == entryId) {
497                        iter.remove();
498                    }
499                }
500    
501                laborCorrectionForm.updateLaborEntryForManualEdit();
502                laborCorrectionForm.getAllEntries().add(laborCorrectionForm.getLaborEntryForManualEdit());
503    
504                // we've modified the list of all entries, so repersist it
505                SpringContext.getBean(GlCorrectionProcessOriginEntryService.class).persistAllEntries(laborCorrectionForm.getGlcpSearchResultsSequenceNumber(), laborCorrectionForm.getAllEntries());
506                // laborCorrectionForm.setDisplayEntries(null);

902    
903            LaborCorrectionForm laborCorrectionForm = (LaborCorrectionForm) correctionForm;
904            LaborCorrectionDocument document = laborCorrectionForm.getLaborCorrectionDocument();
905    
906            // Is there a description?
907            if (StringUtils.isEmpty(document.getDocumentHeader().getDocumentDescription())) {
908                GlobalVariables.getMessageMap().putError("document.documentHeader.documentDescription", KFSKeyConstants.ERROR_DOCUMENT_NO_DESCRIPTION);
909                return false;
910            }
911    
912            if (laborCorrectionForm.isPersistedOriginEntriesMissing()) {
913                GlobalVariables.getMessageMap().putError("searchResults", KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_PERSISTED_ORIGIN_ENTRIES_MISSING);
914                return false;
915            }
916    
917            // Did they pick the edit method and system?
918            if (!checkMainDropdown(laborCorrectionForm)) {
919                return false;
920            }
921    
922            if (laborCorrectionForm.getDataLoadedFlag() || laborCorrectionForm.isRestrictedFunctionalityMode()) {
923                document.setCorrectionInputFileName(correctionForm.getInputGroupId());
924            }
925            else {
926                document.setCorrectionInputFileName(null);
927            }
928            if (!checkOriginEntryGroupSelectionBeforeRouting(document)) {
929                return false;
930            }
931    
932            // were the system and edit methods inappropriately changed?
933            if (GlobalVariables.getMessageMap().containsMessageKey(KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_INVALID_SYSTEM_OR_EDIT_METHOD_CHANGE)) {
934                return false;
935            }
936    
937            // was the input group inappropriately changed?
938            if (GlobalVariables.getMessageMap().containsMessageKey(KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_INVALID_INPUT_GROUP_CHANGE)) {

943                return false;
944            }
945    
946            // If it is criteria, are all the criteria valid?
947            if (CorrectionDocumentService.CORRECTION_TYPE_CRITERIA.equals(laborCorrectionForm.getEditMethod())) {
948                if (!validChangeGroups(laborCorrectionForm)) {
949                    return false;
950                }
951            }
952    
953            if (!checkRestrictedFunctionalityModeForManualEdit(laborCorrectionForm)) {
954                return false;
955            }
956    
957            if (!checkInputGroupPersistedForDocumentSave(laborCorrectionForm)) {
958                return false;
959            }
960            // Get the output group if necessary
961            if (CorrectionDocumentService.CORRECTION_TYPE_CRITERIA.equals(laborCorrectionForm.getEditMethod())) {
962                if (!laborCorrectionForm.isRestrictedFunctionalityMode() && laborCorrectionForm.getDataLoadedFlag() && !laborCorrectionForm.getShowOutputFlag()) {
963                    // we're going to force the user to view the output group upon routing, so apply the criteria
964                    // if the user wasn't in show output mode.
965                    updateEntriesFromCriteria(laborCorrectionForm, false);
966                }
967                laborCorrectionForm.setShowOutputFlag(true);
968            }
969            else {
970                // If it is manual edit, we don't need to save any correction groups
971                document.getCorrectionChangeGroup().clear();
972            }
973    
974            // Populate document
975            document.setCorrectionTypeCode(laborCorrectionForm.getEditMethod());
976            document.setCorrectionSelection(laborCorrectionForm.getMatchCriteriaOnly());
977            document.setCorrectionFileDelete(!laborCorrectionForm.getProcessInBatch());
978            document.setCorrectionInputFileName(laborCorrectionForm.getInputGroupId());
979            document.setCorrectionOutputFileName(null); // this field is never used
980    
981            // we'll populate the output group id when the doc has a route level change
982            document.setCorrectionOutputFileName(null);
983    
984            SpringContext.getBean(LaborCorrectionDocumentService.class).persistOriginEntryGroupsForDocumentSave(document, laborCorrectionForm);
985    
986            return true;
987        }
