016    package org.kuali.kfs.module.ld.document.validation.impl;
017    
018    import org.kuali.kfs.module.ld.document.LaborJournalVoucherDocument;
019    import org.kuali.kfs.sys.KFSKeyConstants;
020    import org.kuali.kfs.sys.KFSPropertyConstants;
021    import org.kuali.kfs.sys.document.service.FinancialSystemDocumentTypeService;
022    import org.kuali.kfs.sys.document.validation.GenericValidation;
023    import org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent;
024    import org.kuali.rice.kns.util.GlobalVariables;
025    
026    /**
027     * Validation which checks that the offset type for the document is a valid active, current accounting document type
028     */
029    public class LaborJournalVoucherOffsetTypeCodeValidation extends GenericValidation {
030        private LaborJournalVoucherDocument laborJournalVoucherDocumentForValidation;
031        private FinancialSystemDocumentTypeService financialSystemDocumentTypeService;
032    
033        /**
034         * 
035         * @see org.kuali.kfs.sys.document.validation.Validation#validate(org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent)
036         */
037        public boolean validate(AttributedDocumentEvent event) {
038            final String offsetTypeCode = getLaborJournalVoucherDocumentForValidation().getOffsetTypeCode();
039            if (!getFinancialSystemDocumentTypeService().isCurrentActiveAccountingDocumentType(offsetTypeCode)) {
040                GlobalVariables.getMessageMap().putError(KFSPropertyConstants.OFFSET_TYPE_CODE, KFSKeyConstants.ERROR_DOCUMENT_LABOR_JOURNAL_VOUCHER_OFFSET_TYPE_CODE_NON_ACTIVE_CURRENT_ACCOUNTING_DOCUMENT_TYPE, offsetTypeCode);
041                return false;
042            }
043            return true;
044        }
045    
046        /**
047         * Gets the laborJournalVoucherDocumentForValidation attribute. 
048         * @return Returns the laborJournalVoucherDocumentForValidation.
049         */
050        public LaborJournalVoucherDocument getLaborJournalVoucherDocumentForValidation() {
051            return laborJournalVoucherDocumentForValidation;
052        }
053    
054        /**
055         * Sets the laborJournalVoucherDocumentForValidation attribute value.
056         * @param laborJournalVoucherDocumentForValidation The laborJournalVoucherDocumentForValidation to set.
057         */
058        public void setLaborJournalVoucherDocumentForValidation(LaborJournalVoucherDocument laborJournalVoucherDocumentForValidation) {
059            this.laborJournalVoucherDocumentForValidation = laborJournalVoucherDocumentForValidation;
060        }
061    
062        /**
063         * Gets the financialSystemDocumentTypeService attribute. 
064         * @return Returns the financialSystemDocumentTypeService.
065         */
066        public FinancialSystemDocumentTypeService getFinancialSystemDocumentTypeService() {
067            return financialSystemDocumentTypeService;
068        }
069    
070        /**
071         * Sets the financialSystemDocumentTypeService attribute value.
072         * @param financialSystemDocumentTypeService The financialSystemDocumentTypeService to set.
073         */
074        public void setFinancialSystemDocumentTypeService(FinancialSystemDocumentTypeService financialSystemDocumentTypeService) {
075            this.financialSystemDocumentTypeService = financialSystemDocumentTypeService;
076        }
077    }
