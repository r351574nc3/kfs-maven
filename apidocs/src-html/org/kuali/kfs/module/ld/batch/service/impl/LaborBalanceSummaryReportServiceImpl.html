072        /**
073         * @see org.kuali.kfs.module.ld.batch.service.LaborBalanceSummaryReportService#generateBalanceSummaryReports()
074         */
075        public void generateBalanceSummaryReports() {
076            LOG.info("generateBalanceSummaryReports() started");
077    
078            Date runDate = dateTimeService.getCurrentSqlDate();
079            this.generatePosterOutputSummaryReport(runDate);
080            this.generateBalanceSummaryReports(runDate);
081        }

083        /**
084         * @see org.kuali.kfs.module.ld.batch.service.LaborBalanceSummaryReportService#generateBalanceSummaryReports(java.sql.Date)
085         */
086        public void generateBalanceSummaryReports(Date runDate) {
087            LOG.info("generateBalanceSummaryReports(Date) started");
088    
089            String yearEndPeriodLowerBound = parameterService.getParameterValue(PosterSummaryReportStep.class, CURRENT_YEAR_LOWER);
090            String lastDayOfFiscalYear = parameterService.getParameterValue(PosterSummaryReportStep.class, CURRENT_AND_LAST_YEAR);
091            String yearEndPeriodUpperBound = parameterService.getParameterValue(PosterSummaryReportStep.class, CURRENT_YEAR_UPPER);
092    
093            Integer currentYear = optionsService.getCurrentYearOptions().getUniversityFiscalYear();
094            this.generateBalanceSummaryReports(currentYear, runDate);
095    
096            // if today is within the lower bound of the year end period, then generate reports for the next fiscal year
097            if (this.isInYearEndLowerBound(runDate, yearEndPeriodLowerBound, lastDayOfFiscalYear)) {
098                this.generateBalanceSummaryReports(currentYear + 1, runDate);
099            }
100    
101            // if today is within the upper bound of the year end period, then generate reports for the last fiscal year
102            if (this.isInYearEndUpperBound(runDate, yearEndPeriodUpperBound, lastDayOfFiscalYear)) {
103                this.generateBalanceSummaryReports(currentYear - 1, runDate);
104            }
105        }
106    
107        // generate a set of balance summary reports for actual, budget and encumbrance balances
108        protected void generateBalanceSummaryReports(Integer fiscalYear, Date runDate) {
109            SystemOptions options = optionsService.getOptions(fiscalYear);       
110            if(options == null) {
111                LOG.fatal("The data for " + fiscalYear + "have NOT been setup.");
112                return;
113            }
114            
115            List<String> actualsBalanceTypes = this.getActualBalanceTypes(fiscalYear);   
116            this.writeSummaryReport(fiscalYear, actualsBalanceTypes, laborActualBalanceSummaryReportWriterService);
117    
118            List<String> budgetBalanceTypes = this.getBudgetBalanceTypes(fiscalYear);
119            this.writeSummaryReport(fiscalYear, budgetBalanceTypes, laborBudgetBalanceSummaryReportWriterService);
120    
121            List<String> encumbranceBalanceTypes = this.getEncumbranceBalanceTypes(fiscalYear);
122            this.writeSummaryReport(fiscalYear, encumbranceBalanceTypes, laborEncumbranceSummaryReportWriterService);
123        }
124    
125        protected void writeSummaryReport(Integer fiscalYear, List<String> balanceTypes, FiscalYearAwareReportWriterService reportWriterService) {        
126            List<LaborBalanceSummary> balanceSummary = laborLedgerBalanceService.findBalanceSummary(fiscalYear, balanceTypes);
127            List<GlSummary> summaryList = new ArrayList<GlSummary>(balanceSummary);
128            
129            GlSummary totals = new LaborBalanceSummary();
130            for(GlSummary summaryLine : summaryList) {
131                totals.add(summaryLine);
132            }        
133            totals.setFundGroup("Total");
134            
135            try { 
136                reportWriterService.setFiscalYear(fiscalYear);
137                ((WrappingBatchService)reportWriterService).initialize();
138                reportWriterService.writeSubTitle("Balance Type of " + balanceTypes + " for Fiscal Year " + fiscalYear);
139                reportWriterService.writeNewLines(1);
140                
141                reportWriterService.writeTableRowSeparationLine(totals);
142                reportWriterService.writeTable(summaryList, true, false);
143                
144                reportWriterService.writeTableRowSeparationLine(totals);
145                reportWriterService.writeTableRow(totals);
146            }
147            finally {
148                ((WrappingBatchService)reportWriterService).destroy();
149            }
150        }

152        /**
153         * Generates reports about the output of a poster run.
154         * 
155         * @param runDate the date the poster was run.
156         */
157        protected void generatePosterOutputSummaryReport(Date runDate) {
158            PosterOutputSummaryReport posterOutputSummaryReport = new PosterOutputSummaryReport();
159            
160            // summarize all the entries for the main poster
161            File mainPosterFile = FileUtil.getNewestFile(new File(batchFileDirectoryName), new RegexFileFilter((LaborConstants.BatchFileSystem.POSTER_INPUT_FILE + "\\.[0-9_\\-]+\\" + GeneralLedgerConstants.BatchFileSystem.EXTENSION)));        
162            if (mainPosterFile != null && mainPosterFile.exists()) {
163                LaborOriginEntryFileIterator mainPosterIterator = new LaborOriginEntryFileIterator(mainPosterFile);
164                while (mainPosterIterator.hasNext()) {
165                    OriginEntryInformation originEntry = mainPosterIterator.next();
166                    posterOutputSummaryReport.summarize(originEntry);
167                }
168            } else {
169                LOG.warn("Could not Main Poster Input file, "+ LaborConstants.BatchFileSystem.POSTER_INPUT_FILE + ", for tabulation in the Poster Output Summary Report");
170            }
171            
172            posterOutputSummaryReport.writeReport(laborPosterOutputSummaryReportWriterService);
173        }

223        /**
224         * determine if the given date is within the year end period
225         * 
226         * @param runDate the given date
227         * @param yearEndPeriodLowerBound the lower bound date of year end period
228         * @param yearEndPeriodUpperBound the upper bound date of year end period
229         * @param lastDayOfFiscalYear the last day of the current fiscal year
230         * @return true if the given date is within the lower bound of year end period; otherwise, false
231         */
232        protected boolean isInYearEndPeriod(Date runDate, String yearEndPeriodLowerBound, String yearEndPeriodUpperBound, String lastDayOfFiscalYear) {
233            return isInYearEndLowerBound(runDate, yearEndPeriodLowerBound, lastDayOfFiscalYear) || isInYearEndUpperBound(runDate, yearEndPeriodUpperBound, lastDayOfFiscalYear);
234        }
235    
236        /**
237         * determine if the given date is within the lower bound of year end period
238         * 
239         * @param runDate the given date
240         * @param yearEndPeriodLowerBound the lower bound date of year end period
241         * @param lastDayOfFiscalYear the last day of the current fiscal year
242         * @return true if the given date is within the lower bound of year end period; otherwise, false
243         */
244        protected boolean isInYearEndLowerBound(Date runDate, String yearEndPeriodLowerBound, String lastDayOfFiscalYear) {
245            SimpleDateFormat sdf = new SimpleDateFormat("MMdd");
246            String today = sdf.format(runDate);
247            return today.compareTo(yearEndPeriodLowerBound) >= 0 && today.compareTo(lastDayOfFiscalYear) <= 0;
248        }
249    
250        /**
251         * determine if the given date is within the upper bound of year end period
252         * 
253         * @param runDate the given date
254         * @param yearEndPeriodUpperBound the upper bound date of year end period
255         * @param lastDayOfFiscalYear the last day of the current fiscal year
256         * @return true if the given date is within the upper bound of year end period; otherwise, false
257         */
258        protected boolean isInYearEndUpperBound(Date runDate, String yearEndPeriodUpperBound, String lastDayOfFiscalYear) {
259            SimpleDateFormat sdf = new SimpleDateFormat("MMdd");
260            String today = sdf.format(runDate);
261    
262            String month = StringUtils.mid(lastDayOfFiscalYear, 0, 2);
263            String date = StringUtils.mid(lastDayOfFiscalYear, 2, 2);
264    
265            Calendar calendar = Calendar.getInstance();
266            calendar.set(Calendar.MONTH, Integer.parseInt(month) - 1);
267            calendar.set(Calendar.DATE, Integer.parseInt(date));
268            calendar.add(Calendar.DATE, 1);
269            String firstDayOfNewFiscalYear = sdf.format(calendar.getTime());
270    
271            return today.compareTo(yearEndPeriodUpperBound) <= 0 && today.compareTo(firstDayOfNewFiscalYear) >= 0;
272        }
