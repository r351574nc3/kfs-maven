099        /**
100         * @see org.kuali.kfs.module.ld.batch.service.LaborYearEndBalanceForwardService#forwardBalance(java.lang.Integer,
101         *      java.lang.Integer)
102         */
103        public void forwardBalance(Integer fiscalYear, Integer newFiscalYear) {
104            SystemOptions options = optionsService.getOptions(fiscalYear);
105            Date runDate = dateTimeService.getCurrentSqlDate();
106    
107            Map<String, Integer> reportSummary = this.constructReportSummary();
108            PosterOutputSummaryReport posterOutputSummaryReport = new PosterOutputSummaryReport();
109    
110            List<String> processableBalanceTypeCodes = this.getProcessableBalanceTypeCode(options);
111            List<String> processableObjectTypeCodes = this.getProcessableObjectTypeCodes(options);
112            List<String> subFundGroupCodes = this.getSubFundGroupProcessed();
113            List<String> fundGroupCodes = this.getFundGroupProcessed();
114    
115            // create files
116            String balanceForwardsFileName = batchFileDirectoryName + File.separator + LaborConstants.BatchFileSystem.BALANCE_FORWARDS_FILE + GeneralLedgerConstants.BatchFileSystem.EXTENSION;
117            File balanceForwardsFile = new File(balanceForwardsFileName);
118            PrintStream balanceForwardsPs = null;
119    
120            try {
121                balanceForwardsPs = new PrintStream(balanceForwardsFile);
122            }
123            catch (FileNotFoundException e) {
124                throw new RuntimeException("balanceForwardsFile Files doesn't exist " + balanceForwardsFileName);
125            }
126    
127            // process the selected balances by balance type and object type
128            Map<String, String> fieldValues = new HashMap<String, String>();
129            int numberOfSelectedBalance = 0;
130            for (String balanceTypeCode : processableBalanceTypeCodes) {
131                fieldValues.put(KFSPropertyConstants.FINANCIAL_BALANCE_TYPE_CODE, balanceTypeCode);
132    
133                for (String objectTypeCode : processableObjectTypeCodes) {
134                    fieldValues.put(KFSPropertyConstants.FINANCIAL_OBJECT_TYPE_CODE, objectTypeCode);
135    
136                    fieldValues.remove(KFSPropertyConstants.CHART_OF_ACCOUNTS_CODE);
137                    fieldValues.remove(KFSPropertyConstants.ACCOUNT_NUMBER);
138    
139                    Iterator<LedgerBalanceForYearEndBalanceForward> balanceIterator = laborLedgerBalanceService.findBalancesForFiscalYear(fiscalYear, fieldValues, subFundGroupCodes, fundGroupCodes);
140                    numberOfSelectedBalance += postSelectedBalancesAsOriginEntries(balanceIterator, newFiscalYear, balanceForwardsPs, runDate, posterOutputSummaryReport, reportSummary);
141                }
142            }
143    
144            this.fillStatisticsReportWriter(reportSummary);
145            fillParametersReportWriter(runDate, fiscalYear, fundGroupCodes, subFundGroupCodes, getOriginationCode() , processableBalanceTypeCodes, processableObjectTypeCodes, getDocumentTypeCode());
146            posterOutputSummaryReport.writeReport(laborBalanceForwardReportWriterService);
147    
148            balanceForwardsPs.close();
149            this.createDoneFile(balanceForwardsFileName);
150        }
151    
152        // create a done file after balance forward completes.
153        protected void createDoneFile(String originEntryFileName) {
154            String doneFileName = originEntryFileName.replace(GeneralLedgerConstants.BatchFileSystem.EXTENSION, GeneralLedgerConstants.BatchFileSystem.DONE_FILE_EXTENSION);
155            File doneFile = new File(doneFileName);
156    
157            if (!doneFile.exists()) {
158                try {
159                    doneFile.createNewFile();
160                }
161                catch (IOException e) {
162                    throw new RuntimeException();
163                }
164            }
165        }

167        /**
168         * post the qualified balances into origin entry table for the further labor ledger processing
169         * 
170         * @param balanceIterator the given ledger balances that will be carried forward
171         * @param newFiscalYear the new fiscal year
172         * @param validGroup the group that the posted transaction belongs to
173         * @param errorMap the map that records the error messages
174         * @param runDate the date the transaction is posted
175         * @return the number of qualified balances
176         */
177        protected int postSelectedBalancesAsOriginEntries(Iterator<LedgerBalanceForYearEndBalanceForward> balanceIterator, Integer newFiscalYear, PrintStream balanceForwardsPs, Date runDate, PosterOutputSummaryReport posterOutputSummaryReport, Map<String, Integer> reportSummary) {
178            int numberOfSelectedBalance = 0;
179            String description = this.getDescription();
180            String originationCode = this.getOriginationCode();
181            String documentTypeCode = this.getDocumentTypeCode();
182    
183            while (balanceIterator != null && balanceIterator.hasNext()) {
184                LedgerBalanceForYearEndBalanceForward balance = balanceIterator.next();
185                this.updateReportSummary(reportSummary, LEDGER_BALANCE, KFSConstants.OperationType.READ);
186    
187                List<Message> errors = null;
188    
189                boolean isValidBalance = validateBalance(balance, errors);
190                LaborOriginEntry laborOriginEntry = new LaborOriginEntry();
191                if (isValidBalance) {
192                    laborOriginEntry.setUniversityFiscalYear(newFiscalYear);
193                    laborOriginEntry.setFinancialDocumentTypeCode(documentTypeCode);
194                    laborOriginEntry.setFinancialSystemOriginationCode(originationCode);
195                    laborOriginEntry.setTransactionLedgerEntryDescription(description);
196    
197                    this.postAsOriginEntry(balance, laborOriginEntry, balanceForwardsPs, runDate);
198                    numberOfSelectedBalance++;
199    
200                    posterOutputSummaryReport.summarize((OriginEntryInformation) laborOriginEntry);
201                    this.updateReportSummary(reportSummary, LEDGER_BALANCE, KFSConstants.OperationType.SELECT);
202                    this.updateReportSummary(reportSummary, LEDGER_BALANCE, KFSConstants.OperationType.INSERT);
203                }
204                else if (errors != null && !errors.isEmpty()) {
205                    ObjectUtil.buildObject(laborOriginEntry, balance);
206    
207                    laborBalanceForwardReportWriterService.writeError(laborOriginEntry, errors);
208                    this.updateReportSummary(reportSummary, LEDGER_BALANCE, KFSConstants.OperationType.REPORT_ERROR);
209                }
210            }
211    
212            return numberOfSelectedBalance;
213        }

227        /**
228         * post the qualified balance into origin entry table for the further labor ledger processing
229         * 
230         * @param balance the given ledger balance that will be carried forward
231         * @param newFiscalYear the new fiscal year
232         * @param validGroup the group that the posted transaction belongs to
233         * @param postingDate the date the transaction is posted
234         */
235        protected void postAsOriginEntry(LedgerBalanceForYearEndBalanceForward balance, LaborOriginEntry originEntry, PrintStream balanceForwardsPs, Date postingDate) {
236            try {
237                originEntry.setAccountNumber(balance.getAccountNumber());
238                originEntry.setChartOfAccountsCode(balance.getChartOfAccountsCode());
239                originEntry.setSubAccountNumber(balance.getSubAccountNumber());
240                originEntry.setFinancialObjectCode(balance.getFinancialObjectCode());
241                originEntry.setFinancialSubObjectCode(balance.getFinancialSubObjectCode());
242                originEntry.setFinancialBalanceTypeCode(balance.getFinancialBalanceTypeCode());
243                originEntry.setFinancialObjectTypeCode(balance.getFinancialObjectTypeCode());
244    
245                originEntry.setPositionNumber(balance.getPositionNumber());
246                originEntry.setEmplid(balance.getEmplid());
247                originEntry.setDocumentNumber(balance.getFinancialBalanceTypeCode() + balance.getAccountNumber());
248    
249                originEntry.setProjectCode(KFSConstants.getDashProjectCode());
250                originEntry.setUniversityFiscalPeriodCode(KFSConstants.PERIOD_CODE_CG_BEGINNING_BALANCE);
251    
252                KualiDecimal transactionAmount = balance.getAccountLineAnnualBalanceAmount();
253                transactionAmount = transactionAmount.add(balance.getContractsGrantsBeginningBalanceAmount());
254    
255                originEntry.setTransactionLedgerEntryAmount(transactionAmount.abs());
256                originEntry.setTransactionDebitCreditCode(DebitCreditUtil.getDebitCreditCode(transactionAmount, false));
257    
258                originEntry.setTransactionLedgerEntrySequenceNumber(null);
259                originEntry.setTransactionTotalHours(BigDecimal.ZERO);
260                originEntry.setTransactionDate(postingDate);
261    
262                try {
263                    balanceForwardsPs.printf("%s\n", originEntry.getLine());
264                }
265                catch (Exception e) {
266                    throw new RuntimeException(e.toString());
267                }
268            }
269            catch (Exception e) {
270                LOG.error(e);
271            }
272        }

318        // fill the report writer with the collected data
319        protected Map<String, Integer> constructReportSummary() {
320            Map<String, Integer> reportSummary = new HashMap<String, Integer>();
321            reportSummary.put(LEDGER_BALANCE + "," + KFSConstants.OperationType.READ, 0);
322            reportSummary.put(LEDGER_BALANCE + "," + KFSConstants.OperationType.SELECT, 0);
323            reportSummary.put(LEDGER_BALANCE + "," + KFSConstants.OperationType.REPORT_ERROR, 0);
324            reportSummary.put(ORIGN_ENTRY + "," + KFSConstants.OperationType.INSERT, 0);
325    
326            return reportSummary;
327        }

329        // fill the gl entry report writer with the collected data
330        protected void fillStatisticsReportWriter(Map<String, Integer> glEntryReportSummary) {
331            laborBalanceForwardReportWriterService.writeStatisticLine("NUMBER OF RECORDS READ              %,9d", glEntryReportSummary.get(LEDGER_BALANCE + "," + KFSConstants.OperationType.READ));

333            laborBalanceForwardReportWriterService.writeStatisticLine("NUMBER OF RECORDS IN ERROR          %,9d", glEntryReportSummary.get(LEDGER_BALANCE + "," + KFSConstants.OperationType.REPORT_ERROR));
334            laborBalanceForwardReportWriterService.writeStatisticLine("NUMBER OF RECORDS INSERTED          %,9d", glEntryReportSummary.get(ORIGN_ENTRY + "," + KFSConstants.OperationType.INSERT));
335        }

337        /**
338         * Report out which significant parameters were used in this report.
339         * @param runDate the date when this job was run
340         * @param closingYear the fiscal year closed by this job
341         * @param processedFundGroups the fund groups processed by this job
342         * @param processedSubFundGroups the sub-fund groups processed by this job
343         * @param originationCode the origination code of the posted entries
344         * @param processedBalanceTypeCodes the balance type codes processed by this job
345         * @param processedObjectTypeCodes the object type codes processed by this job
346         * @param documentTypeCode the document type code of posted entries
347         */
348        protected void fillParametersReportWriter(Date runDate, Integer closingYear, List<String> processedFundGroups, List<String> processedSubFundGroups, String originationCode, List<String> processedBalanceTypeCodes, List<String> processedObjectTypeCodes, String documentTypeCode) {
349            laborBalanceForwardReportWriterService.writeParameterLine("%32s %10s", GeneralLedgerConstants.ANNUAL_CLOSING_TRANSACTION_DATE_PARM, runDate.toString());
350            laborBalanceForwardReportWriterService.writeParameterLine("%32s %10s", LaborConstants.YearEnd.OLD_FISCAL_YEAR, closingYear);
351            laborBalanceForwardReportWriterService.writeParameterLine("%32s %10s", LaborConstants.YearEnd.FUND_GROUP_PROCESSED, StringUtils.join(processedFundGroups, ", "));
352            laborBalanceForwardReportWriterService.writeParameterLine("%32s %10s", LaborConstants.YearEnd.SUB_FUND_GROUP_PROCESSED, StringUtils.join(processedSubFundGroups, ", "));
353            laborBalanceForwardReportWriterService.writeParameterLine("%32s %10s", LaborConstants.YearEnd.ORIGINATION_CODE, originationCode);
354            laborBalanceForwardReportWriterService.writeParameterLine("%32s %10s", LaborYearEndBalanceForwardServiceImpl.PROCESSED_BALANCE_TYPES_LABEL, StringUtils.join(processedBalanceTypeCodes, ", " ));
355            laborBalanceForwardReportWriterService.writeParameterLine("%32s %10s", LaborYearEndBalanceForwardServiceImpl.PROCESSED_OBJECT_TYPES_LABEL, StringUtils.join(processedObjectTypeCodes, ", " ));
356            laborBalanceForwardReportWriterService.writeParameterLine("%32s %10s", GeneralLedgerConstants.ANNUAL_CLOSING_DOCUMENT_TYPE, documentTypeCode);
357            laborBalanceForwardReportWriterService.pageBreak();
358        }

360        // update the entry in the given report summary
361        protected void updateReportSummary(Map<String, Integer> reportSummary, String destination, String operation) {
362            String key = destination + "," + operation;
363    
364            if (reportSummary.containsKey(key)) {
365                Integer count = reportSummary.get(key);
366                reportSummary.put(key, count + 1);
367            }
368            else {
369                reportSummary.put(key, 1);
370            }
371        }
