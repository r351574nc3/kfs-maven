016    package org.kuali.kfs.sec.datadictionary;
017    
018    import org.kuali.kfs.sec.document.authorization.SecAccountingLineAuthorizer;
019    import org.kuali.kfs.sys.document.authorization.AccountingLineAuthorizer;
020    import org.kuali.kfs.sys.document.datadictionary.AccountingLineGroupDefinition;
021    
022    
023    /**
024     * Override of group definition to wrap access security around accounting objects
025     */
026    public class SecAccountingLineGroupDefinition extends AccountingLineGroupDefinition {
027    
028        /**
029         * Creates a new instance of the security accounting line authorizer and sets the configured authorizer property
030         * 
031         * @see org.kuali.kfs.sys.document.datadictionary.AccountingLineGroupDefinition#createAccountingLineAuthorizer()
032         */
033        @Override
034        protected AccountingLineAuthorizer createAccountingLineAuthorizer() {
035            Class<? extends AccountingLineAuthorizer> authorizerClass = getAccountingLineAuthorizerClass();
036    
037            SecAccountingLineAuthorizer secAuthorizer = null;
038            AccountingLineAuthorizer authorizer = null;
039            try {
040                secAuthorizer = SecAccountingLineAuthorizer.class.newInstance();
041                authorizer = authorizerClass.newInstance();
042    
043                secAuthorizer.setLineAuthorizer(authorizer);
044            }
045            catch (InstantiationException ie) {
046                throw new IllegalArgumentException("InstantiationException while attempting to instantiate AccountingLineAuthorization class", ie);
047            }
048            catch (IllegalAccessException iae) {
049                throw new IllegalArgumentException("IllegalAccessException while attempting to instantiate AccountingLineAuthorization class", iae);
050            }
051            
052            return secAuthorizer;
053        }
054    
055    }
