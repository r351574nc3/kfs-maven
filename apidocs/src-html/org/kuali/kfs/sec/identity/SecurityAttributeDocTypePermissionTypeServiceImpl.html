034    /**
035     * Type service for Access Security Permissions that restrict based on property name and document type
036     */
037    public class SecurityAttributeDocTypePermissionTypeServiceImpl extends SecurityAttributePermissionTypeServiceImpl {
038        private static final org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(SecurityAttributeDocTypePermissionTypeServiceImpl.class);
039    
040        {
041            requiredAttributes.add(SecKimAttributes.DOCUMENT_TYPE_NAME);
042            checkRequiredAttributes = false;
043        }
044    
045        /**
046         * @see org.kuali.rice.kim.service.support.impl.KimPermissionTypeServiceBase#performPermissionMatches(org.kuali.rice.kim.bo.types.dto.AttributeSet, java.util.List)
047         */
048        @Override
049        protected List<KimPermissionInfo> performPermissionMatches(AttributeSet requestedDetails, List<KimPermissionInfo> permissionsList) {
050            List<KimPermissionInfo> matchingPermissions = new ArrayList<KimPermissionInfo>();
051    
052            for (KimPermissionInfo kpi : permissionsList) {
053                String documentTypeNameMatch = requestedDetails.get(SecKimAttributes.DOCUMENT_TYPE_NAME);
054                String documentTypeName = kpi.getDetails().get(SecKimAttributes.DOCUMENT_TYPE_NAME);
055    
056                boolean docTypeMatch = false;
057                if (SecConstants.ALL_DOCUMENT_TYPE_NAME.equals(documentTypeName)) {
058                    docTypeMatch = true;
059                }
060                else if (StringUtils.equals(documentTypeNameMatch, documentTypeName)) {
061                    docTypeMatch = true;
062                }
063                else {
064                    docTypeMatch = isParentDocType(documentTypeNameMatch, documentTypeName);
065                }
066    
067                if (docTypeMatch && isDetailMatch(requestedDetails, kpi.getDetails())) {
068                    matchingPermissions.add(kpi);
069                }
070            }
071    
072            return matchingPermissions;
073        }
074    
075        /**
076         * Determines if a document type is a parent of another
077         * 
078         * @param docTypeName potential child doc type name
079         * @param potientialParentDocTypeName potential parent doc type name
080         * @return boolean true if the first document type is a child of the second
081         */
082        protected boolean isParentDocType(String docTypeName, String potentialParentDocTypeName) {
083            WorkflowInfo workflowInfo = new WorkflowInfo();
084    
085            DocumentTypeDTO documentType = null;
086            try {
087                documentType = workflowInfo.getDocType(docTypeName);
088                String parentDocTypeName = documentType.getDocTypeParentName();
089    
090                if (StringUtils.equals(docTypeName, parentDocTypeName)) {
091                    return true;
092                }
093                else if ((documentType.getDocTypeParentId() == null) || documentType.getDocTypeParentId().equals(documentType.getDocTypeId())) {
094                    return false;
095                }
096                else {
097                    return isParentDocType(parentDocTypeName, potentialParentDocTypeName);
098                }
099            }
100            catch (WorkflowException e) {
101                LOG.error("Invalid document type found " + docTypeName);
102                throw new RuntimeException("Invalid document type found: " + docTypeName);
103            }
104        }
