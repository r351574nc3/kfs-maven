016    package org.kuali.kfs.sec.businessobject.options;
017    
018    import java.util.ArrayList;
019    import java.util.List;
020    
021    import org.kuali.kfs.sec.service.AccessSecurityService;
022    import org.kuali.kfs.sys.context.SpringContext;
023    import org.kuali.rice.core.util.KeyLabelPair;
024    import org.kuali.rice.kns.lookup.keyvalues.KeyValuesBase;
025    
026    
027    /**
028     * Returns list of security template Ids
029     */
030    public class SecurityTemplateIdFinder extends KeyValuesBase {
031    
032        /**
033         * @see org.kuali.keyvalues.KeyValuesFinder#getKeyValues()
034         */
035        public List getKeyValues() {
036            List activeLabels = new ArrayList();
037    
038            activeLabels.add(new KeyLabelPair(SpringContext.getBean(AccessSecurityService.class).getInquiryWithFieldValueTemplateId(), "Balance Inquiry"));
039            activeLabels.add(new KeyLabelPair(SpringContext.getBean(AccessSecurityService.class).getEditAccountingLineWithFieldValueTemplateId(), "Edit Accounting Line"));
040            activeLabels.add(new KeyLabelPair(SpringContext.getBean(AccessSecurityService.class).getEditDocumentWithFieldValueTemplateId(), "Edit Document"));
041            activeLabels.add(new KeyLabelPair(SpringContext.getBean(AccessSecurityService.class).getLookupWithFieldValueTemplateId(), "Lookup Records"));
042            activeLabels.add(new KeyLabelPair(SpringContext.getBean(AccessSecurityService.class).getViewAccountingLineWithFieldValueTemplateId(), "View Accounting Line"));
043            activeLabels.add(new KeyLabelPair(SpringContext.getBean(AccessSecurityService.class).getViewDocumentWithFieldValueTemplateId(), "View Document"));
044            activeLabels.add(new KeyLabelPair(SpringContext.getBean(AccessSecurityService.class).getViewNotesAttachmentsWithFieldValueTemplateId(), "View Notes/Attachments"));
045    
046            return activeLabels;
047        }
048    }
