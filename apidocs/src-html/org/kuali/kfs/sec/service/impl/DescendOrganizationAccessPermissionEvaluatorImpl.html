016    package org.kuali.kfs.sec.service.impl;
017    
018    import org.apache.commons.lang.StringUtils;
019    import org.kuali.kfs.coa.service.OrganizationService;
020    import org.kuali.kfs.sys.KFSPropertyConstants;
021    import org.kuali.kfs.sys.context.SpringContext;
022    
023    /**
024     * Custom access permission evaluator that looks at the organization hierarchy when matching values
025     */
026    public class DescendOrganizationAccessPermissionEvaluatorImpl extends AccessPermissionEvaluatorImpl {
027    
028        /**
029         * Matches org values based on org hierarchy
030         * 
031         * @see org.kuali.kfs.sec.service.impl.AccessPermissionEvaluatorImpl#isMatch(java.lang.String, java.lang.String)
032         */
033        @Override
034        protected boolean isMatch(String matchValue, String value) {
035            boolean match = false;
036    
037            if (StringUtils.equalsIgnoreCase(value, matchValue)) {
038                match = true;
039            }
040            else {
041                String chartCode = (String) otherKeyFieldValues.get(KFSPropertyConstants.CHART_OF_ACCOUNTS_CODE);
042                match = SpringContext.getBean(OrganizationService.class).isParentOrganization(chartCode, value, chartCode, matchValue);
043            }
044    
045            return match;
046        }
047    
048    }
