016    package org.kuali.kfs.pdp.dataaccess;
017    
018    import java.util.Iterator;
019    import java.util.List;
020    
021    import org.kuali.kfs.pdp.businessobject.PaymentGroup;
022    import org.kuali.kfs.pdp.businessobject.PaymentProcess;
023    
024    public interface PaymentGroupDao {
025    
026        /**
027         * Get all the disbursement numbers for a specific process of a certain type
028         * 
029         * @param pid
030         * @param disbursementType
031         * @return
032         */
033        public List<Integer> getDisbursementNumbersByDisbursementType(Integer pid, String disbursementType);
034        
035        /**
036         * Get all the disbursement numbers for a specific process of a certain type for a certain bank
037         * 
038         * @param pid
039         * @param disbursementType
040         * @param bankCode the bank code to find disbursement numbers for
041         * @return
042         */
043        public abstract List<Integer> getDisbursementNumbersByDisbursementTypeAndBankCode(Integer pid, String disbursementType, String bankCode);
044    
045        /**
046         * Gets list of ach payments in which an advice notification has not been sent
047         * 
048         * @return List<PaymentGroup>
049         */
050        public List<PaymentGroup> getAchPaymentsNeedingAdviceNotification();
051        
052        /**
053         * Given a process id and a disbursement type, finds a distinct list of bank codes used by payment groups within that payment process
054         * @param pid payment process to query payment groups of
055         * @param disbursementType the type of disbursements to query
056         * @return a sorted List of bank codes
057         */
058        public abstract List<String> getDistinctBankCodesForProcessAndType(Integer pid, String disbursementType);
059    
060    }
