016    package org.kuali.kfs.fp.businessobject;
017    
018    import org.kuali.rice.kns.bo.KualiCodeBase;
019    
020    
021    /**
022     * This class represents a CashDetailTypeCode, which is used to differentiate records of the same general data structure from one
023     * another. For example, a CashReceiptDocument has checks and so does a BursarDepositDocument. To be able to re-use the same table
024     * and data structure, we need a field that can act as a flag. Another example involves the fact that a CashReceipt contains a coin
025     * record, a currency record, and a change out record (aggregated coin and currency together). Since we use the same tables to
026     * represent this, we need an attribute to help differentiate between the bunch especially when they all three must co-exist within
027     * the same parent CashReceiptDocument instance.
028     */
029    public class CashDetailTypeCode extends KualiCodeBase {
030        private static final long serialVersionUID = -5228907091959656216L;
031        private String description;
032    
033        /**
034         * Constructs a CashDetailTypeCode business object.
035         */
036        public CashDetailTypeCode() {
037            super();
038        }
039    
040        /**
041         * Gets the description attribute.
042         * 
043         * @return Returns the description.
044         */
045        public String getDescription() {
046            return description;
047        }
048    
049        /**
050         * Sets the description attribute value.
051         * 
052         * @param description The description to set.
053         */
054        public void setDescription(String description) {
055            this.description = description;
056        }
057    }
