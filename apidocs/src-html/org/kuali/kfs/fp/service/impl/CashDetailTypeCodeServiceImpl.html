016    package org.kuali.kfs.fp.service.impl;
017    
018    import org.kuali.kfs.fp.businessobject.CashDetailTypeCode;
019    import org.kuali.kfs.fp.service.CashDetailTypeCodeService;
020    
021    /**
022     * This is the default implementation for the CashDetailTypeCodeService interface. This implementation used the KualiCodeService,
023     * which in turn uses the KualiCodeDao for retrieving values from the database.
024     */
025    public class CashDetailTypeCodeServiceImpl implements CashDetailTypeCodeService {
026        // KFSConstants for doing the actual lookups
027        public final static String CASH_RECEIPT_CHECK = "CRCHK";
028        public final static String CASH_RECEIPT_COIN = "CRCOIN";
029    
030        /**
031         * Constructs a CashDetailTypeCodeServiceImpl instance.
032         */
033        public CashDetailTypeCodeServiceImpl() {
034        }
035    
036    
037        /**
038         * Gets the associated check type code for a CashReceipt.
039         * 
040         * @return Returns the CashReceipt check type code.
041         * @see org.kuali.rice.kns.service.CashDetailTypeCode#getCashReceiptCheckTypeCode()
042         */
043        public CashDetailTypeCode getCashReceiptCheckTypeCode() {
044            return getCashDetailTypeCodeByCode(CASH_RECEIPT_CHECK);
045        }
046    
047        /**
048         * Gets the associated coin type code for a CashReceipt.
049         * 
050         * @return Returns the CashReceipt coin type code.
051         * @see org.kuali.rice.kns.service.CashDetailTypeCode#getCashReceiptCoinTypeCode()
052         */
053        public CashDetailTypeCode getCashReceiptCoinTypeCode() {
054            return getCashDetailTypeCodeByCode(CASH_RECEIPT_CHECK);
055        }
056    
057        /**
058         * Retrieves a populated instance corresponding to the code passed into this method. This is retrieved via the KualiCodeService
059         * and in turn from the database. 
060         * 
061         * TODO - uncomment the commented out line and remove the others when the table is in place
062         * 
063         * @param cashDetailTypeCode The identifier used to retrieve the appropriate type code object.
064         * @return A CashDetailTypeCode instance based on the code provided.
065         * 
066         * @see #CASH_RECEIPT_CHECK
067         * @see #CASH_RECEIPT_COIN
068         */
069        protected CashDetailTypeCode getCashDetailTypeCodeByCode(String cashDetailTypeCode) {
070            // return (CashDetailTypeCode) kualiCodeService.getByCode(CashDetailTypeCode.class, cashDetailTypeCode);
071            return getDummyInstance(cashDetailTypeCode);
072        }
073    
074        /**
075         * This method is a temporary helper method. This should be removed when the lookup table for CashDetailTypeCode business
076         * objects is put in place. Then we'll be retrieving the stuff from the database. 
077         * 
078         * TODO - remove this method after the table is in place; this is a temp helper method
079         * 
080         * @param cashDetailTypeCodeCode The code to populate the dummy instance with.
081         * @return A CashDetailTypeCode instance based on the value provided.
082         */
083        protected CashDetailTypeCode getDummyInstance(String cashDetailTypeCodeCode) {
084            CashDetailTypeCode cashDetailTypeCode = new CashDetailTypeCode();
085            cashDetailTypeCode.setCode(cashDetailTypeCodeCode);
086            return cashDetailTypeCode;
087        }
088    }
