016    package org.kuali.kfs.fp.document.validation.impl;
017    
018    import org.apache.commons.lang.StringUtils;
019    import org.kuali.kfs.fp.document.service.TransferOfFundsService;
020    import org.kuali.kfs.sys.KFSKeyConstants;
021    import org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent;
022    import org.kuali.kfs.sys.document.validation.impl.AccountingLineValueAllowedValidation;
023    import org.kuali.rice.kns.util.GlobalVariables;
024    
025    /**
026     * Override of standard accounting line value allowed definition to make sure that any object sub type on a transfer of funds is either mandatory transfer or non-mandatory transfer.
027     */
028    public class TransferOfFundsObjectSubTypeValueAllowedValidation extends AccountingLineValueAllowedValidation {
029        private TransferOfFundsService transferOfFundsService;
030    
031        /**
032         * Overrides the parent to make sure that the chosen object code's object sub-type code is either Mandatory Transfer or
033         * Non-Mandatory Transfer. 
034         * @see org.kuali.kfs.sys.document.validation.impl.AccountingLineValueAllowedValidation#validate(org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent)
035         */
036        @Override
037        public boolean validate(AttributedDocumentEvent event) {
038            getAccountingLineForValidation().refreshReferenceObject("objectCode");
039            String objectSubTypeCode = getAccountingLineForValidation().getObjectCode().getFinancialObjectSubTypeCode();
040    
041            // make sure a object sub type code exists for this object code
042            if (StringUtils.isBlank(objectSubTypeCode)) {
043                GlobalVariables.getMessageMap().putError("financialObjectCode", KFSKeyConstants.ERROR_DOCUMENT_TOF_OBJECT_SUB_TYPE_IS_NULL, getAccountingLineForValidation().getFinancialObjectCode());
044                return false;
045            }
046    
047            if (!transferOfFundsService.isMandatoryTransfersSubType(objectSubTypeCode) && !transferOfFundsService.isNonMandatoryTransfersSubType(objectSubTypeCode)) {
048                GlobalVariables.getMessageMap().putError("financialObjectCode", KFSKeyConstants.ERROR_DOCUMENT_TOF_OBJECT_SUB_TYPE_NOT_MANDATORY_OR_NON_MANDATORY_TRANSFER, new String[] { getAccountingLineForValidation().getObjectCode().getFinancialObjectSubType().getFinancialObjectSubTypeName(), getAccountingLineForValidation().getFinancialObjectCode() });
049                return false;
050            }
051    
052            return true;
053        }
054    
055        /**
056         * Gets the transferOfFundService attribute. 
057         * @return Returns the transferOfFundService.
058         */
059        public TransferOfFundsService getTransferOfFundsService() {
060            return transferOfFundsService;
061        }
062    
063        /**
064         * Sets the transferOfFundService attribute value.
065         * @param transferOfFundService The transferOfFundService to set.
066         */
067        public void setTransferOfFundsService(TransferOfFundsService transferOfFundsService) {
068            this.transferOfFundsService = transferOfFundsService;
069        }
070    }
