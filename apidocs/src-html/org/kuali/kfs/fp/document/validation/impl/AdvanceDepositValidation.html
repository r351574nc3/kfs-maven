034    public class AdvanceDepositValidation extends GenericValidation {
035        private AdvanceDepositDetail advanceDepositDetailForValidation;
036        /**
037         * @see org.kuali.kfs.sys.document.validation.Validation#validate(org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent)
038         */
039        public boolean validate(AttributedDocumentEvent event) {
040            
041            AdvanceDepositDetail advanceDeposit = getAdvanceDepositDetailForValidation();
042            boolean isValid = true;
043    
044            // check that dollar amount is not zero before continuing
045            if (isValid) {
046                isValid = !advanceDeposit.getFinancialDocumentAdvanceDepositAmount().isZero();
047                if (!isValid) {
048                    String label = SpringContext.getBean(DataDictionaryService.class).getAttributeLabel(AdvanceDepositDetail.class, KFSPropertyConstants.ADVANCE_DEPOSIT_AMOUNT);
049                    GlobalVariables.getMessageMap().putError(KFSPropertyConstants.ADVANCE_DEPOSIT_AMOUNT, KFSKeyConstants.AdvanceDeposit.ERROR_DOCUMENT_ADVANCE_DEPOSIT_ZERO_AMOUNT, label);
050                }
051            }
052    
053            if (isValid) {
054                isValid = BankCodeValidation.validate(advanceDeposit.getFinancialDocumentBankCode(), KFSPropertyConstants.FINANCIAL_DOCUMENT_BANK_CODE, true, false);
055            }
056    
057            return isValid;
058        }
059        /**
060         * Gets the advanceDepositDetailForValidation attribute. 
061         * @return Returns the advanceDepositDetailForValidation.
062         */
063        public AdvanceDepositDetail getAdvanceDepositDetailForValidation() {
064            return advanceDepositDetailForValidation;
065        }
066        /**
067         * Sets the advanceDepositDetailForValidation attribute value.
068         * @param advanceDepositDetailForValidation The advanceDepositDetailForValidation to set.
069         */
070        public void setAdvanceDepositDetailForValidation(AdvanceDepositDetail advanceDepositDetailForValidation) {
071            this.advanceDepositDetailForValidation = advanceDepositDetailForValidation;
072        }
