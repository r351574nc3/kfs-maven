016    package org.kuali.kfs.fp.document.validation.impl;
017    
018    import org.apache.commons.lang.StringUtils;
019    import org.kuali.kfs.coa.businessobject.ObjectType;
020    import org.kuali.kfs.fp.businessobject.VoucherSourceAccountingLine;
021    import org.kuali.kfs.sys.businessobject.AccountingLine;
022    import org.kuali.kfs.sys.document.service.AccountingLineRuleHelperService;
023    import org.kuali.kfs.sys.document.validation.GenericValidation;
024    import org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent;
025    import org.kuali.rice.kns.service.DataDictionaryService;
026    
027    /**
028     * Validation which checks at the object type code on a Journal Voucher's accounting line is valid
029     */
030    public class JournalVoucherObjectTypeValidation extends GenericValidation {
031        private VoucherSourceAccountingLine accountingLineForValidation;
032        private AccountingLineRuleHelperService accountingLineRuleHelperService;
033        private DataDictionaryService dataDictionaryService;
034        
035        /**
036         * Checks that the object type code on the Journal Voucher accounting line is valid and active
037         * @see org.kuali.kfs.sys.document.validation.Validation#validate(org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent)
038         */
039        public boolean validate(AttributedDocumentEvent event) {
040            boolean valid = true;
041            if (StringUtils.isNotBlank(accountingLineForValidation.getObjectTypeCode())) {
042                accountingLineForValidation.refreshReferenceObject("objectType");
043                ObjectType objectTypeCode = accountingLineForValidation.getObjectType();
044                valid &= getAccountingLineRuleHelperService().isValidObjectTypeCode(objectTypeCode, getDataDictionaryService().getDataDictionary());
045            }
046            return valid;
047        }
048    
049        /**
050         * Gets the accountingLineForValidation attribute. 
051         * @return Returns the accountingLineForValidation.
052         */
053        public VoucherSourceAccountingLine getAccountingLineForValidation() {
054            return accountingLineForValidation;
055        }
056    
057        /**
058         * Sets the accountingLineForValidation attribute value.
059         * @param accountingLineForValidation The accountingLineForValidation to set.
060         */
061        public void setAccountingLineForValidation(VoucherSourceAccountingLine accountingLineForValidation) {
062            this.accountingLineForValidation = accountingLineForValidation;
063        }
064    
065        /**
066         * Gets the accountingLineRuleHelperService attribute. 
067         * @return Returns the accountingLineRuleHelperService.
068         */
069        public AccountingLineRuleHelperService getAccountingLineRuleHelperService() {
070            return accountingLineRuleHelperService;
071        }
072    
073        /**
074         * Sets the accountingLineRuleHelperService attribute value.
075         * @param accountingLineRuleHelperService The accountingLineRuleHelperService to set.
076         */
077        public void setAccountingLineRuleHelperService(AccountingLineRuleHelperService accountingLineRuleHelperService) {
078            this.accountingLineRuleHelperService = accountingLineRuleHelperService;
079        }
080    
081        /**
082         * Gets the dataDictionaryService attribute. 
083         * @return Returns the dataDictionaryService.
084         */
085        public DataDictionaryService getDataDictionaryService() {
086            return dataDictionaryService;
087        }
088    
089        /**
090         * Sets the dataDictionaryService attribute value.
091         * @param dataDictionaryService The dataDictionaryService to set.
092         */
093        public void setDataDictionaryService(DataDictionaryService dataDictionaryService) {
094            this.dataDictionaryService = dataDictionaryService;
095        }
096    
097    }
