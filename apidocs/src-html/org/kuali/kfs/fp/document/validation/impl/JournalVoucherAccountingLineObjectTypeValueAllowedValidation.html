016    package org.kuali.kfs.fp.document.validation.impl;
017    
018    import static org.kuali.kfs.sys.KFSConstants.GENERIC_CODE_PROPERTY_NAME;
019    import static org.kuali.kfs.sys.KFSConstants.OBJECT_TYPE_CODE_PROPERTY_NAME;
020    import static org.kuali.kfs.sys.KFSKeyConstants.ERROR_REQUIRED;
021    
022    import org.apache.commons.lang.StringUtils;
023    import org.kuali.kfs.coa.businessobject.ObjectType;
024    import org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent;
025    import org.kuali.kfs.sys.document.validation.impl.AccountingLineValueAllowedValidation;
026    import org.kuali.rice.kns.service.DataDictionaryService;
027    import org.kuali.rice.kns.util.GlobalVariables;
028    
029    /**
030     * The Journal Voucher's version of the accounting line Object Type validation.
031     */
032    public class JournalVoucherAccountingLineObjectTypeValueAllowedValidation extends AccountingLineValueAllowedValidation {
033        private DataDictionaryService dataDictionaryService;
034    
035        /**
036         * The JV allows any object type b/c it is up to the user to enter it into the interface, but it is required. The existence
037         * check is done for us automatically by the data dictionary validation if a value exists; beforehand so we can assume that any
038         * value in that field is valid.
039         * @see org.kuali.kfs.sys.document.validation.Validation#validate(org.kuali.kfs.sys.document.validation.event.AttributedDocumentEvent)
040         */
041        public boolean validate(AttributedDocumentEvent event) {
042            String objectTypeCode = getAccountingLineForValidation().getObjectTypeCode();
043            if (StringUtils.isNotBlank(objectTypeCode)) {
044                return true;
045            }
046            else {
047                String label = getDataDictionaryService().getDataDictionary().getBusinessObjectEntry(ObjectType.class.getName()).getAttributeDefinition(GENERIC_CODE_PROPERTY_NAME).getLabel();
048                GlobalVariables.getMessageMap().putError(OBJECT_TYPE_CODE_PROPERTY_NAME, ERROR_REQUIRED, label);
049                return false;
050            }
051        }
052    
053        /**
054         * Gets the dataDictionaryService attribute. 
055         * @return Returns the dataDictionaryService.
056         */
057        public DataDictionaryService getDataDictionaryService() {
058            return dataDictionaryService;
059        }
060    
061        /**
062         * Sets the dataDictionaryService attribute value.
063         * @param dataDictionaryService The dataDictionaryService to set.
064         */
065        public void setDataDictionaryService(DataDictionaryService dataDictionaryService) {
066            this.dataDictionaryService = dataDictionaryService;
067        }
068    }
