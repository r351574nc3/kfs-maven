016    package org.kuali.kfs.fp.batch;
017    
018    import java.io.File;
019    import java.util.List;
020    
021    import org.apache.commons.lang.StringUtils;
022    import org.kuali.kfs.coa.businessobject.Account;
023    import org.kuali.kfs.coa.service.AccountService;
024    import org.kuali.kfs.fp.businessobject.ProcurementCardTransaction;
025    import org.kuali.kfs.sys.KFSConstants;
026    import org.kuali.kfs.sys.KFSKeyConstants;
027    import org.kuali.kfs.sys.batch.XmlBatchInputFileTypeBase;
028    import org.kuali.kfs.sys.context.SpringContext;
029    import org.kuali.rice.kns.service.DateTimeService;
030    import org.kuali.rice.kns.util.GlobalVariables;
031    
032    /**
033     * Batch input type for the procurement card job.
034     */
035    public class ProcurementCardInputFileType extends XmlBatchInputFileTypeBase {
036        private static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(ProcurementCardInputFileType.class);
037    
038        private DateTimeService dateTimeService;
039    
040        /**
041         * @see org.kuali.kfs.sys.batch.BatchInputFileType#getFileTypeIdentifer()
042         */
043        public String getFileTypeIdentifer() {
044            return KFSConstants.PCDO_FILE_TYPE_INDENTIFIER;
045        }
046    
047        /**
048         * No additional information is added to procurment card batch files.
049         * 
050         * @see org.kuali.kfs.sys.batch.BatchInputFileType#getFileName(org.kuali.rice.kim.bo.Person, java.lang.Object,
051         *      java.lang.String)
052         */
053        public String getFileName(String principalName, Object parsedFileContents, String userIdentifier) {
054            String fileName = "pcdo_" + principalName;
055            if (StringUtils.isNotBlank(userIdentifier)) {
056                fileName += "_" + userIdentifier;
057            }
058            fileName += "_" + dateTimeService.toDateTimeStringForFilename(dateTimeService.getCurrentDate());
059    
060            // remove spaces in filename
061            fileName = StringUtils.remove(fileName, " ");
062    
063            return fileName;
064        }
065        
066        public String getAuthorPrincipalName(File file) {
067            String[] fileNameParts = StringUtils.split(file.getName(), "_");
068            if (fileNameParts.length >= 2) {
069                return fileNameParts[1];
070            }
071            return null;
072        }
073    
074        /**
075         * @see org.kuali.kfs.sys.batch.BatchInputFileType#validate(java.lang.Object)
076         */
077        public boolean validate(Object parsedFileContents) {
078            List<ProcurementCardTransaction> pctrans = (List<ProcurementCardTransaction>)parsedFileContents;
079            boolean valid = true;
080            
081            // add validation for chartCode-accountNumber, as chartCode is not required in xsd due to accounts-cant-cross-charts option
082            AccountService acctserv = SpringContext.getBean(AccountService.class);
083            for (ProcurementCardTransaction pctran : pctrans) {
084                // if chart code is empty while accounts cannot cross charts, then derive chart code from account number
085                if (StringUtils.isEmpty(pctran.getChartOfAccountsCode())) {
086                    if (acctserv.accountsCanCrossCharts()) {
087                        GlobalVariables.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, KFSKeyConstants.ERROR_BATCH_UPLOAD_FILE_EMPTY_CHART, pctran.getAccountNumber());
088                        valid = false;
089                    }
090                    else {
091                        // accountNumber shall not be empty, otherwise won't pass schema validation
092                        Account account = acctserv.getUniqueAccountForAccountNumber(pctran.getAccountNumber());
093                        if (account != null) {
094                            pctran.setChartOfAccountsCode(account.getChartOfAccountsCode());
095                        }       
096                        else {
097                            GlobalVariables.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, KFSKeyConstants.ERROR_BATCH_UPLOAD_FILE_INVALID_ACCOUNT, pctran.getAccountNumber());
098                            valid = false;
099                        }
100                    }
101                }
102            }
103    
104            return valid;
105        }
106    
107        /**
108         * @see org.kuali.kfs.sys.batch.BatchInputFileType#getTitleKey()
109         */
110        public String getTitleKey() {
111            return KFSKeyConstants.MESSAGE_BATCH_UPLOAD_TITLE_PCDO;
112        }
113    
114        /**
115         * Gets the dateTimeService attribute.
116         */
117        public DateTimeService getDateTimeService() {
118            return dateTimeService;
119        }
120    
121        /**
122         * Sets the dateTimeService attribute value.
123         */
124        public void setDateTimeService(DateTimeService dateTimeService) {
125            this.dateTimeService = dateTimeService;
126        }
127    
128    }
