016    package org.kuali.kfs.vnd.businessobject.inquiry;
017    
018    import org.kuali.kfs.sys.KFSConstants;
019    import org.kuali.kfs.sys.businessobject.inquiry.KfsInquirableImpl;
020    import org.kuali.kfs.vnd.businessobject.VendorDetail;
021    import org.kuali.rice.kns.bo.BusinessObject;
022    import org.kuali.rice.kns.lookup.HtmlData;
023    import org.kuali.rice.kns.lookup.HtmlData.AnchorHtmlData;
024    import org.kuali.rice.kns.util.KNSConstants;
025    import org.kuali.rice.kns.util.ObjectUtils;
026    
027    /**
028     * This class adds in some new sections for {@link Org} inquiries, specifically Org Hierarchy Org Review Hierarchy
029     */
030    public class VendorInquirable extends KfsInquirableImpl {
031        
032        /**
033         * Overrides the helper method to build an inquiry url for a result field. 
034         * For the Vendor URL field, returns the url address as the inquiry URL, so that Vendor URL functions as a hyperlink. 
035         * 
036         * @param bo the business object instance to build the urls for
037         * @param propertyName the property which links to an inquirable
038         * @return String url to inquiry
039         */
040        public HtmlData getInquiryUrl(BusinessObject businessObject, String attributeName, boolean forceInquiry) {
041            if (businessObject instanceof VendorDetail && attributeName.equalsIgnoreCase("vendorUrlAddress")) {
042                Object objFieldValue = ObjectUtils.getPropertyValue(businessObject, attributeName);
043                String fieldValue = objFieldValue == null ? KFSConstants.EMPTY_STRING : objFieldValue.toString();
044                return new AnchorHtmlData("http://" + fieldValue, KNSConstants.EMPTY_STRING, ((VendorDetail)businessObject).getVendorName()+" ("+fieldValue+")");
045            }
046    
047            return super.getInquiryUrl(businessObject, attributeName, forceInquiry);
048        }
049    
050    }
