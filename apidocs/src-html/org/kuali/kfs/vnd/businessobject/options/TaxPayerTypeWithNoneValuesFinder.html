016    package org.kuali.kfs.vnd.businessobject.options;
017    
018    import java.util.ArrayList;
019    import java.util.List;
020    
021    import org.kuali.kfs.sys.KFSConstants;
022    import org.kuali.kfs.vnd.VendorConstants;
023    import org.kuali.rice.kns.lookup.keyvalues.KeyValuesBase;
024    import org.kuali.rice.core.util.KeyLabelPair;
025    
026    /**
027     * Generator of a list of tax payer type value pairs, including an empty one, as opposed to <code>TaxPayerTypeValuesFinder</code>,
028     * which does not include an empty pair.
029     * 
030     * @see org.kuali.kfs.fp.businessobject.options.TaxPayerTypeValuesFinder
031     */
032    public class TaxPayerTypeWithNoneValuesFinder extends KeyValuesBase {
033    
034        /*
035         * @see org.kuali.keyvalues.KeyValuesFinder#getKeyValues()
036         */
037        public List getKeyValues() {
038            List keyValues = new ArrayList();
039            keyValues.add(new KeyLabelPair(VendorConstants.TAX_TYPE_FEIN, VendorConstants.TAX_TYPE_FEIN));
040            keyValues.add(new KeyLabelPair(VendorConstants.TAX_TYPE_SSN, VendorConstants.TAX_TYPE_SSN));
041            keyValues.add(new KeyLabelPair(KFSConstants.EMPTY_STRING, VendorConstants.NONE));
042            return keyValues;
043        }
044    
045    }
