032    
033    public class TaxRegionTypeLookupableServiceImpl extends KualiLookupableHelperServiceImpl {
034    
035        /**
036         * Overriding the return URL to create a new Tax Region document.  If CREATE_TAX_REGION_FROM_LOOKUP_INDICATOR
037         * exists, that means that the lookup is coming from the "create new" button on the tax region lookup.
038         * If so, create customer link to tax region maint doc. Otherwise, just use the normal returnUrl.
039         * 
040         * @see org.kuali.rice.kns.lookup.AbstractLookupableHelperServiceImpl#getReturnUrl(org.kuali.rice.kns.bo.BusinessObject,
041         *      java.util.Map, java.lang.String, java.util.List)
042         */
043        @SuppressWarnings("rawtypes")
044        // KFSMI-5158
045        @Override
046        public HtmlData getReturnUrl(BusinessObject businessObject, LookupForm lookupForm, List returnKeys, BusinessObjectRestrictions restrictions) {
047    
048            final String docTypeName = getTaxRegionDocumentTypeName();
049            if (docTypeName.equals(lookupForm.getFormKey())) { 
050                Properties parameters = getParameters(businessObject, lookupForm.getFieldConversions(), lookupForm.getLookupableImplServiceName(), returnKeys);
051                parameters.put(KFSConstants.DISPATCH_REQUEST_PARAMETER, KFSConstants.MAINTENANCE_NEWWITHEXISTING_ACTION);
052                parameters.put(KFSConstants.BUSINESS_OBJECT_CLASS_ATTRIBUTE, TaxRegion.class.getName());
053                parameters.put(KFSConstants.OVERRIDE_KEYS, KFSConstants.TaxRegionConstants.TAX_REGION_TYPE_CODE);
054                parameters.put(KFSConstants.REFRESH_CALLER, KFSConstants.TaxRegionConstants.TAX_REGION_TYPE_CODE + "::" + ((TaxRegionType) businessObject).getTaxRegionTypeCode());
055                
056                final String href = UrlFactory.parameterizeUrl(KFSConstants.MAINTENANCE_ACTION, parameters);
057                final String returnUrlAnchorLabel =
058                    KNSServiceLocator.getKualiConfigurationService().getPropertyString(TITLE_RETURN_URL_PREPENDTEXT_PROPERTY);
059                AnchorHtmlData anchor = new AnchorHtmlData(href, HtmlData.getTitleText(returnUrlAnchorLabel, businessObject, returnKeys, restrictions));
060                anchor.setDisplayText(returnUrlAnchorLabel);
061               return anchor;
062           }
063           else {
064               return super.getReturnUrl(businessObject, lookupForm, returnKeys, restrictions);
065           }
066        }
067        
068        @SuppressWarnings("rawtypes")
069        @Override
070        public List<HtmlData> getCustomActionUrls(BusinessObject businessObject, List pkNames) {
071            List<HtmlData> htmlDataList = super.getCustomActionUrls(businessObject, pkNames);
072            
073            Properties parameters = new Properties();
074            parameters.put(KFSConstants.DISPATCH_REQUEST_PARAMETER, KFSConstants.MAINTENANCE_NEWWITHEXISTING_ACTION);
075            parameters.put(KFSConstants.BUSINESS_OBJECT_CLASS_ATTRIBUTE, TaxRegion.class.getName());
076            parameters.put(KFSConstants.OVERRIDE_KEYS, KFSConstants.TaxRegionConstants.TAX_REGION_TYPE_CODE);
077            parameters.put(KFSConstants.REFRESH_CALLER, KFSConstants.TaxRegionConstants.TAX_REGION_TYPE_CODE + "::" + ((TaxRegionType) businessObject).getTaxRegionTypeCode());
078            parameters.put(KFSConstants.TaxRegionConstants.TAX_REGION_TYPE_CODE, ((TaxRegionType) businessObject).getTaxRegionTypeCode());
079            String href = UrlFactory.parameterizeUrl(KFSConstants.MAINTENANCE_ACTION, parameters);
080            
081            AnchorHtmlData anchorHtmlData = new AnchorHtmlData(href, "start", "create tax region");
082            htmlDataList.add(anchorHtmlData);
083            
084            return htmlDataList;
085        }
086        
087        private String getTaxRegionDocumentTypeName() {
088            return getDataDictionaryService().getDataDictionary().getMaintenanceDocumentEntryForBusinessObjectClass(TaxRegion.class).getDocumentTypeName();
089        } 
090    }
