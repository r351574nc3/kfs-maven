016    package org.kuali.kfs.sys.batch;
017    
018    import java.io.File;
019    import java.util.Calendar;
020    
021    import org.apache.commons.io.filefilter.IOFileFilter;
022    import org.kuali.kfs.sys.batch.service.FilePurgeService;
023    import org.kuali.kfs.sys.context.SpringContext;
024    import org.kuali.rice.kns.service.DateTimeService;
025    
026    /**
027     * A file filter which only accepts any directory or files older than a set age
028     */
029    public class MaxAgePurgeFileFilter implements IOFileFilter {
030        private long minTimestamp;
031        
032        /**
033         * Constructs a MaxAgePurgeFileFilter, using the default time to filter older files from
034         */
035        public MaxAgePurgeFileFilter() {
036            final int daysPastToPurge = SpringContext.getBean(FilePurgeService.class).getStandardDaysBeforePurge();
037            this.minTimestamp = calculateMinTimestamp(daysPastToPurge);
038        }
039        
040        /**
041         * Constructs a MaxAgePurgeFileFilter, using the time set with the parameter associated with the given file purge custom age to get rid of files from
042         * @param filePurgeCustomAge the custom age to get a max file age from
043         */
044        public MaxAgePurgeFileFilter(FilePurgeCustomAge filePurgeCustomAge) {
045            final int daysPastToPurge = SpringContext.getBean(FilePurgeService.class).getDaysBeforePurgeForCustomAge(filePurgeCustomAge);
046            this.minTimestamp = calculateMinTimestamp(daysPastToPurge);
047        }
048        
049        /**
050         * Determines the timestamp any older files should be purged on
051         * @param daysPastToPurge the number of days a file should exist before purging
052         * @return the timestamp of that number of days earlier than "now" (as defined by DateTimeService) that is
053         */
054        private long calculateMinTimestamp(int daysPastToPurge) {
055            Calendar purgeFilesBeforeDate = SpringContext.getBean(DateTimeService.class).getCurrentCalendar();
056            purgeFilesBeforeDate.add(Calendar.DATE, -1*daysPastToPurge);
057            return purgeFilesBeforeDate.getTimeInMillis();
058        }
059    
060        /**
061         * Accepts any directory and any file older than our min timestamp
062         * @see org.apache.commons.io.filefilter.IOFileFilter#accept(java.io.File)
063         */
064        public boolean accept(File file) {
065            if (file.isDirectory()) return true;
066            return (file.lastModified() < this.minTimestamp);
067        }
068    
069        /**
070         * Accepts any directory, and any file which is older than our min timestamp
071         * @see org.apache.commons.io.filefilter.IOFileFilter#accept(java.io.File, java.lang.String)
072         */
073        public boolean accept(File directory, String fileName) {
074            final File file = new File(directory.getName()+File.separator+fileName);
075            if (file.isDirectory()) return true;
076            return (file.lastModified() < this.minTimestamp);
077        }
078    
079    }
