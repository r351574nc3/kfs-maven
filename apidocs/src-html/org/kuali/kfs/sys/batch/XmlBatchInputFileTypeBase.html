016    package org.kuali.kfs.sys.batch;
017    
018    import java.io.ByteArrayInputStream;
019    import java.io.IOException;
020    import java.io.InputStream;
021    import java.net.MalformedURLException;
022    import java.net.URL;
023    
024    import javax.xml.XMLConstants;
025    import javax.xml.transform.Source;
026    import javax.xml.transform.stream.StreamSource;
027    import javax.xml.validation.Schema;
028    import javax.xml.validation.SchemaFactory;
029    import javax.xml.validation.Validator;
030    
031    import org.apache.commons.digester.Digester;
032    import org.apache.commons.digester.Rules;
033    import org.apache.commons.digester.xmlrules.DigesterLoader;
034    import org.kuali.kfs.sys.exception.ParseException;
035    import org.kuali.kfs.sys.exception.XmlErrorHandler;
036    import org.springframework.core.io.UrlResource;
037    import org.xml.sax.SAXException;
038    
039    
040    
041    /**
042     * Base class for BatchInputFileType implementations that validate using an XSD schema and parse using a digester file
043     */
044    public abstract class XmlBatchInputFileTypeBase extends BatchInputFileTypeBase {
045        private static final org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(XmlBatchInputFileTypeBase.class);
046    
047        protected String digestorRulesFileName;
048        protected String schemaLocation;
049    
050        /**
051         * Constructs a BatchInputFileTypeBase.java.
052         */
053        public XmlBatchInputFileTypeBase() {
054            super();
055        }
056    
057        /**
058         * Gets the digestorRulesFileName attribute.
059         */
060        public String getDigestorRulesFileName() {
061            return digestorRulesFileName;
062        }
063    
064        /**
065         * Sets the digestorRulesFileName attribute value.
066         */
067        public void setDigestorRulesFileName(String digestorRulesFileName) {
068            this.digestorRulesFileName = digestorRulesFileName;
069        }
070    
071        /**
072         * Gets the schemaLocation attribute.
073         */
074        public String getSchemaLocation() {
075            return schemaLocation;
076        }
077    
078        /**
079         * Sets the schemaLocation attribute value.
080         */
081        public void setSchemaLocation(String schemaLocation) {
082            this.schemaLocation = schemaLocation;
083        }
084    
085        /**
086         * @see org.kuali.kfs.sys.batch.BatchInputFileType#process(java.lang.String, java.lang.Object)
087         */
088        public void process(String fileName, Object parsedFileContents) {
089            // default impl does nothing
090        }
091    
092        /**
093         * @see org.kuali.kfs.sys.batch.BatchInputFileType#parse(byte[])
094         */
095        public Object parse(byte[] fileByteContent) throws ParseException {
096            if (fileByteContent == null) {
097                LOG.error("an invalid(null) argument was given");
098                throw new IllegalArgumentException("an invalid(null) argument was given");
099            }
100    
101            // handle zero byte contents, xml parsers don't deal with them well
102            if (fileByteContent.length == 0) {
103                LOG.error("an invalid argument was given, empty input stream");
104                throw new IllegalArgumentException("an invalid argument was given, empty input stream");
105            }
106    
107            // validate contents against schema
108            ByteArrayInputStream validateFileContents = new ByteArrayInputStream(fileByteContent);
109            validateContentsAgainstSchema(getSchemaLocation(), validateFileContents);
110    
111            // setup digester for parsing the xml file
112            Digester digester = buildDigester(getSchemaLocation(), getDigestorRulesFileName());
113    
114            Object parsedContents = null;
115            try {
116                ByteArrayInputStream parseFileContents = new ByteArrayInputStream(fileByteContent);
117                parsedContents = digester.parse(parseFileContents);
118            }
119            catch (Exception e) {
120                LOG.error("Error parsing xml contents", e);
121                throw new ParseException("Error parsing xml contents: " + e.getMessage(), e);
122            }
123    
124            return parsedContents;    
125        }
126        
127        /**
128         * Validates the xml contents against the batch input type schema using the java 1.5 validation package.
129         * 
130         * @param schemaLocation - location of the schema file
131         * @param fileContents - xml contents to validate against the schema
132         */
133        protected void validateContentsAgainstSchema(String schemaLocation, InputStream fileContents) throws ParseException {
134            // create a SchemaFactory capable of understanding WXS schemas
135            SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
136    
137            // get schemaFile
138            UrlResource schemaResource = null;
139            try {
140                schemaResource = new UrlResource(schemaLocation);
141            }
142            catch (MalformedURLException e2) {
143                LOG.error("error getting schema url: " + e2.getMessage());
144                throw new RuntimeException("error getting schema url:  " + e2.getMessage(), e2);
145            }
146    
147            // load a WXS schema, represented by a Schema instance
148            Source schemaSource = null;
149            try {
150                schemaSource = new StreamSource(schemaResource.getInputStream());
151            }
152            catch (IOException e2) {
153                LOG.error("error getting schema stream from url: " + e2.getMessage());
154                throw new RuntimeException("error getting schema stream from url:   " + e2.getMessage(), e2);
155            }
156            
157            Schema schema = null;
158            try {
159                schema = factory.newSchema(schemaSource);
160            }
161            catch (SAXException e) {
162                LOG.error("error occured while setting schema file: " + e.getMessage());
163                throw new RuntimeException("error occured while setting schema file: " + e.getMessage(), e);
164            }    
165    
166            // create a Validator instance, which can be used to validate an instance document
167            Validator validator = schema.newValidator();
168            validator.setErrorHandler(new XmlErrorHandler());
169    
170            // validate
171            try {
172                validator.validate(new StreamSource(fileContents));
173            }
174            catch (SAXException e) {
175                LOG.error("error encountered while parsing xml " + e.getMessage());
176                throw new ParseException("Schema validation error occured while processing file: " + e.getMessage(), e);
177            }
178            catch (IOException e1) {
179                LOG.error("error occured while validating file contents: " + e1.getMessage());
180                throw new RuntimeException("error occured while validating file contents: " + e1.getMessage(), e1);
181            }
182        }
183        
184    
185        /**
186         * @return fully-initialized Digester used to process entry XML files
187         */
188        protected Digester buildDigester(String schemaLocation, String digestorRulesFileName) {
189            Digester digester = new Digester();
190            digester.setNamespaceAware(false);
191            digester.setValidating(true);
192            digester.setErrorHandler(new XmlErrorHandler());
193            digester.setSchema(schemaLocation);
194    
195            Rules rules = loadRules(digestorRulesFileName);
196    
197            digester.setRules(rules);
198    
199            return digester;
200        }
201    
202        /**
203         * @return Rules loaded from the appropriate XML file
204         */
205        protected Rules loadRules(String digestorRulesFileName) {
206            // locate Digester rules
207            ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
208            URL rulesUrl = classLoader.getResource(digestorRulesFileName);
209            if (rulesUrl == null) {
210                throw new RuntimeException("unable to locate digester rules file " + digestorRulesFileName);
211            }
212    
213            // create and init digester
214            Digester digester = DigesterLoader.createDigester(rulesUrl);
215    
216            return digester.getRules();
217        }
218    }
