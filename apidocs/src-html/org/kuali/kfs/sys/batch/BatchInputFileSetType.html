016    package org.kuali.kfs.sys.batch;
017    
018    import java.io.File;
019    import java.util.Date;
020    import java.util.List;
021    import java.util.Map;
022    import java.util.Set;
023    
024    import org.kuali.rice.kim.bo.Person;
025    
026    /**
027     * Declares methods that must be implemented for batch input file set type classes, which provides functionaliy needed to manage
028     * files of a certain batch type.
029     */
030    public interface BatchInputFileSetType extends BatchInputType {
031        /**
032         * Returns all of the types supported by this file set type
033         * 
034         * @return a list of all file types supported. The values in the list do not have an externally usable meaning, and are meant to
035         *         be used to call other methods of this interface.
036         */
037        public List<String> getFileTypes();
038    
039        /**
040         * Returns the unique identifier (Spring bean id) for the batch input file set type.
041         */
042        public String getFileSetTypeIdentifer();
043    
044        /**
045         * Gives the name of the directory for which batch files of a given type are stored.
046         * 
047         * @param fileType the file type
048         */
049        public String getDirectoryPath(String fileType);
050    
051        /**
052         * Returns a map of file type to file type descriptions, which are intended to be human readable
053         * 
054         * @return the key is the file type, the value is a human-readable description
055         */
056        public Map<String, String> getFileTypeDescription();
057    
058        /**
059         * Constructs a file name for the file type, the file user identifier, and the user.
060         * 
061         * @param user - user who is uploading the file
062         * @param fileUserIdentifer - file identifier given by user through the batch upload UI
063         */
064        public String getFileName(String fileType, String principalName, String fileUserIdentifer, Date creationDate);
065    
066        /**
067         * Returns whether the file must be uploaded
068         * 
069         * @param fileType the type for the file
070         * @return whether it must be uploaded
071         */
072        public boolean isFileRequired(String fileType);
073    
074        /**
075         * Returns the directory name where done files are to be stored. The behavior of this method is defined if
076         * {@link #isSupportsDoneFileCreation()} returns false.
077         * 
078         * @return the done file directory name
079         */
080        public String getDoneFileDirectoryPath();
081    
082        /**
083         * Returns the file name of the done file for a fileset created by the user with the identifier. The behavior of this method is
084         * defined if {@link #isSupportsDoneFileCreation()} returns false.
085         * 
086         * @return the done file name
087         */
088        public String getDoneFileName(Person user, String fileUserIdentifer, Date creationDate);
089    
090        /**
091         * Returns the set of file user identifiers parsed from the provided list of files for the user.
092         * 
093         * @return a set of file user identifiers
094         */
095        public Set<String> extractFileUserIdentifiers(Person user, List<File> files);
096        
097        /**
098         * Runs validation upon uploaded files.  Note the files passed in the Map may be located in a temporary directory rather than the
099         * directory returned by {@link #getDirectoryPath(String)}
100         * 
101         * If validation fails, the implementation is responsible for adding error messages to the {@link KFSConstants#GLOBAL_ERRORS} property
102         * string in the ErrorMap
103         * 
104         * If validation requires opening up input streams/readers/etc. on the files, implementations of this method must
105         * close all input streams/readers on files contained within the map.  Failure to do so may cause the files to be undeletable.
106         * 
107         * @param typeToFiles a map consisting of file type Strings (see {@link #getFileTypes()}) to file mappings
108         * 
109         * @return true if validation succeeds, false otherwise
110         */
111        public boolean validate(Map<String, File> typeToFiles);
112        
113        
114        /**
115         * This method will invoke some processing on this file
116         * 
117         * There is no error handling provided by the framework.  All error reporting must be done by the implementation of this method
118         * 
119         * @param typeToFiles a map consisting of file type Strings (see {@link #getFileTypes()}) to file mappings
120         *        Note that the map may contain the file handle for {@link KFSConstants#DONE_FILE_TYPE} for the done file,
121         *        if it was created 
122         */
123        public void process(Map<String, File> typeToFiles);
124    }
