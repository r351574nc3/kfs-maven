016    package org.kuali.kfs.sys.document.web;
017    
018    import java.util.ArrayList;
019    import java.util.List;
020    
021    import javax.servlet.jsp.JspException;
022    import javax.servlet.jsp.tagext.BodyTagSupport;
023    
024    import org.kuali.kfs.sys.context.SpringContext;
025    import org.kuali.kfs.sys.document.AccountingDocument;
026    import org.kuali.kfs.sys.document.service.AccountingLineRenderingService;
027    import org.kuali.kfs.sys.document.web.renderers.AccountingLineTableFooterRenderer;
028    import org.kuali.kfs.sys.document.web.renderers.AccountingLineTableHeaderRenderer;
029    import org.kuali.kfs.sys.document.web.renderers.Renderer;
030    import org.kuali.kfs.sys.web.struts.KualiAccountingDocumentFormBase;
031    import org.kuali.rice.kns.web.struts.form.KualiDocumentFormBase;
032    
033    /**
034     * This tag...how to describe what it does?  It takes these groups of accounting lines and, you know,
035     * it, um...it renders them.
036     */
037    public class AccountingLinesTag extends BodyTagSupport {
038        private AccountingDocument document;
039        private KualiAccountingDocumentFormBase form;
040        private List<AccountingLineGroup> groupsToRender;
041    
042        /**
043         * Finds the KualiForm wherever it may be hiding
044         * @return the KualiForm
045         */
046        public KualiAccountingDocumentFormBase getForm() {
047            if (form == null) {
048                form = SpringContext.getBean(AccountingLineRenderingService.class).findForm(pageContext);
049            }
050            return form;
051        }
052        
053        /**
054         * Retrieves the document
055         * @return the document which is currently being operated on
056         */
057        public AccountingDocument getDocument() {
058            if (document == null) {
059                document = (AccountingDocument)getForm().getDocument();
060            }
061            return document;
062        }
063        
064        /**
065         * Adds a group to render for this go-round
066         * @param group the group to render
067         */
068        public void addGroupToRender(AccountingLineGroup group) {
069            if (groupsToRender == null) {
070                groupsToRender = new ArrayList<AccountingLineGroup>();
071            }
072            groupsToRender.add(group);
073        }
074        
075        /**
076         * @see javax.servlet.jsp.tagext.BodyTagSupport#doStartTag()
077         */
078        @Override
079        public int doStartTag() throws JspException {
080            return EVAL_BODY_BUFFERED;
081        }
082    
083        /**
084         * @see javax.servlet.jsp.tagext.BodyTagSupport#doEndTag()
085         */
086        @Override
087        public int doEndTag() throws JspException {
088            int cellCount = getEnforcedCellCount();
089            enforceCellCount(cellCount);
090            renderHeader(cellCount);
091            renderGroups();
092            renderFooter();
093            resetTag();
094            return EVAL_PAGE;
095        }
096        
097        /**
098         * Clears out all state variables in this tag
099         */
100        protected void resetTag() {
101            document = null;
102            groupsToRender = null;
103            form = null;
104        }
105        
106        /**
107         * Determines the cell count we're going to enforce, which is the maximum width in cell of any child group
108         * @return the cell count we'll enforce
109         */
110        protected int getEnforcedCellCount() {
111            int count = 0;
112            for (AccountingLineGroup group : groupsToRender) {
113                if (group.getWidthInCells() > count) {
114                    count = group.getWidthInCells();
115                }
116            }
117            return count;
118        }
119        
120        /**
121         * Oh yeah, we're enforcing!  Oh yes we are!  We're going through all the accounting line groups and we're saying to them,
122         * Hey!  Hey you group!  Are you going to render the number of cells that we've decided to enforce?  And if they're like: Oh, no,
123         * I don't have that many cells, then we'll reset that group to render that many of cells!
124         * @param cellCount the number of cells to uphold
125         */
126        protected void enforceCellCount(int cellCount) {
127            for (AccountingLineGroup group : groupsToRender) {
128                if (group.getWidthInCells() < cellCount) {
129                    group.setCellCount(cellCount);
130                }
131            }
132        }
133        
134        /**
135         * Renders the beginning of the accounting line table, including the tab-container div
136         * @param cellCount the number of cells this table is wide
137         * @throws JspException thrown if rendering fails
138         */
139        protected void renderHeader(int cellCount) throws JspException {
140            AccountingLineTableHeaderRenderer renderer = new AccountingLineTableHeaderRenderer();
141            renderer.setCellCount(cellCount);
142            KualiAccountingDocumentFormBase form = (KualiAccountingDocumentFormBase)getForm();
143            renderer.setHideDetails(form.getHideDetails());
144            renderer.setAccountingLineImportInstructionsUrl(form.getAccountingLineImportInstructionsUrl());
145            renderer.render(pageContext, this);
146        }
147        
148        /**
149         * Renders all the accounting line groups within this accounting line tag
150         * @throws JspException thrown if something goes wrong
151         */
152        protected void renderGroups() throws JspException {
153            final KualiDocumentFormBase form = getForm();
154            final int startingTabIndex = form.getCurrentTabIndex();
155            final int arbitrarilyHighIndex = form.getNextArbitrarilyHighIndex();
156            for (AccountingLineGroup group : groupsToRender) {
157                group.setArbitrarilyHighIndex(arbitrarilyHighIndex);
158                group.renderEverything(pageContext, this);
159            }
160        }
161        
162        /**
163         * Renders the footer for the accounting line table
164         * @throws JspException thrown if something mysterious failed
165         */
166        protected void renderFooter() throws JspException {
167            Renderer renderer = new AccountingLineTableFooterRenderer();
168            renderer.render(pageContext, this);
169        }
170    
171        /**
172         * Removes the document from the tag
173         * @see javax.servlet.jsp.tagext.BodyTagSupport#release()
174         */
175        @Override
176        public void release() {
177            super.release();
178            resetTag();
179        }
180    }
