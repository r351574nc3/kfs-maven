128            if (StringUtils.isBlank(collectionItemPropertyName)) {
129                collectionItemPropertyName = generateItemPropertyFromCollectionNameTheDumbWay(collectionPropertyName);
130            }
131            this.collectionItemPropertyName = collectionItemPropertyName;

134        /**
135         * Automagically generates teh name of the collection item property from the collection property by taking any extra "s" off the end.  Note:
136         * I never claimed this method was smart.  You'd be surprised at how often it'll work, though.
137         * @param collectionName the collection name to generate the collection item name from
138         * @return the collection item property name
139         */
140        protected String generateItemPropertyFromCollectionNameTheDumbWay(String collectionName) {
141            int subStringEnd = collectionName.length();
142            if (collectionName.endsWith("s")) {
143                subStringEnd -= 1;
144            }
145            return collectionName.substring(0, subStringEnd);
146        }

148        /**
149         * @see javax.servlet.jsp.tagext.TagSupport#doStartTag()
150         */
151        @Override
152        public int doStartTag() throws JspException {
153            super.doStartTag();
154            final List<RenderableAccountingLineContainer> containers = generateContainersForAllLines();
155            group = groupDefinition.createAccountingLineGroup(getDocument(), containers, collectionPropertyName, collectionItemPropertyName, getForm().getDisplayedErrors(), getForm().getDisplayedWarnings(), getForm().getDisplayedInfo(), getForm().getDocumentActions().containsKey(KNSConstants.KUALI_ACTION_CAN_EDIT));
156            group.updateDeletabilityOfAllLines();
157            if (getParent() instanceof AccountingLinesTag) {
158                ((AccountingLinesTag)getParent()).addGroupToRender(group);
159                resetTag();
160            }
161            return Tag.SKIP_BODY;
162        }
163        
164        /**
165         * If our parent isn't AccountingLinesTag, then we should render all the group
166         * @see javax.servlet.jsp.tagext.TagSupport#doEndTag()
167         */
168        @Override
169        public int doEndTag() throws JspException {
170            super.doEndTag();
171            if (!(getParent() instanceof AccountingLinesTag)) {
172                group.renderEverything(pageContext, getParent());
173                resetTag();
174            }
175            return Tag.EVAL_PAGE;
176        }

194        /**
195         * Gets the correct accounting line group definition for this 
196         * @return the accounting line group definition to be used by this tag
197         */
198        protected AccountingLineGroupDefinition getGroupDefinition() {
199            if (groupDefinition == null) {
200                final String documentTypeClassName = getDocument().getClass().getName();
201                final FinancialSystemTransactionalDocumentEntry documentEntry = (FinancialSystemTransactionalDocumentEntry)SpringContext.getBean(DataDictionaryService.class).getDataDictionary().getDictionaryObjectEntry(documentTypeClassName);
202                final Map<String, AccountingLineGroupDefinition> groupDefinitions = documentEntry.getAccountingLineGroups();
203                groupDefinition = groupDefinitions.get(attributeGroupName);
204            }
205            return groupDefinition;
206        }

208        /**
209         * Creates a List of renderable elements, with all rendering transformations accomplished, to be rendered 
210         * @param groupDefinition the data dictionary definition of the accounting line group this line is in
211         * @param accountingLine the accounting line itself
212         * @param newLine true if the line is new and not added yet to the document, false otherwise
213         * @return a List of AccountingLineTableRows to render
214         */
215        protected List<AccountingLineTableRow> getRenderableElementsForLine(AccountingLineGroupDefinition groupDefinition, AccountingLine accountingLine, boolean newLine, boolean topLine, String accountingLinePropertyName) {
216            List<TableJoining> layoutElements = groupDefinition.getAccountingLineView().getAccountingLineLayoutElements(accountingLine.getClass());
217            AccountingLineRenderingService renderingService = SpringContext.getBean(AccountingLineRenderingService.class);
218            renderingService.performPreTablificationTransformations(layoutElements, groupDefinition, getDocument(), accountingLine, newLine, getForm().getUnconvertedValues(), accountingLinePropertyName);
219            List<AccountingLineTableRow> renderableElements = renderingService.tablify(layoutElements);
220            removeTopRowIfNecessary(groupDefinition, topLine, renderableElements);
221            renderingService.performPostTablificationTransformations(renderableElements, groupDefinition, getDocument(), accountingLine, newLine);
222            return renderableElements;
223        }
224        
225        /**
226         * If it is determined that removing the first row (presumably a header row) is necessary, removes it 
227         * @param rows the rows to possibly remove the top row from
228         */
229        protected void removeTopRowIfNecessary(AccountingLineGroupDefinition definition, boolean topLine, List<AccountingLineTableRow> rows) {
230            if (definition.isTopHeadersAfterFirstLineHiding() && !topLine) {
231                safelyRemoveTopRow(rows);
232            }
233        }
234        
235        /**
236         * Looks through the top row.  If there's any fields in the top row, it quietly fails; it also won't remove the top row if that's the only row there is
237         * @param rows the List of rows to remove the top one of if possible
238         */
239        protected void safelyRemoveTopRow(List<AccountingLineTableRow> rows) {
240            if (rows != null && rows.size() > 1 && rows.get(0).safeToRemove()) {
241                rows.remove(0);
242            }
243        }

259        /**
260         * @return a List of accounting line table rows to be rendered for all the accounting lines available for rendering within the group
261         */
262        protected List<RenderableAccountingLineContainer> generateContainersForAllLines() {
263            List<RenderableAccountingLineContainer> containers = new ArrayList<RenderableAccountingLineContainer>();
264            
265            final AccountingLineGroupDefinition groupDefinition = getGroupDefinition();
266            final AccountingDocument document = getDocument();
267            final Person currentUser = GlobalVariables.getUserSession().getPerson();
268            boolean addedTopLine = false;
269            
270            // add all existing lines
271            int count = 0;
272            boolean anyEditableLines = false;
273            List<AccountingLine> lines = getAccountingLineCollection();
274            Collections.sort(lines, getGroupDefinition().getAccountingLineComparator());
275            for (AccountingLine accountingLine : lines) {
276                final RenderableAccountingLineContainer container = buildContainerForLine(groupDefinition, document, accountingLine, currentUser, new Integer(count), (addedTopLine ? false : true));
277                containers.add(container);
278                anyEditableLines = anyEditableLines || container.isEditableLine() || isErrorMapContainingErrorsOnLine(container.getAccountingLinePropertyPath());
279                count += 1;
280                addedTopLine = true;
281            }
282            // add the new line
283            if (StringUtils.isNotBlank(newLinePropertyName) && ((getForm().getDocumentActions().containsKey(KNSConstants.KUALI_ACTION_CAN_EDIT) && groupDefinition.getAccountingLineAuthorizer().renderNewLine(document, collectionPropertyName)) || anyEditableLines)) {
284                containers.add(0, buildContainerForLine(groupDefinition, document, getNewAccountingLine(), currentUser, null, true));
285            }
286            
287            return containers;
288        }
289        
290        /**
291         * Determines if the error map contains any errors which exist on the currently rendered accounting line
292         * @param accountingLinePropertyName the property name of the accounting line
293         * @return true if there are errors on the line, false otherwise
294         */
295        protected boolean isErrorMapContainingErrorsOnLine(String accountingLinePropertyName) {
296            for (Object errorKeyAsObject : GlobalVariables.getMessageMap().keySet()) {
297                if (((String)errorKeyAsObject).startsWith(accountingLinePropertyName)) return true;
298            }
299            return false;
300        }

302        /**
303         * Builds an accounting line container for a given accounting line to render
304         * @param groupDefinition the data dictionary definition of the group this line is a part of
305         * @param accountingDocument the accounting document the line is or will someday live on
306         * @param accountingLine the accounting line to render
307         * @param currentUser the currently logged in user
308         * @param count the count of this line within the collection represented by the group; null if this is a new line for the group
309         * @return the container created
310         */
311        protected RenderableAccountingLineContainer buildContainerForLine(AccountingLineGroupDefinition groupDefinition, AccountingDocument accountingDocument, AccountingLine accountingLine, Person currentUser, Integer count, boolean topLine) {
312            final String accountingLinePropertyName = count == null ? newLinePropertyName : collectionItemPropertyName+"["+count.toString()+"]";
313            final boolean newLine = (count == null);
314            final List<AccountingLineTableRow> rows = getRenderableElementsForLine(groupDefinition, accountingLine, newLine, topLine, accountingLinePropertyName);
315    
316            final boolean pageIsEditable = getForm().getDocumentActions().containsKey(KNSConstants.KUALI_ACTION_CAN_EDIT);
317            return new RenderableAccountingLineContainer(getForm(), accountingLine, accountingLinePropertyName, rows, count, groupDefinition.getGroupLabel(), getErrors(), groupDefinition.getAccountingLineAuthorizer(), groupDefinition.getAccountingLineAuthorizer().hasEditPermissionOnAccountingLine(getDocument(), accountingLine, collectionPropertyName, currentUser, pageIsEditable));
318        }
