016    package org.kuali.kfs.sys.service.impl;
017    
018    import java.io.BufferedReader;
019    import java.io.BufferedWriter;
020    import java.io.File;
021    import java.io.FileReader;
022    import java.io.FileWriter;
023    import java.io.IOException;
024    import java.io.PrintWriter;
025    import java.io.Writer;
026    import java.util.Date;
027    import java.util.List;
028    
029    import org.apache.commons.lang.StringUtils;
030    import org.kuali.kfs.sys.KFSConstants;
031    import org.kuali.kfs.sys.context.SpringContext;
032    import org.kuali.kfs.sys.service.ReportAggregatorService;
033    import org.kuali.kfs.sys.service.ReportWriterService;
034    import org.kuali.rice.kns.service.DateTimeService;
035    
036    public class ReportAggregatorServiceTextImpl implements ReportAggregatorService {
037        protected String newLineCharacter;
038    
039        public void aggregateReports(File outputFile, List<File> files) {
040            PrintWriter aggregateReportWriter = null;
041            int pageNumber = 1;
042            try {
043                aggregateReportWriter = new PrintWriter(new FileWriter(outputFile));
044            }
045            catch (IOException e) {
046                throw new RuntimeException("Error opening output file.", e);
047            }
048            for (int i = 0; i < files.size(); i++) {
049                File file = files.get(i);
050                if (i != 0) {
051                    prepareForNewFile(aggregateReportWriter);
052                }
053                pageNumber = dumpFileContents(aggregateReportWriter, file, pageNumber);
054                file.delete();
055            }
056            aggregateReportWriter.close();
057        }
058    
059        protected int dumpFileContents(Writer outputWriter, File file, int currentPageNumber) {
060            try {
061                BufferedReader reader = new BufferedReader(new FileReader(file));
062                while (reader.ready()) {
063                    String line = reader.readLine();
064                    while (line.contains(KFSConstants.REPORT_WRITER_SERVICE_PAGE_NUMBER_PLACEHOLDER)) {
065                        line = StringUtils.replaceOnce(line, KFSConstants.REPORT_WRITER_SERVICE_PAGE_NUMBER_PLACEHOLDER, String.valueOf(currentPageNumber));
066                        currentPageNumber++;
067                    }
068                    outputWriter.write(line);
069                    outputWriter.write(newLineCharacter);
070                }
071                reader.close();
072                return currentPageNumber;
073            }
074            catch (IOException e) {
075                throw new RuntimeException("Error reading or writing file", e);
076            }
077        }
078        
079        protected void prepareForNewFile(PrintWriter outputWriter) {
080            // 12 represents the ASCII Form Feed character
081            outputWriter.printf("%c" + newLineCharacter, 12);
082        }
083    
084        /**
085         * Sets the newLineCharacter attribute value.
086         * @param newLineCharacter The newLineCharacter to set.
087         */
088        public void setNewLineCharacter(String newLineCharacter) {
089            this.newLineCharacter = newLineCharacter;
090        }
091    }
