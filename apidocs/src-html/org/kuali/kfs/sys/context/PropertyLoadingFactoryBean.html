016    package org.kuali.kfs.sys.context;
017    
018    import java.io.IOException;
019    import java.io.InputStream;
020    import java.util.ArrayList;
021    import java.util.Arrays;
022    import java.util.List;
023    import java.util.Properties;
024    
025    import org.apache.commons.lang.StringUtils;
026    import org.kuali.kfs.sys.KFSConstants;
027    import org.kuali.rice.core.config.JAXBConfigImpl;
028    import org.kuali.rice.core.util.ClassLoaderUtils;
029    import org.springframework.beans.factory.FactoryBean;
030    import org.springframework.core.io.DefaultResourceLoader;
031    
032    public class PropertyLoadingFactoryBean implements FactoryBean {
033        private static final String PROPERTY_FILE_NAMES_KEY = "property.files";
034        private static final String PROPERTY_TEST_FILE_NAMES_KEY = "property.test.files";
035        private static final String SECURITY_PROPERTY_FILE_NAME_KEY = "security.property.file";
036        private static final String CONFIGURATION_FILE_NAME = "configuration";
037        private static final Properties BASE_PROPERTIES = new Properties();
038        private static final String HTTP_URL_PROPERTY_NAME = "http.url";
039        private static final String KSB_REMOTING_URL_PROPERTY_NAME = "ksb.remoting.url";
040        private static final String REMOTING_URL_SUFFIX = "/remoting";
041        private Properties props = new Properties();
042        private boolean testMode;
043        private boolean secureMode;
044    
045        public Object getObject() throws Exception {
046            loadBaseProperties();
047            props.putAll(BASE_PROPERTIES);
048            if (secureMode) {
049                loadPropertyList(props,SECURITY_PROPERTY_FILE_NAME_KEY);
050            } else {
051                loadPropertyList(props,PROPERTY_FILE_NAMES_KEY);
052                if (testMode) {
053                    loadPropertyList(props,PROPERTY_TEST_FILE_NAMES_KEY);
054                }            
055            }
056            if (StringUtils.isBlank(System.getProperty(HTTP_URL_PROPERTY_NAME))) {
057                props.put(KSB_REMOTING_URL_PROPERTY_NAME, props.getProperty(KFSConstants.APPLICATION_URL_KEY) + REMOTING_URL_SUFFIX);
058            }
059            else {
060                props.put(KSB_REMOTING_URL_PROPERTY_NAME, new StringBuffer("http://").append(System.getProperty(HTTP_URL_PROPERTY_NAME)).append("/kfs-").append(props.getProperty(KFSConstants.ENVIRONMENT_KEY)).append(REMOTING_URL_SUFFIX).toString());
061            }
062            System.out.println(KSB_REMOTING_URL_PROPERTY_NAME + " set to " + props.getProperty(KSB_REMOTING_URL_PROPERTY_NAME));
063            return props;
064        }
065    
066        public Class getObjectType() {
067            return Properties.class;
068        }
069    
070        public boolean isSingleton() {
071            return true;
072        }
073    
074        private static void loadPropertyList(Properties props, String listPropertyName) {
075            for (String propertyFileName : getBaseListProperty(listPropertyName)) {
076                loadProperties(props,propertyFileName);
077            }
078        }
079    
080        private static void loadProperties( Properties props, String propertyFileName) {
081            InputStream propertyFileInputStream = null;
082            try {
083                try {
084                    propertyFileInputStream = new DefaultResourceLoader(ClassLoaderUtils.getDefaultClassLoader()).getResource(propertyFileName).getInputStream();
085                    props.load(propertyFileInputStream);
086                }
087                finally {
088                    if (propertyFileInputStream != null) {
089                        propertyFileInputStream.close();
090                    }
091                }
092            }
093            catch (IOException e) {
094                throw new RuntimeException("PropertyLoadingFactoryBean unable to load property file: " + propertyFileName);
095            }
096        }
097    
098        public static String getBaseProperty(String propertyName) {
099            loadBaseProperties();
100            return BASE_PROPERTIES.getProperty(propertyName);
101        }
102    
103        protected static List<String> getBaseListProperty(String propertyName) {
104            loadBaseProperties();
105            return Arrays.asList(BASE_PROPERTIES.getProperty(propertyName).split(","));
106        }
107    
108        protected static void loadBaseProperties() {
109            if (BASE_PROPERTIES.isEmpty()) {
110                List<String> riceXmlConfigurations = new ArrayList<String>();
111                riceXmlConfigurations.add("classpath:META-INF/common-config-defaults.xml");
112                JAXBConfigImpl riceXmlConfigurer = new JAXBConfigImpl(riceXmlConfigurations);
113                BASE_PROPERTIES.putAll(riceXmlConfigurer.getProperties());
114                
115                loadProperties(BASE_PROPERTIES, new StringBuffer("classpath:").append(CONFIGURATION_FILE_NAME).append(".properties").toString());
116            }
117        }
118    
119        public void setTestMode(boolean testMode) {
120            this.testMode = testMode;
121        }
122    
123        public void setSecureMode(boolean secureMode) {
124            this.secureMode = secureMode;
125        }
126        
127        public static void clear() {
128            BASE_PROPERTIES.clear();
129        }
130    }
