016    package org.kuali.kfs.sys.context;
017    
018    import javax.servlet.ServletContextEvent;
019    import javax.servlet.ServletContextListener;
020    
021    import org.apache.log4j.Logger;
022    import org.kuali.rice.kns.web.listener.JstlConstantsInitListener;
023    import org.kuali.rice.ksb.messaging.MessageFetcher;
024    import org.kuali.rice.ksb.service.KSBServiceLocator;
025    
026    public class WebApplicationInitListener extends JstlConstantsInitListener implements ServletContextListener {
027        private static final String JSTL_CONSTANTS_CLASSNAMES_KEY = "jstl.constants.classnames";
028        private static final String JSTL_CONSTANTS_MAIN_CLASS = "jstl.constants.main.class";
029        private static final String JSTL_MAIN_CLASS_CONTEXT_NAME = "Constants";
030        private static Logger LOG;
031    
032        @SuppressWarnings("unchecked")
033        public void contextInitialized(ServletContextEvent sce) {
034            Log4jConfigurer.configureLogging(true);
035            LOG = Logger.getLogger(WebApplicationInitListener.class);
036            SpringContext.initializeApplicationContext();
037            MessageFetcher messageFetcher = new MessageFetcher((Integer)null); 
038            KSBServiceLocator.getThreadPool().execute(messageFetcher); 
039            for (String jstlConstantsClassname : PropertyLoadingFactoryBean.getBaseListProperty(JSTL_CONSTANTS_CLASSNAMES_KEY)) {
040                try {
041                    Class jstlConstantsClass = Class.forName(jstlConstantsClassname);
042                    Object jstlConstantsObj = jstlConstantsClass.newInstance();
043                    sce.getServletContext().setAttribute(jstlConstantsClass.getSimpleName(), jstlConstantsObj);
044                    if (jstlConstantsClassname.equals(JSTL_CONSTANTS_MAIN_CLASS)) {
045                        sce.getServletContext().setAttribute(JSTL_MAIN_CLASS_CONTEXT_NAME, jstlConstantsObj);
046                    }
047                }
048                catch (Exception e) {
049                    LOG.warn("Unable to load jstl constants class: " + jstlConstantsClassname, e);
050                }
051            }
052            super.contextInitialized(sce);
053            LOG.info("Finished web application context initialization");
054        }
055    
056        public void contextDestroyed(ServletContextEvent sce) {
057            LOG.info("Started web application context destruction");
058            try {
059                SpringContext.close();
060            } catch ( Exception ex ) {
061                LOG.error( "Unable to close down Spring Context." );
062            }
063            super.contextDestroyed(sce);
064        }
065    }
