018    // This class is a comparator for sorting AccountType with Code name.
019    package org.kuali.kfs.coa.businessobject.options;
020    
021    import java.util.Comparator;
022    
023    import org.kuali.kfs.coa.businessobject.AccountType;
024    
025    /**
026     * This class is a comparator for Account Type Codes
027     */
028    public class AccountTypeCodeComparator implements Comparator {
029    
030        /**
031         * If these two account type codes are the same codes
032         * 
033         * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
034         */
035        public int compare(Object o1, Object o2) {
036            AccountType acctType1 = (AccountType) o1;
037            AccountType acctType2 = (AccountType) o2;
038    
039            int acctTypeComp = acctType1.getAccountTypeCode().compareTo(acctType2.getAccountTypeCode());
040    
041            return acctTypeComp;
042        }
043    
044    }
