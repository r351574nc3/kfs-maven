016    package org.kuali.kfs.coa.batch.dataaccess.impl;
017    
018    import org.apache.ojb.broker.query.Criteria;
019    import org.kuali.kfs.coa.businessobject.ObjectCode;
020    import org.kuali.kfs.sys.KFSConstants;
021    import org.kuali.kfs.sys.KFSPropertyConstants;
022    import org.kuali.kfs.sys.batch.dataaccess.impl.FiscalYearMakerImpl;
023    import org.kuali.rice.kns.bo.PersistableBusinessObject;
024    
025    /**
026     * Performs custom criteria of object code records for a new year being created in the fiscal year maker process
027     */
028    public class ObjectCodeFiscalYearMakerImpl extends FiscalYearMakerImpl {
029    
030        /**
031         * @see org.kuali.kfs.coa.batch.dataaccess.impl.FiscalYearMakerHelperImpl#createSelectionCriteria(java.lang.Integer)
032         */
033        @Override
034        public Criteria createSelectionCriteria(Integer baseFiscalYear) {
035            // get default criteria equal to base year and active
036            Criteria criteria = super.createSelectionCriteria(baseFiscalYear);
037    
038            // or with criteria to pick up budget dummy object for base year (active or inactive)
039            Criteria criteriaBdg = new Criteria();
040            criteriaBdg.addEqualTo(KFSPropertyConstants.UNIVERSITY_FISCAL_YEAR, baseFiscalYear);
041            criteriaBdg.addEqualTo(KFSPropertyConstants.FINANCIAL_OBJECT_CODE, KFSConstants.BudgetConstructionConstants.OBJECT_CODE_2PLG);
042            criteria.addOrCriteria(criteriaBdg);
043    
044            return criteria;
045        }
046    
047    }
