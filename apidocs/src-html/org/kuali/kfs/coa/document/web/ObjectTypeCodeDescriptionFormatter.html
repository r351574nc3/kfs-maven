016    package org.kuali.kfs.coa.document.web;
017    
018    import java.util.Collection;
019    import java.util.HashMap;
020    import java.util.Map;
021    import java.util.Set;
022    
023    import org.kuali.kfs.coa.businessobject.ObjectType;
024    import org.kuali.kfs.sys.KFSConstants;
025    import org.kuali.kfs.sys.context.SpringContext;
026    import org.kuali.kfs.sys.document.web.CodeDescriptionFormatterBase;
027    import org.kuali.rice.kns.bo.PersistableBusinessObject;
028    import org.kuali.rice.kns.service.BusinessObjectService;
029    
030    public class ObjectTypeCodeDescriptionFormatter extends CodeDescriptionFormatterBase {
031    
032        public ObjectTypeCodeDescriptionFormatter() {
033        }
034    
035        @Override
036        protected String getDescriptionOfBO(PersistableBusinessObject bo) {
037            return ((ObjectType) bo).getName();
038        }
039    
040        @Override
041        protected Map<String, PersistableBusinessObject> getValuesToBusinessObjectsMap(Set values) {
042            Map<String, Object> criteria = new HashMap<String, Object>();
043            criteria.put(KFSConstants.GENERIC_CODE_PROPERTY_NAME, values);
044    
045            Map<String, PersistableBusinessObject> results = new HashMap<String, PersistableBusinessObject>();
046            Collection<ObjectType> coll = SpringContext.getBean(BusinessObjectService.class).findMatchingOrderBy(ObjectType.class, criteria, "versionNumber", true);
047            // by sorting on ver #, we can guarantee that the most recent value will remain in the map (assuming the iterator returns
048            // BOs in order)
049            for (ObjectType ot : coll) {
050                results.put(ot.getCode(), ot);
051            }
052            return null;
053        }
054    
055    }
