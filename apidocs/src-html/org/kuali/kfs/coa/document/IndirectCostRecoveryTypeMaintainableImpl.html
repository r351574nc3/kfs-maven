037    public class IndirectCostRecoveryTypeMaintainableImpl extends FinancialSystemMaintainable {
038    
039        private static final org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(IndirectCostRecoveryTypeMaintainableImpl.class);
040        public static final String DOCUMENT_ERROR_PREFIX = "document.";
041        public static final String MAINTAINABLE_ERROR_PATH = DOCUMENT_ERROR_PREFIX + "newMaintainableObject";
042        public static final String DETAIL_ERROR_PATH = MAINTAINABLE_ERROR_PATH + ".add.indirectCostRecoveryExclusionTypeDetails";
043    
044        @Override
045        public void addMultipleValueLookupResults(MaintenanceDocument document, String collectionName, Collection<PersistableBusinessObject> rawValues, boolean needsBlank, PersistableBusinessObject bo) {
046            // PersistableBusinessObject bo = document.getNewMaintainableObject().getBusinessObject();
047            Collection maintCollection = (Collection) ObjectUtils.getPropertyValue(bo, collectionName);
048            String docTypeName = document.getDocumentHeader().getWorkflowDocument().getDocumentType();
049            
050            List<String> duplicateIdentifierFieldsFromDataDictionary = getDuplicateIdentifierFieldsFromDataDictionary(docTypeName, collectionName);
051            
052            List<String> existingIdentifierList = getMultiValueIdentifierList(maintCollection, duplicateIdentifierFieldsFromDataDictionary);
053            
054            Class collectionClass = getMaintenanceDocumentDictionaryService().getCollectionBusinessObjectClass(docTypeName, collectionName);
055    
056            List<MaintainableSectionDefinition> sections = getMaintenanceDocumentDictionaryService().getMaintainableSections(docTypeName);
057            Map<String, String> template = MaintenanceUtils.generateMultipleValueLookupBOTemplate(sections, collectionName);
058            try {
059                int collectionItemNumber = 0; // is there a better way to do this? ie- old school i=0;i<rawValues.size();i++ and rawValues.get(i)?
060                boolean isValid = true;
061                GlobalVariables.getMessageMap().addToErrorPath(DETAIL_ERROR_PATH);
062                for (PersistableBusinessObject nextBo : rawValues) {
063                    IndirectCostRecoveryExclusionType templatedBo = (IndirectCostRecoveryExclusionType) ObjectUtils.createHybridBusinessObject(collectionClass, nextBo, template);
064                    templatedBo.setNewCollectionRecord(true);
065                    prepareBusinessObjectForAdditionFromMultipleValueLookup(collectionName, templatedBo);
066                    if(!hasBusinessObjectExisted(templatedBo, existingIdentifierList, duplicateIdentifierFieldsFromDataDictionary)) {
067                        maintCollection.add(templatedBo);
068                    }
069                    collectionItemNumber++;
070                    templatedBo.setActive(true); // TODO remove after active indicator work is complete
071                }
072                GlobalVariables.getMessageMap().removeFromErrorPath(DETAIL_ERROR_PATH);
073                // putGlobalError(KFSKeyConstants.ERROR_DOCUMENT_ACCTDELEGATEMAINT_PRIMARY_ROUTE_ALREADY_EXISTS_FOR_DOCTYPE);
074            } 
075            catch (Exception e) {
076                LOG.error("Unable to add multiple value lookup results " + e.getMessage());
077                throw new RuntimeException("Unable to add multiple value lookup results " + e.getMessage());
078            }
079        }
080        
081    }
