016    package org.kuali.kfs.coa.document.validation.impl;
017    
018    import java.util.HashMap;
019    import java.util.Map;
020    
021    import org.kuali.kfs.coa.businessobject.BalanceType;
022    import org.kuali.kfs.coa.businessobject.OffsetDefinition;
023    import org.kuali.kfs.sys.KFSKeyConstants;
024    import org.kuali.kfs.sys.context.SpringContext;
025    import org.kuali.kfs.sys.service.UniversityDateService;
026    import org.kuali.rice.kns.bo.Inactivateable;
027    import org.kuali.rice.kns.document.Document;
028    import org.kuali.rice.kns.document.MaintenanceDocument;
029    import org.kuali.rice.kns.maintenance.rules.MaintenanceDocumentRuleBase;
030    import org.kuali.rice.kns.service.BusinessObjectService;
031    import org.kuali.rice.kns.util.GlobalVariables;
032    
033    public class BalanceTypeRule extends MaintenanceDocumentRuleBase {
034    
035        /**
036         * @see org.kuali.rice.kns.maintenance.rules.MaintenanceDocumentRuleBase#processRouteDocument(org.kuali.rice.kns.document.Document)
037         */
038        @Override
039        public boolean processRouteDocument(Document document) {
040            boolean result = super.processRouteDocument(document);
041            result &= checkForBlockingOffsetDefinitions((BalanceType)(((MaintenanceDocument)document).getNewMaintainableObject().getBusinessObject()));
042            return result;
043        }
044    
045        /**
046         * @see org.kuali.rice.kns.maintenance.rules.MaintenanceDocumentRuleBase#processSaveDocument(org.kuali.rice.kns.document.Document)
047         */
048        @Override
049        public boolean processSaveDocument(Document document) {
050            boolean result = super.processSaveDocument(document);
051            result &= checkForBlockingOffsetDefinitions((BalanceType)(((MaintenanceDocument)document).getNewMaintainableObject().getBusinessObject()));
052            return result;
053        }
054        
055        /**
056         * Determines if the given document is inactivating the balance type being maintained
057         * @param document the maintenance document maintaining a balance type
058         * @return true if the document is inactivating that balance type, false otherwise
059         */
060        protected boolean isInactivating(MaintenanceDocument document) {
061            if (!document.isEdit() || document.getOldMaintainableObject() == null) return false;
062            return ((Inactivateable)document.getOldMaintainableObject().getBusinessObject()).isActive() && !((Inactivateable)document.getNewMaintainableObject().getBusinessObject()).isActive();
063        }
064    
065        /**
066         * Determines if the inactivating balance type should be blocked by offset definitions
067         * @param balanceType the balance type to check
068         * @return true if the balance type shouldn't be blocked by any existing offset definitions, false otherwise
069         */
070        protected boolean checkForBlockingOffsetDefinitions(BalanceType balanceType) {
071            final BusinessObjectService businessObjectService = SpringContext.getBean(BusinessObjectService.class);
072            boolean result = true;
073            
074            Map<String, Object> keys = new HashMap<String, Object>();
075            keys.put("financialBalanceTypeCode", balanceType.getFinancialBalanceTypeCode());
076            keys.put("universityFiscalYear", SpringContext.getBean(UniversityDateService.class).getCurrentFiscalYear());
077            
078            final int matchingCount = businessObjectService.countMatching(OffsetDefinition.class, keys);
079            if (matchingCount > 0) {
080                GlobalVariables.getMessageMap().putErrorForSectionId("Edit Balance Type",KFSKeyConstants.ERROR_DOCUMENT_BALANCETYPMAINT_INACTIVATION_BLOCKING,new String[] {balanceType.getFinancialBalanceTypeCode(), Integer.toString(matchingCount), OffsetDefinition.class.getName()});
081                result = false;
082            }
083            return result;
084        }
085    }
