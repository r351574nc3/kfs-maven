016    package org.kuali.kfs.coa.document.validation.impl;
017    
018    import java.util.List;
019    
020    import org.apache.commons.lang.StringUtils;
021    import org.kuali.kfs.coa.businessobject.Account;
022    import org.kuali.kfs.coa.businessobject.Chart;
023    import org.kuali.kfs.coa.businessobject.IndirectCostRecoveryExclusionType;
024    import org.kuali.kfs.coa.businessobject.IndirectCostRecoveryType;
025    import org.kuali.kfs.coa.service.ChartService;
026    import org.kuali.kfs.sys.KFSKeyConstants;
027    import org.kuali.kfs.sys.KFSPropertyConstants;
028    import org.kuali.kfs.sys.context.SpringContext;
029    import org.kuali.rice.kns.bo.PersistableBusinessObject;
030    import org.kuali.rice.kns.document.MaintenanceDocument;
031    import org.kuali.rice.kns.maintenance.rules.MaintenanceDocumentRuleBase;
032    import org.kuali.rice.kns.util.GlobalVariables;
033    import org.kuali.rice.kns.util.ObjectUtils;
034    
035    public class IndirectCostRecoveryTypeRule extends MaintenanceDocumentRuleBase {
036        protected static org.apache.log4j.Logger LOG = org.apache.log4j.Logger.getLogger(Account.class);
037    
038        protected IndirectCostRecoveryType indirectCostRecoveryType;
039        protected List indirectCostRecoveryExclusionTypeDetails;
040        protected ChartService chartService;
041    
042        
043        public IndirectCostRecoveryTypeRule () {
044            chartService = SpringContext.getBean(ChartService.class);
045        }
046        
047        public void setupConvenienceObjects() {
048            indirectCostRecoveryType = (IndirectCostRecoveryType) super.getNewBo();
049            indirectCostRecoveryExclusionTypeDetails = indirectCostRecoveryType.getIndirectCostRecoveryExclusionTypeDetails();
050        }
051        
052        @Override
053        public boolean processCustomAddCollectionLineBusinessRules(MaintenanceDocument document, String collectionName, PersistableBusinessObject line) {
054            boolean isValid = true;
055            isValid &= itemIsValid((IndirectCostRecoveryExclusionType) line);
056            return isValid;
057        }
058        
059        @Override
060        protected boolean processCustomRouteDocumentBusinessRules(MaintenanceDocument document) {
061            boolean isValid = true;
062            GlobalVariables.getMessageMap().addToErrorPath(MAINTAINABLE_ERROR_PATH);
063            isValid &= checkCollectionItems();
064            GlobalVariables.getMessageMap().removeFromErrorPath(MAINTAINABLE_ERROR_PATH);
065            return isValid;
066        }
067        
068        public boolean checkCollectionItems() {
069            boolean isValid = true;
070            for(int i=0;i<indirectCostRecoveryExclusionTypeDetails.size();i++) {
071                String collectionElementPath = "indirectCostRecoveryExclusionTypeDetails[" + i + "]";
072                GlobalVariables.getMessageMap().addToErrorPath(collectionElementPath);
073                isValid &= itemIsValid((IndirectCostRecoveryExclusionType) indirectCostRecoveryExclusionTypeDetails.get(i));
074                GlobalVariables.getMessageMap().removeFromErrorPath(collectionElementPath);
075            }
076            return isValid;
077        }
078        
079        public boolean itemIsValid(IndirectCostRecoveryExclusionType item) {
080            boolean isValid = true;
081            if(ObjectUtils.isNotNull(chartService.getUniversityChart().getChartOfAccountsCode())) {
082                if(StringUtils.isBlank(item.getChartOfAccountsCode())) {
083                    GlobalVariables.getMessageMap().putError(KFSPropertyConstants.FINANCIAL_OBJECT_CODE, KFSKeyConstants.IndirectCostRecovery.ERROR_DOCUMENT_ICR_EXSISTENCE_CHART_CODE, item.getChartOfAccountsCode());
084                    isValid = false;
085                } else if (item.isActive()) {
086                    item.refreshReferenceObject(KFSPropertyConstants.OBJECT_CODE_CURRENT);
087                    if(ObjectUtils.isNull(item.getObjectCodeCurrent())) {
088                        if(item.isNewCollectionRecord()) {
089                            GlobalVariables.getMessageMap().putError(KFSPropertyConstants.FINANCIAL_OBJECT_CODE, KFSKeyConstants.IndirectCostRecovery.ERROR_DOCUMENT_ICR_EXSISTENCE_OBJECT_CODE_DELETE, item.getChartOfAccountsCode(), item.getFinancialObjectCode()); 
090                        } else {
091                            GlobalVariables.getMessageMap().putError(KFSPropertyConstants.FINANCIAL_OBJECT_CODE, KFSKeyConstants.IndirectCostRecovery.ERROR_DOCUMENT_ICR_EXSISTENCE_OBJECT_CODE, item.getChartOfAccountsCode(), item.getFinancialObjectCode()); 
092                        }
093                        isValid = false;
094                    }                
095                }
096                
097            } else {
098                LOG.info("The Chart Service found no Charts of University type.");
099            }
100            return isValid;
101        }
102        
103    }
