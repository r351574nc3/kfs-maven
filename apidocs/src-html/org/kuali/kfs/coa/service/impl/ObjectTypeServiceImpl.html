016    package org.kuali.kfs.coa.service.impl;
017    
018    import java.util.ArrayList;
019    import java.util.List;
020    
021    import org.kuali.kfs.coa.businessobject.ObjectType;
022    import org.kuali.kfs.coa.dataaccess.ObjectTypeDao;
023    import org.kuali.kfs.coa.service.ObjectTypeService;
024    import org.kuali.kfs.sys.businessobject.SystemOptions;
025    import org.kuali.kfs.sys.dataaccess.OptionsDao;
026    import org.kuali.kfs.sys.service.NonTransactional;
027    import org.kuali.kfs.sys.service.UniversityDateService;
028    import org.kuali.rice.kns.util.spring.CacheNoCopy;
029    
030    /**
031     * This service implementation is the default implementation of the ObjectType service that is delivered with Kuali.
032     */
033    
034    @NonTransactional
035    public class ObjectTypeServiceImpl implements ObjectTypeService {
036    
037        private UniversityDateService universityDateService;
038        private ObjectTypeDao objectTypeDao;
039        private OptionsDao optionsDao;
040    
041        /**
042         * @see org.kuali.kfs.coa.service.ObjectTypeService#getByPrimaryKey(java.lang.String)
043         */
044        public ObjectType getByPrimaryKey(String objectTypeCode) {
045            return objectTypeDao.getByPrimaryKey(objectTypeCode);
046        }
047    
048        /**
049         * @see org.kuali.kfs.coa.service.ObjectTypeService#getAssetObjectType(java.lang.Integer)
050         */
051        public String getAssetObjectType(Integer universityFiscalYear) {
052            return optionsDao.getByPrimaryId(universityFiscalYear).getFinancialObjectTypeAssetsCd();
053        }
054    
055        /**
056         * @see org.kuali.kfs.coa.service.ObjectTypeService#getBasicExpenseObjectTypes(java.lang.Integer)
057         */
058        public List<String> getBasicExpenseObjectTypes(Integer universityFiscalYear) {
059    
060            List<String> basicExpenseObjectTypes = new ArrayList<String>();
061            SystemOptions option = optionsDao.getByPrimaryId(universityFiscalYear);
062            basicExpenseObjectTypes.add(option.getFinObjTypeExpenditureexpCd());
063            basicExpenseObjectTypes.add(option.getFinObjTypeExpendNotExpCode());
064            basicExpenseObjectTypes.add(option.getFinObjTypeExpNotExpendCode());
065    
066            return basicExpenseObjectTypes;
067        }
068    
069        /**
070         * @see org.kuali.kfs.coa.service.ObjectTypeService#getExpenseObjectTypes(java.lang.Integer)
071         */
072        public List<String> getExpenseObjectTypes(Integer universityFiscalYear) {
073            List<String> expenseObjectTypes = new ArrayList<String>();
074            SystemOptions option = optionsDao.getByPrimaryId(universityFiscalYear);
075            expenseObjectTypes.add(option.getFinObjTypeExpenditureexpCd());
076            expenseObjectTypes.add(option.getFinObjTypeExpendNotExpCode());
077            expenseObjectTypes.add(option.getFinObjTypeExpNotExpendCode());
078            expenseObjectTypes.add(option.getFinancialObjectTypeTransferExpenseCd());
079    
080            return expenseObjectTypes;
081        }
082    
083        /**
084         * @see org.kuali.kfs.coa.service.ObjectTypeService#getBasicIncomeObjectTypes(java.lang.Integer)
085         */
086        public List<String> getBasicIncomeObjectTypes(Integer universityFiscalYear) {
087    
088            List<String> basicIncomeObjectTypes = new ArrayList<String>();
089            SystemOptions option = optionsDao.getByPrimaryId(universityFiscalYear);
090            basicIncomeObjectTypes.add(option.getFinObjectTypeIncomecashCode());
091            basicIncomeObjectTypes.add(option.getFinObjTypeIncomeNotCashCd());
092            basicIncomeObjectTypes.add(option.getFinObjTypeCshNotIncomeCd());
093    
094            return basicIncomeObjectTypes;
095        }
096    
097        /**
098         * @see org.kuali.kfs.coa.service.ObjectTypeService#getExpenseTransferObjectType(java.lang.Integer)
099         */
100        public String getExpenseTransferObjectType(Integer universityFiscalYear) {
101            return optionsDao.getByPrimaryId(universityFiscalYear).getFinancialObjectTypeTransferExpenseCd();
102        }
103    
104        /**
105         * @see org.kuali.kfs.coa.service.ObjectTypeService#getIncomeTransferObjectType(java.lang.Integer)
106         */
107        public String getIncomeTransferObjectType(Integer universityFiscalYear) {
108            return optionsDao.getByPrimaryId(universityFiscalYear).getFinancialObjectTypeTransferIncomeCd();
109        }
110    
111        /**
112         * @see org.kuali.kfs.coa.service.ObjectTypeService#getCurrentYearAssetObjectType()
113         */
114        public String getCurrentYearAssetObjectType() {
115            return getAssetObjectType(universityDateService.getCurrentFiscalYear());
116        }
117    
118        /**
119         * @see org.kuali.kfs.coa.service.ObjectTypeService#getCurrentYearBasicExpenseObjectTypes()
120         */
121        public List<String> getCurrentYearBasicExpenseObjectTypes() {
122            return getBasicExpenseObjectTypes(universityDateService.getCurrentFiscalYear());
123        }
124    
125        /**
126         * @see org.kuali.kfs.coa.service.ObjectTypeService#getCurrentYearExpenseObjectTypes()
127         */
128        public List<String> getCurrentYearExpenseObjectTypes() {
129            return getExpenseObjectTypes(universityDateService.getCurrentFiscalYear());
130        }
131    
132        /**
133         * @see org.kuali.kfs.coa.service.ObjectTypeService#getCurrentYearBasicIncomeObjectTypes()
134         */
135        @CacheNoCopy
136        public List<String> getCurrentYearBasicIncomeObjectTypes() {
137            return getBasicIncomeObjectTypes(universityDateService.getCurrentFiscalYear());
138        }
139    
140        /**
141         * @see org.kuali.kfs.coa.service.ObjectTypeService#getCurrentYearExpenseTransferObjectType()
142         */
143        public String getCurrentYearExpenseTransferObjectType() {
144            return getExpenseTransferObjectType(universityDateService.getCurrentFiscalYear());
145        }
146    
147        /**
148         * @see org.kuali.kfs.coa.service.ObjectTypeService#getCurrentYearIncomeTransferObjectType()
149         */
150        public String getCurrentYearIncomeTransferObjectType() {
151            return getIncomeTransferObjectType(universityDateService.getCurrentFiscalYear());
152        }
153    
154        /**
155         * @see org.kuali.kfs.coa.service.ObjectTypeService#getNominalActivityClosingAllowedObjectTypes(java.lang.Integer)
156         */
157        public List<String> getNominalActivityClosingAllowedObjectTypes(Integer fiscalYear) {
158            List<String> nominalClosingObjectTypes = new ArrayList<String>();
159            SystemOptions option = optionsDao.getByPrimaryId(fiscalYear);
160            nominalClosingObjectTypes.add(option.getFinObjTypeExpNotExpendCode());
161            nominalClosingObjectTypes.add(option.getFinObjTypeExpenditureexpCd());
162            nominalClosingObjectTypes.add(option.getFinancialObjectTypeTransferExpenseCd());
163            nominalClosingObjectTypes.add(option.getFinancialObjectTypeTransferIncomeCd());
164            nominalClosingObjectTypes.add(option.getFinObjTypeExpendNotExpCode());
165            nominalClosingObjectTypes.add(option.getFinObjTypeCshNotIncomeCd());
166            nominalClosingObjectTypes.add(option.getFinObjTypeIncomeNotCashCd());
167            nominalClosingObjectTypes.add(option.getFinObjectTypeIncomecashCode());
168            return nominalClosingObjectTypes;
169        }
170    
171        /**
172         * @see org.kuali.kfs.coa.service.ObjectTypeService#getGeneralForwardBalanceObjectTypes(java.lang.Integer)
173         */
174        public List<String> getGeneralForwardBalanceObjectTypes(Integer fiscalYear) {
175            SystemOptions option = optionsDao.getByPrimaryId(fiscalYear);
176            List<String> generalBalanceForwardObjectTypes = new ArrayList<String>();
177            generalBalanceForwardObjectTypes.add(option.getFinancialObjectTypeAssetsCd());
178            generalBalanceForwardObjectTypes.add(option.getFinObjectTypeLiabilitiesCode());
179            generalBalanceForwardObjectTypes.add(option.getFinObjectTypeFundBalanceCd());
180            return generalBalanceForwardObjectTypes;
181        }
182    
183        /**
184         * @see org.kuali.kfs.coa.service.ObjectTypeService#getCumulativeForwardBalanceObjectTypes(java.lang.Integer)
185         */
186        public List<String> getCumulativeForwardBalanceObjectTypes(Integer fiscalYear) {
187            SystemOptions option = optionsDao.getByPrimaryId(fiscalYear);
188            List<String> cumulativeBalanceForwardsObjectTypes = new ArrayList<String>();
189            cumulativeBalanceForwardsObjectTypes.add(option.getFinObjTypeExpendNotExpCode());
190            cumulativeBalanceForwardsObjectTypes.add(option.getFinObjTypeExpNotExpendCode());
191            cumulativeBalanceForwardsObjectTypes.add(option.getFinObjTypeExpenditureexpCd());
192            cumulativeBalanceForwardsObjectTypes.add(option.getFinObjTypeIncomeNotCashCd());
193            cumulativeBalanceForwardsObjectTypes.add(option.getFinancialObjectTypeTransferExpenseCd());
194            cumulativeBalanceForwardsObjectTypes.add(option.getFinancialObjectTypeTransferIncomeCd());
195            cumulativeBalanceForwardsObjectTypes.add(option.getFinObjectTypeIncomecashCode());
196            cumulativeBalanceForwardsObjectTypes.add(option.getFinObjTypeCshNotIncomeCd());
197            return cumulativeBalanceForwardsObjectTypes;
198        }
199    
200        /**
201         * @param objectTypeDao The objectTypeDao to set.
202         */
203        public void setObjectTypeDao(ObjectTypeDao objectTypeDao) {
204            this.objectTypeDao = objectTypeDao;
205        }
206    
207        public OptionsDao getOptionsDao() {
208            return optionsDao;
209        }
210    
211        public void setOptionsDao(OptionsDao optionsDao) {
212            this.optionsDao = optionsDao;
213        }
214    
215        public UniversityDateService getUniversityDateService() {
216            return universityDateService;
217        }
218    
219        public void setUniversityDateService(UniversityDateService universityDateService) {
220            this.universityDateService = universityDateService;
221        }
222    }
