016    package org.kuali.kfs.coa.identity;
017    
018    import java.util.HashSet;
019    import java.util.Set;
020    
021    import org.apache.commons.lang.StringUtils;
022    import org.kuali.kfs.sys.identity.KfsKimAttributes;
023    import org.kuali.rice.kew.doctype.service.DocumentTypeService;
024    import org.kuali.rice.kew.service.KEWServiceLocator;
025    import org.kuali.rice.kim.bo.types.dto.AttributeDefinitionMap;
026    import org.kuali.rice.kim.bo.types.dto.AttributeSet;
027    import org.kuali.rice.kim.util.KimCommonUtils;
028    import org.kuali.rice.kns.datadictionary.AttributeDefinition;
029    
030    public class OrganizationHierarchyReviewRoleTypeServiceImpl extends OrganizationHierarchyAwareRoleTypeServiceBase {
031    
032        private DocumentTypeService documentTypeService;
033    
034        /**
035         * Attributes: Chart Code (required) Organization Code Document Type Name Requirement - Traverse the org hierarchy but not the
036         * document type hierarchy
037         * 
038         * @see org.kuali.kfs.coa.identity.OrganizationOptionalHierarchyRoleTypeServiceImpl#performMatch(org.kuali.rice.kim.bo.types.dto.AttributeSet,
039         *      org.kuali.rice.kim.bo.types.dto.AttributeSet)
040         */
041        @Override
042        protected boolean performMatch(AttributeSet qualification, AttributeSet roleQualifier) {
043            if (isParentOrg(qualification.get(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE), qualification.get(KfsKimAttributes.ORGANIZATION_CODE), roleQualifier.get(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE), roleQualifier.get(KfsKimAttributes.ORGANIZATION_CODE), true)) {
044                Set<String> potentialParentDocumentTypeNames = new HashSet<String>(1);
045                if (roleQualifier.containsKey(KfsKimAttributes.DOCUMENT_TYPE_NAME)) {
046                    potentialParentDocumentTypeNames.add(roleQualifier.get(KfsKimAttributes.DOCUMENT_TYPE_NAME));
047                }
048                return potentialParentDocumentTypeNames.isEmpty() 
049                        || StringUtils.equalsIgnoreCase( qualification.get(KfsKimAttributes.DOCUMENT_TYPE_NAME), roleQualifier.get(KfsKimAttributes.DOCUMENT_TYPE_NAME)) 
050                        || (KimCommonUtils.getClosestParentDocumentTypeName(getDocumentTypeService().findByName(qualification.get(KfsKimAttributes.DOCUMENT_TYPE_NAME)), potentialParentDocumentTypeNames) != null);
051            }
052            return false;
053        }
054    
055        protected DocumentTypeService getDocumentTypeService() {
056            if (documentTypeService == null) {
057                documentTypeService = KEWServiceLocator.getDocumentTypeService();
058            }
059            return this.documentTypeService;
060        }
061    
062        /**
063         * @see org.kuali.rice.kim.service.support.impl.KimTypeServiceBase#getAttributeDefinitions(java.lang.String)
064         */
065        @Override
066        public AttributeDefinitionMap getAttributeDefinitions(String kimTypeId) {
067            AttributeDefinitionMap map = super.getAttributeDefinitions(kimTypeId);
068            for (AttributeDefinition definition : map.values()) {
069                if (KfsKimAttributes.ORGANIZATION_CODE.equals(definition.getName())) {
070                    definition.setRequired(Boolean.FALSE);
071                }
072            }
073            return map;
074        }
075    }
