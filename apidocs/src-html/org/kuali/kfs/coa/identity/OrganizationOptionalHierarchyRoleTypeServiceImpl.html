016    package org.kuali.kfs.coa.identity;
017    
018    import org.kuali.kfs.sys.identity.KfsKimAttributes;
019    import org.kuali.rice.kim.bo.types.dto.AttributeSet;
020    
021    public class OrganizationOptionalHierarchyRoleTypeServiceImpl extends OrganizationHierarchyAwareRoleTypeServiceBase {
022        public static final String DESCEND_HIERARCHY_TRUE_VALUE = "Y";
023        public static final String DESCEND_HIERARCHY_FALSE_VALUE = "N";
024        private boolean qualificationDeterminesDescendHierarchy;
025    
026        /***
027         * @see org.kuali.rice.kim.service.support.impl.KimTypeServiceBase#performMatch(org.kuali.rice.kim.bo.types.dto.AttributeSet,
028         *      org.kuali.rice.kim.bo.types.dto.AttributeSet)
029         */
030        @Override
031        protected boolean performMatch(AttributeSet qualification, AttributeSet roleQualifier) {
032            // if no qualification is passed, then we have no basis to reject this
033            // (if a null is let through, then we get an NPE below) 
034            if ( qualification == null || qualification.isEmpty() || roleQualifier == null || roleQualifier.isEmpty() ) {
035                return true;
036            }
037            String descendHierarchy = null;
038            if (qualificationDeterminesDescendHierarchy) {
039                descendHierarchy = qualification.get(KfsKimAttributes.DESCEND_HIERARCHY);
040            }
041            else {
042                descendHierarchy = roleQualifier.get(KfsKimAttributes.DESCEND_HIERARCHY);
043            }
044            return isParentOrg(qualification.get(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE), 
045                    qualification.get(KfsKimAttributes.ORGANIZATION_CODE), 
046                    roleQualifier.get(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE), 
047                    roleQualifier.get(KfsKimAttributes.ORGANIZATION_CODE), DESCEND_HIERARCHY_TRUE_VALUE.equals(descendHierarchy));
048        }
049    
050        public void setQualificationDeterminesDescendHierarchy(boolean qualificationDeterminesDescendHierarchy) {
051            this.qualificationDeterminesDescendHierarchy = qualificationDeterminesDescendHierarchy;
052        }
053    }
