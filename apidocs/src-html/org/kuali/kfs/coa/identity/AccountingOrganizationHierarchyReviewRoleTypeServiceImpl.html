037    public class AccountingOrganizationHierarchyReviewRoleTypeServiceImpl extends OrganizationHierarchyReviewRoleTypeServiceImpl {
038        private static final Logger LOG = Logger.getLogger(AccountingOrganizationHierarchyReviewRoleTypeServiceImpl.class);
039        /**
040         * Create role type service - org.kuali.kfs.coa.identity.AccountingOrganizationHierarchyReviewRoleTypeService for
041         * KFS-COA/"Organization: Always Hierarchical, Document Type & Accounting" Attributes: Chart Code (required) Organization Code
042         * Document Type Name From Amount To Amount Override Code - total amount will be passed in as qualification and role type
043         * service will need to compare it to from and to amount qualifier values for assignees Requirements: - Traverse the org
044         * hierarchy but not the document type hierarchy - from amount must be null or <= total amount supplied / to amount must be null
045         * or >= total amount supplied, and null override code on assignment matches all override codes
046         * 
047         * @see org.kuali.kfs.coa.identity.OrganizationOptionalHierarchyRoleTypeServiceImpl#performMatch(org.kuali.rice.kim.bo.types.dto.AttributeSet,
048         *      org.kuali.rice.kim.bo.types.dto.AttributeSet)
049         */
050        public boolean performMatch(AttributeSet qualification, AttributeSet roleQualifier) {
051            return doesOverrideCodeMatch(qualification, roleQualifier) 
052                    && isValidTotalAmount(qualification, roleQualifier) 
053                    && super.performMatch(qualification, roleQualifier);
054        }
055    
056        protected boolean doesOverrideCodeMatch(AttributeSet qualification, AttributeSet roleQualifier) {
057            return qualification==null || roleQualifier==null || StringUtils.isBlank(qualification.get(KfsKimAttributes.ACCOUNTING_LINE_OVERRIDE_CODE)) 
058                    || StringUtils.isBlank(roleQualifier.get(KfsKimAttributes.ACCOUNTING_LINE_OVERRIDE_CODE)) 
059                    || qualification.get(KfsKimAttributes.ACCOUNTING_LINE_OVERRIDE_CODE).equals(roleQualifier.get(KfsKimAttributes.ACCOUNTING_LINE_OVERRIDE_CODE));
060        }
061    
062        protected boolean isValidTotalAmount(AttributeSet qualification, AttributeSet roleQualifier) {
063            boolean isValidTotalAmount = false;
064            if(qualification==null || roleQualifier==null) {
065                return false;
066            }
067            String totalAmountStr = qualification.get(KfsKimAttributes.FINANCIAL_DOCUMENT_TOTAL_AMOUNT);
068            if ( StringUtils.isBlank(totalAmountStr) ) {
069                return false;
070            }
071            try {
072                KualiDecimal totalAmount = new KualiDecimal(totalAmountStr);
073                String toAmountStr = roleQualifier.get(KfsKimAttributes.TO_AMOUNT);
074                String fromAmountStr = roleQualifier.get(KfsKimAttributes.FROM_AMOUNT);
075                if ((StringUtils.isBlank(toAmountStr) || new KualiDecimal(toAmountStr).isGreaterEqual(totalAmount) ) 
076                        && (StringUtils.isBlank(fromAmountStr) || new KualiDecimal(fromAmountStr).isLessEqual(totalAmount) )) {
077                    isValidTotalAmount = true;
078                }
079            } catch (Exception ex) {
080                isValidTotalAmount = false;
081                LOG.error( "Exception comparing document amount to role qualifiers.", ex );
082            }
083            return isValidTotalAmount;
084        }

086        private List<String> uniqueAttributes = new ArrayList<String>();
087        {
088            uniqueAttributes.add(KimAttributes.DOCUMENT_TYPE_NAME);
089            uniqueAttributes.add(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE);
090            uniqueAttributes.add(KfsKimAttributes.ORGANIZATION_CODE);
091            uniqueAttributes = Collections.unmodifiableList(uniqueAttributes);
092        }
093        
094        @Override
095        public List<String> getUniqueAttributes(String kimTypeId){
096            return uniqueAttributes;
097        }
098    
099        @Override
100        public AttributeSet validateUnmodifiableAttributes(
101                String kimTypeId, AttributeSet originalAttributeSet, AttributeSet newAttributeSet){
102            AttributeSet validationErrors = super.validateUnmodifiableAttributes(kimTypeId, originalAttributeSet, newAttributeSet);
103            List<String> attributeErrors = null;
104            KimAttributeImpl attributeImpl;
105    
106            String fromAmountRoleMember = getAttributeValue(originalAttributeSet, KfsKimAttributes.FROM_AMOUNT);
107            String fromAmountDelegationMember = getAttributeValue(newAttributeSet, KfsKimAttributes.FROM_AMOUNT);
108            KimTypeInfo kimType = getTypeInfoService().getKimType(kimTypeId);
109            KimTypeAttributeInfo attributeInfo;
110            if(isLesserNumber(fromAmountDelegationMember, fromAmountRoleMember)){
111                attributeInfo = kimType.getAttributeDefinitionByName(KfsKimAttributes.FROM_AMOUNT);
112                GlobalVariables.getMessageMap().putError(
113                        KfsKimAttributes.FROM_AMOUNT, RiceKeyConstants.ERROR_DELEGATION_FROM_AMOUNT_LESSER, 
114                        getDataDictionaryService().getAttributeLabel(attributeInfo.getComponentName(), KfsKimAttributes.FROM_AMOUNT));
115                attributeErrors = extractErrorsFromGlobalVariablesErrorMap(KfsKimAttributes.FROM_AMOUNT);
116            }
117            if(attributeErrors!=null){
118                for(String err: attributeErrors){
119                    validationErrors.put(KfsKimAttributes.FROM_AMOUNT, err);
120                }
121                attributeErrors = null;
122            }
123            
124            String toAmountRoleMember = getAttributeValue(originalAttributeSet, KfsKimAttributes.TO_AMOUNT);
125            String toAmountDelegationMember = getAttributeValue(newAttributeSet, KfsKimAttributes.TO_AMOUNT);
126            if(StringUtils.isNotEmpty(toAmountRoleMember) && isGreaterNumber(toAmountDelegationMember, toAmountRoleMember)){
127                attributeInfo = kimType.getAttributeDefinitionByName(KfsKimAttributes.TO_AMOUNT);
128                GlobalVariables.getMessageMap().putError(
129                        KfsKimAttributes.TO_AMOUNT, RiceKeyConstants.ERROR_DELEGATION_TO_AMOUNT_GREATER, 
130                        getDataDictionaryService().getAttributeLabel(attributeInfo.getComponentName(), KfsKimAttributes.TO_AMOUNT));
131                attributeErrors = extractErrorsFromGlobalVariablesErrorMap(KfsKimAttributes.TO_AMOUNT);
132            }
133            if(attributeErrors!=null){
134                for(String err: attributeErrors){
135                    validationErrors.put(KfsKimAttributes.TO_AMOUNT, err);
136                }
137                attributeErrors = null;
138            }
139    
140            return validationErrors;
141        }
142    
143        protected boolean isLesserNumber(String numberStr1, String numberStr2){
144            if(StringUtils.isBlank(numberStr1) ) {
145                numberStr1 = "0";
146            }
147            if(StringUtils.isBlank(numberStr2) ) {
148                numberStr2 = "0";
149            }
150            int number1 = KNSUtils.getIntegerValue(numberStr1);
151            int number2 = KNSUtils.getIntegerValue(numberStr2);
152            return number1 < number2;
153        }
154    
155        protected boolean isGreaterNumber(String numberStr1, String numberStr2){
156            if(StringUtils.isBlank(numberStr1) ) {
157                numberStr1 = "0";
158            }
159            if(StringUtils.isBlank(numberStr2) ) {
160                numberStr2 = "0";
161            }
162            int number1 = KNSUtils.getIntegerValue(numberStr1);
163            int number2 = KNSUtils.getIntegerValue(numberStr2);
164            return number1 > number2;
165        }
