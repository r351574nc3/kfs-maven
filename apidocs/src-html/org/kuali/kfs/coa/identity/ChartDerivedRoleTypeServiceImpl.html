016    package org.kuali.kfs.coa.identity;
017    
018    import org.kuali.kfs.coa.service.OrganizationService;
019    import org.kuali.kfs.sys.identity.KfsKimAttributes;
020    import org.kuali.rice.kim.bo.types.dto.AttributeSet;
021    import org.kuali.rice.kim.service.support.impl.KimDerivedRoleTypeServiceBase;
022    
023    public class ChartDerivedRoleTypeServiceImpl extends KimDerivedRoleTypeServiceBase {
024    
025        private OrganizationService organizationService;
026    
027        /**
028         * This service takes the following attributes:
029         *     Chart Code
030         * Requirements:
031         *     - KFS-SYS University Chart Manger: 
032         *     KFS-SYS Chart Manager where for the chart that matches the one on CA_ORG_T where CA_ORG_T.ORG_TYP_CD = U
033         * 
034         * @see org.kuali.rice.kim.service.support.impl.KimRoleTypeServiceBase#getPrincipalIdsFromApplicationRole(java.lang.String, java.lang.String, org.kuali.rice.kim.bo.types.dto.AttributeSet)
035         */
036        @Override
037        public AttributeSet convertQualificationForMemberRoles( String namespaceCode, String roleName, String memberRoleNamespaceCode, String memberRoleName, AttributeSet qualification){
038            String[] chartOrg = getOrganizationService().getRootOrganizationCode();
039            if(chartOrg != null){
040                String rootChartOfAccountCode = chartOrg[0];
041                // copy all the other qualification attributes
042                AttributeSet processingChartQualification = new AttributeSet( qualification );
043                // now, override the chart
044                processingChartQualification.put(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE, rootChartOfAccountCode);
045                return processingChartQualification;
046            } else {
047                return qualification;
048            }
049        }
050    
051        @Override
052        protected boolean performMatch(AttributeSet inputAttributeSet, AttributeSet storedAttributeSet) {
053            // don't perform any matching - really just a pass-thru role for the embedded KFS-SYS User role
054            return true;
055        }
056    
057        /**
058         * Gets the organizationService attribute. 
059         * @return Returns the organizationService.
060         */
061        protected OrganizationService getOrganizationService() {
062            return organizationService;
063        }
064    
065        /**
066         * Sets the organizationService attribute value.
067         * @param organizationService The organizationService to set.
068         */
069        public void setOrganizationService(OrganizationService organizationService) {
070            this.organizationService = organizationService;
071        }
072    
073    }
