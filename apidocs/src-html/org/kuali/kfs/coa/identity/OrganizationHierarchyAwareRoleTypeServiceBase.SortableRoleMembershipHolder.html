051        protected boolean isParentOrg(String qualificationChartCode, String qualificationOrgCode, String roleChartCode, String roleOrgCode, boolean descendHierarchy) {
052            if ( StringUtils.isBlank(qualificationChartCode) || StringUtils.isBlank(qualificationOrgCode) ) {
053                if ( LOG.isInfoEnabled() ) {
054                    LOG.info("No chart/org qualifications passed into isParentOrg()");
055                }
056                return false;
057            }
058            if (roleChartCode == null && roleOrgCode == null) {
059                LOG.warn("Call to "+this.getClass().getName()+" with no organization role qualifiers; both chart and organization code are null.  Please ensure that qualification data has organization information for this role.");
060                return false;
061            }
062            if (roleOrgCode == null) {
063                return roleChartCode.equals(qualificationChartCode) 
064                        || (descendHierarchy && chartService.isParentChart(qualificationChartCode, roleChartCode));
065            }
066            return (roleChartCode.equals(qualificationChartCode) && roleOrgCode.equals(qualificationOrgCode)) 
067                    || (descendHierarchy && organizationService.isParentOrganization(qualificationChartCode, qualificationOrgCode, roleChartCode, roleOrgCode));
068        }

070        @Override
071        public AttributeSet convertQualificationForMemberRoles(String namespaceCode, String roleName, String memberRoleNamespaceCode, String memberRoleName, AttributeSet qualification) {
072            if ( qualification == null ) {
073                return null;
074            }
075            // only attempt the conversion if :
076            // (a) there is not already a campus code provided by the document 
077            // and (b) we have a chart and organization to resolve
078            if ( StringUtils.isBlank( qualification.get(KfsKimAttributes.CAMPUS_CODE ) )
079                    && StringUtils.isNotBlank( KfsKimAttributes.CHART_OF_ACCOUNTS_CODE)
080                    && StringUtils.isNotBlank( KfsKimAttributes.ORGANIZATION_CODE ) ) {
081                AttributeSet newQualification = new AttributeSet(qualification);
082                try {
083                    Organization org = organizationService.getByPrimaryId(qualification.get(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE), qualification.get(KfsKimAttributes.ORGANIZATION_CODE));
084                    if ( org != null ) {
085                        newQualification.put(KfsKimAttributes.CAMPUS_CODE, org.getOrganizationPhysicalCampusCode());
086                    } else {
087                        if ( LOG.isDebugEnabled() ) {
088                            LOG.debug( "Invalid Chart/Org passed to convertQualificationForMemberRoles: " + namespaceCode + "/" + roleName + "/" + memberRoleNamespaceCode + "/" + memberRoleName + "/" + qualification );
089                        }
090                    }
091                } catch (Exception ex) {
092                    LOG.warn("Unable to convert organization qualification to physical campus", ex);
093                }
094                return newQualification;
095            } else {
096                return qualification;
097            }
098        }

104        public List<DelegateInfo> doDelegationQualifiersMatchQualification(AttributeSet qualification, List<DelegateInfo> delegationMemberList) {
105            AttributeSet translatedQualification = translateInputAttributeSet(qualification);
106            List<DelegateInfo> matchingMemberships = new ArrayList<DelegateInfo>();
107            for (DelegateInfo dmi : delegationMemberList) {
108                if (performMatch(translatedQualification, dmi.getQualifier())) {
109                    matchingMemberships.add(dmi);
110                }
111            }
112            return matchingMemberships;
113        }

124        @Override
125        public List<RoleMembershipInfo> sortRoleMembers(List<RoleMembershipInfo> roleMembers) {
126            List<SortableRoleMembershipHolder> listToSort = new ArrayList<SortableRoleMembershipHolder>( roleMembers.size() );
127            // build the sortable list
128            for ( RoleMembershipInfo rmi : roleMembers ) {
129                listToSort.add( new SortableRoleMembershipHolder( rmi ) );
130            }
131            // sort it
132            Collections.sort(listToSort);
133            // restore them to the list in their sorted order
134            roleMembers.clear();
135            int group = 0; // counter for the group number to add to the roleSortingCode
136            String lastRoleSortingCode = "";
137            for ( SortableRoleMembershipHolder srmh : listToSort ) {
138                if ( !srmh.rmi.getRoleSortingCode().equals( lastRoleSortingCode ) ) {
139                    group++;
140                    lastRoleSortingCode = srmh.rmi.getRoleSortingCode();
141                }
142                
143                srmh.rmi.setRoleSortingCode( StringUtils.leftPad(Integer.toString(group), 3, '0') + "/" + srmh.rmi.getRoleSortingCode() );
144                roleMembers.add( srmh.rmi );
145            }
146            return roleMembers;
147        }
148    
149        protected class SortableRoleMembershipHolder implements Comparable<SortableRoleMembershipHolder> {
150            
151            public String chart;
152            public String org;
153            public RoleMembershipInfo rmi;
154            
155            public SortableRoleMembershipHolder( RoleMembershipInfo rmi ) {
156                this.rmi = rmi;
157                chart = rmi.getQualifier().get(KfsKimAttributes.CHART_OF_ACCOUNTS_CODE);
158                org = rmi.getQualifier().get(KfsKimAttributes.ORGANIZATION_CODE);
159                rmi.setRoleSortingCode( chart+"-"+org );
160            }
161            
162            public int compareTo(SortableRoleMembershipHolder o) {
163                if ( chart == null || org == null || o.chart == null || o.org == null || (o.chart == chart && o.org == org) ) {
164                    return 0;
165                }
166                if ( organizationService.isParentOrganization(o.chart, o.org, chart, org) ) {
167                    return 1;
168                } else {
169                    return -1;
170                }
171            }
172        }
