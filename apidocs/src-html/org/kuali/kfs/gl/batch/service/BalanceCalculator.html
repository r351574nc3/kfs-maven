016    package org.kuali.kfs.gl.batch.service;
017    
018    import java.util.Collection;
019    
020    import org.kuali.kfs.gl.businessobject.Balance;
021    import org.kuali.kfs.gl.businessobject.Transaction;
022    
023    /**
024     * This interface declares methods needed for posting transactions to the appropriate balance records.
025     */
026    public interface BalanceCalculator {
027        /**
028         * Given a collection of balance records, returns the balance that the given transaction would post to
029         * or creates a new balance record
030         * 
031         * @param balanceList a Collection of balance records
032         * @param t the transaction to post
033         * @return the balance to post against
034         */
035        public Balance findBalance(Collection balanceList, Transaction t);
036    
037        /**
038         * Updates the balance based on the given transaction
039         * 
040         * @param t the transaction to post
041         * @param b the balance being posted against
042         */
043        public void updateBalance(Transaction t, Balance b);
044    }
