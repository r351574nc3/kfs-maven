016    package org.kuali.kfs.gl.batch.service.impl.exception;
017    
018    /**
019     * Represents an exception, often occurring in batch jobs, that should be reported
020     * but which shouldn't end the process
021     */
022    public class NonFatalErrorException extends Exception {
023        /**
024         * Constructs a NonFatalErrorException instance
025         */
026        public NonFatalErrorException() {
027            super();
028        }
029    
030        /**
031         * Constructs a NonFatalErrorException instance
032         * @param message the message this exception should use to report itself in the logs
033         * @param cause the original problem
034         */
035        public NonFatalErrorException(String message, Throwable cause) {
036            super(message, cause);
037        }
038    
039        /**
040         * Constructs a NonFatalErrorException instance
041         * @param message the message this exception should use to report itself in the logs
042         */
043        public NonFatalErrorException(String message) {
044            super(message);
045        }
046    
047        /**
048         * Constructs a NonFatalErrorException instance
049         * @param cause the original problem
050         */
051        public NonFatalErrorException(Throwable cause) {
052            super(cause);
053        }
054    }
