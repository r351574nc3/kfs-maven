061        }

063        /**
064         * Builds the file name using the following construction: All collector files start with gl_collectorflatfile_ append the chartorg
065         * from the batch header append the username of the user who is uploading the file then the user supplied indentifier finally
066         * the timestamp
067         * 
068         * @param user who uploaded the file
069         * @param parsedFileContents represents collector batch object
070         * @param userIdentifier user identifier for user who uploaded file
071         * @return String returns file name using the convention mentioned in the description
072         * 
073         * @see org.kuali.kfs.sys.batch.BatchInputFileType#getFileName(org.kuali.rice.kim.bo.Person, java.lang.Object,
074         *      java.lang.String)
075         */
076        public String getFileName(String principalName, Object parsedFileContents, String fileUserIdentifer) {
077            String fileName = FILE_NAME_PREFIX;
078            fileName += principalName;
079            if (org.apache.commons.lang.StringUtils.isNotBlank(fileUserIdentifer)) {
080                fileName += "_" + fileUserIdentifer;
081            }
082            fileName += "_" + dateTimeService.toDateTimeStringForFilename(dateTimeService.getCurrentDate());
083    
084            // remove spaces in filename
085            fileName = org.apache.commons.lang.StringUtils.remove(fileName, " ");
086    
087            return fileName;
088        }

104        /**
105         * @see org.kuali.kfs.sys.batch.BatchInputFileType#parse(byte[])
106         */
107        public Object parse(byte[] fileByteContent) throws ParseException {
108            List<CollectorBatch> batchList = new ArrayList<CollectorBatch>();
109            CollectorBatch currentBatch = null;
110            BufferedReader bufferedFileReader = new BufferedReader(new InputStreamReader(new ByteArrayInputStream(fileByteContent)));
111            String fileLine;
112            Date curDate = SpringContext.getBean(DateTimeService.class).getCurrentSqlDate();
113            UniversityDate universityDate = SpringContext.getBean(UniversityDateService.class).getCurrentUniversityDate();
114            int lineNumber = 0;
115            
116            try {
117                while ((fileLine = bufferedFileReader.readLine()) != null) {
118                    lineNumber++;
119                    String preprocessedLine = preprocessLine(fileLine);
120                    if (fileLine.length() >= 27) {  //if no rec_type, probably a blank or almost blank line at end of file
121                        String recordType = extractRecordType(preprocessedLine);
122                        if ("HD".equals(recordType)) {  // this is a header line
123                            ensurePreviousBatchTerminated(currentBatch, lineNumber);
124                            try {
125                                currentBatch = createCollectorBatch(preprocessedLine, batchList);
126                            }
127                            catch (RuntimeException e) {
128                                if (currentBatch == null){
129                                    currentBatch = new CollectorBatch();
130                                    batchList.add(currentBatch);
131                                }
132                                currentBatch.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, KFSKeyConstants.ERROR_CUSTOM, e.getMessage());
133                            }
134                            
135                        }
136                        else if ("DT".equals(recordType)) {  //ID billing detail
137                            currentBatch = createHeaderlessBatchIfNecessary(currentBatch, batchList, lineNumber);
138                            try {
139                                CollectorDetail collectorDetail = createCollectorDetail(preprocessedLine, curDate, universityDate, lineNumber, currentBatch.getMessageMap());
140                                currentBatch.addCollectorDetail(collectorDetail);
141                            }
142                            catch (RuntimeException e) {
143                                currentBatch.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, KFSKeyConstants.ERROR_CUSTOM, e.getMessage());
144                            }
145                        }
146                        else if ("TL".equals(recordType)) {  //trailer record
147                            currentBatch = createHeaderlessBatchIfNecessary(currentBatch, batchList, lineNumber);
148                            try {
149                                updateCollectorDetailWithTrailerRecords(currentBatch, preprocessedLine, lineNumber);
150                            }
151                            catch (RuntimeException e) {
152                                currentBatch.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, KFSKeyConstants.ERROR_CUSTOM, e.getMessage());
153                            }
154                            currentBatch = null;
155                        }
156                        else {  // accounting record/origin entry
157                            currentBatch = createHeaderlessBatchIfNecessary(currentBatch, batchList, lineNumber);
158                            try {
159                                OriginEntryFull originEntry = createOriginEntry(preprocessedLine, curDate, universityDate, lineNumber, currentBatch.getMessageMap());
160                                currentBatch.addOriginEntry(originEntry);
161                            }
162                            catch (RuntimeException e) {
163                                currentBatch.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, KFSKeyConstants.ERROR_CUSTOM, e.getMessage());
164                            }
165                        }
166                    }
167                }
168                // in case we come across a batch that didn't have a trailer record
169                ensurePreviousBatchTerminated(currentBatch, lineNumber);
170            } catch (Exception e) {
171                LOG.error(e.getMessage() + " happend in CollectorFlatFileInputType.parse.", e);
172                throw new ParseException(e.getMessage() + " happend in CollectorFlatFileInputType.parse.", e);
173            }
174    
175            
176            for (CollectorBatch batch : batchList) {
177                OriginEntryTotals totals = new OriginEntryTotals();
178                totals.addToTotals(batch.getOriginEntries().iterator());
179                batch.setOriginEntryTotals(totals);
180                
181                // now copy all the messages from the per-batch message map into the global map, so that we can display the messages on the upload screen
182                copyAllMessages(batch.getMessageMap(), GlobalVariables.getMessageMap());
183            }
184            return batchList;
185        }
186    
187        private void copyAllMessages(MessageMap sourceMap, MessageMap destMap) {
188            copyAllMessagesHelper(sourceMap.getInfoMessages(), "info", destMap);
189            copyAllMessagesHelper(sourceMap.getWarningMessages(), "warning", destMap);
190            copyAllMessagesHelper(sourceMap.getErrorMessages(), "error", destMap);
191        }
192        
193        private void copyAllMessagesHelper(Map<String, TypedArrayList> sourceMessages, String type, MessageMap destMap) {
194            for (String key : sourceMessages.keySet()) {
195                TypedArrayList messages = sourceMessages.get(key);
196                if (messages != null) {
197                    for (Object o : messages) {
198                        ErrorMessage message = (ErrorMessage) o;
199                        if ("info".equals(type)) {
200                            destMap.putInfoWithoutFullErrorPath(key, message.getErrorKey(), message.getMessageParameters());
201                        }
202                        else if ("warning".equals(type)) {
203                            destMap.putWarningWithoutFullErrorPath(key, message.getErrorKey(), message.getMessageParameters());
204                        }
205                        else if ("error".equals(type)) {
206                            destMap.putErrorWithoutFullErrorPath(key, message.getErrorKey(), message.getMessageParameters());
207                        }
208                        else {
209                            throw new IllegalArgumentException();
210                        }
211                    }
212                }
213            }
214        }
215        protected void ensurePreviousBatchTerminated(CollectorBatch currentBatch, int lineNumber) {
216            if (currentBatch != null) {
217                // we've encountered a new header, when we're still not done parsing the previous batch (i.e. trailer not found).  This is an error, and mark the old batch as such
218                currentBatch.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, KFSKeyConstants.Collector.MISSING_TRAILER_RECORD, Integer.toString(lineNumber));
219                currentBatch.setTotalAmount("0");
220                currentBatch.setTotalRecords(0);
221            }
222        }
223        
224        protected CollectorBatch createHeaderlessBatchIfNecessary(CollectorBatch currentBatch, List<CollectorBatch> batchList, int lineNumber) {
225            if (currentBatch != null) {
226                return currentBatch;
227            }
228            CollectorBatch headerlessBatch = new CollectorBatch();
229            headerlessBatch.setHeaderlessBatch(true);
230            headerlessBatch.getMessageMap().putError(KFSConstants.GLOBAL_ERRORS, KFSKeyConstants.Collector.MISSING_HEADER_RECORD, Integer.toString(lineNumber));
231            batchList.add(headerlessBatch);
232            return headerlessBatch;
233        }

241        protected String addDecimalPoint (String amount) {
242            if (!amount.contains(".")) {  //have to add decimal point if it's missing
243                int length = amount.length();
244                amount = amount.substring(0, length - 2) + "." + amount.substring(length - 2, length);
245            }
246            return amount;
247        }

249        protected CollectorBatch createCollectorBatch(String headerLine, List<CollectorBatch> batchList) {
250            CollectorBatch newBatch = new CollectorBatch();
251            newBatch.setFromTextFileForCollectorBatch(headerLine);
252            batchList.add(newBatch);
253            
254            return newBatch;
255        }

268        protected OriginEntryFull createOriginEntry(String fileLine, Date curDate, UniversityDate universityDate, int lineNumber, MessageMap messageMap) {
269            OriginEntryFull originEntry = new OriginEntryFull();
270            try{        
271                List<Message> originEntryErrorMessages = originEntry.setFromTextFileForBatch(fileLine, lineNumber);
272    
273                if (null == originEntry.getTransactionDate()) {
274                    originEntry.setTransactionDate(curDate);
275                }
276                
277                if (StringUtils.isBlank(originEntry.getUniversityFiscalPeriodCode())) {
278                    originEntry.setUniversityFiscalPeriodCode(universityDate.getUniversityFiscalAccountingPeriod());
279                }
280                
281                if ( null == originEntry.getTransactionLedgerEntrySequenceNumber() ) {
282                    originEntry.setTransactionLedgerEntrySequenceNumber(new Integer(1));
283                }
284                    
285                if (StringUtils.isBlank(originEntry.getSubAccountNumber())) {
286                    originEntry.setSubAccountNumber(" ");
287                }
288                if (StringUtils.isBlank(originEntry.getFinancialSubObjectCode())) {
289                    originEntry.setFinancialSubObjectCode(" ");
290                }
291                
292                for (Message orginEntryError : originEntryErrorMessages) {
293                   messageMap.putError(KFSConstants.GLOBAL_ERRORS, KFSKeyConstants.ERROR_CUSTOM, orginEntryError.getMessage()); 
294                }
295            
296            } catch (Exception e){
297                throw new RuntimeException(e + " occurred in CollectorFlatFileInputType.createOriginEntry()",e);
298            }
299    
300            originEntry.setTransactionLedgerEntryAmount(addDecimalPoint(originEntry.getTransactionLedgerEntryAmount().toString()));
301    
302            return originEntry;
303        }

305        public boolean validate(Object parsedFileContents) {
306            List<CollectorBatch> collectorBatches = (List<CollectorBatch>) parsedFileContents;
307            for (CollectorBatch collectorBatch : collectorBatches) {
308                boolean isValid = collectorHelperService.performValidation(collectorBatch);
309                if (isValid) {
310                    isValid = collectorHelperService.checkTrailerTotals(collectorBatch, null);
311                }
312                if (!isValid) {
313                    return false;
314                }
315            }
316        
317            return true;
318        }

336        protected String extractRecordType(String line) {
337            if (line == null || line.length() < 27) {
338                return null;
339            }
340            
341            String recordType = null;
342            CollectorBatch collectorBatch = new CollectorBatch();
343            final Map<String, Integer> pMap = CollectorBatch.getCollectorBatchHeaderFieldUtil().getFieldBeginningPositionMap();
344            recordType = (collectorBatch.getValue(line, pMap.get(KFSPropertyConstants.COLLECTOR_BATCH_RECORD_TYPE), pMap.get(KFSPropertyConstants.BATCH_SEQUENCE_NUMBER)));
345    
346            return recordType;
347        }
