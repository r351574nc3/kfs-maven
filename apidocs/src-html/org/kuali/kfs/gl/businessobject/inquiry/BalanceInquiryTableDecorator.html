016    package org.kuali.kfs.gl.businessobject.inquiry;
017    
018    import java.util.Iterator;
019    import java.util.List;
020    
021    import javax.servlet.jsp.PageContext;
022    
023    import org.displaytag.decorator.TableDecorator;
024    import org.displaytag.properties.MediaTypeEnum;
025    import org.kuali.rice.kns.web.ui.Column;
026    import org.kuali.rice.kns.web.ui.ResultRow;
027    
028    /**
029     * A decorator meant to help display balance inquiry information for the displaytag tag library.
030     */
031    public class BalanceInquiryTableDecorator extends TableDecorator {
032    
033        private int numOfNonMonthField;
034        private int numOfMonthField = 13;
035        private int rowCounter = 0;
036    
037        /**
038         * Generates the first row of the table, which acts as the headers for the data.
039         * 
040         * @return the String to display as the first row in the table
041         * @see org.displaytag.decorator.TableDecorator#startRow()
042         */
043        @Override
044        public String startRow() {
045            // TableTagParameters.
046            PageContext pageContext = getPageContext();
047            MediaTypeEnum mediaType = (MediaTypeEnum) pageContext.getAttribute("mediaType");
048            ResultRow row = (ResultRow) getCurrentRowObject();
049    
050            if (MediaTypeEnum.HTML.equals(mediaType)) { // Display the nested table.
051    
052                StringBuffer rowBuffer = new StringBuffer("<tr>");
053                rowBuffer.append("<tr>");
054    
055                List columns = row.getColumns();
056                int columnCount = 0;
057                numOfNonMonthField = columns.size() - numOfMonthField;
058                for (Iterator i = columns.iterator(); i.hasNext() && columnCount++ < numOfNonMonthField;) {
059                    Column column = (Column) i.next();
060    
061                    if (rowCounter > 0) {
062                        rowBuffer.append("<th>");
063                        rowBuffer.append(column.getColumnTitle());
064                        rowBuffer.append("</th>");
065                    }
066                }
067                rowBuffer.append("</tr>");
068                return rowBuffer.toString();
069            }
070            return super.startRow();
071        }
072    
073        /**
074         * Generates the last row of the displayed table...which displays some kind of footer...
075         * @return a String representing the last row of the table
076         * @see org.displaytag.decorator.TableDecorator#finishRow()
077         */
078        @Override
079        public String finishRow() {
080            rowCounter++;
081            PageContext pageContext = getPageContext();
082            MediaTypeEnum mediaType = (MediaTypeEnum) pageContext.getAttribute("mediaType");
083            ResultRow row = (ResultRow) getCurrentRowObject();
084    
085            if (MediaTypeEnum.HTML.equals(mediaType)) {
086    
087                // Display the nested table.
088                StringBuffer rowBuffer = new StringBuffer("<tr>");
089    
090                rowBuffer.append("<td colspan='" + numOfNonMonthField + "' class=\"infocell\"><br><center>");
091                rowBuffer.append("<table class=\"datatable-80\" cellspacing=\"0\" cellpadding=\"0\">");
092    
093                for (int o = 0; o < 3; o++) {
094                    rowBuffer.append("<tr>");
095    
096                    for (int i = 0; i < 4; i++) {
097                        int index = this.numOfNonMonthField + o + (3 * i);
098                        Column column = (Column) row.getColumns().get(index);
099    
100                        rowBuffer.append("<th class=\"infocell\" width=\"10%\">");
101                        rowBuffer.append(column.getColumnTitle());
102                        rowBuffer.append("</th>");
103                        rowBuffer.append("<td class=\"numbercell\" width=\"15%\">");
104                        rowBuffer.append("<a href=\"").append(column.getPropertyURL()).append("\" target=\"blank\">");
105                        rowBuffer.append(column.getPropertyValue()).append("</a>");
106                        rowBuffer.append("</td>");
107    
108                    }
109                    rowBuffer.append("</tr>");
110                }
111    
112                rowBuffer.append("<tr>");
113                rowBuffer.append("<td colspan='6'></td>");
114    
115                Column column = (Column) row.getColumns().get(numOfNonMonthField + numOfMonthField - 1);
116    
117                rowBuffer.append("<th class=\"infocell\" width=\"10%\">");
118                rowBuffer.append(column.getColumnTitle());
119                rowBuffer.append("</th>");
120                rowBuffer.append("<td class=\"numbercell\" width=\"15%\">");
121                rowBuffer.append("<a href=\"").append(column.getPropertyURL()).append("\" target=\"blank\">");
122                rowBuffer.append(column.getPropertyValue()).append("</a>");
123                rowBuffer.append("</td>");
124    
125                rowBuffer.append("</tr>");
126                return rowBuffer.append("</table></center><br /></td></tr>").toString();
127            }
128            return super.finishRow();
129        }
130    }
