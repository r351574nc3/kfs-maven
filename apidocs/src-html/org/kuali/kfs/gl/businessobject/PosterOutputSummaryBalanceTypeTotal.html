016    package org.kuali.kfs.gl.businessobject;
017    
018    import java.text.MessageFormat;
019    import java.util.LinkedHashMap;
020    
021    import org.kuali.kfs.sys.KFSKeyConstants;
022    import org.kuali.kfs.sys.context.SpringContext;
023    import org.kuali.rice.kns.service.KualiConfigurationService;
024    
025    /**
026     * Represents a poster output summary line which holds values for a specific balance type
027     */
028    public class PosterOutputSummaryBalanceTypeTotal extends PosterOutputSummaryTotal {
029        private String balanceTypeCode;
030        
031        /**
032         * Constructs a PosterOutputSummaryBalanceTypeTotal
033         * @param balanceTypeCode the balance type code totalled by this totaller
034         */
035        public PosterOutputSummaryBalanceTypeTotal(String balanceTypeCode) {
036            super();
037            this.balanceTypeCode = balanceTypeCode;
038        }
039        
040        /**
041         * @return the balance type code associated with this total line
042         */
043        public String getBalanceTypeCode() {
044            return balanceTypeCode;
045        }
046        
047        /**
048         * Returns the summary for this total line
049         * @see org.kuali.kfs.gl.businessobject.PosterOutputSummaryTotal#getSummary()
050         */
051        @Override
052        public String getSummary() {
053            final String message = SpringContext.getBean(KualiConfigurationService.class).getPropertyString(getSummaryMessageName());
054            return MessageFormat.format(message, (Object[])getSummaryMessageParameters());
055        }
056        
057        /**
058         * @return the name of the summary message
059         */
060        protected String getSummaryMessageName() {
061            return KFSKeyConstants.MESSAGE_REPORT_POSTER_OUTPUT_SUMMARY_BALANCE_TYPE_TOTAL;
062        }
063        
064        /**
065         * @return the values that should be formatted into the message
066         */
067        protected String[] getSummaryMessageParameters() {
068            return new String[] { getBalanceTypeCode() };
069        }
070        
071        /**
072         * A map of the "keys" of this transient business object
073         * @see org.kuali.rice.kns.bo.BusinessObjectBase#toStringMapper()
074         */
075        @Override
076        protected LinkedHashMap toStringMapper() {
077            LinkedHashMap pks = new LinkedHashMap<String, Object>();
078            pks.put("balanceTypeCode",this.getBalanceTypeCode());
079            pks.put("objectTypeCode",this.getObjectTypeCode());
080            pks.put("creditAmount",this.getCreditAmount());
081            pks.put("debitAmount",this.getDebitAmount());
082            pks.put("budgetAmount",this.getBudgetAmount());
083            pks.put("netAmount",this.getNetAmount());
084            return pks;
085        }
086    }
