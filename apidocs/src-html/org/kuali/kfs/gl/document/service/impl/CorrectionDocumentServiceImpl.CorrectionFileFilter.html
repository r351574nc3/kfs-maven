426        /**
427         * Saves an interator of Origin Entry records to the given file name
428         * 
429         * @param fullPathUniqueFileName the name of the file to write entries to
430         * @param entries entries to write
431         */
432        protected void persistOriginEntries(String fullPathUniqueFileName, Iterator<OriginEntryFull> entries) {
433            File fileOut = new File(fullPathUniqueFileName);
434            FileOutputStream streamOut = null;
435            BufferedOutputStream bufferedStreamOut = null;
436            try {
437                streamOut = new FileOutputStream(fileOut);
438                bufferedStreamOut = new BufferedOutputStream(streamOut);
439    
440                byte[] newLine = "\n".getBytes();
441                while (entries.hasNext()) {
442                    OriginEntryFull entry = entries.next();
443                    bufferedStreamOut.write(entry.getLine().getBytes());
444                    bufferedStreamOut.write(newLine);
445                }
446            }
447            catch (IOException e) {
448                LOG.error("unable to persist origin entries to file: " + fullPathUniqueFileName, e);
449                throw new RuntimeException("unable to persist origin entries to file.");
450            }
451            finally {
452                try {
453                    bufferedStreamOut.close();
454                    streamOut.close();
455                }
456                catch (IOException e) {
457                    LOG.error("unable to close output streams for file: " + fullPathUniqueFileName, e);
458                    throw new RuntimeException("unable to close output streams");
459                }
460            }
461        }

565         * @param abortThreshold if more entries than this need to be read...well, they just won't get read
566         * @return a List of OriginEntryFulls
567         */
568        protected List<OriginEntryFull> retrievePersistedOriginEntries(String fullPathUniqueFileName, int abortThreshold) {
569            File fileIn = new File(fullPathUniqueFileName);
570            if (!fileIn.exists()) {
571                LOG.error("File " + fullPathUniqueFileName + " does not exist.");
572                throw new RuntimeException("File does not exist");
573            }
574            BufferedReader reader = null;
575            FileReader fReader = null;
576    
577            List<OriginEntryFull> entries = new ArrayList<OriginEntryFull>();
578            int lineNumber = 0;
579            try {
580                fReader = new FileReader(fileIn);
581                reader = new BufferedReader(fReader);
582                String line;
583                while ((line = reader.readLine()) != null) {
584                    OriginEntryFull entry = new OriginEntryFull();
585                    entry.setFromTextFileForBatch(line, lineNumber);
586                    if (abortThreshold != UNLIMITED_ABORT_THRESHOLD && lineNumber >= abortThreshold) {
587                        return null;
588                    }
589                    lineNumber++;
590                    entries.add(entry);
591                }
592            }
593            catch (IOException e) {
594                LOG.error("retrievePersistedOriginEntries() Error reading file " + fileIn.getAbsolutePath(), e);
595                throw new RuntimeException("Error reading file");
596            }
597            finally {
598                try {
599                    if (fReader != null) {
600                        fReader.close();
601                    }
602                    if (reader != null) {
603                        reader.close();
604                    }
605                }
606                catch (IOException e) {
607                    LOG.error("Unable to close file " + fileIn.getAbsolutePath(), e);
608                    throw new RuntimeException("Error closing file");
609                }
610            }
611            return entries;
612        }

743        public String createOutputFileForProcessing(String docId, java.util.Date today) {
744            File outputFile = new File(glcpDirectoryName + File.separator + docId + OUTPUT_ORIGIN_ENTRIES_FILE_SUFFIX);
745            String newFileName = batchFileDirectoryName + File.separator + GLCP_OUTPUT_PREFIX  + "." + docId + buildFileExtensionWithDate(today);
746            File newFile = new File(newFileName);
747            FileReader inputFileReader;
748            FileWriter newFileWriter;
749    
750            try {
751                // copy output file and put in OriginEntryInformation directory
752                inputFileReader = new FileReader(outputFile);
753                newFileWriter = new FileWriter(newFile);
754                int c;
755                while ((c = inputFileReader.read()) != -1) {
756                    newFileWriter.write(c);
757                }
758    
759                inputFileReader.close();
760                newFileWriter.close();
761    
762                // create done file, after successfully copying output file
763                String doneFileName = newFileName.replace(GeneralLedgerConstants.BatchFileSystem.EXTENSION, GeneralLedgerConstants.BatchFileSystem.DONE_FILE_EXTENSION);
764                File doneFile = new File(doneFileName);
765                if (!doneFile.exists()) {
766                    doneFile.createNewFile();
767                }
768    
769            }
770            catch (IOException e) {
771                throw new RuntimeException(e);
772            }
773    
774            return newFileName;
775        }

784        /**
785         * Saves the input and output origin entry groups for a document prior to saving the document
786         * 
787         * @param document a GLCP document
788         * @param correctionDocumentEntryMetadata metadata about this GLCP document
789         * @see org.kuali.kfs.gl.document.service.CorrectionDocumentService#persistOriginEntryGroupsForDocumentSave(org.kuali.kfs.gl.document.CorrectionDocument,
790         *      org.kuali.kfs.gl.document.web.CorrectionDocumentEntryMetadata)
791         */
792        public void persistOriginEntryGroupsForDocumentSave(GeneralLedgerCorrectionProcessDocument document, CorrectionDocumentEntryMetadata correctionDocumentEntryMetadata) {
793            if (correctionDocumentEntryMetadata.getAllEntries() == null && !correctionDocumentEntryMetadata.isRestrictedFunctionalityMode()) {
794                // if we don't have origin entries loaded and not in restricted functionality mode, then there's nothing worth
795                // persisting
796                removePersistedInputOriginEntries(document);
797                removePersistedOutputOriginEntries(document);
798                return;
799            }
800    
801            if (!correctionDocumentEntryMetadata.getDataLoadedFlag() && !correctionDocumentEntryMetadata.isRestrictedFunctionalityMode()) {
802                // data is not loaded (maybe user selected a new group with no rows)
803                // clear out existing data
804                removePersistedInputOriginEntries(document);
805                removePersistedOutputOriginEntries(document);
806                return;
807            }
808    
809            // reload the group from the origin entry service
810            Iterator<OriginEntryFull> inputGroupEntries; 
811            KualiWorkflowDocument workflowDocument = document.getDocumentHeader().getWorkflowDocument();
812            if ((workflowDocument.stateIsSaved() && !(correctionDocumentEntryMetadata.getInputGroupIdFromLastDocumentLoad() != null && correctionDocumentEntryMetadata.getInputGroupIdFromLastDocumentLoad().equals(document.getCorrectionInputFileName()))) || workflowDocument.stateIsInitiated()) {
813                // we haven't saved the origin entry group yet, so let's load the entries from the DB and persist them for the document
814                // this could be because we've previously saved the doc, but now we are now using a new input group, so we have to
815                // repersist the input group
816    
817    
818                // OriginEntryGroup group = originEntryGroupService.getExactMatchingEntryGroup(document.getCorrectionInputGroupId());
819                
820                File file = new File(document.getCorrectionInputFileName());
821                
822                inputGroupEntries = new OriginEntryFileIterator(file);
823                persistInputOriginEntriesForInitiatedOrSavedDocument(document, inputGroupEntries);
824    
825                // we've exhausted the iterator for the origin entries group
826                // reload the iterator from the file
827                inputGroupEntries = retrievePersistedInputOriginEntriesAsIterator(document);
828            }
829            else if (workflowDocument.stateIsSaved() && correctionDocumentEntryMetadata.getInputGroupIdFromLastDocumentLoad().equals(document.getCorrectionInputFileName())) {
830                // we've saved the origin entries before, so just retrieve them
831                inputGroupEntries = retrievePersistedInputOriginEntriesAsIterator(document);
832            }
833            else {
834                LOG.error("Unexpected state while trying to persist/retrieve GLCP origin entries during document save: document status is " + workflowDocument.getStatusDisplayValue() + " selected input group: " + document.getCorrectionInputFileName() + " last saved input group: " + correctionDocumentEntryMetadata.getInputGroupIdFromLastDocumentLoad());
835                throw new RuntimeException("Error persisting GLCP document origin entries.");
836            }
837    
838            OriginEntryStatistics statistics;
839            if (CorrectionDocumentService.CORRECTION_TYPE_MANUAL.equals(correctionDocumentEntryMetadata.getEditMethod())) {
840                // persist the allEntries element as the output group, since it has all of the modifications made by during the manual
841                // edits
842                persistOutputOriginEntriesForInitiatedOrSavedDocument(document, correctionDocumentEntryMetadata.getAllEntries().iterator());
843    
844                // even though the struts action handler may have computed the doc totals, let's recompute them
845                statistics = CorrectionDocumentUtils.getStatistics(correctionDocumentEntryMetadata.getAllEntries());
846            }
847            else if (CorrectionDocumentService.CORRECTION_TYPE_CRITERIA.equals(correctionDocumentEntryMetadata.getEditMethod())) {
848                // we want to persist the values of the output group. So reapply all of the criteria on each entry, one at a time
849    
850                BufferedOutputStream bufferedOutputStream = null;
851                try {
852                    bufferedOutputStream = openEntryOutputStreamForOutputGroup(document);
853                    statistics = new OriginEntryStatistics();
854                    byte[] newLine = "\n".getBytes();
855    
856                    while (inputGroupEntries.hasNext()) {
857                        OriginEntryFull entry = inputGroupEntries.next();
858    
859                        entry = CorrectionDocumentUtils.applyCriteriaToEntry(entry, correctionDocumentEntryMetadata.getMatchCriteriaOnly(), document.getCorrectionChangeGroup());
860                        if (entry != null) {
861                            CorrectionDocumentUtils.updateStatisticsWithEntry(entry, statistics);
862                            bufferedOutputStream.write(entry.getLine().getBytes());
863                            bufferedOutputStream.write(newLine);
864                        }
865                        // else it was null, which means that the match criteria only flag was set, and the entry didn't match the
866                        // criteria
867                    }
868                }
869                catch (IOException e) {
870                    LOG.error("Unable to persist persisted output entry", e);
871                    throw new RuntimeException("Unable to persist output entry");
872                }
873                finally {
874                    if (bufferedOutputStream != null) {
875                        try {
876                            bufferedOutputStream.close();
877                        }
878                        catch (IOException e) {
879                            LOG.error("Unable to close output stream for persisted output entries", e);
880                            throw new RuntimeException("Unable to close output entry file");
881                        }
882                    }
883                }
884            }
885            else if (CorrectionDocumentService.CORRECTION_TYPE_REMOVE_GROUP_FROM_PROCESSING.equals(correctionDocumentEntryMetadata.getEditMethod())) {
886                // just wipe out the previous output entries
887                removePersistedOutputOriginEntries(document);
888                statistics = new OriginEntryStatistics();
889            }
890            else {
891                throw new RuntimeException("Unrecognized edit method: " + correctionDocumentEntryMetadata.getEditMethod());
892            }
893    
894            CorrectionDocumentUtils.copyStatisticsToDocument(statistics, document);
895        }

1043        protected List<File> getReportsToAggregateIntoReport(String documentNumber) {
1044            File inputDirectory = new File(temporaryReportsDirectory);
1045            if (!inputDirectory.exists() || !inputDirectory.isDirectory()) {
1046                LOG.error(temporaryReportsDirectory + " does not exist or is not a directory.");
1047                throw new RuntimeException("Unable to locate temporary reports directory");
1048            }
1049            String filePrefix = documentNumber + "_" + temporaryReportFilenameComponent;
1050            FileFilter filter = FileFilterUtils.andFileFilter(
1051                    new PrefixFileFilter(filePrefix), new SuffixFileFilter(temporaryReportFilenameSuffix));
1052            
1053            // FSKD-244, KFSMI-5424 sort with filename, just in case 
1054            List<File> fileList = Arrays.asList(inputDirectory.listFiles(filter));
1055            
1056            Comparator fileNameComparator = new Comparator() {
1057                public int compare(Object obj1, Object obj2) {
1058                    if (obj1 == null) {
1059                        return -1;
1060                    }
1061                    if (obj2 == null) {
1062                        return 1;
1063                    }
1064                    File file1 = (File) obj1;
1065                    File file2 = (File) obj2;
1066                    
1067                    return ((Comparable) file1.getName()).compareTo(file2.getName());
1068                }
1069            };
1070            
1071            Collections.sort(fileList, fileNameComparator);
1072            return fileList ;
1073        }
