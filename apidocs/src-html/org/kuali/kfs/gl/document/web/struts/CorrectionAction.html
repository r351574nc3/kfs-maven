095        @Override
096        public ActionForward execute(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
097            LOG.debug("execute() started");
098    
099            CorrectionForm correctionForm = (CorrectionForm) form;
100    
101            // Init our services once
102            if (originEntryGroupService == null) {
103                CorrectionAction.originEntryGroupService = SpringContext.getBean(OriginEntryGroupService.class);
104                CorrectionAction.originEntryService = SpringContext.getBean(OriginEntryService.class);
105                CorrectionAction.dateTimeService = SpringContext.getBean(DateTimeService.class);
106                CorrectionAction.kualiConfigurationService = SpringContext.getBean(KualiConfigurationService.class);
107            }
108    
109            CorrectionForm rForm = (CorrectionForm) form;
110            LOG.debug("execute() methodToCall: " + rForm.getMethodToCall());
111    
112            Collection<OriginEntryFull> persistedOriginEntries = null;
113    
114            // If we are called from the docHandler or reload, ignore the persisted origin entries because we are either creating a new
115            // document or loading an old one
116            if (!(KFSConstants.DOC_HANDLER_METHOD.equals(rForm.getMethodToCall()) || KFSConstants.RELOAD_METHOD_TO_CALL.equals(rForm.getMethodToCall()))) {
117                restoreSystemAndEditMethod(rForm);
118                restoreInputGroupSelectionForDatabaseEdits(rForm);
119                if (!rForm.isRestrictedFunctionalityMode()) {
120                    if (StringUtils.isNotBlank(rForm.getGlcpSearchResultsSequenceNumber())) {
121                        rForm.setAllEntries(SpringContext.getBean(GlCorrectionProcessOriginEntryService.class).retrieveAllEntries(rForm.getGlcpSearchResultsSequenceNumber()));
122                        if (rForm.getAllEntries() == null) {
123                            rForm.setDisplayEntries(null);
124                        }
125                        else {
126                            rForm.setDisplayEntries(new ArrayList<OriginEntryFull>(rForm.getAllEntries()));
127                        }
128    
129                        if ((!"showOutputGroup".equals(rForm.getMethodToCall())) && rForm.getShowOutputFlag()) {
130                            // reapply the any criteria to pare down the list if the match criteria only flag is checked
131                            GeneralLedgerCorrectionProcessDocument document = rForm.getCorrectionDocument();
132                            List<CorrectionChangeGroup> groups = document.getCorrectionChangeGroup();
133                            updateEntriesFromCriteria(rForm, rForm.isRestrictedFunctionalityMode());
134                        }
135    
136                        if (!KFSConstants.TableRenderConstants.SORT_METHOD.equals(rForm.getMethodToCall())) {
137                            // if sorting, we'll let the action take care of the sorting
138                            KualiTableRenderFormMetadata originEntrySearchResultTableMetadata = rForm.getOriginEntrySearchResultTableMetadata();
139                            if (originEntrySearchResultTableMetadata.getPreviouslySortedColumnIndex() != -1) {
140                                List<Column> columns = SpringContext.getBean(CorrectionDocumentService.class).getTableRenderColumnMetadata(rForm.getDocument().getDocumentNumber());
141    
142                                String propertyToSortName = columns.get(originEntrySearchResultTableMetadata.getPreviouslySortedColumnIndex()).getPropertyName();
143                                Comparator valueComparator = columns.get(originEntrySearchResultTableMetadata.getPreviouslySortedColumnIndex()).getValueComparator();
144                                sortList(rForm.getDisplayEntries(), propertyToSortName, valueComparator, originEntrySearchResultTableMetadata.isSortDescending());
145                            }
146                            if (rForm.getAllEntries() != null) {
147                                int maxRowsPerPage = CorrectionDocumentUtils.getRecordsPerPage();
148                                originEntrySearchResultTableMetadata.jumpToPage(originEntrySearchResultTableMetadata.getViewedPageNumber(), rForm.getDisplayEntries().size(), maxRowsPerPage);
149                                originEntrySearchResultTableMetadata.setColumnToSortIndex(originEntrySearchResultTableMetadata.getPreviouslySortedColumnIndex());
150                            }
151                        }
152                    }
153                }
154            }
155    
156            ActionForward af = super.execute(mapping, form, request, response);
157            return af;
158        }

232        /**
233         * Prepare for routing. Return true if we're good to route, false if not
234         * 
235         * @param correctionForm
236         * @return
237         */
238        protected boolean prepareForRoute(CorrectionForm correctionForm) throws Exception {
239            GeneralLedgerCorrectionProcessDocument document = correctionForm.getCorrectionDocument();
240    
241            // Is there a description?
242            if (StringUtils.isEmpty(document.getDocumentHeader().getDocumentDescription())) {
243                GlobalVariables.getMessageMap().putError("document.documentHeader.documentDescription", KFSKeyConstants.ERROR_DOCUMENT_NO_DESCRIPTION);
244                return false;
245            }
246    
247            if (correctionForm.isPersistedOriginEntriesMissing()) {
248                GlobalVariables.getMessageMap().putError("searchResults", KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_PERSISTED_ORIGIN_ENTRIES_MISSING);
249                return false;
250            }
251    
252            // Did they pick the edit method and system?
253            if (!checkMainDropdown(correctionForm)) {
254                return false;
255            }
256    
257            if (correctionForm.getDataLoadedFlag() || correctionForm.isRestrictedFunctionalityMode()) {
258                document.setCorrectionInputFileName(correctionForm.getInputGroupId());
259            }
260            else {
261                document.setCorrectionInputFileName(null);
262            }
263            if (!checkOriginEntryGroupSelectionBeforeRouting(document)) {
264                return false;
265            }
266    
267            // were the system and edit methods inappropriately changed?
268            if (GlobalVariables.getMessageMap().containsMessageKey(KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_INVALID_SYSTEM_OR_EDIT_METHOD_CHANGE)) {
269                return false;
270            }
271    
272            // was the input group inappropriately changed?
273            if (GlobalVariables.getMessageMap().containsMessageKey(KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_INVALID_INPUT_GROUP_CHANGE)) {
274                return false;
275            }
276    
277            if (!validGroupsItemsForDocumentSave(correctionForm)) {
278                return false;
279            }
280    
281            // If it is criteria, are all the criteria valid?
282            if (CorrectionDocumentService.CORRECTION_TYPE_CRITERIA.equals(correctionForm.getEditMethod())) {
283                if (!validChangeGroups(correctionForm)) {
284                    return false;
285                }
286            }
287    
288            if (!checkRestrictedFunctionalityModeForManualEdit(correctionForm)) {
289                return false;
290            }
291    
292            if (!checkInputGroupPersistedForDocumentSave(correctionForm)) {
293                return false;
294            }
295            // Get the output group if necessary
296            if (CorrectionDocumentService.CORRECTION_TYPE_CRITERIA.equals(correctionForm.getEditMethod())) {
297                if (!correctionForm.isRestrictedFunctionalityMode() && correctionForm.getDataLoadedFlag() && !correctionForm.getShowOutputFlag()) {
298                    // we're going to force the user to view the output group upon routing, so apply the criteria
299                    // if the user wasn't in show output mode.
300                    updateEntriesFromCriteria(correctionForm, false);
301                }
302                correctionForm.setShowOutputFlag(true);
303            }
304            else {
305                // If it is manual edit, we don't need to save any correction groups
306                document.getCorrectionChangeGroup().clear();
307            }
308    
309            // Populate document
310            document.setCorrectionTypeCode(correctionForm.getEditMethod());
311            document.setCorrectionSelection(correctionForm.getMatchCriteriaOnly());
312            document.setCorrectionFileDelete(!correctionForm.getProcessInBatch());
313            document.setCorrectionInputFileName(correctionForm.getInputGroupId());
314            document.setCorrectionOutputFileName(null); // this field is never used
315    
316            // we'll populate the output group id when the doc has a route level change
317            document.setCorrectionOutputFileName(null);
318    
319            SpringContext.getBean(CorrectionDocumentService.class).persistOriginEntryGroupsForDocumentSave(document, correctionForm);
320    
321            return true;
322        }

384        /**
385         * Called when the document is loaded from action list or doc search or a new document is created.
386         * 
387         * @see org.kuali.rice.kns.web.struts.action.KualiDocumentActionBase#docHandler(org.apache.struts.action.ActionMapping,
388         *      org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
389         */
390        @Override
391        public ActionForward docHandler(ActionMapping mapping, ActionForm form, HttpServletRequest request, HttpServletResponse response) throws Exception {
392            LOG.debug("docHandler() started");
393    
394            CorrectionForm correctionForm = (CorrectionForm) form;
395            String command = correctionForm.getCommand();
396    
397            if (KEWConstants.INITIATE_COMMAND.equals(command)) {
398                correctionForm.clearForm();
399                createDocument(correctionForm);
400            }
401            else {
402                loadDocument(correctionForm);
403    
404                GeneralLedgerCorrectionProcessDocument document = correctionForm.getCorrectionDocument();
405                correctionForm.setInputGroupIdFromLastDocumentLoad(document.getCorrectionInputFileName());
406                populateAuthorizationFields(correctionForm);
407                Map<String, String> documentActions = correctionForm.getDocumentActions();
408                if (documentActions.containsKey(KNSConstants.KUALI_ACTION_CAN_EDIT)) {
409                    // They have saved the document and they are retreiving it to be completed
410                    correctionForm.setProcessInBatch(!document.getCorrectionFileDelete());
411                    correctionForm.setMatchCriteriaOnly(document.getCorrectionSelection());
412                    correctionForm.setEditMethod(document.getCorrectionTypeCode());
413                    
414                    if (document.getCorrectionInputFileName() != null) {
415                        if (CorrectionDocumentService.CORRECTION_TYPE_CRITERIA.equals(document.getCorrectionTypeCode())) {
416                            loadPersistedInputGroup(correctionForm);
417                            correctionForm.setDeleteFileFlag(false);
418                        }
419                        else if (CorrectionDocumentService.CORRECTION_TYPE_MANUAL.equals(document.getCorrectionTypeCode())) {
420                            // for the "true" param below, when the origin entries are persisted in the CorrectionDocumentService, they
421                            // // are likely
422                            // // not to have origin entry IDs assigned to them. So, we create pseudo entry IDs that are
423                            // // unique within the allEntries list, but not necessarily within the DB. The persistence layer
424                            // // is responsible for auto-incrementing entry IDs in the DB.
425                            loadPersistedOutputGroup(correctionForm, true);
426    
427                            correctionForm.setManualEditFlag(true);
428                            correctionForm.setEditableFlag(false);
429                            correctionForm.setDeleteFileFlag(false);
430                        }
431                        else if (CorrectionDocumentService.CORRECTION_TYPE_REMOVE_GROUP_FROM_PROCESSING.equals(document.getCorrectionTypeCode())) {
432                            loadPersistedInputGroup(correctionForm);
433                            correctionForm.setDeleteFileFlag(true);
434                        }
435                        else {
436                            throw new RuntimeException("Unknown edit method " + document.getCorrectionTypeCode());
437                        }
438                        correctionForm.setDataLoadedFlag(true);
439                    }
440                    else {
441                        correctionForm.setDataLoadedFlag(false);
442                    }
443                    correctionForm.setShowOutputFlag(false);
444                    correctionForm.setInputFileName(document.getCorrectionInputFileName());
445                    if (document.getCorrectionInputFileName() != null) {
446                        correctionForm.setChooseSystem(CorrectionDocumentService.SYSTEM_UPLOAD);
447                    }
448                    else {
449                        correctionForm.setChooseSystem(CorrectionDocumentService.SYSTEM_DATABASE);
450                    }
451    
452                    correctionForm.setPreviousChooseSystem(correctionForm.getChooseSystem());
453                    correctionForm.setPreviousEditMethod(correctionForm.getEditMethod());
454                    correctionForm.setPreviousInputGroupId(correctionForm.getInputGroupId());
455                }
456                else {
457                    // They are calling this from their action list to look at it or approve it
458                    correctionForm.setProcessInBatch(!document.getCorrectionFileDelete());
459                    correctionForm.setMatchCriteriaOnly(document.getCorrectionSelection());
460    
461                    // we don't care about setting entry IDs for the records below, so the param is false below
462                    loadPersistedOutputGroup(correctionForm, false);
463                    correctionForm.setShowOutputFlag(true);
464                }
465                correctionForm.setInputGroupIdFromLastDocumentLoadIsMissing(!originEntryGroupService.getGroupExists(document.getCorrectionInputFileName()));
466            }
467    
468            return mapping.findForward(KFSConstants.MAPPING_BASIC);
469        }

715            //creat file after all enries loaded well

720            String fullFileName = glcpDirectory + File.separator + sourceFile.getFileName() + "-" + CorrectionAction.dateTimeService.toDateTimeStringForFilename(now);
721            
722            BufferedReader br = new BufferedReader(new InputStreamReader(sourceFile.getInputStream()));
723            //create a file
724            File uploadedFile = new File(fullFileName);
725            PrintStream uploadedFilePrintStream;
726            try {
727                uploadedFilePrintStream = new PrintStream(uploadedFile);
728            } catch (IOException e) {
729                throw new RuntimeException(e);
730            }
731            
732            //write entries to file
733            int loadedCount = 0;
734            String stringLine;
735            while ((stringLine= br.readLine()) != null){
736                try {
737                    uploadedFilePrintStream.printf("%s\n", stringLine);
738                    loadedCount++;
739                } catch (Exception e) {
740                    throw new IOException(e.toString());
741                }
742            }
743            uploadedFilePrintStream.close();
744            
745            
746            int recordCountFunctionalityLimit = CorrectionDocumentUtils.getRecordCountFunctionalityLimit();
747            if (CorrectionDocumentUtils.isRestrictedFunctionalityMode(loadedCount, recordCountFunctionalityLimit)) {
748                correctionForm.setRestrictedFunctionalityMode(true);
749                correctionForm.setDataLoadedFlag(false);
750                document.setCorrectionInputFileName(fullFileName);
751                correctionForm.setInputFileName(fullFileName);
752    
753                if (CorrectionDocumentService.CORRECTION_TYPE_MANUAL.equals(correctionForm.getEditMethod())) {
754                    // the group size is not suitable for manual editing because it is too large
755                    if (recordCountFunctionalityLimit == CorrectionDocumentUtils.RECORD_COUNT_FUNCTIONALITY_LIMIT_IS_NONE) {
756                        GlobalVariables.getMessageMap().putError(SYSTEM_AND_EDIT_METHOD_ERROR_KEY, KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_UNABLE_TO_MANUAL_EDIT_ANY_GROUP);
757                    }
758                    else {
759                        GlobalVariables.getMessageMap().putError(SYSTEM_AND_EDIT_METHOD_ERROR_KEY, KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_UNABLE_TO_MANUAL_EDIT_LARGE_GROUP, String.valueOf(recordCountFunctionalityLimit));
760                    }
761                }
762            }
763            else {
764                correctionForm.setRestrictedFunctionalityMode(false);
765                if (loadedCount > 0) {
766                    //now we can load all data from file
767                    List<OriginEntryFull> originEntryList = new ArrayList();
768                    Map loadErrorMap = originEntryService.getEntriesByGroupIdWithPath(uploadedFile.getAbsolutePath(), originEntryList);
769                    //put errors on GlobalVariables
770                    if (loadErrorMap.size() > 0){
771                        Iterator iter = loadErrorMap.keySet().iterator();
772                        while(iter.hasNext()){
773                            Integer lineNumber = (Integer) iter.next();
774                            List<Message> messageList = (List<Message>) loadErrorMap.get(lineNumber);
775                            if (messageList.size() > 0){
776                                for (Message errorMmessage : messageList){
777                                    GlobalVariables.getMessageMap().putError("fileUpload", KFSKeyConstants.ERROR_INVALID_FORMAT_ORIGIN_ENTRY_FROM_TEXT_FILE, new String[] {lineNumber.toString(), errorMmessage.toString()});
778                                }    
779                            }
780                        }
781                        return mapping.findForward(KFSConstants.MAPPING_BASIC);
782                    }
783                    
784                    // Set all the data that we know
785                    correctionForm.setDataLoadedFlag(true);
786                    correctionForm.setInputFileName(fullFileName);
787                    document.setCorrectionInputFileName(fullFileName);
788                    loadAllEntries(originEntryList, correctionForm);
789    
790                    if (CorrectionDocumentService.CORRECTION_TYPE_MANUAL.equals(correctionForm.getEditMethod())) {
791                        correctionForm.setEditableFlag(false);
792                        correctionForm.setManualEditFlag(true);
793                    }
794                }
795                else {
796                    GlobalVariables.getMessageMap().putError("fileUpload", KFSKeyConstants.ERROR_GL_ERROR_CORRECTION_NO_RECORDS);
797                }
798            }
799    
800            if (document.getCorrectionChangeGroup().isEmpty()) {
801                document.addCorrectionChangeGroup(new CorrectionChangeGroup());
802            }
803            return mapping.findForward(KFSConstants.MAPPING_BASIC);
804        }

1599        /**
1600         * For criteria based edits, this method will generate the output group
1601         * 
1602         * @param entries a Collection of OriginEntryFull BOs, this collection and its elements may be directly modified as a result of
1603         *        this method
1604         * @param matchCriteriaOnly if true, only those entries that matched the criteria before changes were applied will remain in the
1605         *        collection
1606         * @param changeCriteriaGroups a list of criteria and change groups.
1607         */
1608        protected void applyCriteriaOnEntries(Collection<OriginEntryFull> entries, boolean matchCriteriaOnly, List<CorrectionChangeGroup> changeCriteriaGroups) {
1609            // Now, if they only want matches in the output group, go through them again and delete items that don't match any of the
1610            // groups
1611            // This means that matches within a group are ANDed and each group is ORed
1612            if (matchCriteriaOnly) {
1613                removeNonMatchingEntries(entries, changeCriteriaGroups);
1614            }
1615    
1616            for (OriginEntryFull oe : entries) {
1617                for (CorrectionChangeGroup ccg : changeCriteriaGroups) {
1618                    int matches = 0;
1619                    for (CorrectionCriteria cc : ccg.getCorrectionCriteria()) {
1620                        if (CorrectionDocumentUtils.entryMatchesCriteria(cc, oe)) {
1621                            matches++;
1622                        }
1623                    }
1624    
1625                    // If they all match, change it
1626                    if (matches == ccg.getCorrectionCriteria().size()) {

1691        protected void sortList(List<OriginEntryFull> list, String propertyToSortName, Comparator valueComparator, boolean sortDescending) {
1692            if (list != null) {
1693                if (valueComparator instanceof NumericValueComparator || valueComparator instanceof TemporalValueComparator) {
1694                    // hack alert: NumericValueComparator can only compare strings, so we use the KualiDecimal and Date's built in
1695                    // mechanism compare values using
1696                    // the comparable comparator
1697                    valueComparator = new Comparator() {
1698                        public int compare(Object obj1, Object obj2) {
1699                            if (obj1 == null) {
1700                                return -1;
1701                            }
1702                            if (obj2 == null) {
1703                                return 1;
1704                            }
1705                            return ((Comparable) obj1).compareTo(obj2);
1706                        }
1707                    };
1708                }
1709                if (sortDescending) {
1710                    valueComparator = new ReverseComparator(valueComparator);
1711                }
1712                Collections.sort(list, new BeanComparator(propertyToSortName, valueComparator));
1713            }
