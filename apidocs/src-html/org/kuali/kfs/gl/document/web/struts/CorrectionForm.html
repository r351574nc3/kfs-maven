135        /**
136         * @see org.kuali.rice.kns.web.struts.form.KualiDocumentFormBase#populate(javax.servlet.http.HttpServletRequest)
137         */
138        @Override
139        public void populate(HttpServletRequest request) {
140            super.populate(request);
141    
142            // Sync up the groups
143            syncGroups();
144    
145            originEntrySearchResultTableMetadata = new KualiTableRenderFormMetadata();
146    
147            if (KFSConstants.TableRenderConstants.SWITCH_TO_PAGE_METHOD.equals(getMethodToCall())) {
148                // look for the page number to switch to
149                originEntrySearchResultTableMetadata.setSwitchToPageNumber(-1);
150    
151                // the param we're looking for looks like: methodToCall.switchToPage.1.x , where 1 is the page nbr
152                String paramPrefix = KFSConstants.DISPATCH_REQUEST_PARAMETER + "." + KFSConstants.TableRenderConstants.SWITCH_TO_PAGE_METHOD + ".";
153                for (Enumeration i = request.getParameterNames(); i.hasMoreElements();) {
154                    String parameterName = (String) i.nextElement();
155                    if (parameterName.startsWith(paramPrefix)) {
156                        String switchToPageNumberStr = StringUtils.substringBetween(parameterName, paramPrefix, ".");
157                        originEntrySearchResultTableMetadata.setSwitchToPageNumber(Integer.parseInt(switchToPageNumberStr));
158                    }
159                }
160                if (originEntrySearchResultTableMetadata.getSwitchToPageNumber() == -1) {
161                    throw new RuntimeException("Couldn't find page number");
162                }
163            }
164    
165            if (KFSConstants.TableRenderConstants.SORT_METHOD.equals(getMethodToCall())) {
166                originEntrySearchResultTableMetadata.setColumnToSortIndex(-1);
167    
168                // the param we're looking for looks like: methodToCall.sort.1.x , where 1 is the column to sort on
169                String paramPrefix = KFSConstants.DISPATCH_REQUEST_PARAMETER + "." + KFSConstants.TableRenderConstants.SORT_METHOD + ".";
170                for (Enumeration i = request.getParameterNames(); i.hasMoreElements();) {
171                    String parameterName = (String) i.nextElement();
172                    if (parameterName.startsWith(paramPrefix) && parameterName.endsWith(".x")) {
173                        String columnToSortStr = StringUtils.substringBetween(parameterName, paramPrefix, ".");
174                        originEntrySearchResultTableMetadata.setColumnToSortIndex(Integer.parseInt(columnToSortStr));
175                    }
176                }
177                if (originEntrySearchResultTableMetadata.getColumnToSortIndex() == -1) {
178                    throw new RuntimeException("Couldn't find column to sort");
179                }
180            }
181            
182            // since the processInBatch option defaults to true, there's no built in POJO way to detect whether it's been unchecked
183            // this code takes care of that
184            if (StringUtils.isNotBlank(request.getParameter("processInBatch" + KFSConstants.CHECKBOX_PRESENT_ON_FORM_ANNOTATION)) && StringUtils.isBlank(request.getParameter("processInBatch"))) {
185                setProcessInBatch(false);
186            }
187            
188            if (StringUtils.isNotBlank(request.getParameter("matchCriteriaOnly" + KFSConstants.CHECKBOX_PRESENT_ON_FORM_ANNOTATION)) && StringUtils.isBlank(request.getParameter("matchCriteriaOnly"))) {
189                setMatchCriteriaOnly(false);
190            }
191        }

193        /**
194         * This method synchronizes number of group holders added with the group count
195         * 
196         */
197        public void syncGroups() {
198            int groupCount = getCorrectionDocument().getCorrectionChangeGroup().size();
199            getGroupsItem(groupCount);
200        }

205         * @return

208            return groups.size();
209        }

211        /**
212         * Returns group item with given ID
213         * 
214         * @param i index of group
215         * @return
216         */
217        public GroupHolder getGroupsItem(int i) {
218            while (i >= groups.size()) {
219                groups.add(new GroupHolder());
220            }
221            return groups.get(i);
222        }

224        /**
225         * Clears correction document form
226         */
227        public void clearForm() {
228            chooseSystem = "";
229            editMethod = "";
230            inputFileName = "";
231            outputGroupId = null;
232            processInBatch = true;
233            matchCriteriaOnly = false;
234            dataLoadedFlag = false;
235            editableFlag = false;
236            manualEditFlag = false;
237            deleteFileFlag = false;
238            showOutputFlag = false;
239            allEntries = new ArrayList<OriginEntryFull>();
240            displayEntries = new ArrayList<OriginEntryFull>();
241    
242            restrictedFunctionalityMode = false;
243    
244            setDocument(null);
245            instantiateDocument();
246            
247            // These are for the blank rows that are used to add criteria/changes
248            groups = new ArrayList<GroupHolder>();
249    
250            inputGroupIdFromLastDocumentLoad = null;
251            inputGroupIdFromLastDocumentLoadIsMissing = false;
252            persistedOriginEntriesMissing = false;
253    
254            // Sync up the groups
255            syncGroups();
256    
257            entryForManualEdit = new OriginEntryFull();
258            entryForManualEdit.setEntryId(0);
259        }

261        /**
262         * Update origin entry for manual edit with correction document attributes (i.e. university fiscal year, entry transaction ledger 
263         * sequence number, entry transaction ledger entry amount, entry transaction date, entry financial document reversal date)
264         */
265        public void updateEntryForManualEdit() {
266            entryForManualEdit.setFieldValue("universityFiscalYear", getEntryUniversityFiscalYear());
267            entryForManualEdit.setFieldValue("transactionLedgerEntrySequenceNumber", getEntryTransactionLedgerEntrySequenceNumber());
268            entryForManualEdit.setFieldValue("transactionLedgerEntryAmount", getEntryTransactionLedgerEntryAmount());
269            entryForManualEdit.setFieldValue("transactionDate", getEntryTransactionDate());
270            entryForManualEdit.setFieldValue("financialDocumentReversalDate", getEntryFinancialDocumentReversalDate());
271        }

273        /**
274         * Clears origin entry for manual edit
275         * 
276         */
277        public void clearEntryForManualEdit() {
278            OriginEntryFull oe = new OriginEntryFull();
279            oe.setEntryId(0);
280            oe.setSubAccountNumber("");
281            oe.setFinancialSubObjectCode("");
282            oe.setProjectCode("");
283            setEntryFinancialDocumentReversalDate("");
284            setEntryTransactionDate("");
285            setEntryTransactionLedgerEntryAmount("");
286            setEntryTransactionLedgerEntrySequenceNumber("");
287            setEntryUniversityFiscalYear("");
288            setEntryForManualEdit(oe);
289        }
