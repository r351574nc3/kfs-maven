016    package org.kuali.kfs.gl.dataaccess;
017    
018    import java.util.List;
019    
020    import org.kuali.kfs.gl.businessobject.CorrectionChange;
021    
022    /**
023     * A DAO interface for CorrectionChange business objects to interact with the databse
024     */
025    public interface CorrectionChangeDao {
026        /**
027         * This method saves a GLCP correction
028         * 
029         * @param spec the GLCP correction to save in the database
030         */
031        void save(CorrectionChange spec);
032    
033        /**
034         * Surprisingly, this method deletes a GLCP correction
035         * 
036         * @param spec the GLCP correction to delete
037         */
038        void delete(CorrectionChange spec);
039    
040        /**
041         * Finds CorrectionChanges associated with the given document and group
042         * 
043         * @param documentHeaderId the document number of a GLCP document
044         * @param correctionGroupLineNumber the line number of the group within the GLCP document to find correction chagnes for
045         * @return a List of correction changes
046         */
047        List findByDocumentHeaderIdAndCorrectionGroupNumber(String documentHeaderId, Integer correctionGroupLineNumber);
048    }
