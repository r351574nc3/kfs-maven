016    package org.kuali.kfs.gl.dataaccess;
017    
018    import java.util.Iterator;
019    
020    import org.kuali.kfs.gl.businessobject.ExpenditureTransaction;
021    import org.kuali.kfs.gl.businessobject.Transaction;
022    
023    /**
024     * An DAO interface to deal help expenditure transactions to deal with the database 
025     */
026    public interface ExpenditureTransactionDao {
027        /**
028         * Returns the expenditure transaction in the database that would be affected if the given transaction is posted
029         * 
030         * @param t a transaction to find a related expenditure transaction for
031         * @return the expenditure transaction if found, null otherwise
032         */
033        public ExpenditureTransaction getByTransaction(Transaction t);
034    
035        /**
036         * Returns all expenditure transactions currently in the database
037         * 
038         * @return an Iterator with all expenditure transactions from the database
039         */
040        public Iterator getAllExpenditureTransactions();
041    
042        /**
043         * Deletes the given expenditure transaction
044         * 
045         * @param et the expenditure transaction that will be removed, as such, from the database
046         */
047        public void delete(ExpenditureTransaction et);
048    
049        /**
050         * Saves an expenditure transaction
051         * @param et the expenditure transaction to save
052         */
053        public void save(ExpenditureTransaction et);
054    
055        /**
056         * Since expenditure transactions are temporary, this method removes all of the currently existing
057         * expenditure transactions from the database
058         */
059        public void deleteAllExpenditureTransactions();
060    }
