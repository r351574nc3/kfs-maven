016    package org.kuali.kfs.gl.web.struts;
017    
018    import java.util.Collection;
019    import java.util.List;
020    
021    import org.kuali.rice.kns.web.struts.form.LookupForm;
022    import org.kuali.rice.kns.web.ui.ResultRow;
023    
024    /**
025     * This class serves as the struts action for implementing multiple value lookups
026     */
027    public interface LookupDisplayTagSurrogate {
028    
029        /**
030         * This method performs the lookup and returns a collection of lookup items. Also initializes values in the form that will allow
031         * the multiple value lookup page to render
032         * 
033         * @param selectable <code>{@link LookupResultsSelectable}</code> since checkboxes are used to select multiple things
034         * @param form an instance of <code>{@link LookupForm}</code>
035         * @param resultTable a list of result rows (used to generate what's shown in the UI). This list will be modified by this method
036         * @param maxRowsPerPage
037         * @param bounded whether the results will be bounded
038         * @return the list of result BOs, possibly qbounded by size
039         */
040        public Collection performMultipleValueLookup(LookupResultsSelectable selectable, LookupForm form, List<ResultRow> resultTable, boolean bounded);
041    
042        /**
043         * This method performs the operations necessary for a multiple value lookup to switch to another page of results and rerender
044         * the page
045         * 
046         * @param multipleValueLookupForm
047         * @param maxRowsPerPage
048         * @return a list of result rows, used by the UI to render the page
049         */
050        public List<ResultRow> switchToPage(LookupResultsSelectable selectable, int maxRowsPerPage);
051    
052        /**
053         * This method performs the operations necessary for a multiple value lookup to sort results and rerender the page
054         * 
055         * @param multipleValueLookupForm
056         * @param maxRowsPerPage
057         * @return a list of result rows, used by the UI to render the page
058         */
059        public List<ResultRow> sort(LookupResultsSelectable selectable, int maxRowsPerPage);
060    
061        /**
062         * This method performs the operations necessary for a multiple value lookup keep track of which results have been selected to
063         * be returned to the calling document. Note, this method does not actually requery for the results.
064         * 
065         * @param multipleValueLookupForm
066         */
067        public void prepareToReturnSelectedResultBOs(LookupResultsSelectable selectable);
068    
069        /**
070         * This method performs the operations necessary for a multiple value lookup to return no results to the calling page
071         * 
072         * @param multipleValueLookupForm
073         */
074        public void prepareToReturnNone(LookupResultsSelectable selectable);
075    
076        /**
077         * This method performs the operations necessary for a multiple value lookup to export the rows via display tag Note: this
078         * method assumes that the export will be opened in a new browser window, therefore, persisting the selected checkboxes will not
079         * be needed.
080         * 
081         * @param multipleValueLookupForm
082         * @return a list of result rows, to be used by display tag to render the results
083         */
084        public List<ResultRow> prepareToExport(LookupResultsSelectable selectable);
085    
086    
087        /**
088         * This method performs the operations necessary for a multiple value lookup to select all of the results and rerender the page
089         * 
090         * @param multipleValueLookupForm
091         * @param maxRowsPerPage
092         * @return a list of result rows, used by the UI to render the page
093         */
094        public List<ResultRow> selectAll(LookupResultsSelectable selectable, int maxRowsPerPage);
095    
096        /**
097         * This method performs the operations necessary for a multiple value lookup to unselect all of the results and rerender the
098         * page
099         * 
100         * @param multipleValueLookupForm
101         * @param maxRowsPerPage
102         * @return a list of result rows, used by the UI to render the page
103         */
104        public List<ResultRow> unselectAll(LookupResultsSelectable selectable, int maxRowsPerPage);
105    
106        /**
107         * This method computes the max number of rows that should be rendered per page for a multiple value lookup. This method first
108         * looks for an application parameter in FS_PARM_T, group SYSTEM, multipleValueLookupResultsPerPage if someone wants to
109         * implement something where a user can decide how many results to display per page, this method is the place to do it. Make
110         * this method read form values to determine the max rows per page based on the user inputs
111         * 
112         * @see org.kuali.kfs.sys.KFSConstants.SystemGroupParameterNames#MULTIPLE_VALUE_LOOKUP_RESULTS_PER_PAGE
113         * @see #DEFAULT_MAX_ROWS_PER_PAGE
114         * @param multipleValueLookupForm the form
115         * @return
116         */
117        public int getMaxRowsPerPage(LookupResultsSelectable selectable);
118    }
