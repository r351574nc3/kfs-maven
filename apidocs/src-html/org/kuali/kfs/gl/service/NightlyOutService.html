016    package org.kuali.kfs.gl.service;
017    
018    /**
019     * This interface defines the batch jobs that would be run nightly against the pending general ledger entries. These jobs are
020     * performed to ensure the correctness of the entries. As to date, the jobs involve two functions: finding unbalanced approved
021     * documents and finding bad approved documents.
022     */
023    public interface NightlyOutService {
024    
025        /**
026         * Deletes all the records that were copied to the GL.
027         */
028        public void deleteCopiedPendingLedgerEntries();
029    
030        /**
031         * Copies the approved pending ledger entries to orign entry table
032         */
033        public void copyApprovedPendingLedgerEntries();
034    }
