016    package org.kuali.kfs.gl.report;
017    
018    import org.kuali.kfs.gl.businessobject.Transaction;
019    import org.kuali.kfs.sys.KFSConstants;
020    import org.kuali.kfs.sys.KFSPropertyConstants;
021    import org.kuali.kfs.sys.report.BusinessObjectReportHelper;
022    import org.kuali.rice.kns.bo.BusinessObject;
023    import org.kuali.rice.kns.util.KualiDecimal;
024    import org.kuali.rice.kns.util.ObjectUtils;
025    
026    public class TransactionListingReportBusinessObjectReportHelper extends BusinessObjectReportHelper {
027        protected static final int TRANSACTION_LEDGER_ENTRY_DESCRIPTION_MAX_LENGTH = 31;
028        
029        /**
030         * @see org.kuali.kfs.sys.report.BusinessObjectReportHelper#retrievePropertyValue(org.kuali.rice.kns.bo.BusinessObject, java.lang.String)
031         */
032        @Override
033        protected Object retrievePropertyValue(BusinessObject businessObject, String propertyName) {
034            if ("debitOrBudgetAmount".equals(propertyName) && businessObject instanceof Transaction) {
035                Transaction e = (Transaction) businessObject;
036                if (KFSConstants.GL_DEBIT_CODE.equals(e.getTransactionDebitCreditCode())) {
037                    // return the debit amount
038                    return e.getTransactionLedgerEntryAmount();
039                } else if (!KFSConstants.GL_DEBIT_CODE.equals(e.getTransactionDebitCreditCode()) &&
040                        !KFSConstants.GL_CREDIT_CODE.equals(e.getTransactionDebitCreditCode())) {
041                    // return the budget amount
042                    return e.getTransactionLedgerEntryAmount();
043                }
044                return KualiDecimal.ZERO;
045            }
046            if ("creditAmount".equals(propertyName) && businessObject instanceof Transaction) {
047                Transaction e = (Transaction) businessObject;
048                if (KFSConstants.GL_CREDIT_CODE.equals(e.getTransactionDebitCreditCode())) {
049                    return e.getTransactionLedgerEntryAmount();
050                }
051                return KualiDecimal.ZERO;
052            }
053            // Truncate the description
054            if ("transactionLedgerEntryDescription".equals(propertyName) && businessObject instanceof Transaction) {
055                Transaction e = (Transaction) businessObject;
056                if (ObjectUtils.isNull(e.getTransactionLedgerEntryDescription()) ||
057                        (e.getTransactionLedgerEntryDescription().length() <= TRANSACTION_LEDGER_ENTRY_DESCRIPTION_MAX_LENGTH)) {
058                    return e.getTransactionLedgerEntryDescription();
059                } else {
060                    return e.getTransactionLedgerEntryDescription().substring(0, TRANSACTION_LEDGER_ENTRY_DESCRIPTION_MAX_LENGTH);
061                }
062            }
063            return super.retrievePropertyValue(businessObject, propertyName);
064        }
065    
066        /**
067         * @see org.kuali.kfs.sys.report.BusinessObjectReportHelper#retrievePropertyValueMaximumLength(java.lang.Class, java.lang.String)
068         */
069        @Override
070        protected int retrievePropertyValueMaximumLength(Class<? extends BusinessObject> businessObjectClass, String propertyName) {
071            if ("debitOrBudgetAmount".equals(propertyName) || "creditAmount".equals(propertyName)) {
072                return super.retrievePropertyValueMaximumLength(businessObjectClass, KFSPropertyConstants.TRANSACTION_LEDGER_ENTRY_AMOUNT);
073            }
074            if ("transactionLedgerEntryDescription".equals(propertyName)) {
075                return TRANSACTION_LEDGER_ENTRY_DESCRIPTION_MAX_LENGTH;
076            }
077            return super.retrievePropertyValueMaximumLength(businessObjectClass, propertyName);
078        }
079    
080    }
