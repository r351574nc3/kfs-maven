016    package org.kuali.kfs.gl.report;
017    
018    import java.util.Set;
019    
020    import org.kuali.kfs.sys.service.ReportWriterService;
021    
022    public class PreScrubberReport {
023        public void generateReport(PreScrubberReportData preScrubberReportData, ReportWriterService reportWriterService) {
024            reportWriterService.writeFormattedMessageLine("Origin Entries In %,20d", preScrubberReportData.getInputRecords());
025            reportWriterService.writeFormattedMessageLine("Origin Entries Out%,20d", preScrubberReportData.getOutputRecords());
026            if (!preScrubberReportData.getAccountsWithNoCharts().isEmpty()) {
027                reportWriterService.writeNewLines(2);
028                reportWriterService.writeFormattedMessageLine("The following account numbers were not associated with any chart of accounts code:");
029                for (String accountNumber : preScrubberReportData.getAccountsWithNoCharts()) {
030                    reportWriterService.writeFormattedMessageLine("%15s", accountNumber);
031                }
032            }
033            if (!preScrubberReportData.getAccountsWithMultipleCharts().isEmpty()) {
034                reportWriterService.writeNewLines(2);
035                reportWriterService.writeFormattedMessageLine("The following account numbers were associated with multiple chart of accounts codes:");
036                for (String accountNumber : preScrubberReportData.getAccountsWithMultipleCharts()) {
037                    reportWriterService.writeFormattedMessageLine("%15s", accountNumber);
038                }
039            }
040        }
041    }
